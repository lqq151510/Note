# Spring AI Function Calling 完整实现

Spring AI 支持 OpenAI 的 **Function Calling**​ 功能，允许 AI 模型调用外部函数/工具。以下是完整的实现方案。

## 一、基础概念

Function Calling 的工作流程：

1. 定义函数（工具）的描述
    
2. 模型决定是否需要调用函数
    
3. 调用相应的外部函数
    
4. 将结果返回给模型继续处理
    
5. 模型生成最终回复

### 1. 定义 Function Bean

```
import org.springframework.ai.model.function.FunctionCallback;
import org.springframework.ai.model.function.FunctionCallbackContext;
import org.springframework.context.annotation.Bean;
import org.springframework.context.annotation.Configuration;
import org.springframework.context.annotation.Description;
import org.springframework.stereotype.Component;
import java.util.function.Function;

@Component
public class FunctionDefinitions {
    
    // 1. 天气查询函数
    @Bean
    public FunctionCallback weatherFunction() {
        return FunctionCallback.builder()
            .name("getCurrentWeather")  // 函数名，与描述中的 function.name 对应
            .description("获取指定城市的当前天气")  // 函数描述
            .responseConverter((response) -> "" + response)  // 可选：响应转换器
            .function(new Function<WeatherRequest, WeatherResponse>() {
                @Override
                public WeatherResponse apply(WeatherRequest request) {
                    return getCurrentWeather(request.location(), request.unit());
                }
            })
            .build();
    }
    
    // 2. 计算器函数
    @Bean
    @Description("执行数学计算，支持加、减、乘、除")  // 通过注解添加描述
    public Function<CalculatorRequest, CalculatorResponse> calculatorFunction() {
        return request -> {
            double result = 0;
            switch (request.operation()) {
                case "add":
                    result = request.a() + request.b();
                    break;
                case "subtract":
                    result = request.a() - request.b();
                    break;
                case "multiply":
                    result = request.a() * request.b();
                    break;
                case "divide":
                    if (request.b() == 0) {
                        throw new IllegalArgumentException("除数不能为0");
                    }
                    result = request.a() / request.b();
                    break;
                default:
                    throw new IllegalArgumentException("不支持的操作: " + request.operation());
            }
            return new CalculatorResponse(result);
        };
    }
    
    // 3. 数据库查询函数
    @Bean
    public FunctionCallback databaseQueryFunction(FunctionCallbackContext context) {
        return FunctionCallback.builder()
            .name("queryDatabase")
            .description("查询数据库获取信息，支持用户、订单、产品等")
            .function(context.getBean(DatabaseService.class)::query)
            .build();
    }
    
    // 4. 外部API调用函数
    @Bean
    public Function<StockPriceRequest, StockPriceResponse> stockPriceFunction() {
        return request -> {
            // 调用外部API获取股票价格
            String apiUrl = String.format("https://api.example.com/stock/%s", request.symbol());
            // 实现HTTP调用...
            return new StockPriceResponse(/* 价格数据 */);
        };
    }
    
    // 5. 系统工具函数
    @Bean
    @Description("获取当前系统时间和日期")
    public Function<SystemInfoRequest, SystemInfoResponse> systemInfoFunction() {
        return request -> {
            LocalDateTime now = LocalDateTime.now();
            return new SystemInfoResponse(
                now.toString(),
                System.getProperty("os.name"),
                Runtime.getRuntime().availableProcessors()
            );
        };
    }
    
    // 天气查询实现
    private WeatherResponse getCurrentWeather(String location, String unit) {
        // 模拟天气查询
        WeatherResponse response = new WeatherResponse();
        response.setLocation(location);
        response.setTemperature(25.0);
        response.setUnit("celsius".equals(unit) ? "°C" : "°F");
        response.setDescription("晴朗");
        response.setHumidity(65);
        response.setWindSpeed(10.5);
        return response;
    }
}
```

## 四、使用 Function Calling

### 1. 基本使用方式

```
import org.springframework.ai.chat.client.ChatClient;
import org.springframework.ai.chat.model.ChatResponse;
import org.springframework.ai.chat.model.Generation;
import org.springframework.ai.chat.prompt.Prompt;
import org.springframework.ai.model.function.FunctionCallbackWrapper;
import org.springframework.stereotype.Service;
import java.util.List;
import java.util.Map;

@Service
public class FunctionCallingService {
    
    private final ChatClient chatClient;
    private final List<FunctionCallback> functionCallbacks;
    
    public FunctionCallingService(ChatClient.Builder chatClientBuilder,
                                 List<FunctionCallback> functionCallbacks) {
        this.functionCallbacks = functionCallbacks;
        
        // 构建带Function Calling的ChatClient
        this.chatClient = chatClientBuilder
            .defaultFunctions("getCurrentWeather", "calculatorFunction")  // 默认启用的函数
            .defaultSystem("你是一个有用的助手，可以调用各种工具来帮助用户。")
            .build();
    }
    
    /**
     * 方式1：通过ChatClient自动调用
     */
    public String chatWithFunctions(String userMessage) {
        return chatClient.prompt()
            .user(userMessage)
            .call()
            .content();
    }
    
    /**
     * 方式2：手动处理Function Calling流程
     */
    public ChatResult handleFunctionCall(String userMessage) {
        // 创建初始Prompt
        Prompt prompt = new Prompt(userMessage);
        
        // 调用ChatClient
        ChatResponse response = chatClient.prompt(prompt)
            .call()
            .chatResponse();
        
        Generation generation = response.getResult();
        
        // 检查是否有Function Call
        if (generation.getToolCalls() != null && !generation.getToolCalls().isEmpty()) {
            return handleToolCalls(generation, userMessage);
        }
        
        // 没有Function Call，直接返回
        return new ChatResult(generation.getOutput().getContent());
    }
    
    /**
     * 处理工具调用
     */
    private ChatResult handleToolCalls(Generation generation, String originalMessage) {
        List<ToolCall> toolCalls = generation.getToolCalls();
        List<ToolResponse> toolResponses = new ArrayList<>();
        
        // 处理每个工具调用
        for (ToolCall toolCall : toolCalls) {
            String functionName = toolCall.getFunction().getName();
            Map<String, Object> arguments = toolCall.getFunction().getArguments();
            
            // 查找对应的函数
            Optional<FunctionCallback> functionOpt = functionCallbacks.stream()
                .filter(fc -> fc.getName().equals(functionName))
                .findFirst();
            
            if (functionOpt.isPresent()) {
                try {
                    // 执行函数
                    Object result = functionOpt.get().call(arguments);
                    toolResponses.add(new ToolResponse(toolCall.getId(), result));
                } catch (Exception e) {
                    toolResponses.add(new ToolResponse(toolCall.getId(), 
                        "调用函数出错: " + e.getMessage()));
                }
            } else {
                toolResponses.add(new ToolResponse(toolCall.getId(), 
                    "未找到函数: " + functionName));
            }
        }
        
        // 将函数执行结果返回给AI
        ChatResponse finalResponse = chatClient.prompt()
            .user(originalMessage)
            .advisors(advisor -> advisor
                .preProcess(prompt -> {
                    // 添加工具调用的上下文
                    prompt.getInstructions().addAll(toolResponses.stream()
                        .map(tr -> new SystemMessage("函数调用结果: " + tr.content()))
                        .collect(Collectors.toList()));
                    return prompt;
                }))
            .call()
            .chatResponse();
        
        String finalContent = finalResponse.getResult().getOutput().getContent();
        return new ChatResult(finalContent, toolResponses);
    }
    
    /**
     * 方式3：流式Function Calling
     */
    public Flux<String> streamWithFunctions(String userMessage) {
        return chatClient.prompt()
            .user(userMessage)
            .stream()
            .content()
            .concatMap(chunk -> {
                // 处理流式输出，可以在这里检测Function Call
                return Flux.just(chunk);
            });
    }
}
```

```
@Component
public class ContextAwareFunctionService {
    
    private final ChatClient chatClient;
    private final UserSessionService sessionService;
    
    // 带用户上下文的函数
    @Bean
    public FunctionCallback getUserInfoFunction(UserSessionService sessionService) {
        return FunctionCallback.builder()
            .name("getUserInfo")
            .description("获取当前用户的基本信息")
            .function(new Function<Map<String, Object>, Map<String, Object>>() {
                @Override
                public Map<String, Object> apply(Map<String, Object> request) {
                    String userId = (String) request.getOrDefault("userId", "anonymous");
                    return sessionService.getUserInfo(userId);
                }
            })
            .build();
    }
    
    // 带数据库访问的函数
    @Bean
    public FunctionCallback queryOrdersFunction(OrderRepository orderRepository) {
        return FunctionCallback.builder()
            .name("queryOrders")
            .description("查询用户的订单信息")
            .inputType(OrderQuery.class)  // 指定输入类型
            .function(orderRepository::findByCriteria)
            .build();
    }
    
    /**
     * 智能路由：根据上下文动态选择函数
     */
    public String intelligentChat(String userId, String userMessage) {
        // 获取用户上下文
        UserContext context = sessionService.getUserContext(userId);
        
        // 动态构建可用的函数列表
        List<String> availableFunctions = getAvailableFunctions(context);
        
        String response = chatClient.prompt()
            .user(userMessage)
            .functions(availableFunctions)  // 动态传入可用的函数
            .call()
            .content();
        
        // 记录函数调用历史
        recordFunctionCall(userId, response);
        
        return response;
    }
    
    private List<String> getAvailableFunctions(UserContext context) {
        List<String> functions = new ArrayList<>();
        functions.add("getCurrentWeather");
        functions.add("calculatorFunction");
        
        // 根据用户角色添加特定函数
        if (context.hasPermission("query_orders")) {
            functions.add("queryOrders");
        }
        if (context.hasPermission("manage_users")) {
            functions.add("getUserInfo");
        }
        
        return functions;
    }
}
```

```
@Component
class DataAnalysisFunctions {
    
    private final JdbcTemplate jdbcTemplate;
    private final ObjectMapper objectMapper;
    
    // 1. SQL查询函数
    @Bean
    public FunctionCallback queryDataFunction() {
        return FunctionCallback.builder()
            .name("executeSQLQuery")
            .description("执行SQL查询并返回结果")
            .inputType(SQLQueryRequest.class)
            .function(this::executeQuery)
            .build();
    }
    
    private QueryResult executeQuery(SQLQueryRequest request) {
        // 安全验证
        validateSQL(request.getSql());
        
        // 执行查询
        List<Map<String, Object>> results = jdbcTemplate.queryForList(request.getSql());
        
        return new QueryResult(results, results.size());
    }
    
    // 2. 数据聚合函数
    @Bean
    public FunctionCallback aggregateDataFunction() {
        return FunctionCallback.builder()
            .name("aggregateData")
            .description("对数据进行聚合计算：求和、平均、计数等")
            .inputType(AggregationRequest.class)
            .function(this::aggregateData)
            .build();
    }
    
    private AggregationResult aggregateData(AggregationRequest request) {
        String sql = String.format("""
            SELECT %s(%s) as result 
            FROM %s 
            WHERE %s
            """, request.getOperation(), request.getColumn(), 
            request.getTable(), request.getCondition());
        
        Double result = jdbcTemplate.queryForObject(sql, Double.class);
        return new AggregationResult(request.getOperation(), result);
    }
    
    // 3. 图表生成函数
    @Bean
    public FunctionCallback generateChartFunction() {
        return FunctionCallback.builder()
            .name("generateChart")
            .description("根据数据生成图表（折线图、柱状图、饼图等）")
            .inputType(ChartRequest.class)
            .function(this::generateChart)
            .build();
    }
    
    private ChartResult generateChart(ChartRequest request) {
        // 使用JFreeChart或其他库生成图表
        byte[] chartImage = generateChartImage(
            request.getChartType(),
            request.getData(),
            request.getOptions()
        );
        
        return new ChartResult(chartImage, request.getChartType());
    }
    
    // 安全验证
    private void validateSQL(String sql) {
        // 防止SQL注入
        if (sql.toLowerCase().contains("delete") || 
            sql.toLowerCase().contains("drop") ||
            sql.toLowerCase().contains("insert") ||
            sql.toLowerCase().contains("update")) {
            throw new SecurityException("不允许执行DML语句");
        }
    }
```

```
@Component
@Slf4j
public class FunctionCallingMonitor {
    
    @Around("@annotation(LogFunctionCall)")
    public Object logFunctionCall(ProceedingJoinPoint joinPoint) throws Throwable {
        String functionName = joinPoint.getSignature().getName();
        Object[] args = joinPoint.getArgs();
        
        log.info("开始调用函数: {}，参数: {}", functionName, args);
        long startTime = System.currentTimeMillis();
        
        try {
            Object result = joinPoint.proceed();
            long duration = System.currentTimeMillis() - startTime;
            
            log.info("函数调用成功: {}，耗时: {}ms，结果: {}", 
                    functionName, duration, 
                    result != null ? result.toString() : "null");
            
            // 记录指标
            Metrics.counter("function.calls", "name", functionName, "status", "success")
                  .increment();
            
            return result;
        } catch (Exception e) {
            long duration = System.currentTimeMillis() - startTime;
            log.error("函数调用失败: {}，耗时: {}ms，错误: {}", 
                     functionName, duration, e.getMessage());
            
            Metrics.counter("function.calls", "name", functionName, "status", "error")
                  .increment();
            
            throw e;
        }
    }
}

// 自定义注解
@Target(ElementType.METHOD)
@Retention(RetentionPolicy.RUNTIME)
public @interface LogFunctionCall {
    String value() default "";
}
```

```
// Function Calling 前端客户端
class FunctionCallingClient {
    constructor(baseUrl = '/api/chat') {
        this.baseUrl = baseUrl;
        this.sessionId = this.getSessionId();
    }
    
    // 发送消息并处理Function Calling
    async sendMessage(message, options = {}) {
        const response = await fetch(`${this.baseUrl}/function-call`, {
            method: 'POST',
            headers: {'Content-Type': 'application/json'},
            body: JSON.stringify({
                sessionId: this.sessionId,
                message: message,
                options: {
                    functions: options.functions || [],  // 指定可用的函数
                    temperature: options.temperature || 0.1,
                    stream: options.stream || false
                }
            })
        });
        
        if (options.stream) {
            return this.handleStreamResponse(response);
        }
        
        const data = await response.json();
        
        // 检查是否有函数调用
        if (data.toolCalls && data.toolCalls.length > 0) {
            return await this.handleToolCalls(data);
        }
        
        return data;
    }
    
    // 处理工具调用
    async handleToolCalls(data) {
        const toolCalls = data.toolCalls;
        const toolResponses = [];
        
        for (const toolCall of toolCalls) {
            // 执行对应的前端函数或调用后端API
            const result = await this.executeTool(toolCall);
            toolResponses.push({
                toolCallId: toolCall.id,
                result: result
            });
        }
        
        // 将结果发送回服务器
        const finalResponse = await fetch(`${this.baseUrl}/function-result`, {
            method: 'POST',
            headers: {'Content-Type': 'application/json'},
            body: JSON.stringify({
                sessionId: this.sessionId,
                toolResponses: toolResponses,
                originalMessage: data.originalMessage
            })
        });
        
        return await finalResponse.json();
    }
    
    // 执行工具函数
    async executeTool(toolCall) {
        const functionName = toolCall.function.name;
        const args = toolCall.function.arguments;
        
        switch (functionName) {
            case 'getCurrentWeather':
                // 调用天气API
                return await this.getWeather(args.location, args.unit);
                
            case 'openBrowser':
                // 前端操作：打开浏览器
                window.open(args.url, '_blank');
                return `已打开链接: ${args.url}`;
                
            case 'showNotification':
                // 前端操作：显示通知
                if (Notification.permission === 'granted') {
                    new Notification(args.title, {body: args.body});
                }
                return '通知已发送';
                
            default:
                // 调用后端函数执行
                return await this.callBackendFunction(functionName, args);
        }
    }
    
    // 流式响应处理
    async handleStreamResponse(response) {
        const reader = response.body.getReader();
        const decoder = new TextDecoder();
        
        return new ReadableStream({
            async start(controller) {
                try {
                    while (true) {
                        const {done, value} = await reader.read();
                        if (done) break;
                        
                        const chunk = decoder.decode(value);
                        const lines = chunk.split('\n');
                        
                        for (const line of lines) {
                            if (line.startsWith('data: ')) {
                                const data = line.substring(6);
                                if (data === '[DONE]') {
                                    controller.close();
                                    return;
                                }
                                
                                try {
                                    const parsed = JSON.parse(data);
                                    controller.enqueue(parsed);
                                } catch (e) {
                                    // 忽略解析错误
                                }
                            }
                        }
                    }
                } finally {
                    reader.releaseLock();
                }
            }
        });
    }
}
```
