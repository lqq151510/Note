import lombok.Data;
import java.util.List;

@Data
public class ChatRequest {
    private String sessionId;     // 会话ID（前端维护或后端生成）
    private String message;       // 用户消息
    private String model;         // 可选：指定模型
    private Boolean stream = false; // 是否流式响应
}

@Data
public class ChatResponse {
    private String reply;         // AI回复
    private String sessionId;     // 会话ID
    private Usage usage;          // Token使用量
    private Long timestamp;       // 响应时间戳
}

@Data
public class Usage {
    private Integer promptTokens;
    private Integer completionTokens;
    private Integer totalTokens;
}

@Data
public class ChatHistoryResponse {
    private String sessionId;
    private List<Message> messages;
}

@Data
public class Message {
    private String role;          // "user" 或 "assistant"
    private String content;
    private Long timestamp;
}

#### 2. 核心控制器

```
import org.springframework.ai.chat.client.ChatClient;
import org.springframework.ai.chat.model.ChatResponse;
import org.springframework.ai.chat.model.Usage;
import org.springframework.http.HttpStatus;
import org.springframework.http.MediaType;
import org.springframework.http.ResponseEntity;
import org.springframework.web.bind.annotation.*;
import org.springframework.web.servlet.mvc.method.annotation.SseEmitter;
import java.io.IOException;
import java.time.LocalDateTime;
import java.util.*;
import java.util.concurrent.ConcurrentHashMap;

@RestController
@RequestMapping("/api/ai")
@CrossOrigin(origins = {"http://localhost:3000", "https://your-frontend.com"}) // 允许跨域
public class AIController {
    
    private final ChatClient chatClient;
    private final LoggingAIService loggingService; // 上一节的服务
    
    // 内存存储会话（生产环境建议用Redis）
    private final Map<String, List<Message>> sessionStore = new ConcurrentHashMap<>();
    
    public AIController(ChatClient chatClient, LoggingAIService loggingService) {
        this.chatClient = chatClient;
        this.loggingService = loggingService;
    }
    
    /**
     * 普通对话接口（阻塞式，等待完整响应）
     */
    @PostMapping("/chat")
    public ResponseEntity<ChatResponse> chat(@RequestBody ChatRequest request) {
        // 1. 记录用户消息
        String sessionId = getOrCreateSessionId(request.getSessionId());
        addMessageToSession(sessionId, "user", request.getMessage());
        
        // 2. 调用AI
        ChatResponse aiResponse = chatClient.prompt()
                .user(request.getMessage())
                .call()
                .chatResponse();
        
        String reply = aiResponse.getResult().getOutput().getContent();
        
        // 3. 记录AI回复
        addMessageToSession(sessionId, "assistant", reply);
        
        // 4. 可选：持久化到数据库
        loggingService.saveChatLog(sessionId, "user", request.getMessage());
        loggingService.saveChatLog(sessionId, "assistant", reply, aiResponse.getUsage());
        
        // 5. 返回响应
        ChatResponse response = new ChatResponse();
        response.setReply(reply);
        response.setSessionId(sessionId);
        response.setTimestamp(System.currentTimeMillis());
        if (aiResponse.getUsage() != null) {
            Usage usage = new Usage();
            usage.setPromptTokens(aiResponse.getUsage().getPromptTokens());
            usage.setCompletionTokens(aiResponse.getUsage().getCompletionTokens());
            usage.setTotalTokens(aiResponse.getUsage().getTotalTokens());
            response.setUsage(usage);
        }
        
        return ResponseEntity.ok(response);
    }
    
    /**
     * 流式对话接口（Server-Sent Events）
     * 前端通过 EventSource 连接
     */
    @GetMapping(value = "/chat/stream", produces = MediaType.TEXT_EVENT_STREAM_VALUE)
    public SseEmitter streamChat(@RequestParam String message, 
                                 @RequestParam(required = false) String sessionId) {
        SseEmitter emitter = new SseEmitter(60_000L); // 60秒超时
        
        String finalSessionId = getOrCreateSessionId(sessionId);
        addMessageToSession(finalSessionId, "user", message);
        
        // 异步处理流式响应
        CompletableFuture.runAsync(() -> {
            try {
                StringBuilder fullResponse = new StringBuilder();
                
                chatClient.prompt()
                    .user(message)
                    .stream()
                    .contentStream()
                    .forEach(chunk -> {
                        try {
                            // 发送每个chunk
                            SseEmitter.SseEventBuilder event = SseEmitter.event()
                                    .data(chunk) // 只发送纯文本
                                    .id(UUID.randomUUID().toString())
                                    .name("message");
                            emitter.send(event);
                            fullResponse.append(chunk);
                        } catch (IOException e) {
                            emitter.completeWithError(e);
                        }
                    });
                
                // 流式完成
                addMessageToSession(finalSessionId, "assistant", fullResponse.toString());
                loggingService.saveChatLog(finalSessionId, "assistant", fullResponse.toString());
                emitter.complete();
                
            } catch (Exception e) {
                emitter.completeWithError(e);
            }
        });
        
        emitter.onCompletion(() -> System.out.println("SSE 连接完成"));
        emitter.onTimeout(() -> {
            System.out.println("SSE 连接超时");
            emitter.complete();
        });
        
        return emitter;
    }
    
    /**
     * 获取会话历史
     */
    @GetMapping("/history/{sessionId}")
    public ResponseEntity<ChatHistoryResponse> getHistory(@PathVariable String sessionId) {
        List<Message> messages = sessionStore.getOrDefault(sessionId, new ArrayList<>());
        
        // 如果内存中没有，从数据库查询
        if (messages.isEmpty()) {
            messages = loggingService.getSessionHistory(sessionId).stream()
                    .map(log -> {
                        Message msg = new Message();
                        msg.setRole(log.getRole());
                        msg.setContent(log.getContent());
                        msg.setTimestamp(log.getTimestamp().toEpochSecond());
                        return msg;
                    })
                    .collect(Collectors.toList());
        }
        
        ChatHistoryResponse response = new ChatHistoryResponse();
        response.setSessionId(sessionId);
        response.setMessages(messages);
        
        return ResponseEntity.ok(response);
    }
    
    /**
     * 创建新会话
     */
    @PostMapping("/session/new")
    public ResponseEntity<Map<String, String>> createSession() {
        String sessionId = UUID.randomUUID().toString();
        sessionStore.put(sessionId, new ArrayList<>());
        return ResponseEntity.ok(Map.of("sessionId", sessionId));
    }
    
    // 辅助方法
    private String getOrCreateSessionId(String sessionId) {
        if (sessionId == null || sessionId.isEmpty()) {
            sessionId = UUID.randomUUID().toString();
            sessionStore.put(sessionId, new ArrayList<>());
        } else if (!sessionStore.containsKey(sessionId)) {
            sessionStore.put(sessionId, new ArrayList<>());
        }
        return sessionId;
    }
    
    private void addMessageToSession(String sessionId, String role, String content) {
        List<Message> messages = sessionStore.computeIfAbsent(sessionId, k -> new ArrayList<>());
        Message message = new Message();
        message.setRole(role);
        message.setContent(content);
        message.setTimestamp(System.currentTimeMillis() / 1000);
        messages.add(message);
    }
}

// React/Vue 组件中
async function sendMessage(message, sessionId = null) {
    try {
        const response = await fetch('http://localhost:8080/api/ai/chat', {
            method: 'POST',
            headers: {
                'Content-Type': 'application/json',
                'Authorization': `Bearer ${localStorage.getItem('token')}` // 如果有认证
            },
            body: JSON.stringify({
                sessionId: sessionId,
                message: message,
                stream: false
            })
        });
        
        if (!response.ok) throw new Error('请求失败');
        const data = await response.json();
        
        console.log('AI回复:', data.reply);
        console.log('使用Token:', data.usage);
        console.log('会话ID:', data.sessionId);
        
        return data;
    } catch (error) {
        console.error('请求出错:', error);
    }
}

#### 2. 流式请求（SSE）

```
// 流式对话
function streamChat(message, sessionId, onChunk, onComplete, onError) {
    const eventSource = new EventSource(
        `http://localhost:8080/api/ai/chat/stream?message=${encodeURIComponent(message)}&sessionId=${sessionId || ''}`
    );
    
    let fullResponse = '';
    
    eventSource.onmessage = (event) => {
        const chunk = event.data;
        fullResponse += chunk;
        onChunk?.(chunk, fullResponse);
    };
    
    eventSource.onerror = (error) => {
        console.error('SSE错误:', error);
        eventSource.close();
        onError?.(error);
    };
    
    // 自定义事件（如果需要结构化数据）
    eventSource.addEventListener('usage', (event) => {
        const usage = JSON.parse(event.data);
        console.log('Token使用:', usage);
    });
    
    eventSource.addEventListener('complete', () => {
        eventSource.close();
        onComplete?.(fullResponse);
    });
    
    return eventSource; // 用于手动关闭
}

// 使用示例
const sse = streamChat(
    '你好，介绍一下Spring AI',
    'session-123',
    (chunk, full) => {
        // 实时更新UI
        document.getElementById('output').innerHTML += chunk;
    },
    (fullResponse) => {
        console.log('完整回复:', fullResponse);
    },
    (error) => {
        console.error('流式请求失败:', error);
    }
);

### 三、完整配置建议

#### 1. 跨域配置（可选全局配置）

```
import org.springframework.context.annotation.Configuration;
import org.springframework.web.servlet.config.annotation.CorsRegistry;
import org.springframework.web.servlet.config.annotation.WebMvcConfigurer;

@Configuration
public class WebConfig implements WebMvcConfigurer {
    @Override
    public void addCorsMappings(CorsRegistry registry) {
        registry.addMapping("/api/**")
                .allowedOrigins("http://localhost:3000", "https://your-app.com")
                .allowedMethods("GET", "POST", "PUT", "DELETE", "OPTIONS")
                .allowedHeaders("*")
                .allowCredentials(true)
                .maxAge(3600);
    }
}

@Data
public class ApiResponse<T> {
    private Integer code;
    private String message;
    private T data;
    private Long timestamp;
    
    public static <T> ApiResponse<T> success(T data) {
        ApiResponse<T> response = new ApiResponse<>();
        response.setCode(200);
        response.setMessage("success");
        response.setData(data);
        response.setTimestamp(System.currentTimeMillis());
        return response;
    }
    
    public static ApiResponse<?> error(Integer code, String message) {
        ApiResponse<?> response = new ApiResponse<>();
        response.setCode(code);
        response.setMessage(message);
        response.setTimestamp(System.currentTimeMillis());
        return response;
    }
}

#### 3. 异常处理

```
@ControllerAdvice
public class GlobalExceptionHandler {
    
    @ExceptionHandler(Exception.class)
    public ResponseEntity<ApiResponse<?>> handleException(Exception e) {
        return ResponseEntity.status(500)
                .body(ApiResponse.error(500, "服务异常: " + e.getMessage()));
    }
    
    @ExceptionHandler(IllegalArgumentException.class)
    public ResponseEntity<ApiResponse<?>> handleBadRequest(Exception e) {
        return ResponseEntity.status(400)
                .body(ApiResponse.error(400, "请求参数错误: " + e.getMessage()));
    }
}

### 四、前后端对接要点

1. **会话管理**：建议前端在 localStorage 中存储 `sessionId`，每次对话携带
    
2. **加载状态**：前端在请求时显示加载动画，流式响应时实时显示文字
    
3. **错误处理**：前端需要处理网络错误、超时、服务端错误等
    
4. **安全性**：
    
    - API Key 必须保存在后端
        
    - 生产环境启用 HTTPS
        
    - 添加 API 调用频率限制
        
    - 敏感内容过滤
        
    
5. **部署**：
    
    - 后端部署到云服务器
        
    - 前端打包后部署到 Nginx 或对象存储
        
    - 配置正确的跨域和反向代理
        
    

通过以上完整的接口设计和配置，前端可以轻松实现与 Spring AI 后端的对话交互、流式输出、历史记录管理等功能。
