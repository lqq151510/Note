# RestHighLevelClient 查询操作详解

`RestHighLevelClient`提供了完整的查询 API，支持从简单到复杂的各种查询场景。

---

## 1. 查询基础结构

```
// 基本查询流程
SearchRequest request = new SearchRequest("index_name");  // 指定索引
SearchSourceBuilder sourceBuilder = new SearchSourceBuilder();  // 构建查询条件
sourceBuilder.query(QueryBuilders.matchAllQuery());  // 设置查询条件
request.source(sourceBuilder);  // 将查询条件加入请求
SearchResponse response = client.search(request, RequestOptions.DEFAULT);  // 执行查询
```

---

    // 6. 复合查询（Bool Query）
    public void boolQuery() throws IOException {
        SearchRequest request = new SearchRequest("products");
        SearchSourceBuilder sourceBuilder = new SearchSourceBuilder();
        
        // 构建bool查询
        BoolQueryBuilder boolQuery = QueryBuilders.boolQuery();
        
        // must条件：必须满足
        boolQuery.must(QueryBuilders.matchQuery("name", "手机"));
        
        // filter条件：必须满足，但不参与评分
        boolQuery.filter(QueryBuilders.termQuery("status", "published"));
        boolQuery.filter(QueryBuilders.rangeQuery("price").gte(1000).lte(5000));
        
        // should条件：应该满足
        boolQuery.should(QueryBuilders.termQuery("brand", "华为"));
        boolQuery.should(QueryBuilders.termQuery("brand", "小米"));
        boolQuery.minimumShouldMatch(1);  // 至少满足一个should条件
        
        // must_not条件：必须不满足
        boolQuery.mustNot(QueryBuilders.termQuery("category", "二手"));
        
        sourceBuilder.query(boolQuery);
        request.source(sourceBuilder);
        
        SearchResponse response = client.search(request, RequestOptions.DEFAULT);
        processSearchResponse(response);
    }

    public void searchWithHighlight() throws IOException {
        SearchRequest request = new SearchRequest("products");
        SearchSourceBuilder sourceBuilder = new SearchSourceBuilder();
        
        // 查询条件
        sourceBuilder.query(QueryBuilders.matchQuery("description", "高性能"));

        nameField.preTags("<span style='color:red'>");
        nameField.postTags("</span>");
        nameField.fragmentSize(0);  // 0表示不截断
        
        HighlightBuilder.Field descField = new HighlightBuilder.Field("description");
        descField.preTags("<em>");
        descField.postTags("</em>");
        descField.fragmentSize(150);  // 每个片段长度
        descField.numOfFragments(3);  // 最多返回3个片段
        
        highlightBuilder.field(nameField);
        highlightBuilder.field(descField);
        highlightBuilder.requireFieldMatch(false);  // 不需要所有字段都匹配
        
        sourceBuilder.highlighter(highlightBuilder);
        request.source(sourceBuilder);
        
        SearchResponse response = client.search(request, RequestOptions.DEFAULT);
        
        // 处理高亮结果
        for (SearchHit hit : response.getHits().getHits()) {
            Map<String, Object> source = hit.getSourceAsMap();
            
            // 获取高亮字段
            Map<String, org.elasticsearch.search.fetch.subphase.highlight.HighlightField> highlightFields = 
                hit.getHighlightFields();
            
            if (highlightFields.containsKey("name")) {
                String highlightedName = highlightFields.get("name").fragments()[0].string();
                System.out.println("高亮名称: " + highlightedName);
            }
            
            if (highlightFields.containsKey("description")) {
                org.elasticsearch.search.fetch.subphase.highlight.HighlightField descHighlight = 
                    highlightFields.get("description");
                for (org.elasticsearch.common.text.Text fragment : descHighlight.getFragments()) {
                    System.out.println("描述片段: " + fragment.string());
                }
            }
        }
    }
    
    // 9. 聚合查询
    public void searchWithAggregation() throws IOException {
        SearchRequest request = new SearchRequest("products");
        SearchSourceBuilder sourceBuilder = new SearchSourceBuilder();
        
        // 基础查询
        sourceBuilder.query(QueryBuilders.matchAllQuery());
        sourceBuilder.size(0);  // 不返回具体文档，只返回聚合结果
        
        // 1. 按品牌分组聚合
        TermsAggregationBuilder brandAgg = AggregationBuilders.terms("brand_agg")
            .field("brand.keyword")
            .size(10)  // 返回前10个品牌
            .order(Terms.Order.aggregation("avg_price", false));  // 按平均价格降序
        
        // 子聚合：计算每个品牌的平均价格
        brandAgg.subAggregation(
            AggregationBuilders.avg("avg_price").field("price")
        );
        
        // 子聚合：计算每个品牌的总销量
        brandAgg.subAggregation(
            AggregationBuilders.sum("total_sales").field("sales")
        );
        
        // 2. 价格范围聚合
        TermsAggregationBuilder priceRangeAgg = AggregationBuilders.range("price_ranges")
            .field("price")
            .addRange(0, 1000)
            .addRange(1000, 5000)
            .addRange(5000, 10000)
            .addRange(10000, Double.MAX_VALUE);
        
        sourceBuilder.aggregation(brandAgg);
        sourceBuilder.aggregation(priceRangeAgg);
        
        // 3. 指标聚合
        sourceBuilder.aggregation(
            AggregationBuilders.stats("price_stats").field("price")
        );
        
        request.source(sourceBuilder);
        SearchResponse response = client.search(request, RequestOptions.DEFAULT);
        
        // 处理聚合结果
        Terms brandTerms = response.getAggregations().get("brand_agg");
        List<? extends Terms.Bucket> brandBuckets = brandTerms.getBuckets();
        
        for (Terms.Bucket bucket : brandBuckets) {
            System.out.println("品牌: " + bucket.getKey() + 
                ", 商品数: " + bucket.getDocCount());
            
            // 获取子聚合结果
            org.elasticsearch.search.aggregations.metrics.Avg avgPrice = 
                bucket.getAggregations().get("avg_price");
            org.elasticsearch.search.aggregations.metrics.Sum totalSales = 
                bucket.getAggregations().get("total_sales");
            
            System.out.println("平均价格: " + avgPrice.getValue());
            System.out.println("总销量: " + totalSales.getValue());
        }
        
        // 处理价格范围聚合
        org.elasticsearch.search.aggregations.bucket.range.Range priceRanges = 
            response.getAggregations().get("price_ranges");
        for (org.elasticsearch.search.aggregations.bucket.range.Range.Bucket bucket : 
                priceRanges.getBuckets()) {
            System.out.println("价格范围: " + bucket.getKey() + 
                ", 商品数: " + bucket.getDocCount());
        }
        
        // 处理统计聚合
        org.elasticsearch.search.aggregations.metrics.Stats priceStats = 
            response.getAggregations().get("price_stats");
        System.out.println("价格统计 - 最小值: " + priceStats.getMin());
        System.out.println("价格统计 - 最大值: " + priceStats.getMax());
        System.out.println("价格统计 - 平均值: " + priceStats.getAvg());
        System.out.println("价格统计 - 总和: " + priceStats.getSum());
    }
    
    // 10. 滚动查询（深度分页）
    public void scrollSearch() throws IOException {
        SearchRequest request = new SearchRequest("products");
        SearchSourceBuilder sourceBuilder = new SearchSourceBuilder();
        
        sourceBuilder.query(QueryBuilders.matchAllQuery());
        sourceBuilder.size(100);  // 每批获取100条
        
        request.source(sourceBuilder);
        request.scroll(TimeValue.timeValueMinutes(1L));  // 滚动上下文保持1分钟
        
        // 第一次查询
        SearchResponse response = client.search(request, RequestOptions.DEFAULT);
        String scrollId = response.getScrollId();
        SearchHit[] hits = response.getHits().getHits();
        
        int total = 0;
        while (hits != null && hits.length > 0) {
            total += hits.length;
            System.out.println("已获取: " + total + " 条数据");
            
            // 处理当前批次
            for (SearchHit hit : hits) {
                Map<String, Object> source = hit.getSourceAsMap();
                // 处理数据...
            }
            
            // 继续滚动
            SearchScrollRequest scrollRequest = new SearchScrollRequest(scrollId);
            scrollRequest.scroll(TimeValue.timeValueSeconds(30));
            
            response = client.scroll(scrollRequest, RequestOptions.DEFAULT);
            scrollId = response.getScrollId();
            hits = response.getHits().getHits();
        }
        
        // 清除滚动上下文
        ClearScrollRequest clearScrollRequest = new ClearScrollRequest();
        clearScrollRequest.addScrollId(scrollId);
        client.clearScroll(clearScrollRequest, RequestOptions.DEFAULT);
    }

        SearchSourceBuilder sourceBuilder = new SearchSourceBuilder();
        
        // 构建建议查询
        SuggestBuilder suggestBuilder = new SuggestBuilder();
        suggestBuilder.addSuggestion("name_suggest",

                .prefixLength(3) // 前缀长度

        );

        request.source(sourceBuilder);
        
        SearchResponse response = client.search(request, RequestOptions.DEFAULT);
        
        // 处理建议结果
        org.elasticsearch.search.suggest.Suggest suggest = response.getSuggest();
        if (suggest != null) {
            org.elasticsearch.search.suggest.Suggest.Suggestion<?> suggestion = 
                suggest.getSuggestion("name_suggest");
            
            for (org.elasticsearch.search.suggest.Suggestion.Entry<?> entry : suggestion.getEntries()) {
                for (org.elasticsearch.search.suggest.Suggestion.Entry.Option option : entry) {
                    System.out.println("建议: " + option.getText().string());
                }
            }
        }
    }
    
    // 13. 函数评分查询
    public void functionScoreQuery() throws IOException {
        SearchRequest request = new SearchRequest("products");
        SearchSourceBuilder sourceBuilder = new SearchSourceBuilder();
        
        // 构建函数评分查询
        FunctionScoreQueryBuilder.FilterFunctionBuilder[] functions = {
            new FunctionScoreQueryBuilder.FilterFunctionBuilder(
                QueryBuilders.termQuery("is_featured", true),
                ScoreFunctionBuilders.weightFactorFunction(2.0f)  // 特征商品权重*2
            ),
            new FunctionScoreQueryBuilder.FilterFunctionBuilder(
                QueryBuilders.rangeQuery("sales").gte(1000),
                ScoreFunctionBuilders.fieldValueFactorFunction("sales")  // 销量影响评分
                    .modifier(FieldValueFactorFunction.Modifier.LOG1P)
                    .factor(0.1f)
            ),
            new FunctionScoreQueryBuilder.FilterFunctionBuilder(
                ScoreFunctionBuilders.gaussDecayFunction("create_time", "now", "30d", "15d")
                    .setWeight(1.5f)  // 新商品权重更高
            )
        };
        
        FunctionScoreQueryBuilder functionScoreQuery = QueryBuilders.functionScoreQuery(
            QueryBuilders.matchQuery("name", "手机"),  // 基础查询
            functions
        )
        .scoreMode(FunctionScoreQuery.ScoreMode.SUM)  // 评分模式
        .boostMode(CombineFunction.SUM)               // 提升模式
        .maxBoost(10.0f);                             // 最大提升
        
        sourceBuilder.query(functionScoreQuery);
        request.source(sourceBuilder);
        
        SearchResponse response = client.search(request, RequestOptions.DEFAULT);
        processSearchResponse(response);
    }

    // 17. 复合查询工具类
    public class AdvancedSearchBuilder {
        private final SearchSourceBuilder sourceBuilder = new SearchSourceBuilder();
        private final BoolQueryBuilder boolQuery = QueryBuilders.boolQuery();
        
        public AdvancedSearchBuilder() {
            sourceBuilder.query(boolQuery);
        }
        
        public AdvancedSearchBuilder addMustQuery(QueryBuilder query) {
            boolQuery.must(query);
            return this;
        }
        
        public AdvancedSearchBuilder addFilterQuery(QueryBuilder query) {
            boolQuery.filter(query);
            return this;
        }
        
        public AdvancedSearchBuilder addShouldQuery(QueryBuilder query) {
            boolQuery.should(query);
            return this;
        }
        
        public AdvancedSearchBuilder addMustNotQuery(QueryBuilder query) {
            boolQuery.mustNot(query);
            return this;
        }
        
        public AdvancedSearchBuilder setPaging(int from, int size) {
            sourceBuilder.from(from);
            sourceBuilder.size(size);
            return this;
        }
        
        public AdvancedSearchBuilder addSort(String field, SortOrder order) {
            sourceBuilder.sort(SortBuilders.fieldSort(field).order(order));
            return this;
        }
        
        public AdvancedSearchBuilder addHighlight(String field, String preTags, String postTags) {
            HighlightBuilder highlightBuilder = sourceBuilder.highlighter();
            if (highlightBuilder == null) {
                highlightBuilder = new HighlightBuilder();
            }
            highlightBuilder.field(new HighlightBuilder.Field(field)
                .preTags(preTags)
                .postTags(postTags));
            sourceBuilder.highlighter(highlightBuilder);
            return this;
        }
        
        public AdvancedSearchBuilder addAggregation(String name, TermsAggregationBuilder agg) {
            sourceBuilder.aggregation(agg);
            return this;
        }
        
        public SearchRequest build(String... indices) {
            SearchRequest request = new SearchRequest(indices);
            request.source(sourceBuilder);
            return request;
        }
    }

    // 处理查询结果的通用方法
    private void processSearchResponse(SearchResponse response) {
        SearchHits hits = response.getHits();
        
        System.out.println("总命中数: " + hits.getTotalHits().value);
        System.out.println("最大得分: " + hits.getMaxScore());
        System.out.println("查询耗时: " + response.getTook());
        
        for (SearchHit hit : hits.getHits()) {
            System.out.println("\n=== 文档ID: " + hit.getId() + " ===");
            System.out.println("得分: " + hit.getScore());
            System.out.println("源数据: " + hit.getSourceAsString());
            
            // 获取文档字段
            Map<String, Object> source = hit.getSourceAsMap();
            source.forEach((key, value) -> {
                System.out.println(key + ": " + value);
            });
            
            // 获取排序值
            if (hit.getSortValues() != null && hit.getSortValues().length > 0) {
                System.out.println("排序值: " + Arrays.toString(hit.getSortValues()));
            }
            
            // 获取高亮
            Map<String, org.elasticsearch.search.fetch.subphase.highlight.HighlightField> 
                highlightFields = hit.getHighlightFields();
            if (highlightFields != null && !highlightFields.isEmpty()) {
                System.out.println("高亮字段:");
                highlightFields.forEach((field, highlight) -> {
                    System.out.println("  " + field + ": " + 
                        Arrays.stream(highlight.getFragments())
                              .map(fragment -> fragment.string())
                              .collect(java.util.stream.Collectors.joining("...")));
                });
            }
        }
        
        // 处理聚合结果
        if (response.getAggregations() != null) {
            System.out.println("\n=== 聚合结果 ===");
            // 聚合处理逻辑...
        }
        
        // 处理建议
        if (response.getSuggest() != null) {
            System.out.println("\n=== 建议结果 ===");
            // 建议处理逻辑...
        }
    }

    public void asyncSearch() {
        SearchRequest request = new SearchRequest("products");
        SearchSourceBuilder sourceBuilder = new SearchSourceBuilder();
        sourceBuilder.query(QueryBuilders.matchAllQuery());
        request.source(sourceBuilder);
        
        // 异步执行查询
        client.searchAsync(request, RequestOptions.DEFAULT, 
            new ActionListener<SearchResponse>() {
                @Override
                public void onResponse(SearchResponse response) {
                    System.out.println("异步查询成功");
                    processSearchResponse(response);
                }
                
                @Override
                public void onFailure(Exception e) {
                    System.err.println("异步查询失败: " + e.getMessage());
                }
            });
        
        // 等待异步查询完成
        try {
            Thread.sleep(2000);
        } catch (InterruptedException e) {
            Thread.currentThread().interrupt();
        }
    }
    
    // 19. 批量查询（Multi Search）
    public void multiSearch() throws IOException {
        MultiSearchRequest request = new MultiSearchRequest();
        
        // 第一个查询
        SearchRequest search1 = new SearchRequest("products");
        SearchSourceBuilder source1 = new SearchSourceBuilder();
        source1.query(QueryBuilders.matchQuery("name", "手机"));
        search1.source(source1);
        request.add(search1);
        
        // 第二个查询
        SearchRequest search2 = new SearchRequest("orders");
        SearchSourceBuilder source2 = new SearchSourceBuilder();
        source2.query(QueryBuilders.rangeQuery("amount").gte(1000));
        search2.source(source2);
        request.add(search2);
        
        // 执行批量查询
        MultiSearchResponse response = client.msearch(request, RequestOptions.DEFAULT);
        
        for (int i = 0; i < response.getResponses().length; i++) {
            MultiSearchResponse.Item item = response.getResponses()[i];
            if (item.isFailure()) {
                System.err.println("查询" + i + "失败: " + item.getFailureMessage());
            } else {
                System.out.println("查询" + i + "结果:");
                processSearchResponse(item.getResponse());
            }
        }
    }
    
    // 20. 查询超时设置
    public void searchWithTimeout() throws IOException {
        SearchRequest request = new SearchRequest("products");
        SearchSourceBuilder sourceBuilder = new SearchSourceBuilder();
        
        sourceBuilder.query(QueryBuilders.matchAllQuery());
        sourceBuilder.timeout(TimeValue.timeValueSeconds(5));  // 查询超时5秒
        
        request.source(sourceBuilder);
        
        // 设置请求超时
        request.setIndicesOptions(IndicesOptions.fromOptions(true, true, true, true));
        
        try {
            SearchResponse response = client.search(request, RequestOptions.DEFAULT);
            processSearchResponse(response);
        } catch (org.elasticsearch.ElasticsearchTimeoutException e) {
            System.err.println("查询超时: " + e.getMessage());
        }
    }

public class ElasticsearchQueryUtils {
    private final RestHighLevelClient client;
    
    public ElasticsearchQueryUtils(RestHighLevelClient client) {
        this.client = client;
    }
    
    /**
     * 通用查询方法
     */
    public SearchResponse search(String index, QueryBuilder query, 
                                int from, int size, 
                                SortBuilder<?>... sorts) throws IOException {
        SearchRequest request = new SearchRequest(index);
        SearchSourceBuilder sourceBuilder = new SearchSourceBuilder();
        
        sourceBuilder.query(query);
        sourceBuilder.from(from);
        sourceBuilder.size(size);
        
        if (sorts != null) {
            for (SortBuilder<?> sort : sorts) {
                sourceBuilder.sort(sort);
            }
        }
        
        request.source(sourceBuilder);
        return client.search(request, RequestOptions.DEFAULT);
    }
    
    /**
     * 分页查询
     */
    public <T> PageResult<T> searchPage(String index, QueryBuilder query, 
                                        int pageNum, int pageSize,
                                        Class<T> clazz) throws IOException {
        SearchRequest request = new SearchRequest(index);
        SearchSourceBuilder sourceBuilder = new SearchSourceBuilder();
        
        sourceBuilder.query(query);
        sourceBuilder.from((pageNum - 1) * pageSize);
        sourceBuilder.size(pageSize);
        
        request.source(sourceBuilder);
        SearchResponse response = client.search(request, RequestOptions.DEFAULT);
        
        List<T> list = new ArrayList<>();
        for (SearchHit hit : response.getHits().getHits()) {
            T obj = JSON.parseObject(hit.getSourceAsString(), clazz);
            list.add(obj);
        }
        
        return new PageResult<>(list, pageNum, pageSize, 
            response.getHits().getTotalHits().value);
    }
    
    /**
     * 统计数量
     */
    public long count(String index, QueryBuilder query) throws IOException {
        SearchRequest request = new SearchRequest(index);
        SearchSourceBuilder sourceBuilder = new SearchSourceBuilder();
        
        sourceBuilder.query(query);
        sourceBuilder.size(0);  // 不返回文档
        
        request.source(sourceBuilder);
        SearchResponse response = client.search(request, RequestOptions.DEFAULT);
        
        return response.getHits().getTotalHits().value;
    }
    
    /**
     * 判断文档是否存在
     */
    public boolean exists(String index, String id) throws IOException {
        GetRequest request = new GetRequest(index, id);
        request.fetchSourceContext(FetchSourceContext.DO_NOT_FETCH_SOURCE);
        
        return client.exists(request, RequestOptions.DEFAULT);
    }
    
    /**
     * 批量查询
     */
    public Map<String, Map<String, Object>> multiGet(String index, 
                                                     List<String> ids) throws IOException {
        MultiGetRequest request = new MultiGetRequest();
        
        for (String id : ids) {
            request.add(new MultiGetRequest.Item(index, id));
        }
        
        MultiGetResponse response = client.mget(request, RequestOptions.DEFAULT);
        
        Map<String, Map<String, Object>> result = new HashMap<>();
        for (MultiGetItemResponse item : response.getResponses()) {
            if (item.isFailed()) {
                result.put(item.getId(), null);
            } else {
                result.put(item.getId(), item.getResponse().getSourceAsMap());
            }
        }
        
        return result;
    }
    
    /**
     * 查询去重
     */
    public List<String> distinct(String index, String field, 
                                QueryBuilder query, int size) throws IOException {
        SearchRequest request = new SearchRequest(index);
        SearchSourceBuilder sourceBuilder = new SearchSourceBuilder();
        
        sourceBuilder.query(query);
        sourceBuilder.size(0);
        
        // 使用聚合实现去重
        TermsAggregationBuilder agg = AggregationBuilders.terms("distinct_" + field)
            .field(field)
            .size(size);
        sourceBuilder.aggregation(agg);
        
        request.source(sourceBuilder);
        SearchResponse response = client.search(request, RequestOptions.DEFAULT);
        
        Terms terms = response.getAggregations().get("distinct_" + field);
        return terms.getBuckets().stream()
            .map(Terms.Bucket::getKeyAsString)
            .collect(Collectors.toList());
    }
    
    /**
     * 查询结果转换
     */
    public <T> List<T> convertHits(SearchResponse response, Class<T> clazz) {
        return Arrays.stream(response.getHits().getHits())
            .map(hit -> {
                try {
                    return JSON.parseObject(hit.getSourceAsString(), clazz);
                } catch (Exception e) {
                    return null;
                }
            })
            .filter(Objects::nonNull)
            .collect(Collectors.toList());
    }

        private int pageNum;
        private int pageSize;
        private long total;
        
        public PageResult(List<T> list, int pageNum, int pageSize, long total) {
            this.list = list;
            this.pageNum = pageNum;
            this.pageSize = pageSize;
            this.total = total;
        }
        
        // getter/setter...
    }
}

## 关键点总结

1. **查询构建**：使用 `QueryBuilders`构建各种查询条件
    
2. **复合查询**：`BoolQueryBuilder`是最常用的复合查询构造器
    
3. **性能优化**：
    
    - 使用 `filter`替代 `must`提高性能
        
    - 合理设置分页大小
        
    - 使用滚动查询处理大量数据
        
    
4. **结果处理**：
    
    - 注意处理分页和高亮
        
    - 合理使用聚合统计
        
    - 异步查询提高并发性能
        
    
5. **异常处理**：所有IO操作都需要异常处理
    
6. **资源管理**：及时关闭响应和客户端
