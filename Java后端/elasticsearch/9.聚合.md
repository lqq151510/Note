# Elasticsearch Java 客户端聚合操作详解

以下是使用 **RestHighLevelClient**​ 和 **Java API Client**​ 进行聚合查询的完整指南。

---

## 1. 聚合基础

### 聚合的类型

```
// 聚合分类
1. 桶聚合（Bucket Aggregations） - 分组
2. 指标聚合（Metrics Aggregations） - 计算
3. 管道聚合（Pipeline Aggregations） - 基于其他聚合结果
4. 矩阵聚合（Matrix Aggregations） - 多字段矩阵计算
```

---

## 2. 使用 RestHighLevelClient 的聚合操作

### 2.1 基础聚合查询示例

```
import org.elasticsearch.action.search.SearchRequest;
import org.elasticsearch.action.search.SearchResponse;
import org.elasticsearch.client.RequestOptions;
import org.elasticsearch.client.RestHighLevelClient;
import org.elasticsearch.index.query.QueryBuilders;
import org.elasticsearch.search.aggregations.*;
import org.elasticsearch.search.aggregations.bucket.histogram.*;
import org.elasticsearch.search.aggregations.bucket.range.*;
import org.elasticsearch.search.aggregations.bucket.terms.*;
import org.elasticsearch.search.aggregations.metrics.*;
import org.elasticsearch.search.builder.SearchSourceBuilder;
import org.elasticsearch.search.sort.SortOrder;

import java.io.IOException;
import java.util.List;

public class RestHighLevelAggregationDemo {
    private final RestHighLevelClient client;
    
    public RestHighLevelAggregationDemo(RestHighLevelClient client) {
        this.client = client;
    }
    
    // 1. 词项聚合（Terms Aggregation）
    public void termsAggregation() throws IOException {
        SearchRequest request = new SearchRequest("products");
        SearchSourceBuilder sourceBuilder = new SearchSourceBuilder();
        
        // 构建查询
        sourceBuilder.query(QueryBuilders.matchAllQuery());
        sourceBuilder.size(0);  // 不返回文档，只返回聚合结果
        
        // 创建词项聚合
        TermsAggregationBuilder aggregation = AggregationBuilders.terms("brand_agg")
            .field("brand.keyword")  // 聚合字段
            .size(10)                 // 返回前10个桶
            .order(BucketOrder.count(false))  // 按文档数降序
            .minDocCount(1)           // 最小文档数
            .shardSize(100)           // 每个分片返回的桶数
            .executionHint("map");    // 执行提示
        
        sourceBuilder.aggregation(aggregation);
        request.source(sourceBuilder);
        
        SearchResponse response = client.search(request, RequestOptions.DEFAULT);
        
        // 处理聚合结果
        Terms terms = response.getAggregations().get("brand_agg");
        List<? extends Terms.Bucket> buckets = terms.getBuckets();
        
        System.out.println("=== 品牌统计 ===");
        for (Terms.Bucket bucket : buckets) {
            System.out.printf("品牌: %s, 商品数: %d, 比例: %.2f%%%n",
                bucket.getKeyAsString(),
                bucket.getDocCount(),
                (bucket.getDocCount() * 100.0 / terms.getSumOfOtherDocCounts()));
        }
        
        System.out.println("其他品牌总数: " + terms.getSumOfOtherDocCounts());
    }
    
    // 2. 多级嵌套聚合
    public void nestedAggregation() throws IOException {
        SearchRequest request = new SearchRequest("products");
        SearchSourceBuilder sourceBuilder = new SearchSourceBuilder();
        
        sourceBuilder.query(QueryBuilders.matchAllQuery());
        sourceBuilder.size(0);
        
        // 第一级：按品牌分组
        TermsAggregationBuilder brandAgg = AggregationBuilders.terms("brand_agg")
            .field("brand.keyword")
            .size(5);
        
        // 第二级：每个品牌下按品类分组
        TermsAggregationBuilder categoryAgg = AggregationBuilders.terms("category_agg")
            .field("category.keyword")
            .size(3);
        
        // 第三级：每个品类计算价格统计
        StatsAggregationBuilder priceStatsAgg = AggregationBuilders.stats("price_stats")
            .field("price");
        
        // 构建聚合嵌套关系
        brandAgg.subAggregation(categoryAgg);
        categoryAgg.subAggregation(priceStatsAgg);
        
        sourceBuilder.aggregation(brandAgg);
        request.source(sourceBuilder);
        
        SearchResponse response = client.search(request, RequestOptions.DEFAULT);
        
        // 处理多级聚合结果
        Terms brandTerms = response.getAggregations().get("brand_agg");
        
        for (Terms.Bucket brandBucket : brandTerms.getBuckets()) {
            System.out.printf("\n=== 品牌: %s (商品数: %d) ===%n",
                brandBucket.getKeyAsString(), brandBucket.getDocCount());
            
            Terms categoryTerms = brandBucket.getAggregations().get("category_agg");
            
            for (Terms.Bucket categoryBucket : categoryTerms.getBuckets()) {
                Stats priceStats = categoryBucket.getAggregations().get("price_stats");
                
                System.out.printf("  品类: %s (商品数: %d)%n",
                    categoryBucket.getKeyAsString(), categoryBucket.getDocCount());
                System.out.printf("    价格统计: 最低%.2f, 最高%.2f, 平均%.2f, 总计%.2f%n",
                    priceStats.getMin(), priceStats.getMax(),
                    priceStats.getAvg(), priceStats.getSum());
            }
        }
    }
    
    // 3. 指标聚合
    public void metricsAggregation() throws IOException {
        SearchRequest request = new SearchRequest("products");
        SearchSourceBuilder sourceBuilder = new SearchSourceBuilder();
        
        sourceBuilder.query(QueryBuilders.matchAllQuery());
        sourceBuilder.size(0);
        
        // 多种指标聚合
        sourceBuilder.aggregation(
            AggregationBuilders.avg("avg_price").field("price")
        );
        
        sourceBuilder.aggregation(
            AggregationBuilders.max("max_price").field("price")
        );
        
        sourceBuilder.aggregation(
            AggregationBuilders.min("min_price").field("price")
        );
        
        sourceBuilder.aggregation(
            AggregationBuilders.sum("total_sales").field("sales")
        );
        
        sourceBuilder.aggregation(
            AggregationBuilders.count("total_count").field("_id")
        );
        
        sourceBuilder.aggregation(
            AggregationBuilders.stats("price_stats").field("price")
        );
        
        sourceBuilder.aggregation(
            AggregationBuilders.extendedStats("price_extended_stats").field("price")
        );
        
        sourceBuilder.aggregation(
            AggregationBuilders.percentiles("price_percentiles")
                .field("price")
                .percentiles(1.0, 5.0, 25.0, 50.0, 75.0, 95.0, 99.0)
        );
        
        sourceBuilder.aggregation(
            AggregationBuilders.cardinality("unique_brands")
                .field("brand.keyword")
        );
        
        request.source(sourceBuilder);
        SearchResponse response = client.search(request, RequestOptions.DEFAULT);
        
        // 处理指标聚合结果
        Avg avgPrice = response.getAggregations().get("avg_price");
        Max maxPrice = response.getAggregations().get("max_price");
        Min minPrice = response.getAggregations().get("min_price");
        Sum totalSales = response.getAggregations().get("total_sales");
        ValueCount totalCount = response.getAggregations().get("total_count");
        Stats priceStats = response.getAggregations().get("price_stats");
        ExtendedStats extStats = response.getAggregations().get("price_extended_stats");
        Percentiles percentiles = response.getAggregations().get("price_percentiles");
        Cardinality uniqueBrands = response.getAggregations().get("unique_brands");
        
        System.out.println("=== 指标统计 ===");
        System.out.printf("平均价格: %.2f%n", avgPrice.getValue());
        System.out.printf("最高价格: %.2f%n", maxPrice.getValue());
        System.out.printf("最低价格: %.2f%n", minPrice.getValue());
        System.out.printf("总销量: %.2f%n", totalSales.getValue());
        System.out.printf("商品总数: %d%n", totalCount.getValue());
        System.out.printf("独立品牌数: %d%n", uniqueBrands.getValue());
        
        System.out.println("\n=== 价格统计详情 ===");
        System.out.printf("计数: %d, 最小值: %.2f, 最大值: %.2f%n",
            priceStats.getCount(), priceStats.getMin(), priceStats.getMax());
        System.out.printf("平均值: %.2f, 总和: %.2f%n",
            priceStats.getAvg(), priceStats.getSum());
        
        System.out.println("\n=== 扩展统计 ===");
        System.out.printf("方差: %.2f, 标准差: %.2f%n",
            extStats.getVariance(), extStats.getStdDeviation());
        
        System.out.println("\n=== 百分位数 ===");
        for (Percentile percentile : percentiles) {
            System.out.printf("P%.1f: %.2f%n", percentile.getPercent(), percentile.getValue());
        }
    }
    
    // 4. 范围聚合
    public void rangeAggregation() throws IOException {
        SearchRequest request = new SearchRequest("products");
        SearchSourceBuilder sourceBuilder = new SearchSourceBuilder();
        
        sourceBuilder.query(QueryBuilders.matchAllQuery());
        sourceBuilder.size(0);
        
        // 价格范围聚合
        RangeAggregationBuilder priceRangeAgg = AggregationBuilders.range("price_ranges")
            .field("price")
            .addRange(0, 100)         // 0-100
            .addRange(100, 500)       // 100-500
            .addRange(500, 1000)      // 500-1000
            .addRange(1000, 5000)     // 1000-5000
            .addRange(5000, Double.MAX_VALUE)  // 5000以上
            .keyed(true);  // 使用key而不是匿名范围
        
        // 日期范围聚合
        DateRangeAggregationBuilder dateRangeAgg = AggregationBuilders.dateRange("date_ranges")
            .field("create_time")
            .format("yyyy-MM-dd")
            .addRange("2023-01-01", "2023-12-31")
            .addRange("2024-01-01", "2024-12-31");
        
        sourceBuilder.aggregation(priceRangeAgg);
        sourceBuilder.aggregation(dateRangeAgg);
        
        request.source(sourceBuilder);
        SearchResponse response = client.search(request, RequestOptions.DEFAULT);
        
        // 处理范围聚合结果
        Range priceRanges = response.getAggregations().get("price_ranges");
        System.out.println("=== 价格范围分布 ===");
        for (Range.Bucket bucket : priceRanges.getBuckets()) {
            System.out.printf("价格范围 %s: %d 件商品%n",
                bucket.getKeyAsString(), bucket.getDocCount());
        }
        
        Range dateRanges = response.getAggregations().get("date_ranges");
        System.out.println("\n=== 创建时间分布 ===");
        for (Range.Bucket bucket : dateRanges.getBuckets()) {
            System.out.printf("时间范围 %s: %d 件商品%n",
                bucket.getKeyAsString(), bucket.getDocCount());
        }
    }
    
    // 5. 直方图聚合
    public void histogramAggregation() throws IOException {
        SearchRequest request = new SearchRequest("products");
        SearchSourceBuilder sourceBuilder = new SearchSourceBuilder();
        
        sourceBuilder.query(QueryBuilders.matchAllQuery());
        sourceBuilder.size(0);
        
        // 数值直方图
        HistogramAggregationBuilder priceHistogram = AggregationBuilders.histogram("price_histogram")
            .field("price")
            .interval(500.0)  // 间隔500
            .extendedBounds(0.0, 5000.0)  // 扩展边界
            .minDocCount(0);  // 显示空桶
        
        // 日期直方图
        DateHistogramAggregationBuilder dateHistogram = AggregationBuilders.dateHistogram("sales_by_month")
            .field("sale_date")
            .calendarInterval(DateHistogramInterval.MONTH)  // 按月聚合
            .format("yyyy-MM")
            .minDocCount(0)
            .order(BucketOrder.key(true));  // 按时间顺序
        
        sourceBuilder.aggregation(priceHistogram);
        sourceBuilder.aggregation(dateHistogram);
        
        request.source(sourceBuilder);
        SearchResponse response = client.search(request, RequestOptions.DEFAULT);
        
        // 处理直方图结果
        Histogram priceHistogramResult = response.getAggregations().get("price_histogram");
        System.out.println("=== 价格分布直方图 ===");
        for (Histogram.Bucket bucket : priceHistogramResult.getBuckets()) {
            System.out.printf("价格区间 [%.0f-%.0f): %d 件商品%n",
                bucket.getKeyAsNumber().doubleValue(),
                bucket.getKeyAsNumber().doubleValue() + 500,
                bucket.getDocCount());
        }
        
        Histogram dateHistogramResult = response.getAggregations().get("sales_by_month");
        System.out.println("\n=== 月度销售分布 ===");

    }
    
    // 6. 过滤聚合
    public void filterAggregation() throws IOException {
        SearchRequest request = new SearchRequest("products");
        SearchSourceBuilder sourceBuilder = new SearchSourceBuilder();
        
        sourceBuilder.query(QueryBuilders.matchAllQuery());
        sourceBuilder.size(0);
        
        // 过滤聚合
        FilterAggregationBuilder highPriceFilter = AggregationBuilders.filter("high_price_products",
            QueryBuilders.rangeQuery("price").gte(1000));
        
        highPriceFilter.subAggregation(
            AggregationBuilders.terms("brand_distribution")
                .field("brand.keyword")
        );
        
        // 多个过滤聚合
        FiltersAggregationBuilder filtersAgg = AggregationBuilders.filters("price_filters",
            new FiltersAggregator.KeyedFilter("cheap", 
                QueryBuilders.rangeQuery("price").lt(100)),
            new FiltersAggregator.KeyedFilter("medium", 
                QueryBuilders.rangeQuery("price").gte(100).lt(1000)),
            new FiltersAggregator.KeyedFilter("expensive", 
                QueryBuilders.rangeQuery("price").gte(1000))
        );
        
        sourceBuilder.aggregation(highPriceFilter);
        sourceBuilder.aggregation(filtersAgg);
        
        request.source(sourceBuilder);
        SearchResponse response = client.search(request, RequestOptions.DEFAULT);
        
        // 处理过滤聚合结果
        InternalFilter highPriceFilterResult = response.getAggregations().get("high_price_products");
        System.out.printf("高价商品(≥1000)总数: %d%n", highPriceFilterResult.getDocCount());
        
        InternalFilters filtersResult = response.getAggregations().get("price_filters");
        for (InternalFilters.InternalBucket bucket : filtersResult.getBuckets()) {
            System.out.printf("价格区间 %s: %d 件商品%n",
                bucket.getKey(), bucket.getDocCount());
        }
    }

    // 9. 管道聚合
    public void pipelineAggregation() throws IOException {
        SearchRequest request = new SearchRequest("products");
        SearchSourceBuilder sourceBuilder = new SearchSourceBuilder();
        
        sourceBuilder.query(QueryBuilders.matchAllQuery());
        sourceBuilder.size(0);
        
        // 先按品牌聚合
        TermsAggregationBuilder brandAgg = AggregationBuilders.terms("brand_agg")
            .field("brand.keyword");
        
        // 子聚合：计算平均价格
        brandAgg.subAggregation(
            AggregationBuilders.avg("avg_price").field("price")
        );
        
        // 管道聚合：按平均价格排序
        BucketSortPipelineAggregationBuilder bucketSortAgg = 
            PipelineAggregatorBuilders.bucketSort("sort_by_avg_price", 
                List.of(BucketOrder.aggregation("avg_price", false)));  // 按平均价格降序
        
        brandAgg.subAggregation(bucketSortAgg);
        
        // 管道聚合：统计品牌数量
        StatsBucketPipelineAggregationBuilder statsBucketAgg = 
            PipelineAggregatorBuilders.statsBucket("avg_price_stats", "avg_price");
        
        // 管道聚合：累计和
        CumulativeSumPipelineAggregationBuilder cumulativeSumAgg = 
            PipelineAggregatorBuilders.cumulativeSum("cumulative_sales", "total_sales");
        
        sourceBuilder.aggregation(brandAgg);
        sourceBuilder.aggregation(statsBucketAgg);
        sourceBuilder.aggregation(cumulativeSumAgg);
        
        request.source(sourceBuilder);
        SearchResponse response = client.search(request, RequestOptions.DEFAULT);
        
        // 处理管道聚合结果
        StatsBucket avgPriceStats = response.getAggregations().get("avg_price_stats");
        System.out.println("品牌平均价格统计:");
        System.out.printf("最小值: %.2f, 最大值: %.2f, 平均值: %.2f%n",
            avgPriceStats.getMin(), avgPriceStats.getMax(), avgPriceStats.getAvg());
    }
    
    // 10. 矩阵聚合
    public void matrixAggregation() throws IOException {
        SearchRequest request = new SearchRequest("orders");
        SearchSourceBuilder sourceBuilder = new SearchSourceBuilder();
        
        sourceBuilder.query(QueryBuilders.matchAllQuery());
        sourceBuilder.size(0);
        
        // 矩阵统计
        MatrixStatsAggregationBuilder matrixStatsAgg = 
            AggregationBuilders.matrixStats("price_quantity_stats")
                .fields(List.of("price", "quantity"));
        
        sourceBuilder.aggregation(matrixStatsAgg);
        
        request.source(sourceBuilder);
        SearchResponse response = client.search(request, RequestOptions.DEFAULT);
        
        MatrixStats matrixStats = response.getAggregations().get("price_quantity_stats");
        
        System.out.println("=== 价格-数量矩阵统计 ===");
        for (MatrixStats.MatrixStatsEntry entry : matrixStats) {
            System.out.printf("字段: %s%n", entry.getFieldName());
            System.out.printf("  计数: %d, 平均值: %.2f%n", 
                entry.getCount(), entry.getMean());
            System.out.printf("  方差: %.2f, 偏度: %.2f, 峰度: %.2f%n",
                entry.getVariance(), entry.getSkewness(), entry.getKurtosis());
        }
        
        System.out.printf("协方差矩阵:%n");
        for (int i = 0; i < matrixStats.getFieldCount(); i++) {
            for (int j = 0; j < matrixStats.getFieldCount(); j++) {
                System.out.printf("%.2f\t", matrixStats.getCovariance(i, j));
            }
            System.out.println();
        }
    }
    
    // 11. 复杂聚合查询示例
    public void complexAggregationQuery() throws IOException {
        SearchRequest request = new SearchRequest("sales");
        SearchSourceBuilder sourceBuilder = new SearchSourceBuilder();
        
        // 基础查询条件
        BoolQueryBuilder boolQuery = QueryBuilders.boolQuery()
            .filter(QueryBuilders.rangeQuery("sale_date")
                .gte("2024-01-01")
                .lte("2024-12-31"));
        
        sourceBuilder.query(boolQuery);
        sourceBuilder.size(0);
        
        // 1. 按产品类别聚合
        TermsAggregationBuilder categoryAgg = AggregationBuilders.terms("by_category")
            .field("product.category.keyword")
            .size(20);
        
        // 2. 每个类别下按地区聚合
        TermsAggregationBuilder regionAgg = AggregationBuilders.terms("by_region")
            .field("region.keyword");
        
        // 3. 每个地区计算销售指标
        SumAggregationBuilder totalSalesAgg = AggregationBuilders.sum("total_sales")
            .field("amount");
        
        AvgAggregationBuilder avgPriceAgg = AggregationBuilders.avg("avg_price")
            .field("price");
        
        CardinalityAggregationBuilder uniqueCustomersAgg = 
            AggregationBuilders.cardinality("unique_customers")
                .field("customer_id");
        
        // 4. 管道聚合：按总销售额排序
        BucketSortPipelineAggregationBuilder sortAgg = 
            PipelineAggregatorBuilders.bucketSort("sort_by_sales",
                List.of(BucketOrder.aggregation("total_sales", false)))
            .size(10);
        
        // 5. 顶级聚合：总体统计
        StatsAggregationBuilder overallStats = AggregationBuilders.stats("overall_stats")
            .field("amount");
        
        // 构建聚合树
        categoryAgg.subAggregation(regionAgg);
        regionAgg.subAggregation(totalSalesAgg);
        regionAgg.subAggregation(avgPriceAgg);
        regionAgg.subAggregation(uniqueCustomersAgg);
        categoryAgg.subAggregation(sortAgg);
        
        sourceBuilder.aggregation(categoryAgg);
        sourceBuilder.aggregation(overallStats);
        
        request.source(sourceBuilder);
        SearchResponse response = client.search(request, RequestOptions.DEFAULT);
        
        // 处理聚合结果
        Terms categoryTerms = response.getAggregations().get("by_category");
        Stats overallStatsResult = response.getAggregations().get("overall_stats");
        
        System.out.println("=== 2024年销售分析报告 ===");
        System.out.printf("总销售额: %.2f, 平均单笔: %.2f%n",
            overallStatsResult.getSum(), overallStatsResult.getAvg());
        System.out.printf("最大单笔: %.2f, 最小单笔: %.2f%n",
            overallStatsResult.getMax(), overallStatsResult.getMin());
        
        System.out.println("\n=== 按产品类别分析 ===");
        for (Terms.Bucket categoryBucket : categoryTerms.getBuckets()) {
            System.out.printf("\n品类: %s (订单数: %d)%n",
                categoryBucket.getKeyAsString(), categoryBucket.getDocCount());
            
            Terms regionTerms = categoryBucket.getAggregations().get("by_region");
            
            for (Terms.Bucket regionBucket : regionTerms.getBuckets()) {
                Sum totalSales = regionBucket.getAggregations().get("total_sales");
                Avg avgPrice = regionBucket.getAggregations().get("avg_price");
                Cardinality uniqueCustomers = regionBucket.getAggregations().get("unique_customers");
                
                System.out.printf("  地区: %s%n", regionBucket.getKeyAsString());
                System.out.printf("    销售额: %.2f, 平均价格: %.2f, 客户数: %d%n",
                    totalSales.getValue(), avgPrice.getValue(), uniqueCustomers.getValue());
            }
        }
    }
    
    // 12. 聚合结果工具类
    public class AggregationResultProcessor {
        public void processResponse(SearchResponse response) {
            Aggregations aggregations = response.getAggregations();
            
            if (aggregations == null) {
                System.out.println("无聚合结果");
                return;
            }
            
            // 遍历所有聚合
            for (Aggregation agg : aggregations) {
                processAggregation(agg, 0);
            }
        }
        
        private void processAggregation(Aggregation aggregation, int depth) {
            String indent = "  ".repeat(depth);
            String type = aggregation.getType();
            
            System.out.printf("%s%s: %s%n", indent, aggregation.getName(), type);
            
            switch (type) {
                case "sterms":
                case "lterms":
                case "string_terms":
                    Terms terms = (Terms) aggregation;
                    for (Terms.Bucket bucket : terms.getBuckets()) {
                        System.out.printf("%s  %s: %d%n", 
                            indent, bucket.getKeyAsString(), bucket.getDocCount());
                        
                        // 处理子聚合
                        Aggregations subAggs = bucket.getAggregations();
                        if (subAggs != null) {
                            for (Aggregation subAgg : subAggs) {
                                processAggregation(subAgg, depth + 2);
                            }
                        }
                    }
                    break;
                    
                case "avg":
                    Avg avg = (Avg) aggregation;
                    System.out.printf("%s  值: %.2f%n", indent, avg.getValue());
                    break;
                    
                case "sum":
                    Sum sum = (Sum) aggregation;
                    System.out.printf("%s  值: %.2f%n", indent, sum.getValue());
                    break;
                    
                case "min":
                    Min min = (Min) aggregation;
                    System.out.printf("%s  值: %.2f%n", indent, min.getValue());
                    break;
                    
                case "max":
                    Max max = (Max) aggregation;
                    System.out.printf("%s  值: %.2f%n", indent, max.getValue());
                    break;
                    
                case "stats":
                    Stats stats = (Stats) aggregation;
                    System.out.printf("%s  计数: %d, 最小值: %.2f, 最大值: %.2f, 平均值: %.2f, 总和: %.2f%n",
                        indent, stats.getCount(), stats.getMin(), 
                        stats.getMax(), stats.getAvg(), stats.getSum());
                    break;
                    
                case "histogram":
                    Histogram histogram = (Histogram) aggregation;
                    for (Histogram.Bucket bucket : histogram.getBuckets()) {
                        System.out.printf("%s  [%.2f]: %d%n",
                            indent, bucket.getKeyAsNumber().doubleValue(), bucket.getDocCount());
                    }
                    break;
                    
                // 可以添加更多聚合类型处理...
            }
        }
    }
```

```
public class ElasticsearchAggregationHelper {
    
    /**
     * 创建聚合查询请求
     */
    public static SearchRequest buildAggregationRequest(String index, 
                                                       QueryBuilder query,
                                                       AggregationBuilder... aggregations) {
        SearchRequest request = new SearchRequest(index);
        SearchSourceBuilder sourceBuilder = new SearchSourceBuilder();
        
        sourceBuilder.query(query);
        sourceBuilder.size(0);  // 聚合查询通常不返回文档
        
        for (AggregationBuilder aggregation : aggregations) {
            sourceBuilder.aggregation(aggregation);
        }
        
        request.source(sourceBuilder);
        return request;
    }
    
    /**
     * 解析聚合结果为Map
     */
    public static Map<String, Object> parseAggregations(Aggregations aggregations) {
        Map<String, Object> result = new HashMap<>();
        
        if (aggregations == null) {
            return result;
        }
        
        for (Aggregation aggregation : aggregations) {
            result.put(aggregation.getName(), parseAggregation(aggregation));
        }
        
        return result;
    }
    
    private static Object parseAggregation(Aggregation aggregation) {
        String type = aggregation.getType();
        
        switch (type) {
            case "sterms":
            case "lterms":
                Terms terms = (Terms) aggregation;
                Map<String, Object> termsResult = new LinkedHashMap<>();
                Map<String, Object> buckets = new LinkedHashMap<>();
                
                for (Terms.Bucket bucket : terms.getBuckets()) {
                    Map<String, Object> bucketData = new HashMap<>();
                    bucketData.put("count", bucket.getDocCount());
                    
                    // 处理子聚合
                    Aggregations subAggs = bucket.getAggregations();
                    if (subAggs != null) {
                        bucketData.put("subAggs", parseAggregations(subAggs));
                    }
                    
                    buckets.put(bucket.getKeyAsString(), bucketData);
                }
                
                termsResult.put("buckets", buckets);
                termsResult.put("totalBuckets", terms.getBuckets().size());
                termsResult.put("sumOtherDocCount", terms.getSumOfOtherDocCounts());
                return termsResult;
                
            case "avg":
                return ((Avg) aggregation).getValue();
                
            case "sum":
                return ((Sum) aggregation).getValue();
                
            case "min":
                return ((Min) aggregation).getValue();
                
            case "max":
                return ((Max) aggregation).getValue();
                
            case "stats":
                Stats stats = (Stats) aggregation;
                Map<String, Object> statsMap = new HashMap<>();
                statsMap.put("count", stats.getCount());
                statsMap.put("min", stats.getMin());
                statsMap.put("max", stats.getMax());
                statsMap.put("avg", stats.getAvg());
                statsMap.put("sum", stats.getSum());
                return statsMap;
                
            case "histogram":
            case "date_histogram":
                Histogram histogram = (Histogram) aggregation;
                Map<String, Object> histResult = new LinkedHashMap<>();
                
                for (Histogram.Bucket bucket : histogram.getBuckets()) {
                    String key = bucket.getKeyAsString();
                    histResult.put(key, bucket.getDocCount());
                }
                
                return histResult;
                
            case "range":
                Range range = (Range) aggregation;
                Map<String, Object> rangeResult = new LinkedHashMap<>();
                
                for (Range.Bucket bucket : range.getBuckets()) {
                    rangeResult.put(bucket.getKeyAsString(), bucket.getDocCount());
                }
                
                return rangeResult;
                
            default:
                return aggregation.toString();
        }
    }
    
    /**
     * 生成聚合统计报表
     */
    public static String generateAggregationReport(Aggregations aggregations) {
        StringBuilder report = new StringBuilder();
        
        if (aggregations == null) {
            return "无聚合数据";
        }
        
        for (Aggregation aggregation : aggregations) {
            report.append(generateAggregationDetail(aggregation, 0));
        }
        
        return report.toString();
    }
    
    private static String generateAggregationDetail(Aggregation aggregation, int depth) {
        StringBuilder detail = new StringBuilder();
        String indent = "  ".repeat(depth);
        
        detail.append(String.format("%s[%s] %s%n", 
            indent, aggregation.getType(), aggregation.getName()));
        
        if (aggregation instanceof Terms) {
            Terms terms = (Terms) aggregation;
            for (Terms.Bucket bucket : terms.getBuckets()) {
                detail.append(String.format("%s  %s: %d%n", 
                    indent, bucket.getKeyAsString(), bucket.getDocCount()));
                
                // 递归处理子聚合
                Aggregations subAggs = bucket.getAggregations();
                if (subAggs != null) {
                    for (Aggregation subAgg : subAggs) {
                        detail.append(generateAggregationDetail(subAgg, depth + 2));
                    }
                }
            }
        } else if (aggregation instanceof Stats) {
            Stats stats = (Stats) aggregation;
            detail.append(String.format("%s  计数: %d, 最小值: %.2f, 最大值: %.2f, 平均值: %.2f, 总和: %.2f%n",
                indent, stats.getCount(), stats.getMin(), stats.getMax(), 
                stats.getAvg(), stats.getSum()));
        } else if (aggregation instanceof Avg) {
            detail.append(String.format("%s  平均值: %.2f%n", 
                indent, ((Avg) aggregation).getValue()));
        } else if (aggregation instanceof Sum) {
            detail.append(String.format("%s  总和: %.2f%n", 
                indent, ((Sum) aggregation).getValue()));
        }
        
        return detail.toString();
    }
```

```
public class AggregationBestPractices {

    /**
     * 性能优化的聚合查询
     */
    public SearchResponse optimizedAggregation() throws IOException {
        SearchRequest request = new SearchRequest("large_dataset");
        SearchSourceBuilder sourceBuilder = new SearchSourceBuilder();
        
        // 1. 使用过滤查询而不是match_all
        BoolQueryBuilder boolQuery = QueryBuilders.boolQuery()
            .filter(QueryBuilders.rangeQuery("date").gte("2024-01-01"));
        
        sourceBuilder.query(boolQuery);
        sourceBuilder.size(0);
        
        // 2. 控制聚合桶的数量
        TermsAggregationBuilder termsAgg = AggregationBuilders.terms("category_agg")
            .field("category.keyword")
            .size(100)  // 限制返回桶的数量
            .shardSize(200)  // 设置分片大小
            .executionHint("global_ordinals");  // 执行提示
        
        // 3. 使用近似聚合提高性能
        termsAgg.collectMode(SubAggCollectionMode.DEPTH_FIRST);
        
        // 4. 对于大数据集使用采样聚合
        AggregationBuilder sampledAgg = AggregationBuilders.sampler("sample")
            .shardSize(1000)
            .subAggregation(termsAgg);
        
        sourceBuilder.aggregation(sampledAgg);
        request.source(sourceBuilder);
        
        return client.search(request, RequestOptions.DEFAULT);
    }
    
    /**
     * 聚合查询结果缓存
     */
    public class AggregationCache {
        private final Map<String, Aggregations> cache = new ConcurrentHashMap<>();
        private final long cacheDuration = 5 * 60 * 1000;  // 5分钟
        
        public Aggregations getAggregation(String cacheKey, Supplier<Aggregations> supplier) 
                throws IOException {
            CacheEntry entry = cache.get(cacheKey);
            
            if (entry != null && System.currentTimeMillis() - entry.timestamp < cacheDuration) {
                return entry.aggregations;
            }
            
            // 重新查询
            Aggregations aggregations = supplier.get();
            cache.put(cacheKey, new CacheEntry(aggregations, System.currentTimeMillis()));
            return aggregations;
        }
        
        private static class CacheEntry {
            Aggregations aggregations;
            long timestamp;
            
            CacheEntry(Aggregations aggregations, long timestamp) {
                this.aggregations = aggregations;
                this.timestamp = timestamp;
            }
        }
    }
    
    /**
     * 异步聚合查询
     */
    public void asyncAggregation() {
        SearchRequest request = new SearchRequest("products");
        SearchSourceBuilder sourceBuilder = new SearchSourceBuilder();
        
        sourceBuilder.query(QueryBuilders.matchAllQuery());
```

## 关键总结

1. **聚合类型**：
    
    - 桶聚合：`terms`, `range`, `histogram`, `date_histogram`, `nested`
        
    - 指标聚合：`avg`, `sum`, `min`, `max`, `stats`, `percentiles`, `cardinality`
        
    - 管道聚合：`bucket_sort`, `cumulative_sum`, `stats_bucket`
        
    - 矩阵聚合：`matrix_stats`
        
    
2. **性能优化**：
    
    - 使用 `filter`而不是 `must`进行范围限定
        
    - 控制返回桶的数量（`size`, `shardSize`）
        
    - 使用采样聚合处理大数据集
        
    - 合理使用缓存
        
    
3. **结果处理**：
    
    - 通过 `getAggregations()`获取聚合结果
        
    - 使用类型转换获取具体聚合类型
        
    - 递归处理多级嵌套聚合
        
    
4. **最佳实践**：
    
    - 聚合查询通常设置 `size: 0`避免返回文档
        
    - 使用异步查询提高并发性能
        
    - 实现结果缓存减少重复查询
        
    - 对海量数据使用滚动聚合
