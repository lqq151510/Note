# Elasticsearch DSL 查询详解

DSL（Domain Specific Language）是 Elasticsearch 基于 JSON 的查询语言，用于执行复杂的搜索和聚合操作。

---

## 1. DSL 查询基本结构

```
{
  "query": { ... },      // 查询条件
  "from": 0,             // 分页起始位置
  "size": 10,            // 返回结果数
  "_source": { ... },    // 字段过滤
  "sort": [ ... ],       // 排序
  "highlight": { ... },  // 高亮
  "aggs": { ... }        // 聚合
}
```

---

## 2. 查询类型分类

### 2.1 全文检索查询（Full Text Queries）

#### match 查询

```
{
  "query": {
    "match": {
      "title": {
        "query": "elasticsearch教程",
        "operator": "and",  // 或 "or"
        "minimum_should_match": "75%"
      }
    }
  }
}
```

#### match_phrase 查询

```
{
  "query": {
    "match_phrase": {
      "content": {
        "query": "分布式搜索",
        "slop": 1  // 允许短语中词语的间隔
      }
    }
  }
}
```

#### multi_match 查询

```
{
  "query": {
    "multi_match": {
      "query": "elasticsearch",
      "fields": ["title^2", "content", "description"],  // ^2表示权重
      "type": "best_fields",  // best_fields, most_fields, cross_fields, phrase
      "tie_breaker": 0.3
    }
  }
}
```

#### match_all 查询

```
{
  "query": {
    "match_all": {}
  }
}
```

### 2.2 词项级查询（Term-level Queries）

#### term 查询

```
{
  "query": {
    "term": {
      "status": {
        "value": "published"
      }
    }
  }
}
```

#### terms 查询

```
{
  "query": {
    "terms": {
      "category": ["科技", "教育", "娱乐"]
    }
  }
}
```

#### range 查询

```
{
  "query": {
    "range": {
      "price": {
        "gte": 100,
        "lte": 1000,
        "boost": 2.0
      }
    }
  }
}
// 日期范围
{
  "query": {
    "range": {
      "create_time": {
        "gte": "2024-01-01",
        "lte": "2024-12-31",
        "format": "yyyy-MM-dd"
      }
    }
  }
}
```

#### exists 查询

```
{
  "query": {
    "exists": {
      "field": "description"
    }
  }
}
```

#### prefix 查询

```
{
  "query": {
    "prefix": {
      "title": {
        "value": "elast"
      }
    }
  }
}
```

#### wildcard 查询

```
{
  "query": {
    "wildcard": {
      "title": {
        "value": "elast*ch"
      }
    }
  }
}
```

#### regexp 查询

```
{
  "query": {
    "regexp": {
      "title": {
        "value": "elast.*search",
        "flags": "ALL",
        "max_determinized_states": 10000
      }
    }
  }
}
```

#### fuzzy 查询

```
{
  "query": {
    "fuzzy": {
      "title": {
        "value": "elasticserch",  // 允许拼写错误
        "fuzziness": "AUTO",       // 或 1, 2
        "max_expansions": 50
      }
    }
  }
}
```

## 3. 复合查询（Compound Queries）

### bool 查询（最常用）

```
{
  "query": {
    "bool": {
      "must": [    // 必须满足，参与算分
        { "match": { "title": "搜索" } }
      ],
      "filter": [  // 必须满足，不参与算分，性能更好
        { "term": { "status": "published" } },
        { "range": { "publish_date": { "gte": "2024-01-01" } } }
      ],
      "should": [  // 应该满足，参与算分
        { "match": { "content": "教程" } },
        { "match": { "description": "入门" } }
      ],
      "must_not": [  // 必须不满足
        { "term": { "category": "广告" } }
      ],
      "minimum_should_match": 1
    }
  }
}
```

### boosting 查询

```
{
  "query": {
    "boosting": {
      "positive": {
        "match": { "content": "elasticsearch" }
      },
      "negative": {
        "match": { "content": "教程" }
      },
      "negative_boost": 0.2
    }
  }
}
```

### constant_score 查询

```
{
  "query": {
    "constant_score": {
      "filter": {
        "term": { "status": "active" }
      },
      "boost": 1.2
    }
  }
}
```

## 5. 特殊查询

### script 查询

```
{
  "query": {
    "script": {
      "script": {
        "source": "doc['price'].value * doc['discount'].value > params.min_price",
        "params": {
          "min_price": 50
        }
      }
    }
  }
}
```

### nested 查询

```
{
  "query": {
    "nested": {
      "path": "products",
      "query": {
        "bool": {
          "must": [
            { "match": { "products.name": "手机" } },
            { "range": { "products.price": { "gte": 1000 } } }
          ]
        }
      },
      "score_mode": "avg"
    }
  }
}
```

### has_child / has_parent 查询

```
// 子文档查询
{
  "query": {
    "has_child": {
      "type": "comment",
      "query": {
        "match": { "content": "good" }
      }
    }
  }
}
```

## 7. 聚合查询

### 指标聚合

```
{
  "size": 0,
  "aggs": {
    "avg_price": {
      "avg": { "field": "price" }
    },
    "max_price": {
      "max": { "field": "price" }
    },
    "min_price": {
      "min": { "field": "price" }
    },
    "sum_price": {
      "sum": { "field": "price" }
    },
    "stats_price": {  // 统计所有指标
      "stats": { "field": "price" }
    }
  }
}
```

### 桶聚合

```
{
  "size": 0,
  "aggs": {
    "group_by_category": {
      "terms": {
        "field": "category",
        "size": 10,
        "order": { "_count": "desc" }
      },
      "aggs": {  // 子聚合
        "avg_price": {
          "avg": { "field": "price" }
        },
        "price_stats": {
          "stats": { "field": "price" }
        }
      }
    }
  }
}
```

### 范围聚合

```
{
  "size": 0,
  "aggs": {
    "price_ranges": {
      "range": {
        "field": "price",
        "ranges": [
          { "to": 100 },
          { "from": 100, "to": 500 },
          { "from": 500, "to": 1000 },
          { "from": 1000 }
        ]
      },
      "aggs": {
        "avg_rating": {
          "avg": { "field": "rating" }
        }
      }
    }
  }
}
```

### 日期直方图聚合

```
{
  "size": 0,
  "aggs": {
    "sales_over_time": {
      "date_histogram": {
        "field": "sale_date",
        "calendar_interval": "month",
        "format": "yyyy-MM",
        "min_doc_count": 0
      },
      "aggs": {
        "total_sales": {
          "sum": { "field": "amount" }
        }
      }
    }
  }
}
```

### 地理位置聚合

```
{
  "size": 0,
  "aggs": {
    "viewport": {
      "geo_bounds": {
        "field": "location",
        "wrap_longitude": true
      }
    },
    "avg_location": {
      "geo_centroid": {
        "field": "location"
      }
    }
  }
}
```

## 9. 性能优化技巧

### 9.1 使用过滤器缓存

```
{
  "query": {
    "bool": {
      "filter": [  // filter 不参与算分，可以被缓存
        { "term": { "category": "电子产品" } },
        { "range": { "price": { "gte": 1000 } } }
      ],
      "must": [  // must 参与算分，不使用缓存
        { "match": { "description": "高性能" } }
      ]
    }
  }
}
```

### 9.2 查询重写

```
{
  "query": {
    "prefix": {
      "title": {
        "value": "elast",
        "rewrite": "top_terms_boost_10"  // 控制重写策略
      }
    }
  }
}
```

### 9.3 分页优化

```
{
  "size": 100,
  "sort": ["_doc"],  // 使用 _doc 排序，避免相关性计算
  "search_after": [123456]  // 用于深度分页
}
```

### 9.4 查询折叠

```
{
  "collapse": {
    "field": "brand",  // 按品牌折叠
    "inner_hits": {
      "name": "best_products",
      "size": 3,
      "sort": [{ "sales": "desc" }]
    }
  }
}
```

## 10. 使用 RestHighLevelClient 执行 DSL 查询

```
public class DSLQueryExample {
    
    public void searchWithDSL(RestHighLevelClient client) throws IOException {
        // 1. 构建查询
        SearchSourceBuilder sourceBuilder = new SearchSourceBuilder();
        
        // 构建bool查询
        BoolQueryBuilder boolQuery = QueryBuilders.boolQuery();
        
        // must条件
        boolQuery.must(QueryBuilders.matchQuery("title", "手机"));
        
        // filter条件
        boolQuery.filter(QueryBuilders.termQuery("status", "published"));
        boolQuery.filter(QueryBuilders.rangeQuery("price").gte(1000).lte(5000));
        
        // should条件
        boolQuery.should(QueryBuilders.matchQuery("brand", "华为"));
        boolQuery.should(QueryBuilders.matchQuery("brand", "小米"));
        boolQuery.minimumShouldMatch(1);
        
        sourceBuilder.query(boolQuery);
        
        // 2. 设置分页
        sourceBuilder.from(0);
        sourceBuilder.size(10);
        
        // 3. 设置排序
        sourceBuilder.sort(SortBuilders.fieldSort("sales").order(SortOrder.DESC));
        sourceBuilder.sort(SortBuilders.scoreSort().order(SortOrder.DESC));
        
        // 4. 设置高亮
        HighlightBuilder highlightBuilder = new HighlightBuilder();
        highlightBuilder.field("title");
        highlightBuilder.field("description");
        highlightBuilder.preTags("<em>");
        highlightBuilder.postTags("</em>");
        sourceBuilder.highlighter(highlightBuilder);
        
        // 5. 设置聚合
        TermsAggregationBuilder brandAgg = AggregationBuilders.terms("brand_agg")
            .field("brand")
            .size(10);
        
        brandAgg.subAggregation(AggregationBuilders.avg("avg_price").field("price"));
        sourceBuilder.aggregation(brandAgg);
        
        // 6. 执行查询
        SearchRequest request = new SearchRequest("products");
        request.source(sourceBuilder);
        
        SearchResponse response = client.search(request, RequestOptions.DEFAULT);
        
        // 7. 处理结果
        SearchHit[] hits = response.getHits().getHits();
        for (SearchHit hit : hits) {
            System.out.println("文档ID: " + hit.getId());
            System.out.println("文档内容: " + hit.getSourceAsString());
            System.out.println("得分: " + hit.getScore());
            
            // 处理高亮
            Map<String, HighlightField> highlightFields = hit.getHighlightFields();
            if (highlightFields.containsKey("title")) {
                System.out.println("高亮标题: " + highlightFields.get("title").fragments()[0]);
            }
        }
        
        // 处理聚合结果
        Terms brandTerms = response.getAggregations().get("brand_agg");
        for (Terms.Bucket bucket : brandTerms.getBuckets()) {
            System.out.println("品牌: " + bucket.getKey() + 
                ", 数量: " + bucket.getDocCount());
            
            Avg avgPrice = bucket.getAggregations().get("avg_price");
            System.out.println("平均价格: " + avgPrice.getValue());
        }
    }
}
```

## 关键总结

1. **查询类型选择**：
    
    - 全文检索用 `match`、`multi_match`
        
    - 精确匹配用 `term`、`terms`
        
    - 复杂逻辑用 `bool`查询
        
    
2. **性能优化**：
    
    - 能放 `filter`的就不要放 `must`
        
    - 合理使用缓存
        
    - 避免深度分页
        
    
3. **相关性控制**：
    
    - 通过 `boost`控制字段权重
        
    - 使用 `minimum_should_match`控制匹配度
        
    - 合理使用 `function_score`自定义评分
        
    
4. **聚合分析**：
    
    - 桶聚合用于分组
        
    - 指标聚合用于计算
        
    - 支持多级嵌套聚合
