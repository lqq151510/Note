# Spring Cloud Gateway 动态路由完整实现

动态路由是 API 网关的核心功能，允许在不重启网关的情况下动态添加、修改、删除路由规则。以下是基于 Spring Cloud Gateway 的完整动态路由实现方案。

## 一、动态路由架构设计

### 1. **整体架构**

```
数据库/配置中心 → 路由配置变更 → 事件监听 → 路由刷新 → 路由生效
     ↓               ↓           ↓         ↓         ↓
 路由存储       网关管理API    动态监听   刷新机制   实时生效
```

## 二、基于内存的动态路由

### 1. **内存路由管理器**

```
import org.springframework.cloud.gateway.event.RefreshRoutesEvent;
import org.springframework.cloud.gateway.route.RouteDefinition;
import org.springframework.cloud.gateway.route.RouteDefinitionLocator;
import org.springframework.cloud.gateway.route.RouteDefinitionWriter;
import org.springframework.context.ApplicationEventPublisher;
import org.springframework.stereotype.Component;
import reactor.core.publisher.Flux;
import reactor.core.publisher.Mono;

import java.util.List;
import java.util.Map;
import java.util.concurrent.ConcurrentHashMap;

@Component
public class InMemoryRouteManager {
    
    // 内存存储路由定义
    private final Map<String, RouteDefinition> routeDefinitionMap = 
        new ConcurrentHashMap<>();
    
    @Autowired
    private RouteDefinitionWriter routeDefinitionWriter;
    
    @Autowired
    private ApplicationEventPublisher publisher;
    
    @Autowired
    private RouteDefinitionLocator routeDefinitionLocator;
    
    /**
     * 添加路由
     */
    public Mono<Boolean> add(RouteDefinition routeDefinition) {
        try {
            // 1. 检查路由ID是否已存在
            if (routeDefinitionMap.containsKey(routeDefinition.getId())) {
                return Mono.just(false);
            }
            
            // 2. 保存到内存
            routeDefinitionMap.put(routeDefinition.getId(), routeDefinition);
            
            // 3. 保存到网关
            routeDefinitionWriter.save(Mono.just(routeDefinition)).subscribe();
            
            // 4. 发布路由刷新事件
            notifyChanged();
            
            // 5. 记录日志
            logRouteChange("ADD", routeDefinition);
            
            return Mono.just(true);
            
        } catch (Exception e) {
            log.error("添加路由失败", e);
            return Mono.error(e);
        }
    }
    
    /**
     * 更新路由
     */
    public Mono<Boolean> update(RouteDefinition routeDefinition) {
        try {
            // 1. 检查路由是否存在
            if (!routeDefinitionMap.containsKey(routeDefinition.getId())) {
                return Mono.just(false);
            }
            
            // 2. 先删除旧路由
            delete(routeDefinition.getId());
            
            // 3. 添加新路由
            add(routeDefinition);
            
            // 4. 记录日志
            logRouteChange("UPDATE", routeDefinition);
            
            return Mono.just(true);
            
        } catch (Exception e) {
            log.error("更新路由失败", e);
            return Mono.error(e);
        }
    }
    
    /**
     * 删除路由
     */
    public Mono<Boolean> delete(String routeId) {
        try {
            // 1. 检查路由是否存在
            if (!routeDefinitionMap.containsKey(routeId)) {
                return Mono.just(false);
            }
            
            // 2. 从内存删除
            routeDefinitionMap.remove(routeId);
            
            // 3. 从网关删除
            routeDefinitionWriter.delete(Mono.just(routeId)).subscribe();
            
            // 4. 发布路由刷新事件
            notifyChanged();
            
            // 5. 记录日志
            logRouteChange("DELETE", routeId);
            
            return Mono.just(true);
            
        } catch (Exception e) {
            log.error("删除路由失败", e);
            return Mono.error(e);
        }
    }
    
    /**
     * 获取所有路由
     */
    public Flux<RouteDefinition> getAllRoutes() {
        return Flux.fromIterable(routeDefinitionMap.values());
    }
    
    /**
     * 获取单个路由
     */
    public Mono<RouteDefinition> getRoute(String routeId) {
        RouteDefinition route = routeDefinitionMap.get(routeId);
        return route != null ? Mono.just(route) : Mono.empty();
    }
    
    /**
     * 刷新所有路由
     */
    public void refreshAll() {
        // 清空现有路由
        routeDefinitionMap.values().forEach(route -> 
            routeDefinitionWriter.delete(Mono.just(route.getId())).subscribe()
        );
        
        // 重新加载所有路由
        routeDefinitionMap.clear();
        loadAllRoutes();
        
        // 发布刷新事件
        notifyChanged();
    }
    
    /**
     * 发布路由变更事件
     */
    private void notifyChanged() {
        publisher.publishEvent(new RefreshRoutesEvent(this));
    }
    
    /**
     * 记录路由变更日志
     */
    private void logRouteChange(String action, RouteDefinition route) {
        log.info("路由{}: id={}, uri={}, predicates={}", 
            action, route.getId(), route.getUri(), route.getPredicates());
    }
    
    private void logRouteChange(String action, String routeId) {
        log.info("路由{}: id={}", action, routeId);
    }
}
```

```
    @Column(name = "created_time")
    private LocalDateTime createdTime;

    private String updatedBy;
    
    @Column(name = "updated_time")
    private LocalDateTime updatedTime;

    @Version
    private Integer version = 1;
    
    @Column(name = "description", length = 512)
    private String description;
    
    // JSON转换器
    @Converter
    public static class PredicateListConverter implements AttributeConverter<List<PredicateDefinition>, String> {
        private final ObjectMapper objectMapper = new ObjectMapper();
        
        @Override
        public String convertToDatabaseColumn(List<PredicateDefinition> predicates) {
            try {
                return objectMapper.writeValueAsString(predicates);
            } catch (JsonProcessingException e) {
                throw new RuntimeException("转换断言失败", e);
            }
        }
        
        @Override
        public List<PredicateDefinition> convertToEntityAttribute(String dbData) {
            try {
                return objectMapper.readValue(dbData, 
                    new TypeReference<List<PredicateDefinition>>() {});
            } catch (JsonProcessingException e) {
                throw new RuntimeException("解析断言失败", e);
            }
        }
    }
    
    // 转换为RouteDefinition
    public RouteDefinition toRouteDefinition() {
        RouteDefinition definition = new RouteDefinition();
        definition.setId(this.id);
        definition.setUri(URI.create(this.uri));
        definition.setOrder(this.order);
        definition.setPredicates(this.predicates);
        definition.setFilters(this.filters);
        definition.setMetadata(this.metadata);
        return definition;
    }
    
    // 从RouteDefinition转换
    public static GatewayRoute fromRouteDefinition(RouteDefinition definition) {
        return GatewayRoute.builder()
            .id(definition.getId())
            .routeName(definition.getId())
            .uri(definition.getUri().toString())
            .order(definition.getOrder())
            .predicates(definition.getPredicates())
            .filters(definition.getFilters())
            .metadata(definition.getMetadata())
            .build();
    }
```

### 3. **数据库路由仓库**

```
import org.springframework.cloud.gateway.event.RefreshRoutesEvent;
import org.springframework.cloud.gateway.route.RouteDefinition;
import org.springframework.cloud.gateway.route.RouteDefinitionLocator;
import org.springframework.cloud.gateway.route.RouteDefinitionWriter;
import org.springframework.context.ApplicationEventPublisher;
import org.springframework.data.redis.core.RedisTemplate;
import org.springframework.stereotype.Repository;
import org.springframework.transaction.annotation.Transactional;
import reactor.core.publisher.Flux;
import reactor.core.publisher.Mono;

import java.util.List;
import java.util.concurrent.TimeUnit;
import java.util.stream.Collectors;

@Repository
@Slf4j
public class DatabaseRouteRepository implements RouteDefinitionLocator {
    
    @Autowired
    private GatewayRouteMapper routeMapper;
    
    @Autowired
    private RedisTemplate<String, Object> redisTemplate;
    
    @Autowired
    private RouteDefinitionWriter routeDefinitionWriter;
    
    @Autowired
    private ApplicationEventPublisher publisher;
    
    private static final String ROUTE_CACHE_KEY = "gateway:routes";
    private static final String ROUTE_DETAIL_CACHE_PREFIX = "gateway:route:";
    private static final long CACHE_TTL = 300; // 5分钟
    
    /**
     * 从数据库加载所有路由
     */
    @Override
    public Flux<RouteDefinition> getRouteDefinitions() {
        return Flux.fromIterable(loadAllRoutes());
    }
    
    /**
     * 加载所有启用的路由
     */
    private List<RouteDefinition> loadAllRoutes() {
        // 1. 尝试从缓存获取
        List<RouteDefinition> cachedRoutes = getRoutesFromCache();
        if (cachedRoutes != null && !cachedRoutes.isEmpty()) {
            log.info("从缓存加载路由，数量: {}", cachedRoutes.size());
            return cachedRoutes;
        }
        
        // 2. 从数据库加载
        List<GatewayRoute> gatewayRoutes = routeMapper.findByStatus(1);
        List<RouteDefinition> routeDefinitions = gatewayRoutes.stream()
            .map(GatewayRoute::toRouteDefinition)
            .collect(Collectors.toList());
        
        log.info("从数据库加载路由，数量: {}", routeDefinitions.size());
        
        // 3. 缓存路由
        cacheRoutes(routeDefinitions);
        
        return routeDefinitions;
    }
    
    /**
     * 添加路由
     */
    @Transactional
    public Mono<RouteResult> addRoute(RouteDefinition routeDefinition, String operator) {
        try {
            // 1. 检查路由是否已存在
            if (routeMapper.existsById(routeDefinition.getId())) {
                return Mono.just(RouteResult.error("路由已存在"));
            }
            
            // 2. 保存到数据库
            GatewayRoute gatewayRoute = GatewayRoute.fromRouteDefinition(routeDefinition);
            gatewayRoute.setCreatedBy(operator);
            gatewayRoute.setCreatedTime(LocalDateTime.now());
            gatewayRoute.setUpdatedBy(operator);
            gatewayRoute.setUpdatedTime(LocalDateTime.now());
            
            routeMapper.insert(gatewayRoute);
            
            // 3. 保存到网关
            routeDefinitionWriter.save(Mono.just(routeDefinition)).subscribe();
            
            // 4. 发布刷新事件
            publisher.publishEvent(new RefreshRoutesEvent(this));
            
            // 5. 清除缓存
            clearRouteCache();
            
            // 6. 记录历史
            saveRouteHistory(gatewayRoute, "ADD", operator, "添加路由");
            
            log.info("路由添加成功: {}", routeDefinition.getId());
            return Mono.just(RouteResult.success("路由添加成功"));
            
        } catch (Exception e) {
            log.error("添加路由失败", e);
            return Mono.just(RouteResult.error("添加路由失败: " + e.getMessage()));
        }
    }
    
    /**
     * 更新路由
     */
    @Transactional
    public Mono<RouteResult> updateRoute(RouteDefinition routeDefinition, String operator) {
        try {
            // 1. 检查路由是否存在
            GatewayRoute existingRoute = routeMapper.selectById(routeDefinition.getId());
            if (existingRoute == null) {
                return Mono.just(RouteResult.error("路由不存在"));
            }
            
            // 2. 保存历史快照
            saveRouteSnapshot(existingRoute, "UPDATE_BEFORE");
            
            // 3. 更新数据库
            GatewayRoute gatewayRoute = GatewayRoute.fromRouteDefinition(routeDefinition);
            gatewayRoute.setUpdatedBy(operator);
            gatewayRoute.setUpdatedTime(LocalDateTime.now());
            gatewayRoute.setVersion(existingRoute.getVersion() + 1);
            
            int updated = routeMapper.updateById(gatewayRoute);
            if (updated == 0) {
                return Mono.just(RouteResult.error("路由更新失败，版本冲突"));
            }
            
            // 4. 先删除旧路由
            routeDefinitionWriter.delete(Mono.just(routeDefinition.getId())).subscribe();
            
            // 5. 添加新路由
            routeDefinitionWriter.save(Mono.just(routeDefinition)).subscribe();
            
            // 6. 发布刷新事件
            publisher.publishEvent(new RefreshRoutesEvent(this));
            
            // 7. 清除缓存
            clearRouteCache();
            
            // 8. 记录历史
            saveRouteHistory(gatewayRoute, "UPDATE", operator, "更新路由");
            
            log.info("路由更新成功: {}", routeDefinition.getId());
            return Mono.just(RouteResult.success("路由更新成功"));
            
        } catch (Exception e) {
            log.error("更新路由失败", e);
            return Mono.just(RouteResult.error("更新路由失败: " + e.getMessage()));
        }
    }
    
    /**
     * 删除路由
     */
    @Transactional
    public Mono<RouteResult> deleteRoute(String routeId, String operator) {
        try {
            // 1. 检查路由是否存在
            GatewayRoute existingRoute = routeMapper.selectById(routeId);
            if (existingRoute == null) {
                return Mono.just(RouteResult.error("路由不存在"));
            }
            
            // 2. 保存历史快照
            saveRouteSnapshot(existingRoute, "DELETE");
            
            // 3. 逻辑删除（更新状态）
            existingRoute.setStatus(0);
            existingRoute.setUpdatedBy(operator);
            existingRoute.setUpdatedTime(LocalDateTime.now());
            
            routeMapper.updateById(existingRoute);
            
            // 4. 从网关删除
            routeDefinitionWriter.delete(Mono.just(routeId)).subscribe();
            
            // 5. 发布刷新事件
            publisher.publishEvent(new RefreshRoutesEvent(this));
            
            // 6. 清除缓存
            clearRouteCache();
            
            // 7. 记录历史
            saveRouteHistory(existingRoute, "DELETE", operator, "删除路由");
            
            log.info("路由删除成功: {}", routeId);
            return Mono.just(RouteResult.success("路由删除成功"));
            
        } catch (Exception e) {
            log.error("删除路由失败", e);
            return Mono.just(RouteResult.error("删除路由失败: " + e.getMessage()));
        }
    }
    
    /**
     * 批量导入路由
     */
    @Transactional
    public Mono<BatchRouteResult> batchImportRoutes(List<RouteDefinition> routeDefinitions, String operator) {
        BatchRouteResult result = new BatchRouteResult();
        
        routeDefinitions.forEach(route -> {
            try {
                RouteResult routeResult = addRoute(route, operator).block();
                if (routeResult.isSuccess()) {
                    result.addSuccess(route.getId());
                } else {
                    result.addFailure(route.getId(), routeResult.getMessage());
                }
            } catch (Exception e) {
                result.addFailure(route.getId(), e.getMessage());
            }
        });
        
        return Mono.just(result);
    }
    
    /**
     * 从缓存获取路由
     */
    @SuppressWarnings("unchecked")
    private List<RouteDefinition> getRoutesFromCache() {
        try {
            return (List<RouteDefinition>) redisTemplate.opsForValue().get(ROUTE_CACHE_KEY);
        } catch (Exception e) {
            log.warn("从缓存获取路由失败", e);
            return null;
        }
    }
    
    /**
     * 缓存路由
     */
    private void cacheRoutes(List<RouteDefinition> routes) {
        try {
            redisTemplate.opsForValue().set(
                ROUTE_CACHE_KEY, 
                routes, 
                CACHE_TTL, 
                TimeUnit.SECONDS
            );
        } catch (Exception e) {
            log.warn("缓存路由失败", e);
        }
    }
    
    /**
     * 清除路由缓存
     */
    private void clearRouteCache() {
        try {
            redisTemplate.delete(ROUTE_CACHE_KEY);
        } catch (Exception e) {
            log.warn("清除路由缓存失败", e);
        }
    }
    
    /**
     * 保存路由历史
     */
    private void saveRouteHistory(GatewayRoute route, String operationType, 
                                 String operator, String remark) {
        try {
            GatewayRouteHistory history = GatewayRouteHistory.builder()
                .routeId(route.getId())
                .operationType(operationType)
                .routeSnapshot(JSON.toJSONString(route))
                .operator(operator)
                .remark(remark)
                .operationTime(LocalDateTime.now())
                .build();
            
            routeHistoryMapper.insert(history);
        } catch (Exception e) {
            log.error("保存路由历史失败", e);
        }
    }
    
    @Data
    @Builder
    public static class RouteResult {
        private boolean success;
        private String message;
        private Object data;
        
        public static RouteResult success(String message) {
            return RouteResult.builder()
                .success(true)
                .message(message)
                .build();
        }
        
        public static RouteResult error(String message) {
            return RouteResult.builder()
                .success(false)
                .message(message)
                .build();
        }
    }
    
    @Data
    public static class BatchRouteResult {
        private boolean success = true;
        private List<String> successRoutes = new ArrayList<>();
        private Map<String, String> failureRoutes = new HashMap<>();
        
        public void addSuccess(String routeId) {
            successRoutes.add(routeId);
        }
        
        public void addFailure(String routeId, String reason) {
            failureRoutes.put(routeId, reason);
            success = false;
        }
    }
}
```

## 四、动态路由管理API

### 1. **RESTful API 设计**

```
@RestController
@RequestMapping("/api/gateway/routes")
@Slf4j
@Validated
public class DynamicRouteController {
    
    @Autowired
    private DatabaseRouteRepository routeRepository;
    
    @Autowired
    private RouteValidator routeValidator;
    
    /**
     * 获取所有路由
     */
    @GetMapping
    public Mono<ApiResponse<List<RouteDTO>>> getAllRoutes(
            @RequestParam(required = false) Integer status) {
        
        List<GatewayRoute> routes = status != null ? 
            routeMapper.findByStatus(status) : 
            routeMapper.findAll();
        
        List<RouteDTO> routeDTOs = routes.stream()
            .map(RouteDTO::fromEntity)
            .collect(Collectors.toList());
        
        return Mono.just(ApiResponse.success(routeDTOs));
    }
    
    /**
     * 获取路由详情
     */
    @GetMapping("/{routeId}")
    public Mono<ApiResponse<RouteDTO>> getRoute(@PathVariable String routeId) {
        GatewayRoute route = routeMapper.selectById(routeId);
        if (route == null) {
            return Mono.just(ApiResponse.error(404, "路由不存在"));
        }
        return Mono.just(ApiResponse.success(RouteDTO.fromEntity(route)));
    }
    
    /**
     * 创建路由
     */
    @PostMapping
    public Mono<ApiResponse<DatabaseRouteRepository.RouteResult>> createRoute(
            @Valid @RequestBody RouteCreateRequest request,
            @RequestAttribute String operator) {
        
        // 1. 验证请求
        ValidationResult validation = routeValidator.validateCreateRequest(request);
        if (!validation.isValid()) {
            return Mono.just(ApiResponse.error(400, validation.getMessage()));
        }
        
        // 2. 构建路由定义
        RouteDefinition routeDefinition = buildRouteDefinition(request);
        
        // 3. 添加路由
        return routeRepository.addRoute(routeDefinition, operator)
            .map(result -> {
                if (result.isSuccess()) {
                    return ApiResponse.success(result);
                } else {
                    return ApiResponse.error(400, result.getMessage());
                }
            });
    }
    
    /**
     * 更新路由
     */
    @PutMapping("/{routeId}")
    public Mono<ApiResponse<DatabaseRouteRepository.RouteResult>> updateRoute(
            @PathVariable String routeId,
            @Valid @RequestBody RouteUpdateRequest request,
            @RequestAttribute String operator) {
        
        // 1. 验证请求
        ValidationResult validation = routeValidator.validateUpdateRequest(request);
        if (!validation.isValid()) {
            return Mono.just(ApiResponse.error(400, validation.getMessage()));
        }
        
        // 2. 构建路由定义
        RouteDefinition routeDefinition = buildRouteDefinition(request);
        routeDefinition.setId(routeId);
        
        // 3. 更新路由
        return routeRepository.updateRoute(routeDefinition, operator)
            .map(result -> {
                if (result.isSuccess()) {
                    return ApiResponse.success(result);
                } else {
                    return ApiResponse.error(400, result.getMessage());
                }
            });
    }
    
    /**
     * 删除路由
     */
    @DeleteMapping("/{routeId}")
    public Mono<ApiResponse<DatabaseRouteRepository.RouteResult>> deleteRoute(
            @PathVariable String routeId,
            @RequestAttribute String operator) {
        
        return routeRepository.deleteRoute(routeId, operator)
            .map(result -> {
                if (result.isSuccess()) {
                    return ApiResponse.success(result);
                } else {
                    return ApiResponse.error(400, result.getMessage());
                }
            });
    }
    
    /**
     * 启用/禁用路由
     */
    @PatchMapping("/{routeId}/status")
    public Mono<ApiResponse<DatabaseRouteRepository.RouteResult>> updateRouteStatus(
            @PathVariable String routeId,
            @RequestParam Integer status,
            @RequestAttribute String operator) {
        
        if (status != 0 && status != 1) {
            return Mono.just(ApiResponse.error(400, "状态值无效"));
        }
        
        GatewayRoute route = routeMapper.selectById(routeId);
        if (route == null) {
            return Mono.just(ApiResponse.error(404, "路由不存在"));
        }
        
        route.setStatus(status);
        route.setUpdatedBy(operator);
        route.setUpdatedTime(LocalDateTime.now());
        
        routeMapper.updateById(route);
        
        // 刷新路由
        refreshRoutes();
        
        return Mono.just(ApiResponse.success("路由状态更新成功"));
    }
    
    /**
     * 批量导入路由
     */
    @PostMapping("/batch-import")
    public Mono<ApiResponse<DatabaseRouteRepository.BatchRouteResult>> batchImport(
            @Valid @RequestBody BatchRouteRequest request,
            @RequestAttribute String operator) {
        
        List<RouteDefinition> routeDefinitions = request.getRoutes().stream()
            .map(this::buildRouteDefinition)
            .collect(Collectors.toList());
        
        return routeRepository.batchImportRoutes(routeDefinitions, operator)
            .map(ApiResponse::success);
    }
    
    /**
     * 导出路由配置
     */
    @GetMapping("/export")
    public Mono<ResponseEntity<String>> exportRoutes(
            @RequestParam(required = false) String format) {
        
        List<GatewayRoute> routes = routeMapper.findAll();
        String exportContent;
        
        if ("yaml".equalsIgnoreCase(format)) {
            exportContent = exportToYaml(routes);
        } else {
            exportContent = exportToJson(routes);
        }
        
        HttpHeaders headers = new HttpHeaders();
        headers.add("Content-Disposition", 
            "attachment; filename=routes_" + System.currentTimeMillis() + 
            ("yaml".equalsIgnoreCase(format) ? ".yaml" : ".json"));
        
        return Mono.just(ResponseEntity.ok()
            .headers(headers)
            .body(exportContent));
    }
    
    /**
     * 刷新所有路由
     */
    @PostMapping("/refresh")
    public Mono<ApiResponse<Void>> refreshRoutes() {
        publisher.publishEvent(new RefreshRoutesEvent(this));
        return Mono.just(ApiResponse.success("路由刷新成功"));
    }
```

```
@Component
@Slf4j
public class RouteValidator {
    
    @Autowired
    private Environment environment;
    
    /**
     * 验证路由定义
     */
    public ValidationResult validateRoute(RouteDefinition routeDefinition) {
        List<String> errors = new ArrayList<>();
        
        // 1. 验证ID
        if (!StringUtils.hasText(routeDefinition.getId())) {
            errors.add("路由ID不能为空");
        } else if (!routeDefinition.getId().matches("^[a-zA-Z0-9_-]+$")) {
            errors.add("路由ID只能包含字母、数字、下划线和横线");
        }
        
        // 2. 验证URI
        try {
            URI uri = routeDefinition.getUri();
            if (uri == null) {
                errors.add("路由URI不能为空");
            } else if (!isValidUri(uri)) {
                errors.add("路由URI格式无效: " + uri);
            }
        } catch (Exception e) {
            errors.add("路由URI格式错误: " + e.getMessage());
        }
        
        // 3. 验证断言
        if (routeDefinition.getPredicates() == null || 
            routeDefinition.getPredicates().isEmpty()) {
            errors.add("路由断言不能为空");
        } else {
            validatePredicates(routeDefinition.getPredicates(), errors);
        }
        
        // 4. 验证过滤器
        if (routeDefinition.getFilters() != null) {
            validateFilters(routeDefinition.getFilters(), errors);
        }
        
        // 5. 验证顺序
        if (routeDefinition.getOrder() != null && routeDefinition.getOrder() < 0) {
            errors.add("路由顺序不能小于0");
        }
        
        if (errors.isEmpty()) {
            return ValidationResult.success();
        } else {
            return ValidationResult.error(String.join("; ", errors));
        }
    }
    
    /**
     * 验证URI是否有效
     */
    private boolean isValidUri(URI uri) {
        String scheme = uri.getScheme();
        String host = uri.getHost();
        
        if (scheme == null || host == null) {
            return false;
        }
        
        // 支持的协议
        Set<String> supportedSchemes = Set.of("http", "https", "lb", "ws", "wss");
        if (!supportedSchemes.contains(scheme)) {
            return false;
        }
        
        // 特殊验证：lb协议
        if ("lb".equals(scheme)) {
            // 验证服务名格式
            String serviceName = host;
            if (!serviceName.matches("^[a-zA-Z0-9-]+$")) {
                return false;
            }
        }
        
        return true;
    }
    
    /**
     * 验证断言
     */
    private void validatePredicates(List<PredicateDefinition> predicates, List<String> errors) {
        Set<String> supportedPredicates = Set.of(
            "Path", "Method", "Header", "Query", "Cookie", 
            "Host", "RemoteAddr", "After", "Before", "Between", 
            "Weight", "CloudFoundryRouteService"
        );
        
        predicates.forEach(predicate -> {
            String name = predicate.getName();
            if (!supportedPredicates.contains(name)) {
                errors.add("不支持的断言类型: " + name);
            }
            
            // 验证Path断言的路径格式
            if ("Path".equals(name)) {
                Map<String, String> args = predicate.getArgs();
                String pattern = args.get("pattern");
                if (pattern == null || pattern.trim().isEmpty()) {
                    errors.add("Path断言缺少pattern参数");
                } else if (!pattern.startsWith("/")) {
                    errors.add("Path断言必须以/开头: " + pattern);
                }
            }
        });
    }
    
    /**
     * 验证过滤器
     */
    private void validateFilters(List<FilterDefinition> filters, List<String> errors) {
        Set<String> supportedFilters = Set.of(
            "StripPrefix", "PrefixPath", "RewritePath", "SetPath",
            "AddRequestHeader", "RemoveRequestHeader", "SetRequestHeader",
            "AddResponseHeader", "RemoveResponseHeader", "SetResponseHeader",
            "AddRequestParameter", "RemoveRequestParameter", 
            "Retry", "CircuitBreaker", "RequestRateLimiter",
            "RequestSize", "RedirectTo", "SaveSession", 
            "SecureHeaders", "SetRequestHostHeader"
        );
        
        filters.forEach(filter -> {
            String name = filter.getName();
            if (!supportedFilters.contains(name)) {
                errors.add("不支持的过滤器类型: " + name);
            }
        });
    }
    
    /**
     * 验证路由是否冲突
     */
    public boolean isRouteConflict(RouteDefinition newRoute, List<RouteDefinition> existingRoutes) {
        for (RouteDefinition existingRoute : existingRoutes) {
            if (existingRoute.getId().equals(newRoute.getId())) {
                return true; // ID冲突
            }
            
            // 检查路径冲突
            if (isPathConflict(newRoute, existingRoute)) {
                return true;
            }
        }
        
        return false;
    }
    
    /**
     * 检查路径冲突
     */
    private boolean isPathConflict(RouteDefinition route1, RouteDefinition route2) {
        // 获取两个路由的Path断言
        List<String> paths1 = extractPaths(route1);
        List<String> paths2 = extractPaths(route2);
        
        // 检查是否有重叠的路径
        for (String path1 : paths1) {
            for (String path2 : paths2) {
                if (isPathOverlap(path1, path2)) {
                    return true;
                }
            }
        }
        
        return false;
    }
    
    @Data
    public static class ValidationResult {
        private boolean valid;
        private String message;
        
        public static ValidationResult success() {
            return new ValidationResult(true, "验证通过");
        }
        
        public static ValidationResult error(String message) {
            return new ValidationResult(false, message);
        }
    }
```

```
        ).increment();
    }
```
