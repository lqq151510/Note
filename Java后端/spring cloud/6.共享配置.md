# Spring Cloud 共享配置最佳实践

共享配置是微服务架构中的关键实践，用于实现配置的统一管理和复用。以下是基于 Spring Cloud Config 和 Nacos Config 的完整共享配置方案。

## 一、配置管理架构

### 1. **三层配置结构**

```
应用级配置 (application.yml) → 服务级配置 (service-xxx.yml) → 共享配置 (shared-xxx.yml)
        ↓                        ↓                        ↓
  环境相关配置              服务特有配置               全局公共配置
  实例特定配置              业务逻辑配置               中间件配置
                          数据库连接配置             安全认证配置
```

## 二、Nacos 配置中心共享配置

### 1. **配置文件命名规范**

```
Data ID格式: {prefix}-{spring.profiles.active}.{file-extension}
前缀: 默认使用 spring.application.name
环境: dev, test, prod
文件扩展名: properties, yaml, yml

示例:
- shared-config.yaml          # 全局共享配置
- shared-config-dev.yaml      # 开发环境共享配置
- user-service.yaml          # 用户服务配置
- user-service-dev.yaml      # 用户服务开发环境配置
```

          data-id: shared-${spring.profiles.active}.yaml
          group: DEFAULT_GROUP
          refresh: true
        shared-configs[2]:
          data-id: service-common.yaml

        # 扩展配置
        extension-configs[0]:
          data-id: extension-config.yaml
          group: EXT_GROUP

        # 配置加载优先级
        # 1. shared-configs[0]
        # 2. shared-configs[1]
        # 3. ${spring.application.name}.yaml
        # 4. ${spring.application.name}-${spring.profiles.active}.yaml
        # 5. extension-configs[0]

import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.cloud.context.refresh.ContextRefresher;
import org.springframework.cloud.endpoint.event.RefreshEvent;
import org.springframework.context.ApplicationEventPublisher;
import org.springframework.stereotype.Component;

@Component
public class ConfigRefreshHandler {
    
    @Autowired
    private ContextRefresher contextRefresher;
    
    @Autowired
    private ApplicationEventPublisher eventPublisher;
    
    /**
     * 手动刷新配置
     */
    public void refreshConfig() {
        // 方式1: 使用ContextRefresher
        Set<String> refreshedKeys = contextRefresher.refresh();
        log.info("刷新配置键: {}", refreshedKeys);
        
        // 方式2: 发布RefreshEvent
        eventPublisher.publishEvent(new RefreshEvent(this, null, "手动刷新配置"));
        
        // 执行自定义刷新逻辑
        onConfigRefresh();
    }
    
    /**
     * 配置变更监听
     */
    @EventListener
    public void onRefreshScopeRefreshed(RefreshScopeRefreshedEvent event) {
        String beanName = event.getName();
        log.info("RefreshScope刷新Bean: {}", beanName);
        
        // 重新加载配置
        reloadBusinessConfig();
    }
    
    /**
     * 业务配置刷新
     */
    @RefreshScope
    @Component
    @ConfigurationProperties(prefix = "business.config")
    @Data
    public class BusinessConfig {
        private Map<String, String> settings = new HashMap<>();
        private List<String> whitelist = new ArrayList<>();
        private Integer timeout = 5000;
        
        @PostConstruct
        public void init() {
            log.info("业务配置初始化: {}", this);
        }
    }
}

### 2. **配置更新监听器**

```
import com.alibaba.nacos.api.config.ConfigService;
import com.alibaba.nacos.api.config.listener.Listener;
import com.alibaba.nacos.api.exception.NacosException;
import lombok.extern.slf4j.Slf4j;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.cloud.context.refresh.ContextRefresher;
import org.springframework.stereotype.Component;

import javax.annotation.PostConstruct;
import java.util.concurrent.Executor;

@Slf4j
@Component
public class ConfigChangeListener {
    
    @Autowired
    private ConfigService configService;
    
    @Autowired
    private ContextRefresher contextRefresher;
    
    // 监听共享配置
    private static final String SHARED_CONFIG_DATA_ID = "shared-config.yaml";
    private static final String SHARED_GROUP = "DEFAULT_GROUP";
    
    @PostConstruct
    public void init() {
        registerListeners();
    }
    
    /**
     * 注册配置变更监听器
     */
    private void registerListeners() {
        try {
            // 监听共享配置
            configService.addListener(SHARED_CONFIG_DATA_ID, SHARED_GROUP, new Listener() {
                @Override
                public Executor getExecutor() {
                    return null; // 使用默认执行器
                }
                
                @Override
                public void receiveConfigInfo(String configInfo) {
                    log.info("共享配置变更: {}", SHARED_CONFIG_DATA_ID);
                    handleConfigChange(configInfo, SHARED_CONFIG_DATA_ID);
                }
            });
            
            // 监听环境配置
            String envConfigDataId = String.format("shared-%s.yaml", 
                SpringContextUtil.getActiveProfile());
            configService.addListener(envConfigDataId, SHARED_GROUP, new Listener() {
                @Override
                public Executor getExecutor() {
                    return null;
                }
                
                @Override
                public void receiveConfigInfo(String configInfo) {
                    log.info("环境配置变更: {}", envConfigDataId);
                    handleConfigChange(configInfo, envConfigDataId);
                }
            });
            
        } catch (NacosException e) {
            log.error("注册配置监听器失败", e);
        }
    }
    
    /**
     * 处理配置变更
     */
    private void handleConfigChange(String configInfo, String dataId) {
        try {
            // 1. 记录变更日志
            log.info("配置变更 - Data ID: {}, 新配置:\n{}", dataId, configInfo);
            
            // 2. 验证配置格式
            validateConfig(configInfo);
            
            // 3. 刷新Spring配置
            Set<String> refreshedKeys = contextRefresher.refresh();
            log.info("刷新配置键: {}", refreshedKeys);
            
            // 4. 发送配置变更事件
            publishConfigChangeEvent(dataId, refreshedKeys);
            
            // 5. 执行自定义刷新逻辑
            executeCustomRefreshLogic(dataId);
            
        } catch (Exception e) {
            log.error("处理配置变更失败: {}", dataId, e);
            // 发送告警
            sendAlert("配置变更异常", dataId, e.getMessage());
        }
    }
    
    /**
     * 验证配置格式
     */
    private void validateConfig(String configInfo) {
        try {
            Yaml yaml = new Yaml();
            yaml.load(configInfo);
        } catch (Exception e) {
            throw new IllegalArgumentException("配置格式错误: " + e.getMessage());
        }
    }
}
```

     * 自动检测并迁移配置
     */

        String latestVersion = getLatestConfigVersion();
        
        if (!currentVersion.equals(latestVersion)) {
            log.info("检测到配置版本变化: {} -> {}", currentVersion, latestVersion);
            migrateConfig(currentVersion, latestVersion);
        }
    }
}

### 2. **配置访问控制**

```
@Component
public class ConfigSecurityFilter {
    
    @EventListener
    public void onBeforeRefresh(EnvironmentChangeEvent event) {
        // 检查配置变更权限
        if (!hasConfigChangePermission()) {
            throw new SecurityException("无权限修改配置");
        }
        
        // 检查敏感配置变更
        Set<String> changedKeys = event.getKeys();
        if (containsSensitiveKeys(changedKeys)) {
            log.warn("敏感配置变更: {}", changedKeys);
            sendSecurityAlert(changedKeys);
        }
    }
    
    /**
     * 敏感配置检查
     */
    private boolean containsSensitiveKeys(Set<String> keys) {
        List<String> sensitivePatterns = Arrays.asList(
            ".*password.*", ".*secret.*", ".*key.*", ".*token.*"
        );
        
        return keys.stream().anyMatch(key ->
            sensitivePatterns.stream().anyMatch(pattern ->
                key.matches(pattern)
            )
        );
    }
}

public class ConfigHealthIndicator implements HealthIndicator {

    private String sharedConfigDataId;

                sharedConfigDataId, 
                "DEFAULT_GROUP", 
                5000
            );
            
            if (StringUtils.isEmpty(content)) {
                return Health.down()
                    .withDetail("config", sharedConfigDataId)
                    .withDetail("error", "配置内容为空")
                    .build();
            }
            
            // 检查配置格式
            Yaml yaml = new Yaml();
            yaml.load(content);
            
            return Health.up()
                .withDetail("config", sharedConfigDataId)
                .withDetail("size", content.length())
                .build();
            
        } catch (Exception e) {
            return Health.down()
                .withDetail("config", sharedConfigDataId)
                .withDetail("error", e.getMessage())
                .build();
        }
    }

@Component
public class ConfigDependencyAnalyzer {
    
    /**
     * 分析配置依赖
     */
    public ConfigDependency analyzeDependencies(String configContent) {
        ConfigDependency dependency = new ConfigDependency();
        
        // 解析配置引用
        Pattern pattern = Pattern.compile("\\$\\{(.+?)\\}");
        Matcher matcher = pattern.matcher(configContent);
        
        while (matcher.find()) {
            String reference = matcher.group(1);
            dependency.addReference(reference);
        }
        
        // 解析配置文件引用
        if (configContent.contains("shared-configs")) {
            dependency.addSharedConfig("shared-config.yaml");
        }
        
        return dependency;
    }
    
    /**
     * 检测配置循环依赖
     */
    public boolean hasCircularDependency(ConfigDependency dependency) {
        Map<String, Set<String>> graph = buildDependencyGraph(dependency);
        return detectCycle(graph);
    }
    
    @Data
    public static class ConfigDependency {
        private Set<String> environmentVariables = new HashSet<>();
        private Set<String> configReferences = new HashSet<>();
        private Set<String> sharedConfigs = new HashSet<>();
        private Set<String> propertySources = new HashSet<>();
        
        public void addReference(String reference) {
            if (reference.contains(":")) {
                // 环境变量引用
                environmentVariables.add(reference.split(":")[0]);
            } else {
                configReferences.add(reference);
            }
        }
    }

## 十一、配置回滚策略

### 1. **自动回滚机制**

```
@Component
public class ConfigRollbackManager {
    
    @Autowired
    private NacosConfigOperations configOperations;
    
    // 配置备份
    private final Map<String, ConfigBackup> configBackups = new ConcurrentHashMap<>();
    
    /**
     * 配置变更前备份
     */
    public void backupBeforeChange(String dataId, String group, String content) {
        ConfigBackup backup = ConfigBackup.builder()
            .dataId(dataId)
            .group(group)
            .content(content)
            .backupTime(System.currentTimeMillis())
            .build();
        
        configBackups.put(getBackupKey(dataId, group), backup);
        
        // 持久化备份
        persistBackup(backup);
    }
    
    /**
     * 自动回滚检查
     */
    @Scheduled(fixedDelay = 30000) // 每30秒检查一次
    public void autoRollbackCheck() {
        configBackups.forEach((key, backup) -> {
            if (needRollback(backup)) {
                log.warn("检测到配置异常，执行自动回滚: {}", key);
                rollback(backup);
            }
        });
    }
    
    /**
     * 判断是否需要回滚
     */
    private boolean needRollback(ConfigBackup backup) {
        long currentTime = System.currentTimeMillis();
        long backupTime = backup.getBackupTime();
        
        // 检查时间窗口（5分钟内）
        if (currentTime - backupTime > 5 * 60 * 1000) {
            return false;
        }
        
        // 检查服务健康状态
        boolean isHealthy = checkServiceHealth();
        
        // 检查错误率
        double errorRate = getErrorRate();
        
        return !isHealthy || errorRate > 0.1; // 错误率超过10%
    }
    
    /**
     * 执行回滚
     */
    private void rollback(ConfigBackup backup) {
        try {
            // 恢复配置
            configOperations.publishConfig(
                backup.getDataId(),
                backup.getGroup(),
                backup.getContent()
            );
            
            // 发送回滚通知
            sendRollbackNotification(backup);
            
            // 记录回滚日志
            logRollback(backup);
            
        } catch (Exception e) {
            log.error("配置回滚失败: {}", backup.getDataId(), e);
            sendAlert("配置回滚失败", backup.getDataId(), e.getMessage());
        }
    }
}
