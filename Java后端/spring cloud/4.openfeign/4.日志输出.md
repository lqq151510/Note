OpenFeign 日志输出是调试和监控微服务间调用的关键工具。以下是详细的配置和最佳实践指南：

## 一、日志级别配置

### 1. **全局日志级别配置**

在 `application.yml`中配置：

```
logging:
  level:
    # Feign 客户端接口包路径设为 DEBUG
    com.example.feign: DEBUG
    # Feign 底层日志
    feign: DEBUG
    # HttpClient 连接池日志
    org.apache.http: WARN
    # 负载均衡器日志
    org.springframework.cloud.loadbalancer: INFO
```

### 2. **Feign 客户端日志级别**

```
feign:
  client:
    config:
      default:  # 全局默认配置
        loggerLevel: FULL
      user-service:  # 指定特定服务
        loggerLevel: BASIC
      payment-service:
        loggerLevel: HEADERS
      report-service:
        loggerLevel: NONE
```

**日志级别说明**：

- **NONE**：不记录任何日志（默认）
    
- **BASIC**：仅记录请求方法、URL、响应状态码和执行时间
    
- **HEADERS**：记录 BASIC 信息 + 请求和响应头
    
- **FULL**：记录请求和响应的头、正文、元数据（性能影响最大）
    

## 二、完整的日志配置示例

### 1. **生产环境推荐配置**

```
# application.yml
logging:
  level:
    # Feign 相关日志
    com.example.feign: INFO
    feign.Logger: DEBUG
    org.springframework.cloud.openfeign: INFO
    
    # HTTP 客户端日志
    org.apache.http.wire: WARN
    org.apache.http.impl.conn: WARN
    org.apache.http.impl.client: WARN
    
    # 连接池日志
    org.apache.http.pool: WARN
    
  pattern:
    console: "%d{yyyy-MM-dd HH:mm:ss.SSS} [%thread] %-5level %logger{36} - %msg%n"
    file: "%d{yyyy-MM-dd HH:mm:ss.SSS} [%thread] %-5level %logger{36} - %msg%n"

feign:
  client:
    config:
      default:
        loggerLevel: BASIC
        connectTimeout: 5000
        readTimeout: 10000
```

### 2. **开发/测试环境详细日志**

```
# application-dev.yml
logging:
  level:
    com.example.feign: DEBUG
    feign.Logger: DEBUG
    feign: TRACE
    org.springframework.cloud.openfeign: DEBUG
    
  file:
    name: logs/feign.log
    max-size: 10MB
    max-history: 30
    
feign:
  client:
    config:
      default:
        loggerLevel: FULL
```

## 三、通过代码配置日志

### 1. **全局配置类**

```
import feign.Logger;
import org.springframework.context.annotation.Bean;
import org.springframework.context.annotation.Configuration;

@Configuration
public class FeignLogConfig {
    
    /**
     * 全局 Feign 日志级别
     */
    @Bean
    public Logger.Level feignLoggerLevel() {
        // 根据环境变量动态设置
        String env = System.getProperty("spring.profiles.active", "dev");
        if ("prod".equals(env)) {
            return Logger.Level.BASIC;
        } else {
            return Logger.Level.FULL;
        }
    }
    
    /**
     * 自定义 Feign 日志记录器
     */
    @Bean
    public Logger feignLogger() {
        return new CustomFeignLogger();
    }
}
```

### 2. **特定客户端配置**

```
import org.springframework.context.annotation.Configuration;

public class UserServiceFeignConfig {
```

## 四、自定义日志记录器

### 1. **高级自定义日志记录器**

```
import feign.Logger;
import feign.Request;
import feign.Response;
import lombok.extern.slf4j.Slf4j;
import org.springframework.stereotype.Component;

import java.io.IOException;
import java.nio.charset.StandardCharsets;
import java.util.Collection;
import java.util.Map;

@Slf4j
@Component
public class CustomFeignLogger extends Logger {
    
    private static final String REQUEST_ID_HEADER = "X-Request-ID";
    private static final String LOG_PREFIX = "[Feign]";
    
    @Override
    protected void log(String configKey, String format, Object... args) {
        if (log.isDebugEnabled()) {
            String message = String.format(methodTag(configKey) + format, args);
            log.debug("{} {}", LOG_PREFIX, message);
        }
    }
    
    @Override
    protected void logRequest(String configKey, Level logLevel, Request request) {
        if (log.isInfoEnabled()) {
            StringBuilder builder = new StringBuilder();
            builder.append("\n=== Feign Request ===\n");
            builder.append("Service: ").append(configKey).append("\n");
            builder.append("Method: ").append(request.httpMethod()).append("\n");
            builder.append("URL: ").append(request.url()).append("\n");
            
            // 请求头
            Map<String, Collection<String>> headers = request.headers();
            if (!headers.isEmpty()) {
                builder.append("Headers:\n");
                headers.forEach((key, values) -> 
                    values.forEach(value -> 
                        builder.append("  ").append(key).append(": ").append(value).append("\n")
                    )
                );
            }
            
            // 请求体（脱敏处理）
            if (request.body() != null && request.length() > 0) {
                String body = new String(request.body(), StandardCharsets.UTF_8);
                builder.append("Body: ").append(maskSensitiveData(body)).append("\n");
            }
            
            builder.append("===================\n");
            log.info("{} {}", LOG_PREFIX, builder.toString());
        }
    }
    
    @Override
    protected Response logAndRebufferResponse(
        String configKey, 
        Level logLevel, 
        Response response, 
        long elapsedTime
    ) throws IOException {
        
        if (log.isInfoEnabled()) {
            StringBuilder builder = new StringBuilder();
            builder.append("\n=== Feign Response ===\n");
            builder.append("Service: ").append(configKey).append("\n");
            builder.append("Status: ").append(response.status()).append("\n");
            builder.append("Time: ").append(elapsedTime).append("ms\n");
            
            // 响应头
            Map<String, Collection<String>> headers = response.headers();
            if (!headers.isEmpty()) {
                builder.append("Headers:\n");
                headers.forEach((key, values) -> 
                    values.forEach(value -> 
                        builder.append("  ").append(key).append(": ").append(value).append("\n")
                    )
                );
            }
            
            // 响应体
            if (response.body() != null && !(response.status() == 204 || response.status() == 205)) {
                byte[] bodyData = Util.toByteArray(response.body().asInputStream());
                String body = new String(bodyData, StandardCharsets.UTF_8);
                
                // 记录到日志（脱敏处理）
                String maskedBody = maskSensitiveData(body);
                builder.append("Body: ").append(maskedBody).append("\n");
                
                // 重新构造响应体
                return response.toBuilder()
                    .body(bodyData)
                    .build();
            }
            
            builder.append("====================\n");
            log.info("{} {}", LOG_PREFIX, builder.toString());
        }
        
        return response;
    }
    
    /**
     * 敏感信息脱敏
     */
    private String maskSensitiveData(String content) {
        if (content == null || content.isEmpty()) {
            return content;
        }
        
        // 脱敏密码
        content = content.replaceAll("\"password\":\"[^\"]*\"", "\"password\":\"******\"");
        content = content.replaceAll("\"token\":\"[^\"]*\"", "\"token\":\"******\"");
        content = content.replaceAll("\"authorization\":\"[^\"]*\"", "\"authorization\":\"******\"");
        
        // 脱敏手机号
        content = content.replaceAll("1[3-9]\\d{9}", "1******");
        
        // 脱敏身份证号
        content = content.replaceAll("\\d{17}[\\dXx]", "***************");
        
        // 脱敏邮箱
        content = content.replaceAll("([A-Za-z0-9_\\-\\.]+)@([A-Za-z0-9_\\-\\.]+)\\.([A-Za-z]{2,5})", "***@***.$3");
        
        return content;
    }
    
    @Override
    protected Response logAndRebufferResponse(
        String configKey, 
        Level logLevel, 
        Response response, 
        long elapsedTime,
        Request request
    ) throws IOException {
        // 记录请求ID用于链路追踪
        String requestId = request.headers().getOrDefault(REQUEST_ID_HEADER, null)
            .stream()
            .findFirst()
            .orElse("N/A");
        
        MDC.put("requestId", requestId);
        
        return logAndRebufferResponse(configKey, logLevel, response, elapsedTime);
    }
}
```
