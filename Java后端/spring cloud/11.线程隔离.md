# 微服务线程隔离全方位解决方案

线程隔离是防止服务雪崩的核心技术，通过将不同服务、不同优先级的请求分配到不同的线程池中执行，避免某个服务的故障影响整个系统。

## 一、线程隔离核心概念

### 1. **线程隔离架构对比**

|隔离方式|原理|优点|缺点|适用场景|
|---|---|---|---|---|
|**线程池隔离**​|为每个服务创建独立线程池|完全隔离，互不影响|上下文切换开销|核心服务、耗时操作|
|**信号量隔离**​|限制并发线程数，共用线程池|轻量级，开销小|无法隔离超时|快速操作、高并发场景|
|**舱壁模式**​|资源分组隔离|防止级联故障|资源浪费|关键资源保护|
|**异步隔离**​|非阻塞IO，事件驱动|高并发，低延迟|编程复杂|IO密集型操作|

## 二、线程池隔离实现

### 1. **Spring Boot 线程池配置**

```
import org.springframework.context.annotation.Bean;
import org.springframework.context.annotation.Configuration;
import org.springframework.scheduling.concurrent.ThreadPoolTaskExecutor;
import org.springframework.scheduling.concurrent.ThreadPoolTaskScheduler;
import java.util.concurrent.*;

@Configuration
public class ThreadPoolIsolationConfig {
    
    /**
     * 核心业务线程池
     */
    @Bean("coreThreadPool")
    public ThreadPoolTaskExecutor coreThreadPool() {
        ThreadPoolTaskExecutor executor = new ThreadPoolTaskExecutor();
        
        // 核心参数
        executor.setCorePoolSize(20);                          // 核心线程数
        executor.setMaxPoolSize(100);                          // 最大线程数
        executor.setQueueCapacity(1000);                       // 队列容量
        executor.setKeepAliveSeconds(60);                      // 空闲线程存活时间
        
        // 线程配置
        executor.setThreadNamePrefix("core-pool-");            // 线程名前缀
        executor.setThreadPriority(Thread.NORM_PRIORITY);      // 线程优先级
        executor.setDaemon(false);                             // 是否守护线程
        
        // 拒绝策略
        executor.setRejectedExecutionHandler(new ThreadPoolExecutor.CallerRunsPolicy() {
            @Override
            public void rejectedExecution(Runnable r, ThreadPoolExecutor e) {
                // 自定义拒绝策略
                log.warn("核心线程池拒绝任务: {}, 活跃线程: {}, 队列大小: {}", 
                    r.toString(), e.getActiveCount(), e.getQueue().size());
                
                // 发送告警
                sendAlert("线程池拒绝", "核心线程池已满");
                
                // 降级处理
                if (r instanceof BusinessTask) {
                    ((BusinessTask) r).onReject();
                }
                
                // 执行默认策略
                super.rejectedExecution(r, e);
            }
        });
        
        // 其他配置
        executor.setWaitForTasksToCompleteOnShutdown(true);    // 优雅关闭
        executor.setAwaitTerminationSeconds(30);               // 关闭等待时间
        executor.setAllowCoreThreadTimeOut(false);             // 核心线程超时
        
        // 初始化
        executor.initialize();
        
        return executor;
    }
    
    /**
     * IO密集型线程池
     */
    @Bean("ioThreadPool")
    public ThreadPoolTaskExecutor ioThreadPool() {
        ThreadPoolTaskExecutor executor = new ThreadPoolTaskExecutor();
        
        // IO密集型：线程数 = CPU核心数 * (1 + 等待时间/计算时间)
        int coreSize = Runtime.getRuntime().availableProcessors() * 2;
        int maxSize = coreSize * 4;
        
        executor.setCorePoolSize(coreSize);
        executor.setMaxPoolSize(maxSize);
        executor.setQueueCapacity(2000);
        executor.setKeepAliveSeconds(30);
        executor.setThreadNamePrefix("io-pool-");
        
        // 使用LinkedTransferQueue提高吞吐量
        executor.setQueueCapacity(0);  // 使用SynchronousQueue
        
        // 拒绝策略：直接丢弃，记录日志
        executor.setRejectedExecutionHandler((r, e) -> {
            log.error("IO线程池拒绝任务，当前活跃: {}", e.getActiveCount());
            // 异步重试
            retryService.retryAsync(r);
        });
        
        return executor;
    }
    
    /**
     * CPU密集型线程池
     */
    @Bean("cpuThreadPool")
    public ThreadPoolTaskExecutor cpuThreadPool() {
        ThreadPoolTaskExecutor executor = new ThreadPoolTaskExecutor();
        
        // CPU密集型：线程数 = CPU核心数 + 1
        int cpuCores = Runtime.getRuntime().availableProcessors();
        executor.setCorePoolSize(cpuCores + 1);
        executor.setMaxPoolSize(cpuCores * 2);
        executor.setQueueCapacity(100);
        executor.setKeepAliveSeconds(60);
        executor.setThreadNamePrefix("cpu-pool-");
        
        // 使用优先级队列
        executor.setQueueCapacity(100);
        
        return executor;
    }
    
    /**
     * 定时任务线程池
     */
    @Bean("scheduledThreadPool")
    public ScheduledExecutorService scheduledThreadPool() {
        ScheduledThreadPoolExecutor executor = new ScheduledThreadPoolExecutor(
            5,  // 核心线程数
            new ThreadFactoryBuilder()
                .setNameFormat("scheduled-pool-%d")
                .setDaemon(true)
                .build(),
            new ThreadPoolExecutor.CallerRunsPolicy()
        );
        
        // 移除取消的任务
        executor.setRemoveOnCancelPolicy(true);
        
        return executor;
    }
    
    /**
     * 虚拟线程池（Java 19+）
     */
    @Bean("virtualThreadPool")
    public ExecutorService virtualThreadPool() {
        // Java 19+ 虚拟线程
        return Executors.newVirtualThreadPerTaskExecutor();
    }
}
```

### 2. **动态线程池配置**

```
import org.springframework.scheduling.concurrent.ThreadPoolTaskExecutor;
import org.springframework.stereotype.Component;
import org.springframework.beans.BeansException;
import org.springframework.context.ApplicationContext;
import org.springframework.context.ApplicationContextAware;
import javax.annotation.PreDestroy;
import java.util.Map;
import java.util.concurrent.ConcurrentHashMap;
import java.util.concurrent.atomic.AtomicInteger;

@Component
@Slf4j
public class DynamicThreadPoolManager implements ApplicationContextAware {
    
    // 线程池容器
    private final Map<String, ThreadPoolExecutor> threadPools = new ConcurrentHashMap<>();
    private final Map<String, ThreadPoolMetrics> poolMetrics = new ConcurrentHashMap<>();
    
    // 配置管理
    private final Map<String, ThreadPoolConfig> poolConfigs = new ConcurrentHashMap<>();
    
    private ApplicationContext applicationContext;
    
    @Override
    public void setApplicationContext(ApplicationContext applicationContext) throws BeansException {
        this.applicationContext = applicationContext;
    }
    
    /**
     * 创建动态线程池
     */
    public ThreadPoolExecutor createThreadPool(String poolName, ThreadPoolConfig config) {
        return threadPools.computeIfAbsent(poolName, name -> {
            // 创建线程池
            ThreadPoolExecutor executor = new ThreadPoolExecutor(
                config.getCorePoolSize(),
                config.getMaximumPoolSize(),
                config.getKeepAliveTime(),
                config.getUnit(),
                new LinkedBlockingQueue<>(config.getQueueCapacity()),
                new ThreadFactoryBuilder()
                    .setNameFormat(poolName + "-%d")
                    .setUncaughtExceptionHandler((t, e) -> 
                        log.error("线程异常: {}", t.getName(), e))
                    .build(),
                config.getRejectedExecutionHandler()
            );
            
            // 初始化监控
            poolMetrics.put(poolName, new ThreadPoolMetrics());
            
            // 启动监控
            startMonitoring(poolName, executor);
            
            log.info("创建动态线程池: {}, 配置: {}", poolName, config);
            return executor;
        });
    }
    
    /**
     * 动态调整线程池参数
     */
    public synchronized boolean adjustThreadPool(String poolName, 
                                                ThreadPoolAdjustment adjustment) {
        ThreadPoolExecutor executor = threadPools.get(poolName);
        if (executor == null) {
            log.warn("线程池不存在: {}", poolName);
            return false;
        }
        
        ThreadPoolConfig oldConfig = poolConfigs.get(poolName);
        
        try {
            // 调整核心线程数
            if (adjustment.getCorePoolSize() != null) {
                executor.setCorePoolSize(adjustment.getCorePoolSize());
                log.info("调整核心线程数: {} -> {}", poolName, adjustment.getCorePoolSize());
            }
            
            // 调整最大线程数
            if (adjustment.getMaximumPoolSize() != null) {
                executor.setMaximumPoolSize(adjustment.getMaximumPoolSize());
                log.info("调整最大线程数: {} -> {}", poolName, adjustment.getMaximumPoolSize());
            }
            
            // 调整空闲时间
            if (adjustment.getKeepAliveTime() != null) {
                executor.setKeepAliveTime(adjustment.getKeepAliveTime(), 
                    adjustment.getUnit());
                log.info("调整空闲时间: {} -> {}", poolName, adjustment.getKeepAliveTime());
            }
            
            // 更新配置
            if (oldConfig != null) {
                oldConfig.updateFromAdjustment(adjustment);
            }
            
            return true;
        } catch (Exception e) {
            log.error("调整线程池失败: {}", poolName, e);
            return false;
        }
    }
    
    /**
     * 根据负载自动调整
     */
    @Scheduled(fixedDelay = 30000)  // 每30秒调整一次
    public void autoAdjustThreadPools() {
        threadPools.forEach((poolName, executor) -> {
            ThreadPoolMetrics metrics = poolMetrics.get(poolName);
            if (metrics == null) {
                return;
            }
            
            // 分析线程池状态
            ThreadPoolAnalysis analysis = analyzeThreadPool(executor, metrics);
            
            // 根据分析结果调整
            ThreadPoolAdjustment adjustment = calculateAdjustment(analysis);
            
            if (adjustment != null) {
                adjustThreadPool(poolName, adjustment);
            }
        });
    }
    
    /**
     * 分析线程池状态
     */
    private ThreadPoolAnalysis analyzeThreadPool(ThreadPoolExecutor executor, 
                                                ThreadPoolMetrics metrics) {
        ThreadPoolAnalysis analysis = new ThreadPoolAnalysis();
        
        // 当前状态
        analysis.setPoolSize(executor.getPoolSize());
        analysis.setActiveCount(executor.getActiveCount());
        analysis.setQueueSize(executor.getQueue().size());
        analysis.setCompletedTaskCount(executor.getCompletedTaskCount());
        analysis.setLargestPoolSize(executor.getLargestPoolSize());
        
        // 计算使用率
        double cpuUsage = getCpuUsage();
        double memoryUsage = getMemoryUsage();
        
        // 队列使用率
        int queueCapacity = executor.getQueue().remainingCapacity() + 
                           executor.getQueue().size();
        double queueUsage = (double) executor.getQueue().size() / queueCapacity;
        
        analysis.setQueueUsage(queueUsage);
        analysis.setCpuUsage(cpuUsage);
        analysis.setMemoryUsage(memoryUsage);
        
        return analysis;
    }
    
    /**
     * 计算调整策略
     */
    private ThreadPoolAdjustment calculateAdjustment(ThreadPoolAnalysis analysis) {
        ThreadPoolAdjustment adjustment = new ThreadPoolAdjustment();
        
        // 队列使用率高，增加线程
        if (analysis.getQueueUsage() > 0.8 && 
            analysis.getPoolSize() < analysis.getLargestPoolSize()) {
            adjustment.setCorePoolSize(analysis.getPoolSize() + 1);
        }
        
        // CPU使用率高，减少线程
        if (analysis.getCpuUsage() > 0.8 && analysis.getPoolSize() > 1) {
            adjustment.setCorePoolSize(Math.max(1, analysis.getPoolSize() - 1));
        }
        
        // 内存使用率高，减少队列大小
        if (analysis.getMemoryUsage() > 0.7) {
            // 可以通过调整队列大小来减少内存占用
        }
        
        return adjustment.hasChanges() ? adjustment : null;
    }
    
    /**
     * 优雅关闭
     */
    @PreDestroy
    public void shutdownGracefully() {
        log.info("开始优雅关闭线程池...");
        
        threadPools.forEach((poolName, executor) -> {
            try {
                // 停止接收新任务
                executor.shutdown();
                
                // 等待现有任务完成
                if (!executor.awaitTermination(30, TimeUnit.SECONDS)) {
                    // 强制关闭
                    executor.shutdownNow();
                    
                    if (!executor.awaitTermination(10, TimeUnit.SECONDS)) {
                        log.error("线程池无法关闭: {}", poolName);
                    }
                }
                
                log.info("线程池已关闭: {}", poolName);
            } catch (InterruptedException e) {
                Thread.currentThread().interrupt();
                executor.shutdownNow();
            }
        });
    }
    
    @Data
    public static class ThreadPoolConfig {
        private int corePoolSize = 5;
        private int maximumPoolSize = 20;
        private long keepAliveTime = 60;
        private TimeUnit unit = TimeUnit.SECONDS;
        private int queueCapacity = 100;
        private RejectedExecutionHandler rejectedExecutionHandler = 
            new ThreadPoolExecutor.AbortPolicy();
            
        public void updateFromAdjustment(ThreadPoolAdjustment adjustment) {
            if (adjustment.getCorePoolSize() != null) {
                this.corePoolSize = adjustment.getCorePoolSize();
            }
            if (adjustment.getMaximumPoolSize() != null) {
                this.maximumPoolSize = adjustment.getMaximumPoolSize();
            }
            if (adjustment.getKeepAliveTime() != null) {
                this.keepAliveTime = adjustment.getKeepAliveTime();
            }
            if (adjustment.getUnit() != null) {
                this.unit = adjustment.getUnit();
            }
        }
    }
    
    @Data
    public static class ThreadPoolAdjustment {
        private Integer corePoolSize;
        private Integer maximumPoolSize;
        private Long keepAliveTime;
        private TimeUnit unit;
        
        public boolean hasChanges() {
            return corePoolSize != null || maximumPoolSize != null || 
                   keepAliveTime != null || unit != null;
        }
    }
}
```

## 三、信号量隔离

### 1. **Semaphore 实现信号量隔离**

```
import org.springframework.stereotype.Component;
import java.util.concurrent.ConcurrentHashMap;
import java.util.concurrent.Semaphore;
import java.util.concurrent.TimeUnit;

@Component
@Slf4j
public class SemaphoreIsolationManager {
    
    // 信号量容器
    private final ConcurrentHashMap<String, SemaphoreWrapper> semaphores = 
        new ConcurrentHashMap<>();
    
    // 配置容器
    private final ConcurrentHashMap<String, SemaphoreConfig> configs = 
        new ConcurrentHashMap<>();
    
    /**
     * 初始化信号量配置
     */
    @PostConstruct
    public void init() {
        // 数据库连接信号量
        configs.put("database", new SemaphoreConfig(50, 1000, 30000));
        
        // Redis连接信号量
        configs.put("redis", new SemaphoreConfig(100, 500, 10000));
        
        // 外部API信号量
        configs.put("external_api", new SemaphoreConfig(20, 200, 5000));
        
        // 文件操作信号量
        configs.put("file_io", new SemaphoreConfig(10, 100, 30000));
    }
    
    /**
     * 执行受保护的代码
     */
    public <T> T executeWithSemaphore(String resourceName, Supplier<T> supplier) {
        return executeWithSemaphore(resourceName, supplier, 1, 5000L, 
            TimeUnit.MILLISECONDS, null);
    }
    
    /**
     * 带超时的信号量保护
     */
    public <T> T executeWithSemaphore(String resourceName, Supplier<T> supplier,
                                     int permits, long timeout, TimeUnit unit,
                                     Supplier<T> fallback) {
        
        SemaphoreWrapper wrapper = getSemaphoreWrapper(resourceName);
        
        try {
            // 尝试获取信号量
            if (wrapper.tryAcquire(permits, timeout, unit)) {
                try {
                    // 执行受保护的代码
                    return supplier.get();
                } finally {
                    // 释放信号量
                    wrapper.release(permits);
                }
            } else {
                // 超时处理
                log.warn("获取信号量超时: {}, permits: {}, timeout: {}ms", 
                    resourceName, permits, unit.toMillis(timeout));
                
                // 执行降级逻辑
                if (fallback != null) {
                    return fallback.get();
                }
                
                throw new ResourceBusyException("系统繁忙，请稍后重试");
            }
        } catch (InterruptedException e) {
            Thread.currentThread().interrupt();
            throw new ServiceException("操作被中断", e);
        } catch (Exception e) {
            // 记录异常
            wrapper.recordException();
            throw e;
        }
    }
    
    /**
     * 批量信号量控制
     */
    public <T> T executeWithMultipleSemaphores(List<String> resources, 
                                              Supplier<T> supplier) {
        // 按顺序获取所有信号量
        List<SemaphoreWrapper> acquired = new ArrayList<>();
        
        try {
            for (String resource : resources) {
                SemaphoreWrapper wrapper = getSemaphoreWrapper(resource);
                if (wrapper.tryAcquire(1, 1000, TimeUnit.MILLISECONDS)) {
                    acquired.add(wrapper);
                } else {
                    // 获取失败，释放已获得的信号量
                    releaseAll(acquired);
                    throw new ResourceBusyException("资源繁忙: " + resource);
                }
            }
            
            // 执行代码
            return supplier.get();
            
        } finally {
            // 释放所有信号量
            releaseAll(acquired);
        }
    }
    
    /**
     * 动态调整信号量
     */
    public synchronized void adjustSemaphore(String resourceName, int newPermits) {
        SemaphoreWrapper wrapper = semaphores.get(resourceName);
        SemaphoreConfig config = configs.get(resourceName);
        
        if (wrapper != null && config != null) {
            int oldPermits = wrapper.getAvailablePermits();
            
            if (newPermits != oldPermits) {
                // 创建新的信号量
                SemaphoreWrapper newWrapper = new SemaphoreWrapper(
                    resourceName, newPermits, config.getAcquireTimeout(), 
                    config.getWaitTimeout()
                );
                
                // 迁移现有许可
                int toRelease = Math.min(newPermits, oldPermits);
                newWrapper.getSemaphore().release(toRelease);
                
                // 替换信号量
                semaphores.put(resourceName, newWrapper);
                
                log.info("调整信号量: {} {} -> {}", 
                    resourceName, oldPermits, newPermits);
            }
        }
    }
    
    /**
     * 监控信号量状态
     */
    public SemaphoreMetrics getSemaphoreMetrics(String resourceName) {
        SemaphoreWrapper wrapper = semaphores.get(resourceName);
        if (wrapper != null) {
            return wrapper.getMetrics();
        }
        return null;
    }
    
    /**
     * 获取所有信号量状态
     */
    public Map<String, SemaphoreMetrics> getAllSemaphoreMetrics() {
        Map<String, SemaphoreMetrics> metrics = new HashMap<>();
        semaphores.forEach((name, wrapper) -> 
            metrics.put(name, wrapper.getMetrics()));
        return metrics;
    }
    
    /**
     * 获取信号量包装器
     */
    private SemaphoreWrapper getSemaphoreWrapper(String resourceName) {
        return semaphores.computeIfAbsent(resourceName, name -> {
            SemaphoreConfig config = configs.getOrDefault(name, 
                new SemaphoreConfig(10, 1000, 10000));
            
            return new SemaphoreWrapper(name, config.getMaxPermits(), 
                config.getAcquireTimeout(), config.getWaitTimeout());
        });
    }
    
    /**
     * 释放所有信号量
     */
    private void releaseAll(List<SemaphoreWrapper> wrappers) {
        for (SemaphoreWrapper wrapper : wrappers) {
            try {
                wrapper.release(1);
            } catch (Exception e) {
                log.error("释放信号量失败", e);
            }
        }
    }
    
    /**
     * 信号量包装器
     */
    @Data
    public static class SemaphoreWrapper {
        private final String resourceName;
        private final Semaphore semaphore;
        private final long acquireTimeout;  // 获取超时时间
        private final long waitTimeout;     // 等待超时时间
        
        // 监控指标
        private final AtomicLong acquireCount = new AtomicLong(0);
        private final AtomicLong timeoutCount = new AtomicLong(0);
        private final AtomicLong successCount = new AtomicLong(0);
        private final AtomicLong exceptionCount = new AtomicLong(0);
        private final AtomicLong totalWaitTime = new AtomicLong(0);
        
        public SemaphoreWrapper(String resourceName, int permits, 
                               long acquireTimeout, long waitTimeout) {
            this.resourceName = resourceName;
            this.semaphore = new Semaphore(permits, true);  // 公平信号量
            this.acquireTimeout = acquireTimeout;
            this.waitTimeout = waitTimeout;
        }
        
        public boolean tryAcquire(int permits, long timeout, TimeUnit unit) 
            throws InterruptedException {
            
            long startTime = System.currentTimeMillis();
            boolean acquired = semaphore.tryAcquire(permits, timeout, unit);
            long waitTime = System.currentTimeMillis() - startTime;
            
            acquireCount.incrementAndGet();
            totalWaitTime.addAndGet(waitTime);
            
            if (!acquired) {
                timeoutCount.incrementAndGet();
            }
            
            return acquired;
        }
        
        public void release(int permits) {
            semaphore.release(permits);
            successCount.incrementAndGet();
        }
        
        public int getAvailablePermits() {
            return semaphore.availablePermits();
        }
        
        public int getQueueLength() {
            return semaphore.getQueueLength();
        }
        
        public void recordException() {
            exceptionCount.incrementAndGet();
        }
        
        public SemaphoreMetrics getMetrics() {
            return SemaphoreMetrics.builder()
                .resourceName(resourceName)
                .availablePermits(getAvailablePermits())
                .queueLength(getQueueLength())
                .acquireCount(acquireCount.get())
                .timeoutCount(timeoutCount.get())
                .successCount(successCount.get())
                .exceptionCount(exceptionCount.get())
                .averageWaitTime(acquireCount.get() > 0 ? 
                    (double) totalWaitTime.get() / acquireCount.get() : 0)
                .timeoutRate(acquireCount.get() > 0 ? 
                    (double) timeoutCount.get() / acquireCount.get() : 0)
                .build();
        }
    }
    
    @Data
    @Builder
    public static class SemaphoreConfig {
        private int maxPermits = 10;           // 最大许可数
        private long acquireTimeout = 1000;    // 获取超时(ms)
        private long waitTimeout = 10000;      // 等待超时(ms)
    }
}
```

## 四、Hystrix 线程隔离

### 1. **Hystrix 线程池配置**

```
import com.netflix.hystrix.HystrixCommand;
import com.netflix.hystrix.HystrixCommandGroupKey;
import com.netflix.hystrix.HystrixCommandKey;
import com.netflix.hystrix.HystrixCommandProperties;
import com.netflix.hystrix.HystrixThreadPoolProperties;
import org.springframework.stereotype.Component;

@Component
public class HystrixThreadIsolation {
    
    /**
     * 自定义Hystrix命令
     */
    public class UserServiceCommand extends HystrixCommand<UserDTO> {
        
        private final Long userId;
        private final UserService userService;
        private final FallbackService fallbackService;
        
        public UserServiceCommand(Long userId, UserService userService, 
                                 FallbackService fallbackService) {
            super(Setter
                .withGroupKey(HystrixCommandGroupKey.Factory.asKey("UserService"))
                .andCommandKey(HystrixCommandKey.Factory.asKey("GetUserById"))
                .andThreadPoolKey(HystrixThreadPoolKey.Factory.asKey("UserServicePool"))
                
                // 线程池配置
                .andThreadPoolPropertiesDefaults(HystrixThreadPoolProperties.Setter()
                    .withCoreSize(20)                       // 核心线程数
                    .withMaximumSize(100)                   // 最大线程数
                    .withAllowMaximumSizeToDivergeFromCoreSize(true)  // 允许扩展
                    .withKeepAliveTimeMinutes(1)            // 空闲线程存活时间
                    .withMaxQueueSize(1000)                 // 队列大小
                    .withQueueSizeRejectionThreshold(500)   // 队列拒绝阈值
                )
                
                // 命令配置
                .andCommandPropertiesDefaults(HystrixCommandProperties.Setter()
                    .withExecutionIsolationStrategy(
                        ExecutionIsolationStrategy.THREAD)  // 线程隔离
                    .withExecutionTimeoutInMilliseconds(3000)  // 执行超时
```

```
                name, newConfig.getMaxConcurrentCalls());
        }
    }
```

### 1. **线程池监控指标**

```
import org.springframework.boot.actuate.health.Health;
import org.springframework.boot.actuate.health.HealthIndicator;
import org.springframework.scheduling.concurrent.ThreadPoolTaskExecutor;
import org.springframework.stereotype.Component;
import javax.annotation.PostConstruct;
import java.lang.management.ManagementFactory;
import java.lang.management.ThreadMXBean;
import java.util.HashMap;
import java.util.Map;
import java.util.concurrent.ThreadPoolExecutor;

@Component
@Slf4j
public class ThreadPoolMonitor implements HealthIndicator {
    
    private final Map<String, ThreadPoolExecutor> threadPools = new HashMap<>();
    private final ThreadMXBean threadMXBean = ManagementFactory.getThreadMXBean();
    
    /**
     * 注册线程池
     */
    public void registerThreadPool(String name, ThreadPoolExecutor executor) {
        threadPools.put(name, executor);
        log.info("注册线程池监控: {}", name);
    }
    
    /**
     * 健康检查
     */
    @Override
    public Health health() {
        Map<String, Object> details = new HashMap<>();
        
        // 系统线程信息
        details.put("system.thread.count", threadMXBean.getThreadCount());
        details.put("system.thread.daemon.count", threadMXBean.getDaemonThreadCount());
        details.put("system.thread.peak.count", threadMXBean.getPeakThreadCount());
        
        // 监控每个线程池
        threadPools.forEach((name, executor) -> {
            Map<String, Object> poolDetails = new HashMap<>();
            
            poolDetails.put("pool.size", executor.getPoolSize());
            poolDetails.put("core.pool.size", executor.getCorePoolSize());
            poolDetails.put("max.pool.size", executor.getMaximumPoolSize());
            poolDetails.put("active.count", executor.getActiveCount());
            poolDetails.put("queue.size", executor.getQueue().size());
            poolDetails.put("queue.remaining", executor.getQueue().remainingCapacity());
            poolDetails.put("completed.task.count", executor.getCompletedTaskCount());
            poolDetails.put("largest.pool.size", executor.getLargestPoolSize());
            poolDetails.put("task.count", executor.getTaskCount());
            
            // 计算使用率
            double queueUsage = calculateQueueUsage(executor);
            double poolUsage = calculatePoolUsage(executor);
            
            poolDetails.put("queue.usage.percent", queueUsage);
            poolDetails.put("pool.usage.percent", poolUsage);
            
            details.put(name, poolDetails);
            
            // 检查告警条件
            checkPoolAlerts(name, executor, queueUsage, poolUsage);
        });
        
        return Health.up().withDetails(details).build();
    }
    
    /**
     * 定期监控
     */
    @Scheduled(fixedDelay = 10000)
    public void monitorThreadPools() {
        threadPools.forEach((name, executor) -> {
            ThreadPoolMetrics metrics = collectMetrics(name, executor);
            
            // 记录到监控系统
            sendToMetrics(metrics);
            
            // 检查是否需要调整
            if (needAdjustment(metrics)) {
                adjustThreadPool(name, executor, metrics);
            }
        });
    }
    
    /**
     * 收集线程池指标
     */
    private ThreadPoolMetrics collectMetrics(String name, ThreadPoolExecutor executor) {
        return ThreadPoolMetrics.builder()
            .name(name)
            .poolSize(executor.getPoolSize())
            .corePoolSize(executor.getCorePoolSize())
            .maximumPoolSize(executor.getMaximumPoolSize())
            .activeCount(executor.getActiveCount())
            .queueSize(executor.getQueue().size())
            .queueCapacity(executor.getQueue().size() + 
                executor.getQueue().remainingCapacity())
            .completedTaskCount(executor.getCompletedTaskCount())
            .largestPoolSize(executor.getLargestPoolSize())
            .taskCount(executor.getTaskCount())
            .timestamp(System.currentTimeMillis())
            .build();
    }
    
    /**
     * 计算队列使用率
     */
    private double calculateQueueUsage(ThreadPoolExecutor executor) {
        int queueSize = executor.getQueue().size();
        int queueCapacity = queueSize + executor.getQueue().remainingCapacity();
        
        if (queueCapacity == 0) {
            return 0.0;
        }
        
        return (double) queueSize / queueCapacity * 100;
    }
    
    /**
     * 计算线程池使用率
     */
    private double calculatePoolUsage(ThreadPoolExecutor executor) {
        int activeCount = executor.getActiveCount();
        int maxPoolSize = executor.getMaximumPoolSize();
        
        if (maxPoolSize == 0) {
            return 0.0;
        }
        
        return (double) activeCount / maxPoolSize * 100;
    }
    
    /**
     * 检查告警条件
     */
    private void checkPoolAlerts(String name, ThreadPoolExecutor executor, 
                                double queueUsage, double poolUsage) {
        
        if (queueUsage > 80) {
            log.warn("线程池队列使用率高: {} - {}%", name, queueUsage);
            sendAlert("队列告警", name, queueUsage);
        }
        
        if (poolUsage > 80) {
            log.warn("线程池活跃度高: {} - {}%", name, poolUsage);
            sendAlert("活跃度告警", name, poolUsage);
        }
        
        if (executor.getQueue().size() > 500) {
            log.error("线程池队列积压严重: {} - {}", name, executor.getQueue().size());
            sendAlert("队列积压", name, executor.getQueue().size());
        }
    }
    
    /**
     * 判断是否需要调整
     */
    private boolean needAdjustment(ThreadPoolMetrics metrics) {
        // 队列使用率持续高
        if (metrics.getQueueUsagePercent() > 80) {
            return true;
        }
        
        // 活跃线程数接近最大值
        if (metrics.getActiveCount() >= metrics.getMaximumPoolSize() * 0.9) {
            return true;
        }
        
        return false;
    }
    
    /**
     * 调整线程池
     */
    private void adjustThreadPool(String name, ThreadPoolExecutor executor, 
                                 ThreadPoolMetrics metrics) {
        
        int currentMaxSize = executor.getMaximumPoolSize();
        int newMaxSize = currentMaxSize + 5;  // 增加5个线程
        
        // 限制最大调整
        newMaxSize = Math.min(newMaxSize, 200);
        
        executor.setMaximumPoolSize(newMaxSize);
        log.info("调整线程池最大线程数: {} {} -> {}", 
            name, currentMaxSize, newMaxSize);
    }
    
    @Data
    @Builder
    public static class ThreadPoolMetrics {
        private String name;
        private int poolSize;
        private int corePoolSize;
        private int maximumPoolSize;
        private int activeCount;
        private int queueSize;
        private int queueCapacity;
        private long completedTaskCount;
        private int largestPoolSize;
        private long taskCount;
        private long timestamp;
        
        public double getQueueUsagePercent() {
            if (queueCapacity == 0) {
                return 0.0;
            }
            return (double) queueSize / queueCapacity * 100;
        }
        
        public double getPoolUsagePercent() {
            if (maximumPoolSize == 0) {
                return 0.0;
            }
            return (double) activeCount / maximumPoolSize * 100;
        }
    }
}
```
