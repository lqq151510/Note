# 服务降级（Fallback）全方位解决方案

服务降级是保证系统可用性的关键策略，当服务不可用或性能下降时，通过提供备选方案保证核心功能可用。以下是完整的服务降级解决方案。

## 一、降级策略分类

### 1. **降级策略矩阵**

| 策略         | 说明         | 适用场景    | 实现复杂度 |
| ---------- | ---------- | ------- | ----- |
| **快速失败**​  | 立即返回错误或默认值 | 非核心功能   | 简单    |
| **静默处理**​  | 忽略异常，返回空值  | 查询类接口   | 简单    |
| **缓存降级**​  | 返回缓存数据     | 可容忍数据延迟 | 中等    |
| **默认值降级**​ | 返回预定义默认值   | 配置类数据   | 简单    |
| **兜底服务**​  | 调用备用服务     | 关键业务    | 复杂    |
| **人工开关**​  | 手动开启降级     | 运维干预    | 中等    |
| **智能降级**​  | 基于指标自动降级   | 高可用系统   | 复杂    |

import com.netflix.hystrix.contrib.javanica.annotation.HystrixCommand;
import com.netflix.hystrix.contrib.javanica.annotation.HystrixProperty;
import org.springframework.stereotype.Service;
import java.util.concurrent.TimeUnit;

@Service
@Slf4j
public class HystrixFallbackService {
    
    /**
     * 基础降级示例
     */
    @HystrixCommand(
        fallbackMethod = "getUserFallback",
        commandProperties = {
            @HystrixProperty(name = "execution.isolation.thread.timeoutInMilliseconds", value = "3000"),
            @HystrixProperty(name = "circuitBreaker.requestVolumeThreshold", value = "20"),
            @HystrixProperty(name = "circuitBreaker.errorThresholdPercentage", value = "50"),
            @HystrixProperty(name = "circuitBreaker.sleepWindowInMilliseconds", value = "5000"),
            @HystrixProperty(name = "metrics.rollingStats.timeInMilliseconds", value = "10000")
        },
        threadPoolProperties = {
            @HystrixProperty(name = "coreSize", value = "20"),
            @HystrixProperty(name = "maxQueueSize", value = "100"),
            @HystrixProperty(name = "queueSizeRejectionThreshold", value = "20")
        }
    )
    public UserDTO getUserById(Long userId) {
        // 模拟耗时操作
        try {
            TimeUnit.MILLISECONDS.sleep(2000);
        } catch (InterruptedException e) {
            Thread.currentThread().interrupt();
        }
        
        return userRepository.findById(userId)
            .orElseThrow(() -> new UserNotFoundException("用户不存在"));
    }
    
    /**
     * 降级方法
     * 参数和返回类型必须与原方法一致
     */
    public UserDTO getUserFallback(Long userId) {
        log.warn("用户服务降级，返回默认用户，userId: {}", userId);
        
        return UserDTO.builder()
            .id(userId)
            .name("默认用户")
            .email("default@example.com")
            .status(0)
            .build();
    }
    
    /**
     * 带异常参数的降级
     */
    @HystrixCommand(fallbackMethod = "getUserFallbackWithException")
    public UserDTO getUserWithException(Long userId) {
        throw new RuntimeException("模拟服务异常");
    }
    
    public UserDTO getUserFallbackWithException(Long userId, Throwable e) {
        log.error("用户服务异常降级，userId: {}, error: {}", userId, e.getMessage());
        
        return UserDTO.builder()
            .id(userId)
            .name("异常降级用户")
            .message("服务暂时不可用: " + e.getMessage())
            .build();
    }
    
    /**
     * 多级降级链
     */
    @HystrixCommand(fallbackMethod = "getUserLevel1Fallback")
    public UserDTO getUserWithMultiLevelFallback(Long userId) {
        // 一级降级
        return userService.getUserById(userId);
    }
    
    public UserDTO getUserLevel1Fallback(Long userId) {
        log.warn("一级降级，尝试从缓存获取");
        
        try {
            // 二级降级：尝试缓存
            return userCacheService.getUserFromCache(userId);
        } catch (Exception e) {
            // 触发三级降级
            return getUserLevel2Fallback(userId, e);
        }
    }
    
    public UserDTO getUserLevel2Fallback(Long userId, Throwable e) {
        log.warn("二级降级，返回默认值");
        
        return UserDTO.builder()
            .id(userId)
            .name("降级用户")
            .email("fallback@example.com")
            .status(0)
            .build();
    }
    
    /**
     * 忽略特定异常
     */
    @HystrixCommand(
        fallbackMethod = "getUserIgnoreExceptionFallback",
        ignoreExceptions = {BusinessException.class, IllegalArgumentException.class}
    )
    public UserDTO getUserIgnoreException(Long userId) {
        if (userId == null) {
            throw new IllegalArgumentException("用户ID不能为空");
        }
        
        if (userId < 0) {
            throw new BusinessException("业务异常");
        }
        
        return userRepository.findById(userId)
            .orElseThrow(() -> new RuntimeException("系统异常"));
    }
    
    public UserDTO getUserIgnoreExceptionFallback(Long userId, Throwable e) {
        log.warn("忽略特定异常的降级，userId: {}", userId);
        
        return UserDTO.builder()
            .id(userId)
            .name("忽略异常降级")
            .message("忽略异常: " + e.getClass().getSimpleName())
            .build();
    }
}

// 1. Feign客户端接口
@FeignClient(
    name = "user-service",
    path = "/api/users",
    fallback = UserServiceFallback.class,
    fallbackFactory = UserServiceFallbackFactory.class
)
public interface UserServiceClient {
    
    @GetMapping("/{id}")
    ApiResult<UserDTO> getUserById(@PathVariable("id") Long id);
    
    @PostMapping("/batch")
    ApiResult<List<UserDTO>> getUsersBatch(@RequestBody List<Long> ids);
    
    @GetMapping("/search")
    ApiResult<PageResult<UserDTO>> searchUsers(
        @RequestParam("keyword") String keyword,
        @RequestParam(value = "page", defaultValue = "1") Integer page,
        @RequestParam(value = "size", defaultValue = "20") Integer size
    );
}

// 2. 简单降级实现
@Component
@Slf4j
public class UserServiceFallback implements UserServiceClient {
    
    @Override
    public ApiResult<UserDTO> getUserById(Long id) {
        log.warn("用户服务降级: getUserById({})", id);
        
        return ApiResult.success(
            UserDTO.builder()
                .id(id)
                .name("默认用户")
                .email("default@example.com")
                .status(0)
                .build()
        );
    }
    
    @Override
    public ApiResult<List<UserDTO>> getUsersBatch(List<Long> ids) {
        log.warn("用户服务批量降级: getUsersBatch({})", ids.size());
        
        List<UserDTO> fallbackUsers = ids.stream()
            .map(id -> UserDTO.builder()
                .id(id)
                .name("批量降级用户")
                .status(0)
                .build())
            .collect(Collectors.toList());
        
        return ApiResult.success(fallbackUsers);
    }
    
    @Override
    public ApiResult<PageResult<UserDTO>> searchUsers(String keyword, Integer page, Integer size) {
        log.warn("用户搜索降级: searchUsers({})", keyword);
        
        return ApiResult.success(
            PageResult.<UserDTO>builder()
                .page(page)
                .size(size)
                .total(0)
                .data(Collections.emptyList())
                .build()
        );
    }
}

// 3. 高级降级工厂（可获取异常信息）
@Component
@Slf4j
public class UserServiceFallbackFactory implements FallbackFactory<UserServiceClient> {
    
    private final UserCacheService userCacheService;
    private final DegradeLogger degradeLogger;
    private final MetricsCollector metricsCollector;
    
    public UserServiceFallbackFactory(UserCacheService userCacheService,
                                     DegradeLogger degradeLogger,
                                     MetricsCollector metricsCollector) {
        this.userCacheService = userCacheService;
        this.degradeLogger = degradeLogger;
        this.metricsCollector = metricsCollector;
    }
    
    @Override
    public UserServiceClient create(Throwable cause) {
        return new UserServiceClient() {
            @Override
            public ApiResult<UserDTO> getUserById(Long id) {
                // 记录降级指标
                metricsCollector.recordFallback("user-service", "getUserById", cause);
                
                // 记录降级日志
                degradeLogger.logFallback("getUserById", id, cause);
                
                // 多级降级策略
                return executeMultiLevelFallback(id, cause);
            }
            
            @Override
            public ApiResult<List<UserDTO>> getUsersBatch(List<Long> ids) {
                metricsCollector.recordFallback("user-service", "getUsersBatch", cause);
                degradeLogger.logFallback("getUsersBatch", ids.size(), cause);
                
                // 批量降级：尝试部分成功
                List<UserDTO> result = new ArrayList<>();
                for (Long id : ids) {
                    try {
                        ApiResult<UserDTO> userResult = getUserById(id);
                        if (userResult.isSuccess()) {
                            result.add(userResult.getData());
                        }
                    } catch (Exception e) {
                        // 单个失败不影响整体
                        result.add(createDefaultUser(id));
                    }
                }
                
                return ApiResult.success(result);
            }
            
            @Override
            public ApiResult<PageResult<UserDTO>> searchUsers(String keyword, 
                                                            Integer page, 
                                                            Integer size) {
                metricsCollector.recordFallback("user-service", "searchUsers", cause);
                degradeLogger.logFallback("searchUsers", keyword, cause);
                
                // 搜索降级：返回热门结果或空结果
                List<UserDTO> fallbackResults = getFallbackSearchResults(keyword);
                
                return ApiResult.success(
                    PageResult.<UserDTO>builder()
                        .page(page)
                        .size(size)
                        .total(fallbackResults.size())
                        .data(fallbackResults)
                        .message("搜索服务降级，返回推荐结果")
                        .build()
                );
            }
            
            /**
             * 多级降级执行
             */
            private ApiResult<UserDTO> executeMultiLevelFallback(Long userId, Throwable cause) {
                // 第一级：尝试本地缓存
                try {
                    UserDTO cachedUser = userCacheService.getUserFromLocalCache(userId);
                    if (cachedUser != null) {
                        log.info("一级降级成功：从本地缓存获取用户");
                        return ApiResult.success(cachedUser);
                    }
                } catch (Exception e) {
                    log.warn("一级降级失败", e);
                }
                
                // 第二级：尝试远程缓存
                try {
                    UserDTO cachedUser = userCacheService.getUserFromRemoteCache(userId);
                    if (cachedUser != null) {
                        log.info("二级降级成功：从远程缓存获取用户");
                        return ApiResult.success(cachedUser);
                    }
                } catch (Exception e) {
                    log.warn("二级降级失败", e);
                }
                
                // 第三级：返回默认值
                log.info("三级降级：返回默认用户");
                return ApiResult.success(createDefaultUser(userId));
            }
            
            private UserDTO createDefaultUser(Long id) {
                return UserDTO.builder()
                    .id(id)
                    .name("降级用户")
                    .email("fallback@example.com")
                    .status(0)
                    .build();
            }
            
            private List<UserDTO> getFallbackSearchResults(String keyword) {
                // 返回热门用户或空列表
                if (StringUtils.hasText(keyword)) {
                    return Arrays.asList(
                        UserDTO.builder().id(1L).name("热门用户1").build(),
                        UserDTO.builder().id(2L).name("热门用户2").build()
                    );
                }
                return Collections.emptyList();
            }
        };
    }
}

## 三、Resilience4j 降级

### 1. **Circuit Breaker 降级**

```
import io.github.resilience4j.circuitbreaker.CircuitBreaker;
import io.github.resilience4j.circuitbreaker.CircuitBreakerConfig;
import io.github.resilience4j.circuitbreaker.CircuitBreakerRegistry;
import io.github.resilience4j.decorators.Decorators;
import org.springframework.stereotype.Service;
import java.time.Duration;
import java.util.concurrent.Callable;
import java.util.concurrent.CompletableFuture;
import java.util.function.Supplier;

@Service
@Slf4j
public class Resilience4jFallbackService {
    
    private final CircuitBreaker circuitBreaker;
    private final UserService userService;
    private final UserCacheService userCacheService;
    
    public Resilience4jFallbackService(UserService userService, 
                                      UserCacheService userCacheService) {
        this.userService = userService;
        this.userCacheService = userCacheService;
        
        // 创建熔断器配置
        CircuitBreakerConfig config = CircuitBreakerConfig.custom()
            .failureRateThreshold(50)                         // 失败率阈值
            .slowCallRateThreshold(100)                       // 慢调用率阈值
            .slowCallDurationThreshold(Duration.ofSeconds(2)) // 慢调用阈值
            .waitDurationInOpenState(Duration.ofSeconds(30))  // 熔断持续时间
            .permittedNumberOfCallsInHalfOpenState(10)        // 半开状态允许调用数
            .minimumNumberOfCalls(20)                         // 最小调用数
            .slidingWindowType(CircuitBreakerConfig.SlidingWindowType.COUNT_BASED) // 滑动窗口类型
            .slidingWindowSize(100)                           // 滑动窗口大小
            .recordExceptions(IOException.class, TimeoutException.class) // 记录异常
            .ignoreExceptions(BusinessException.class)        // 忽略异常
            .build();
        
        this.circuitBreaker = CircuitBreaker.of("user-service", config);
        
        // 注册事件监听器
        registerEventListeners();
    }
    
    /**
     * 基础降级
     */
    public UserDTO getUserWithFallback(Long userId) {
        Supplier<UserDTO> supplier = CircuitBreaker
            .decorateSupplier(circuitBreaker, () -> userService.getUserById(userId));
        
        Try<UserDTO> result = Try.ofSupplier(supplier)
            .recover(throwable -> {
                // 降级逻辑
                return handleFallback(userId, throwable);
            });
        
        return result.get();
    }
    
    /**
     * 多级降级
     */
    public UserDTO getUserWithMultiLevelFallback(Long userId) {
        Supplier<UserDTO> level1 = () -> userService.getUserById(userId);
        Supplier<UserDTO> level2 = () -> userCacheService.getUserFromCache(userId);
        Supplier<UserDTO> level3 = () -> createDefaultUser(userId);
        
        return Decorators.ofSupplier(level1)
            .withCircuitBreaker(circuitBreaker)
            .withFallback(Arrays.asList(
                Exception.class, 
                Throwable.class
            ), throwable -> {
                log.warn("一级降级失败，尝试二级", throwable);
                return level2.get();
            })
            .withFallback(Arrays.asList(
                Exception.class, 
                Throwable.class
            ), throwable -> {
                log.warn("二级降级失败，使用三级", throwable);
                return level3.get();
            })
            .get();
    }
    
    /**
     * 异步降级
     */
    public CompletableFuture<UserDTO> getUserAsync(Long userId) {
        Supplier<CompletableFuture<UserDTO>> supplier = () -> 
            CompletableFuture.supplyAsync(() -> userService.getUserById(userId));
        
        Supplier<CompletableFuture<UserDTO>> decoratedSupplier = 
            CircuitBreaker.decorateSupplier(circuitBreaker, supplier);
        
        return Try.ofSupplier(decoratedSupplier)
            .recover(throwable -> {
                // 异步降级
                return CompletableFuture.completedFuture(
                    handleFallback(userId, throwable)
                );
            })
            .get();
    }
    
    /**
     * 组合降级（CircuitBreaker + Retry + Bulkhead）
     */
    public UserDTO getUserWithCombinedFallback(Long userId) {
        CircuitBreaker circuitBreaker = CircuitBreaker.ofDefaults("user-service");
        Retry retry = Retry.ofDefaults("user-service");
        Bulkhead bulkhead = Bulkhead.ofDefaults("user-service");
        
        Supplier<UserDTO> decoratedSupplier = Decorators
            .ofSupplier(() -> userService.getUserById(userId))
            .withCircuitBreaker(circuitBreaker)
            .withBulkhead(bulkhead)
            .withRetry(retry)
            .withFallback(Arrays.asList(
                Exception.class, 
                Throwable.class
            ), throwable -> handleFallback(userId, throwable))
            .decorate();
        
        return Try.ofSupplier(decoratedSupplier)
            .getOrElseGet(throwable -> createDefaultUser(userId));
    }
    
    /**
     * 处理降级
     */
    private UserDTO handleFallback(Long userId, Throwable throwable) {
        // 记录降级
        log.warn("用户服务降级，userId: {}, cause: {}", userId, throwable.getMessage());
        
        // 根据异常类型选择降级策略
        if (throwable instanceof TimeoutException) {
            return UserDTO.builder()
                .id(userId)
                .name("请求超时")
                .message("用户服务响应超时，请稍后重试")
                .build();
        } else if (throwable instanceof CircuitBreakerOpenException) {
            return UserDTO.builder()
                .id(userId)
                .name("服务熔断")
                .message("用户服务暂时不可用，请稍后重试")
                .build();
        } else if (throwable instanceof IOException) {
            return UserDTO.builder()
                .id(userId)
                .name("网络异常")
                .message("网络连接异常，请检查网络")
                .build();
        }
        
        // 默认降级
        return createDefaultUser(userId);
    }
    
    /**
     * 注册事件监听器
     */
    private void registerEventListeners() {
        circuitBreaker.getEventPublisher()
            .onSuccess(event -> {
                log.debug("CircuitBreaker成功: {}", event.getEventType());
            })
            .onError(event -> {
                log.warn("CircuitBreaker错误: {}", event.getEventType());
            })
            .onStateTransition(event -> {
                log.info("CircuitBreaker状态变更: {} -> {}", 
                    event.getStateTransition().getFromState(),
                    event.getStateTransition().getToState());
                
                // 状态变更时发送告警
                if (event.getStateTransition().getToState() == CircuitBreaker.State.OPEN) {
                    sendAlert("熔断器打开", "user-service");
                } else if (event.getStateTransition().getToState() == CircuitBreaker.State.CLOSED) {
                    sendAlert("熔断器关闭", "user-service");
                }
            })
            .onCallNotPermitted(event -> {
                log.warn("CircuitBreaker拒绝调用");
            });
    }
}

## 四、Sentinel 降级

### 1. **@SentinelResource 降级**

```
import com.alibaba.csp.sentinel.annotation.SentinelResource;
import com.alibaba.csp.sentinel.slots.block.BlockException;
import com.alibaba.csp.sentinel.slots.block.degrade.DegradeException;
import org.springframework.stereotype.Service;
import java.util.concurrent.TimeUnit;

@Service
@Slf4j
public class SentinelFallbackService {
    
    /**
     * 基础降级
     */
    @SentinelResource(
        value = "getUserById",
        blockHandler = "getUserBlockHandler",      // 流控/熔断降级
        fallback = "getUserFallback",             // 业务异常降级
        exceptionsToIgnore = {IllegalArgumentException.class}  // 忽略的异常
    )
    public UserDTO getUserById(Long userId) {
        // 模拟耗时操作
        try {
            TimeUnit.MILLISECONDS.sleep(100);
        } catch (InterruptedException e) {
            Thread.currentThread().interrupt();
        }
        
        if (userId <= 0) {
            throw new IllegalArgumentException("用户ID无效");
        }
        
        return userRepository.findById(userId)
            .orElseThrow(() -> new UserNotFoundException("用户不存在"));
    }
    
    /**
     * 流控/熔断降级处理
     * 参数和返回类型必须与原方法一致，最后加BlockException参数
     */
    public UserDTO getUserBlockHandler(Long userId, BlockException ex) {
        log.warn("用户服务限流/熔断降级，userId: {}, rule: {}", 
            userId, ex.getRule());
        
        // 根据异常类型返回不同提示
        if (ex instanceof DegradeException) {
            return UserDTO.builder()
                .id(userId)
                .name("服务降级")
                .message("用户服务暂时不可用，请稍后重试")
                .suggestRetryTime(System.currentTimeMillis() + 30000)  // 30秒后重试
                .build();
        } else {
            return UserDTO.builder()
                .id(userId)
                .name("系统繁忙")
                .message("当前访问人数过多，请稍后重试")
                .build();
        }
    }
    
    /**
     * 业务异常降级处理
     * 参数和返回类型必须与原方法一致，最后加Throwable参数
     */
    public UserDTO getUserFallback(Long userId, Throwable t) {
        log.error("用户服务业务异常降级，userId: {}, error: {}", 
            userId, t.getMessage(), t);
        
        // 根据异常类型选择降级策略
        if (t instanceof UserNotFoundException) {
            return UserDTO.builder()
                .id(userId)
                .name("用户不存在")
                .status(0)
                .build();
        } else if (t instanceof TimeoutException) {
            return UserDTO.builder()
                .id(userId)
                .name("请求超时")
                .message("用户服务响应超时，请稍后重试")
                .build();
        } else if (t instanceof RuntimeException) {
            return UserDTO.builder()
                .id(userId)
                .name("系统异常")
                .message("系统内部错误，请联系管理员")
                .build();
        }
        
        // 默认降级
        return createDefaultUser(userId);
    }
    
    /**
     * 外部降级类
     */
    @SentinelResource(
        value = "getUserDetail",
        blockHandler = "getUserDetailBlockHandler",
        blockHandlerClass = UserFallbackHandler.class,
        fallback = "getUserDetailFallback",
        fallbackClass = UserFallbackHandler.class
    )
    public UserDetailDTO getUserDetail(Long userId) {
        return userDetailService.getUserDetail(userId);
    }
    
    /**
     * 热点参数降级
     */
    @SentinelResource(
        value = "getHotUser",
        blockHandler = "getHotUserBlockHandler"
    )
    public UserDTO getHotUser(Long userId, String source) {
        // 热点参数限流：针对不同的source进行限流
        return userService.getUserById(userId);
    }
    
    public UserDTO getHotUserBlockHandler(Long userId, String source, BlockException ex) {
        log.warn("热点用户限流降级，userId: {}, source: {}", userId, source);
        
        // 返回简化信息
        return UserDTO.builder()
            .id(userId)
            .name("热门用户")
            .message("当前访问人数过多，展示简化信息")
            .build();
    }
}

/**
 * 外部降级处理器
 */

@Component
@Slf4j
public class SmartFallbackManager {
    
    // 降级级别
    public enum FallbackLevel {
        LEVEL_0,  // 无降级
        LEVEL_1,  // 轻度降级 - 缓存降级
        LEVEL_2,  // 中度降级 - 默认值降级
        LEVEL_3,  // 重度降级 - 空值降级
        LEVEL_4   // 完全降级 - 异常降级
    }
    
    // 当前降级级别
    private volatile FallbackLevel currentLevel = FallbackLevel.LEVEL_0;
    
    // 降级配置
    private final Map<String, ServiceFallbackConfig> configs = new ConcurrentHashMap<>();
    
    /**
     * 初始化降级配置
     */
    @PostConstruct
    public void init() {
        // 用户服务降级配置
        configs.put("user-service", ServiceFallbackConfig.builder()
            .serviceName("user-service")
            .levels(Arrays.asList(
                FallbackLevelConfig.builder()
                    .level(FallbackLevel.LEVEL_1)
                    .condition(metrics -> metrics.getErrorRate() > 0.1)
                    .strategy(new CacheFallbackStrategy())
                    .build(),
                FallbackLevelConfig.builder()
                    .level(FallbackLevel.LEVEL_2)
                    .condition(metrics -> metrics.getErrorRate() > 0.3)
                    .strategy(new DefaultValueFallbackStrategy())
                    .build(),
                FallbackLevelConfig.builder()
                    .level(FallbackLevel.LEVEL_3)
                    .condition(metrics -> metrics.getErrorRate() > 0.5)
                    .strategy(new EmptyValueFallbackStrategy())
                    .build(),
                FallbackLevelConfig.builder()
                    .level(FallbackLevel.LEVEL_4)
                    .condition(metrics -> metrics.getErrorRate() > 0.7)
                    .strategy(new ExceptionFallbackStrategy())
                    .build()
            ))
            .build());
            
        // 启动自动降级调整
        startAutoFallbackAdjustment();
    }
    
    /**
     * 执行多级降级
     */
    public <T> T executeWithSmartFallback(String serviceName, 
                                         Supplier<T> normalLogic,
                                         FallbackContext context) {
        
        ServiceFallbackConfig config = configs.get(serviceName);
        if (config == null) {
            return normalLogic.get();
        }
        
        // 获取当前应该执行的降级级别
        FallbackLevel targetLevel = determineFallbackLevel(serviceName, context);
        
        // 执行对应级别的降级逻辑
        return executeAtLevel(serviceName, normalLogic, context, targetLevel);
    }
    
    /**
     * 确定降级级别
     */
    private FallbackLevel determineFallbackLevel(String serviceName, 
                                                FallbackContext context) {
        // 1. 检查手动开关
        if (hasManualOverride(serviceName)) {
            return getManualLevel(serviceName);
        }
        
        // 2. 检查系统指标
        SystemMetrics metrics = getSystemMetrics();
        ServiceFallbackConfig config = configs.get(serviceName);
        
        // 3. 按顺序检查降级条件
        for (FallbackLevelConfig levelConfig : config.getLevels()) {
            if (levelConfig.getCondition().test(metrics)) {
                return levelConfig.getLevel();
            }
        }
        
        return FallbackLevel.LEVEL_0;
    }
    
    /**
     * 在指定级别执行
     */
    private <T> T executeAtLevel(String serviceName, 
                                Supplier<T> normalLogic,
                                FallbackContext context,
                                FallbackLevel level) {
        
        ServiceFallbackConfig config = configs.get(serviceName);
        FallbackLevelConfig levelConfig = config.getLevelConfig(level);
        
        if (levelConfig == null || levelConfig.getLevel() == FallbackLevel.LEVEL_0) {
            // 无降级，执行正常逻辑
            try {
                return normalLogic.get();
            } catch (Exception e) {
                // 即使无降级配置，也提供默认降级
                return (T) config.getDefaultStrategy().execute(context, e);
            }
        }
        
        // 执行降级策略
        try {
            return (T) levelConfig.getStrategy().execute(context, null);
        } catch (Exception e) {
            log.error("降级策略执行失败", e);
            
            // 降级失败，尝试更高级别的降级
            FallbackLevel nextLevel = getNextLevel(level);
            if (nextLevel != level) {
                return executeAtLevel(serviceName, normalLogic, context, nextLevel);
            }
            
            // 无法降级，抛出异常
            throw new ServiceDegradedException("服务降级失败");
        }
    }
    
    /**
     * 自动调整降级级别
     */
    @Scheduled(fixedDelay = 30000)  // 每30秒调整一次
    public void autoAdjustFallbackLevels() {
        configs.forEach((serviceName, config) -> {
            SystemMetrics metrics = getSystemMetrics();
            FallbackContext context = FallbackContext.builder()
                .serviceName(serviceName)
                .timestamp(System.currentTimeMillis())
                .build();
            
            FallbackLevel newLevel = determineFallbackLevel(serviceName, context);
            
            if (newLevel != currentLevel) {
                log.info("自动调整降级级别: {} {} -> {}", 
                    serviceName, currentLevel, newLevel);
                
                currentLevel = newLevel;
                
                // 发送降级事件
                publishFallbackLevelChangeEvent(serviceName, currentLevel, newLevel);
                
                // 发送告警
                if (newLevel.ordinal() > FallbackLevel.LEVEL_1.ordinal()) {
                    sendFallbackAlert(serviceName, currentLevel, newLevel);
                }
            }
        });
    }
    
    /**
     * 手动触发降级
     */
    public void manualTriggerFallback(String serviceName, FallbackLevel level, 
                                     String reason, String operator) {
        ServiceFallbackConfig config = configs.get(serviceName);
        if (config != null) {
            config.setManualLevel(level);
            config.setManualReason(reason);
            config.setOperator(operator);
            config.setManualTime(System.currentTimeMillis());
            
            log.info("手动触发降级: {} -> {}, 原因: {}, 操作人: {}", 
                serviceName, level, reason, operator);
            
            // 记录操作日志
            auditLogService.logManualFallback(serviceName, level, reason, operator);
        }
    }
    
    /**
     * 恢复降级
     */
    public void recoverFallback(String serviceName, String operator) {
        ServiceFallbackConfig config = configs.get(serviceName);
        if (config != null) {
            config.setManualLevel(null);
            config.setManualReason(null);
            config.setOperator(operator);
            config.setManualTime(null);
            
            log.info("恢复降级: {}, 操作人: {}", serviceName, operator);
        }
    }

    public interface FallbackStrategy<T> {
        T execute(FallbackContext context, Throwable cause);
    }
    
    // 缓存降级策略
    public class CacheFallbackStrategy implements FallbackStrategy<Object> {
        @Override
        public Object execute(FallbackContext context, Throwable cause) {
            String serviceName = context.getServiceName();
            Object params = context.getParams();
            
            // 尝试从缓存获取
            Object cached = cacheService.get(buildCacheKey(serviceName, params));
            if (cached != null) {
                return cached;
            }
            
            throw new FallbackException("缓存降级失败");
        }
    }
    
    // 默认值降级策略
    public class DefaultValueFallbackStrategy implements FallbackStrategy<Object> {
        @Override
        public Object execute(FallbackContext context, Throwable cause) {
            // 返回预定义的默认值
            return createDefaultValue(context);
        }
    }

    @Builder
    public static class ServiceFallbackConfig {
        private String serviceName;
        private List<FallbackLevelConfig> levels;
        private FallbackStrategy defaultStrategy;
        private FallbackLevel manualLevel;     // 手动降级级别
        private String manualReason;           // 手动降级原因
        private String operator;               // 操作人
        private Long manualTime;               // 手动降级时间
    }

    public static class FallbackLevelConfig {

        private Predicate<SystemMetrics> condition;
        private FallbackStrategy strategy;
    }

    @Builder

        private String methodName;
        private Object params;
        private Map<String, Object> metadata;
        private long timestamp;
    }
