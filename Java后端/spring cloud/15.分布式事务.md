# Seata 分布式事务完整解决方案

Seata 是阿里巴巴开源的分布式事务解决方案，支持 AT、TCC、SAGA 和 XA 事务模式，为企业级分布式事务提供了完整解决方案。

## 一、Seata 核心架构

### 1. **Seata 架构组件**

```
┌─────────────────────────────────────────────────────────────┐
│                     应用层 (Application)                      │
├─────────────────────────────────────────────────────────────┤
│                     TM (事务管理器)                          │
│                      ↓                                      │
│                     RM (资源管理器)                          │
├─────────────────────────────────────────────────────────────┤
│                      TC (事务协调器)                          │
│                   ┌──────┐ ┌──────┐                        │
│                   │ 注册中心  │ │ 配置中心  │                  │
│                   └──────┘ └──────┘                        │
├─────────────────────────────────────────────────────────────┤
│                     存储层 (Storage)                         │
│                   ┌──────┐ ┌──────┐                        │
│                   │ 数据库 │ │ 文件存储│                      │
│                   └──────┘ └──────┘                        │
└─────────────────────────────────────────────────────────────┘
```

## 三、AT 模式（自动模式）

### 1. **AT 模式原理**

```
1. 第一阶段：提交前
   ┌─────────┐   ┌─────────┐   ┌─────────┐
   │ 业务SQL  │ → │ 前镜像  │ → │ 后镜像  │
   └─────────┘   └─────────┘   └─────────┘
   
2. 第二阶段：提交时
   ┌─────────┐   ┌─────────┐
   │ 全局提交 │ → │ 删除日志 │
   └─────────┘   └─────────┘
   
3. 第二阶段：回滚时
   ┌─────────┐   ┌─────────┐
   │ 全局回滚 │ → │ 恢复数据 │
   └─────────┘   └─────────┘
```

### 2. **AT 模式实现**(推荐)

```
// 1. 订单服务
@Service
@Slf4j
public class OrderService {
    
    @Autowired
    private OrderMapper orderMapper;
    
    @Autowired
    private AccountFeignClient accountFeignClient;
    
    @Autowired
    private ProductFeignClient productFeignClient;
    
    @Autowired
    private MessageProducer messageProducer;
    
    /**
     * 创建订单（AT模式）
     */
    @GlobalTransactional(name = "createOrder", timeoutMills = 30000)
    public ApiResult<OrderDTO> createOrder(OrderCreateRequest request) {
        // 1. 创建订单
        Order order = buildOrder(request);
        orderMapper.insert(order);
        log.info("订单创建成功: {}", order.getOrderNo());
        
        // 2. 扣减库存
        ProductDeductRequest deductRequest = new ProductDeductRequest();
        deductRequest.setProductId(request.getProductId());
        deductRequest.setQuantity(request.getQuantity());
        
        ApiResult<Void> deductResult = productFeignClient.deductStock(deductRequest);
        if (!deductResult.isSuccess()) {
            throw new BusinessException("库存不足");
        }
        log.info("库存扣减成功");
        
        // 3. 扣减余额
        BalanceDeductRequest balanceRequest = new BalanceDeductRequest();
        balanceRequest.setUserId(request.getUserId());
        balanceRequest.setAmount(request.getTotalAmount());
        
        ApiResult<Void> balanceResult = accountFeignClient.deductBalance(balanceRequest);
        if (!balanceResult.isSuccess()) {
            throw new BusinessException("余额不足");
        }
        log.info("余额扣减成功");
        
        // 4. 更新订单状态
        order.setStatus(OrderStatus.PAID);
        order.setPayTime(new Date());
        orderMapper.updateById(order);
        
        // 5. 发送订单创建消息
        sendOrderCreatedMessage(order);
        
        return ApiResult.success(OrderDTO.fromEntity(order));
    }
    
    /**
     * 批量创建订单
     */
    @GlobalTransactional(name = "batchCreateOrder", timeoutMills = 60000)
    public ApiResult<BatchOrderResult> batchCreateOrder(List<OrderCreateRequest> requests) {
        BatchOrderResult result = new BatchOrderResult();
        
        for (OrderCreateRequest request : requests) {
            try {
                ApiResult<OrderDTO> orderResult = createOrder(request);
                if (orderResult.isSuccess()) {
                    result.addSuccess(request);
                } else {
                    result.addFailure(request, orderResult.getMessage());
                }
            } catch (Exception e) {
                result.addFailure(request, e.getMessage());
                
                // 记录异常，但不中断整个批量操作
                log.error("创建订单失败: {}", request, e);
            }
        }
        
        return ApiResult.success(result);
    }
}

// 2. 账户服务
@Service
@Slf4j
public class AccountService {
    
    @Autowired
    private AccountMapper accountMapper;
    
    /**
     * 扣减余额
     */
    public ApiResult<Void> deductBalance(BalanceDeductRequest request) {
        // 查询账户
        Account account = accountMapper.selectByUserIdForUpdate(request.getUserId());
        if (account == null) {
            return ApiResult.error(404, "账户不存在");
        }
        
        // 检查余额
        BigDecimal newBalance = account.getBalance().subtract(request.getAmount());
        if (newBalance.compareTo(BigDecimal.ZERO) < 0) {
            return ApiResult.error(400, "余额不足");
        }
        
        // 更新余额
        int rows = accountMapper.updateBalance(account.getId(), 
            request.getAmount().negate());
        
        if (rows > 0) {
            // 记录余额变动
            accountMapper.insertBalanceLog(account.getId(), 
                BalanceLogType.DEDUCT, request.getAmount(), "订单支付");
            
            return ApiResult.success();
        }
        
        return ApiResult.error(500, "扣减余额失败");
    }
}

@Service
@Slf4j
public class ProductService {
    
    @Autowired
    private ProductMapper productMapper;
    
    /**
     * 扣减库存
     */
    public ApiResult<Void> deductStock(ProductDeductRequest request) {
        // 查询商品
        Product product = productMapper.selectByIdForUpdate(request.getProductId());
        if (product == null) {
            return ApiResult.error(404, "商品不存在");
        }
        
        // 检查库存
        int newStock = product.getStock() - request.getQuantity();
        if (newStock < 0) {
            return ApiResult.error(400, "库存不足");
        }
        
        // 更新库存
        int rows = productMapper.updateStock(product.getId(), 
            -request.getQuantity());
        
        if (rows > 0) {
            // 记录库存变动
            productMapper.insertStockLog(product.getId(), 
                StockLogType.DEDUCT, request.getQuantity(), "订单扣减");
            
            return ApiResult.success();
        }
        
        return ApiResult.error(500, "扣减库存失败");
    }
}
```

## 四、TCC 模式（补偿模式）

### 1. **TCC 模式原理**

```
Try阶段: 预留资源
Confirm阶段: 确认提交
Cancel阶段: 取消释放
```

### 2. **TCC 模式实现**

```
// 1. TCC接口定义
public interface OrderTccAction {
    
    @TwoPhaseBusinessAction(name = "orderTcc", commitMethod = "commit", rollbackMethod = "rollback")
    boolean prepareOrder(BusinessActionContext actionContext, OrderCreateRequest request);
    
    boolean commit(BusinessActionContext actionContext);
    
    boolean rollback(BusinessActionContext actionContext);
}

// 2. TCC实现
@Service
@Slf4j
public class OrderTccActionImpl implements OrderTccAction {
    
    @Autowired
    private OrderMapper orderMapper;
    
    @Autowired
    private AccountTccAction accountTccAction;
    
    @Autowired
    private ProductTccAction productTccAction;
    
    @Override
    public boolean prepareOrder(BusinessActionContext actionContext, OrderCreateRequest request) {
        log.info("=== 开始Try阶段: 创建订单 ===");
        
        try {
            // 1. 创建订单（状态为待确认）
            Order order = buildOrder(request);
            order.setStatus(OrderStatus.PREPARE);
            orderMapper.insert(order);
            
            // 保存上下文
            actionContext.getActionContext().put("order", order);
            
            // 2. Try阶段 - 预留库存
            BusinessActionContext productContext = new BusinessActionContext();
            productContext.setActionContext(new HashMap<>());
            productContext.getActionContext().put("productId", request.getProductId());
            productContext.getActionContext().put("quantity", request.getQuantity());
            
            boolean productResult = productTccAction.prepareDeductStock(productContext);
            if (!productResult) {
                throw new BusinessException("库存预留失败");
            }
            
            // 保存子事务上下文
            actionContext.getActionContext().put("productContext", productContext);
            
            // 3. Try阶段 - 预留余额
            BusinessActionContext accountContext = new BusinessActionContext();
            accountContext.setActionContext(new HashMap<>());
            accountContext.getActionContext().put("userId", request.getUserId());
            accountContext.getActionContext().put("amount", request.getTotalAmount());
            
            boolean accountResult = accountTccAction.prepareDeductBalance(accountContext);
            if (!accountResult) {
                throw new BusinessException("余额预留失败");
            }
            
            actionContext.getActionContext().put("accountContext", accountContext);
            
            log.info("=== Try阶段完成: 订单号={} ===", order.getOrderNo());
            return true;
            
        } catch (Exception e) {
            log.error("Try阶段失败", e);
            
            // 执行取消操作
            rollback(actionContext);
            throw e;
        }
    }
    
    @Override
    public boolean commit(BusinessActionContext actionContext) {
        log.info("=== 开始Confirm阶段: 确认订单 ===");
        
        try {
            Order order = (Order) actionContext.getActionContext().get("order");
            if (order == null) {
                log.warn("订单不存在，跳过Confirm");
                return true;
            }
            
            // 1. 更新订单状态为已支付
            order.setStatus(OrderStatus.PAID);
            order.setPayTime(new Date());
            orderMapper.updateById(order);
            
            // 2. Confirm库存扣减
            BusinessActionContext productContext = (BusinessActionContext) 
                actionContext.getActionContext().get("productContext");
            if (productContext != null) {
                productTccAction.commit(productContext);
            }
            
            // 3. Confirm余额扣减
            BusinessActionContext accountContext = (BusinessActionContext) 
                actionContext.getActionContext().get("accountContext");
            if (accountContext != null) {
                accountTccAction.commit(accountContext);
            }
            
            // 4. 发送订单创建消息
            sendOrderCreatedMessage(order);
            
            log.info("=== Confirm阶段完成: 订单号={} ===", order.getOrderNo());
            return true;
            
        } catch (Exception e) {
            log.error("Confirm阶段失败", e);
            
            // 重试机制
            handleConfirmRetry(actionContext, e);
            return false;
        }
    }
    
    @Override
    public boolean rollback(BusinessActionContext actionContext) {
        log.info("=== 开始Cancel阶段: 取消订单 ===");
        
        try {
            Order order = (Order) actionContext.getActionContext().get("order");
            
            // 1. Cancel库存预留
            BusinessActionContext productContext = (BusinessActionContext) 
                actionContext.getActionContext().get("productContext");
            if (productContext != null) {
                productTccAction.rollback(productContext);
            }
            
            // 2. Cancel余额预留
            BusinessActionContext accountContext = (BusinessActionContext) 
                actionContext.getActionContext().get("accountContext");
            if (accountContext != null) {
                accountTccAction.rollback(accountContext);
            }
            
            // 3. 删除订单记录
            if (order != null) {
                orderMapper.deleteById(order.getId());
                log.info("订单已删除: {}", order.getOrderNo());
            }
            
            // 4. 发送订单取消消息
            if (order != null) {
                sendOrderCanceledMessage(order);
            }
            
            log.info("=== Cancel阶段完成 ===");
            return true;
            
        } catch (Exception e) {
            log.error("Cancel阶段失败", e);
            
            // 记录异常，后续补偿
            recordCancelException(actionContext, e);
            
            // 重试机制
            handleRollbackRetry(actionContext, e);
            return false;
        }
    }
}

// 3. 账户TCC实现
@Service
@Slf4j
public class AccountTccActionImpl implements AccountTccAction {
    
    @Autowired
    private AccountMapper accountMapper;
    
    @Override
    public boolean prepareDeductBalance(BusinessActionContext actionContext) {
        log.info("=== 账户Try阶段: 预留余额 ===");
        
        Long userId = (Long) actionContext.getActionContext().get("userId");
        BigDecimal amount = (BigDecimal) actionContext.getActionContext().get("amount");
        
        Account account = accountMapper.selectByUserIdForUpdate(userId);
        if (account == null) {
            throw new BusinessException("账户不存在");
        }
        
        // 检查余额
        BigDecimal newBalance = account.getBalance().subtract(amount);
        if (newBalance.compareTo(BigDecimal.ZERO) < 0) {
            throw new BusinessException("余额不足");
        }
        
        // 冻结金额
        BigDecimal newFreezeAmount = account.getFreezeAmount().add(amount);
        
        int rows = accountMapper.updateFreezeAmount(
            account.getId(), amount, newFreezeAmount);
        
        if (rows > 0) {
            // 记录冻结日志
            accountMapper.insertFreezeLog(account.getId(), 
                FreezeLogType.FREEZE, amount, "订单预留");
            
            return true;
        }
        
        return false;
    }
    
    @Override
    public boolean commit(BusinessActionContext actionContext) {
        log.info("=== 账户Confirm阶段: 确认扣减 ===");
        
        Long userId = (Long) actionContext.getActionContext().get("userId");
        BigDecimal amount = (BigDecimal) actionContext.getActionContext().get("amount");
        
        Account account = accountMapper.selectByUserId(userId);
        if (account == null) {
            log.warn("账户不存在，跳过Confirm");
            return true;
        }
        
        // 实际扣减余额
        BigDecimal newBalance = account.getBalance().subtract(amount);
        BigDecimal newFreezeAmount = account.getFreezeAmount().subtract(amount);
        
        int rows = accountMapper.confirmDeduct(
            account.getId(), amount, newBalance, newFreezeAmount);
        
        if (rows > 0) {
            // 记录扣减日志
            accountMapper.insertBalanceLog(account.getId(), 
                BalanceLogType.DEDUCT, amount, "订单支付");
            
            return true;
        }
        
        return false;
    }
    
    @Override
    public boolean rollback(BusinessActionContext actionContext) {
        log.info("=== 账户Cancel阶段: 释放冻结 ===");
        
        Long userId = (Long) actionContext.getActionContext().get("userId");
        BigDecimal amount = (BigDecimal) actionContext.getActionContext().get("amount");
        
        Account account = accountMapper.selectByUserId(userId);
        if (account == null) {
            log.warn("账户不存在，跳过Cancel");
            return true;
        }
        
        // 释放冻结金额
        BigDecimal newFreezeAmount = account.getFreezeAmount().subtract(amount);
        
        int rows = accountMapper.rollbackFreeze(
            account.getId(), amount, newFreezeAmount);
        
        if (rows > 0) {
            // 记录释放日志
            accountMapper.insertFreezeLog(account.getId(), 
                FreezeLogType.UNFREEZE, amount, "订单取消");
            
            return true;
        }
        
        return false;
    }
}
```

## 五、SAGA 模式（长事务模式）

### 1. **SAGA 模式原理**

```
正向操作: O1 → O2 → O3
反向补偿: C3 ← C2 ← C1
```

### 2. **SAGA 状态机实现**

```
// 1. Saga状态机定义
@Service
@Slf4j
public class OrderSagaService {
    
    @Autowired
    private StateMachineEngine stateMachineEngine;
    
    /**
     * 创建Saga状态机
     */
    public void createOrderSaga(OrderCreateRequest request) {
        String stateMachineName = "orderCreateSaga";
        
        // 构建状态机
        StateMachineBuilder builder = StateMachineBuilderFactory.create();
        
        // 1. 创建订单状态
        builder.addState("CREATE_ORDER")
            .setServiceBean("orderService")
            .setServiceMethod("createOrder")
            .setCompensateServiceBean("orderService")
            .setCompensateServiceMethod("cancelOrder");
        
        // 2. 扣减库存状态
        builder.addState("DEDUCT_STOCK")
            .setServiceBean("productService")
            .setServiceMethod("deductStock")
            .setCompensateServiceBean("productService")
            .setCompensateServiceMethod("restoreStock");
        
        // 3. 扣减余额状态
        builder.addState("DEDUCT_BALANCE")
            .setServiceBean("accountService")
            .setServiceMethod("deductBalance")
            .setCompensateServiceBean("accountService")
            .setCompensateServiceMethod("restoreBalance");
        
        // 设置状态流转
        builder.from("CREATE_ORDER").to("DEDUCT_STOCK");
        builder.from("DEDUCT_STOCK").to("DEDUCT_BALANCE");
        builder.from("DEDUCT_BALANCE").to("SUCCESS");
        
        // 设置补偿流转
        builder.from("DEDUCT_BALANCE").to("DEDUCT_STOCK", "COMPENSATE");
        builder.from("DEDUCT_STOCK").to("CREATE_ORDER", "COMPENSATE");
        
        // 注册状态机
        StateMachine stateMachine = builder.build(stateMachineName);
        stateMachineEngine.registerStateMachine(stateMachine);
        
        // 执行状态机
        Map<String, Object> startParams = new HashMap<>();
        startParams.put("request", request);
        
        StateMachineInstance instance = stateMachineEngine.start(
            stateMachineName, null, startParams);
        
        // 监听状态机执行
        monitorSagaInstance(instance);
    }
    
    /**
     * 订单服务正向操作
     */
    public boolean createOrder(StateContext<String, String> context) {
        OrderCreateRequest request = (OrderCreateRequest) context.getRequest();
        
        log.info("Saga正向操作: 创建订单, 订单号: {}", request.getOrderNo());
        
        try {
            Order order = buildOrder(request);
            orderMapper.insert(order);
            
            // 保存订单ID到上下文
            context.setVariable("orderId", order.getId());
            
            return true;
        } catch (Exception e) {
            log.error("创建订单失败", e);
            return false;
        }
    }
    
    /**
     * 订单服务补偿操作
     */
    public boolean cancelOrder(StateContext<String, String> context) {
        Long orderId = (Long) context.getVariable("orderId");
        
        log.info("Saga补偿操作: 取消订单, 订单ID: {}", orderId);
        
        try {
            if (orderId != null) {
                orderMapper.deleteById(orderId);
            }
            
            return true;
        } catch (Exception e) {
            log.error("取消订单失败", e);
            return false;
        }
    }
    
    /**
     * 监控Saga实例
     */
    private void monitorSagaInstance(StateMachineInstance instance) {
        CompletableFuture.runAsync(() -> {
            try {
                // 等待执行完成
                instance.waitForFinish(30000);
                
                StateMachineStatus status = instance.getStatus();
                if (status == StateMachineStatus.FINISHED) {
                    log.info("Saga执行成功: {}", instance.getId());
                } else if (status == StateMachineStatus.FAILED) {
                    log.error("Saga执行失败: {}", instance.getId());
                    
                    // 获取失败信息
                    List<StateMachineExecution> executions = instance.getExecutions();
                    for (StateMachineExecution execution : executions) {
                        if (execution.getStatus() == ExecutionStatus.FAILED) {
                            log.error("失败步骤: {}, 异常: {}", 
                                execution.getStateName(), execution.getException());
                        }
                    }
                    
                    // 发送告警
                    sendSagaFailedAlert(instance);
                }
            } catch (Exception e) {
                log.error("监控Saga失败", e);
            }
        });
    }
```

```
@Component
@Slf4j
public class SagaCompensationService {
    
    @Autowired
    private OrderMapper orderMapper;
    
    @Autowired
    private AccountMapper accountMapper;
    
    @Autowired
    private ProductMapper productMapper;
    
    /**
     * 定期扫描需要补偿的事务
     */
    @Scheduled(fixedDelay = 60000)  // 每分钟执行一次
    public void scanAndCompensate() {
        log.info("开始扫描需要补偿的事务...");
        
        // 1. 扫描超时未完成的事务
        List<SagaTransaction> timeoutTransactions = 
            sagaTransactionMapper.selectTimeoutTransactions(300000);  // 5分钟
        
        for (SagaTransaction transaction : timeoutTransactions) {
            try {
                compensateTransaction(transaction);
            } catch (Exception e) {
                log.error("补偿事务失败: {}", transaction.getId(), e);
                recordCompensationFailure(transaction, e);
            }
        }
        
        // 2. 扫描补偿失败的事务
        List<SagaTransaction> failedCompensations = 
            sagaTransactionMapper.selectFailedCompensations();
        
        for (SagaTransaction transaction : failedCompensations) {
            try {
                retryCompensation(transaction);
            } catch (Exception e) {
                log.error("重试补偿失败: {}", transaction.getId(), e);
                
                if (transaction.getRetryCount() >= 3) {
                    // 超过重试次数，标记为人工处理
                    markForManualHandling(transaction);
                }
            }
        }
    }
    
    /**
     * 补偿事务
     */
    private void compensateTransaction(SagaTransaction transaction) {
        log.info("开始补偿事务: {}", transaction.getId());
        
        // 获取事务执行记录
        List<SagaExecution> executions = sagaExecutionMapper
            .selectByTransactionId(transaction.getId());
        
        // 按反向顺序执行补偿
        Collections.reverse(executions);
        
        for (SagaExecution execution : executions) {
            if (execution.getStatus() == ExecutionStatus.SUCCEEDED) {
                try {
                    executeCompensation(execution);
                    execution.setCompensationStatus(CompensationStatus.SUCCEEDED);
                } catch (Exception e) {
                    log.error("执行补偿失败: {}", execution.getId(), e);
                    execution.setCompensationStatus(CompensationStatus.FAILED);
                    execution.setCompensationError(e.getMessage());
                    
                    // 记录补偿失败
                    sagaExecutionMapper.update(execution);
                    throw e;
                }
            }
        }
        
        // 更新事务状态
        transaction.setStatus(TransactionStatus.COMPENSATED);
        transaction.setCompensateTime(new Date());
        sagaTransactionMapper.update(transaction);
        
        log.info("事务补偿完成: {}", transaction.getId());
    }
    
    /**
     * 执行补偿操作
     */
    private void executeCompensation(SagaExecution execution) {
        switch (execution.getServiceName()) {
            case "orderService":
                if ("cancelOrder".equals(execution.getCompensationMethod())) {
                    orderService.cancelOrder(execution.getBusinessKey());
                }
                break;
            case "accountService":
                if ("restoreBalance".equals(execution.getCompensationMethod())) {
                    accountService.restoreBalance(execution.getBusinessKey());
                }
                break;
            case "productService":
                if ("restoreStock".equals(execution.getCompensationMethod())) {
                    productService.restoreStock(execution.getBusinessKey());
                }
                break;
        }
    }
}
```

```
            .collect(Collectors.toList());

     * 手动提交事务
     */
    @PostMapping("/transactions/{xid}/commit")
    public ApiResult<Void> manualCommit(@PathVariable String xid) {

            GlobalSession session = sessionManager.findGlobalSession(xid);
            if (session == null) {
                return ApiResult.error(404, "事务不存在");
            }
            
            coordinator.doGlobalCommit(session, true);
            return ApiResult.success("事务已提交");
        } catch (Exception e) {
            return ApiResult.error(500, "提交失败: " + e.getMessage());
        }
    }
    
    /**
     * 手动回滚事务
     */
    @PostMapping("/transactions/{xid}/rollback")
    public ApiResult<Void> manualRollback(@PathVariable String xid) {
        try {
            GlobalSession session = sessionManager.findGlobalSession(xid);
            if (session == null) {
                return ApiResult.error(404, "事务不存在");
            }
            
            coordinator.doGlobalRollback(session, true);

     * 获取事务统计
     */
    @GetMapping("/statistics")
    public ApiResult<TransactionStatistics> getStatistics(
            @RequestParam(required = false) Long startTime,
            @RequestParam(required = false) Long endTime) {
        
        TransactionStatistics statistics = new TransactionStatistics();
        
        List<GlobalSession> sessions = sessionManager.allSessions();
        
        long now = System.currentTimeMillis();
        if (startTime == null) {
            startTime = now - 24 * 60 * 60 * 1000;  // 24小时前
        }
        if (endTime == null) {
            endTime = now;
        }
        
        // 过滤时间范围内的会话
        List<GlobalSession> filteredSessions = sessions.stream()
            .filter(session -> {
                long beginTime = session.getBeginTime();
                return beginTime >= startTime && beginTime <= endTime;
            })
            .collect(Collectors.toList());
        
        // 计算统计
        statistics.setTotalCount(filteredSessions.size());
        statistics.setSuccessCount((int) filteredSessions.stream()
            .filter(s -> s.getStatus() == GlobalStatus.Committed)
            .count());
        statistics.setFailedCount((int) filteredSessions.stream()
            .filter(s -> s.getStatus() == GlobalStatus.Rollbacked)
            .count());
        statistics.setTimeoutCount((int) filteredSessions.stream()
            .filter(s -> s.getStatus() == GlobalStatus.TimeoutRollbacked || 
                        s.getStatus() == GlobalStatus.TimeoutRollbacking)
            .count());
        
        // 计算平均耗时
        double avgDuration = filteredSessions.stream()
            .filter(s -> s.getStatus() == GlobalStatus.Committed || 
                        s.getStatus() == GlobalStatus.Rollbacked)
            .mapToLong(s -> {
                Long end = s.getEndTime();
                if (end == null) {
                    end = now;
                }
                return end - s.getBeginTime();
            })
            .average()
            .orElse(0.0);
        
        statistics.setAverageDuration(avgDuration);
        
        return ApiResult.success(statistics);
    }
```

```
            };
            trendChart.setOption(trendOption);
            
            // 状态分布图
            const statusOption = {
                tooltip: {
                    trigger: 'item',
                    formatter: '{a} <br/>{b}: {c} ({d}%)'
                },
                legend: {
                    orient: 'vertical',
                    left: 10,
                    data: ['已提交', '已回滚', '进行中', '超时']

                series: [
                    {
                        name: '事务状态',
                        type: 'pie',

                        avoidLabelOverlap: false,
                        label: {

                        },
                        emphasis: {

                                show: true,
                                fontSize: '20',
                                fontWeight: 'bold'
                            }
                        },
                        labelLine: {
                            show: false
                        },
                        data: [
                            {value: data.successCount, name: '已提交', itemStyle: {color: '#28a745'}},
                            {value: data.failedCount, name: '已回滚', itemStyle: {color: '#dc3545'}},
                            {value: data.totalCount - data.successCount - data.failedCount - data.timeoutCount, 
                             name: '进行中', itemStyle: {color: '#17a2b8'}},

                        ]
                    }
                ]
            };
            statusChart.setOption(statusOption);
        }
        
        // 加载事务列表
        function loadTransactions(page, keyword) {
            currentPage = page;
            let url = `/api/seata/transactions?page=${page}&size=20`;
            if (keyword) {
                url += `&keyword=${encodeURIComponent(keyword)}`;
            }
            
            $.ajax({
                url: url,
                type: 'GET',
                success: function(response) {
                    if (response.success) {
                        renderTransactions(response.data);
                    }
                }
            });
        }
        
        // 渲染事务列表
        function renderTransactions(transactions) {
            const tbody = $('#transactionList');
            tbody.empty();
            
            transactions.forEach(transaction => {
                const statusClass = getStatusClass(transaction.status);
                const duration = transaction.endTime ? 
                    transaction.endTime - transaction.beginTime : 
                    Date.now() - transaction.beginTime;
                
                const row = `
                    <tr class="transaction-card" onclick="showDetail('${transaction.xid}')">
                        <td>
                            <small class="text-muted">${transaction.xid.substring(0, 20)}...</small>
                        </td>
                        <td>${transaction.applicationId}</td>
                        <td>
                            <span class="status-badge ${statusClass}">
                                ${transaction.status}
                            </span>
                        </td>
                        <td>${formatTime(transaction.beginTime)}</td>
                        <td>${duration}ms</td>
                        <td>${transaction.branchCount || 0}</td>
                        <td>
                            <button class="btn btn-sm btn-outline-primary" 
                                    onclick="event.stopPropagation();commitTransaction('${transaction.xid}')">
                                <i class="fas fa-check"></i>
                            </button>
                            <button class="btn btn-sm btn-outline-danger ms-1" 
                                    onclick="event.stopPropagation();rollbackTransaction('${transaction.xid}')">
                                <i class="fas fa-times"></i>
                            </button>
                        </td>
                    </tr>
                `;
                tbody.append(row);
            });
            
            renderPagination();
        }
        
        // 显示事务详情
        function showDetail(xid) {
            $.ajax({
                url: `/api/seata/transactions/${xid}`,
                type: 'GET',
                success: function(response) {
                    if (response.success) {
                        renderTransactionDetail(response.data);
                        $('#detailModal').modal('show');
                    }
                }
            });
        }
        
        // 手动提交事务
        function commitTransaction(xid) {
            if (confirm('确定要手动提交此事务吗？')) {
                $.ajax({
                    url: `/api/seata/transactions/${xid}/commit`,
                    type: 'POST',
                    success: function(response) {
                        alert(response.message);
                        loadTransactions(currentPage);
                    }
                });
            }
        }
        
        // 手动回滚事务
        function rollbackTransaction(xid) {
            if (confirm('确定要手动回滚此事务吗？')) {
                $.ajax({
                    url: `/api/seata/transactions/${xid}/rollback`,
                    type: 'POST',
                    success: function(response) {
                        alert(response.message);
                        loadTransactions(currentPage);
                    }
                });
            }
        }
        
        // WebSocket连接
        function connectWebSocket() {
            const socket = new SockJS('/ws');
            const stompClient = Stomp.over(socket);
            
            stompClient.connect({}, function(frame) {
                console.log('Connected: ' + frame);
                
                // 订阅事务事件
                stompClient.subscribe('/topic/transactions', function(message) {
                    const event = JSON.parse(message.body);
                    handleTransactionEvent(event);
                });
            });
        }
        
        // 处理事务事件
        function handleTransactionEvent(event) {
            const toast = `
                <div class="toast show" role="alert" style="position: fixed; top: 20px; right: 20px; z-index: 9999;">
                    <div class="toast-header">
                        <strong class="me-auto">事务事件</strong>
                        <small>${new Date().toLocaleTimeString()}</small>
                        <button type="button" class="btn-close" data-bs-dismiss="toast"></button>
                    </div>
                    <div class="toast-body">
                        <strong>${event.xid.substring(0, 20)}...</strong><br>
                        状态: ${event.status}<br>
                        时间: ${formatTime(event.timestamp)}
                    </div>
                </div>
            `;
            
            $('body').append(toast);
            
            // 5秒后自动移除
            setTimeout(() => {
                $(toast).remove();
            }, 5000);
            
            // 刷新数据
            loadStatistics();
            loadTransactions(currentPage);
        }
        
        // 工具函数
        function getStatusClass(status) {
            switch(status) {
                case 'Committed': return 'status-committed';
                case 'Rollbacked': return 'status-rollbacked';
                case 'Begin': return 'status-begin';
                default: return '';
            }
        }
        
        function formatTime(timestamp) {
            return new Date(timestamp).toLocaleString();
        }
    </script>
</body>
```
