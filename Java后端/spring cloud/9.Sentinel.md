# Spring Cloud Alibaba Sentinel 完整指南

Sentinel 是阿里巴巴开源的**流量控制、熔断降级、系统负载保护**的分布式系统防护组件。以下是 Sentinel 的全面使用指南。

## 一、Sentinel 核心概念

### 1. **核心概念对比表**

|概念|作用|类比|
|---|---|---|
|**流量控制**​|控制 QPS/并发数，防止服务被突发流量冲垮|高速公路限流|
|**熔断降级**​|当服务不稳定时自动熔断，避免级联故障|电路保险丝|
|**系统负载保护**​|根据系统负载自动限流，保护系统|CPU过载保护|
|**热点参数限流**​|对热点参数单独限流|VIP通道限流|
|**黑白名单控制**​|限制或放行指定来源|访问控制列表|

## 三、流量控制

### 1. **QPS流量控制**

```
import com.alibaba.csp.sentinel.annotation.SentinelResource;
import com.alibaba.csp.sentinel.slots.block.BlockException;
import org.springframework.stereotype.Service;

@Service
public class UserService {
    
    /**
     * QPS流量控制：每秒最多100个请求
     */
    @SentinelResource(
        value = "getUserById", 
        blockHandler = "getUserByIdBlockHandler",  // 流控处理
        fallback = "getUserByIdFallback"          // 降级处理
    )
    public UserDTO getUserById(Long userId) {
        // 业务逻辑
        return userRepository.findById(userId)
            .orElseThrow(() -> new UserNotFoundException("用户不存在"));
    }
    
    /**
     * 流控处理函数（参数和返回类型需与原始方法一致，最后加BlockException参数）
     */
    public UserDTO getUserByIdBlockHandler(Long userId, BlockException ex) {
        // 记录流控日志
        log.warn("触发流控，userId: {}, rule: {}", userId, ex.getRule());
        
        // 返回友好提示
        return UserDTO.builder()
            .id(userId)
            .name("系统繁忙")
            .message("当前访问人数过多，请稍后重试")
            .build();
    }
    
    /**
     * 降级处理函数（参数和返回类型需与原始方法一致，最后加Throwable参数）
     */
    public UserDTO getUserByIdFallback(Long userId, Throwable t) {
        log.error("服务降级，userId: {}, error: {}", userId, t.getMessage());
        
        // 返回降级数据
        return UserDTO.builder()
            .id(userId)
            .name("默认用户")
            .email("default@example.com")
            .build();
    }
    
    /**
     * 并发线程数控制：最多10个线程同时执行
     */
    @SentinelResource(
        value = "batchProcessUsers",
        blockHandler = "batchProcessUsersBlockHandler"
    )
    public ApiResult batchProcessUsers(List<Long> userIds) {
        // 批量处理逻辑
        return processService.batchProcess(userIds);
    }
    
    /**
     * 关联流控：当关联资源getOrder达到阈值时，限制getUser
     */
    @SentinelResource(
        value = "getUserWithOrder",
        blockHandler = "getUserWithOrderBlockHandler"
    )
    public UserDTO getUserWithOrder(Long userId) {
        // 获取用户订单信息
        OrderDTO order = orderService.getOrderByUserId(userId);
        return userRepository.findById(userId)
            .map(user -> UserDTO.fromEntity(user, order))
            .orElse(null);
    }
}

import com.alibaba.csp.sentinel.slots.block.RuleConstant;
import com.alibaba.csp.sentinel.slots.block.flow.FlowRule;
import com.alibaba.csp.sentinel.slots.block.flow.FlowRuleManager;
import org.springframework.boot.CommandLineRunner;
import org.springframework.stereotype.Component;

import java.util.ArrayList;
import java.util.List;

@Component
public class SentinelRuleInitializer implements CommandLineRunner {
    
    @Override
    public void run(String... args) {
        initFlowRules();
        initDegradeRules();
        initSystemRules();
        initAuthorityRules();
    }
    
    /**
     * 初始化流控规则
     */
    private void initFlowRules() {
        List<FlowRule> rules = new ArrayList<>();
        
        // 规则1：用户查询接口QPS限流
        FlowRule userQueryRule = new FlowRule();
        userQueryRule.setResource("getUserById");
        userQueryRule.setGrade(RuleConstant.FLOW_GRADE_QPS);  // QPS限流模式
        userQueryRule.setCount(100);  // 阈值：100 QPS
        userQueryRule.setLimitApp("default");  // 默认应用
        userQueryRule.setStrategy(RuleConstant.STRATEGY_DIRECT);  // 直接限流
        userQueryRule.setControlBehavior(RuleConstant.CONTROL_BEHAVIOR_WARM_UP);  // 预热模式
        userQueryRule.setWarmUpPeriodSec(10);  // 预热时间10秒
        rules.add(userQueryRule);
        
        // 规则2：批量处理接口并发线程数限流
        FlowRule batchProcessRule = new FlowRule();
        batchProcessRule.setResource("batchProcessUsers");
        batchProcessRule.setGrade(RuleConstant.FLOW_GRADE_THREAD);  // 线程数模式
        batchProcessRule.setCount(10);  // 最大并发线程数10
        batchProcessRule.setLimitApp("default");
        rules.add(batchProcessRule);
        
        // 规则3：关联流控 - 当订单查询繁忙时，限制用户查询
        FlowRule relatedRule = new FlowRule();
        relatedRule.setResource("getUserWithOrder");
        relatedRule.setGrade(RuleConstant.FLOW_GRADE_QPS);
        relatedRule.setCount(50);
        relatedRule.setStrategy(RuleConstant.STRATEGY_RELATE);  // 关联限流
        relatedRule.setRefResource("getOrderByUserId");  // 关联资源
        rules.add(relatedRule);
        
        // 规则4：链路流控 - 特定入口限流
        FlowRule linkRule = new FlowRule();
        linkRule.setResource("createOrder");
        linkRule.setGrade(RuleConstant.FLOW_GRADE_QPS);
        linkRule.setCount(20);
        linkRule.setStrategy(RuleConstant.STRATEGY_CHAIN);  // 链路模式
        linkRule.setRefResource("/order/create");  // 入口资源
        rules.add(linkRule);
        
        FlowRuleManager.loadRules(rules);
        log.info("Sentinel流控规则初始化完成，共{}条规则", rules.size());
    }
}

## 四、熔断降级

### 1. **熔断降级规则**

```
import com.alibaba.csp.sentinel.slots.block.degrade.DegradeRule;
import com.alibaba.csp.sentinel.slots.block.degrade.DegradeRuleManager;
import com.alibaba.csp.sentinel.slots.block.degrade.circuitbreaker.CircuitBreakerStrategy;

@Component
public class SentinelDegradeInitializer {
    
    public void initDegradeRules() {
        List<DegradeRule> rules = new ArrayList<>();
        
        // 规则1：慢调用比例熔断
        DegradeRule slowCallRule = new DegradeRule();
        slowCallRule.setResource("queryUserInfo");
        slowCallRule.setGrade(CircuitBreakerStrategy.SLOW_REQUEST_RATIO.getType());
        slowCallRule.setCount(2000);  // RT阈值2000ms
        slowCallRule.setTimeWindow(10);  // 熔断时长10秒
        slowCallRule.setSlowRequestRatio(0.5);  // 慢调用比例阈值50%
        slowCallRule.setMinRequestAmount(5);  // 最小请求数5
        slowCallRule.setStatIntervalMs(10000);  // 统计窗口10秒
        rules.add(slowCallRule);
        
        // 规则2：异常比例熔断
        DegradeRule exceptionRatioRule = new DegradeRule();
        exceptionRatioRule.setResource("updateUser");
        exceptionRatioRule.setGrade(CircuitBreakerStrategy.ERROR_RATIO.getType());
        exceptionRatioRule.setCount(0.5);  // 异常比例阈值50%
        exceptionRatioRule.setTimeWindow(5);  // 熔断时长5秒
        exceptionRatioRule.setMinRequestAmount(10);  // 最小请求数10
        exceptionRatioRule.setStatIntervalMs(60000);  // 统计窗口60秒
        rules.add(exceptionRatioRule);
        
        // 规则3：异常数熔断
        DegradeRule exceptionCountRule = new DegradeRule();
        exceptionCountRule.setResource("deleteUser");
        exceptionCountRule.setGrade(CircuitBreakerStrategy.ERROR_COUNT.getType());
        exceptionCountRule.setCount(5);  // 异常数阈值5
        exceptionCountRule.setTimeWindow(10);  // 熔断时长10秒
        exceptionCountRule.setMinRequestAmount(20);  // 最小请求数20
        exceptionCountRule.setStatIntervalMs(60000);  // 统计窗口60秒
        rules.add(exceptionCountRule);
        
        DegradeRuleManager.loadRules(rules);
        log.info("Sentinel降级规则初始化完成，共{}条规则", rules.size());
    }
    
    /**
     * 手动熔断控制
     */
    @Service
    public class CircuitBreakerService {
        
        // 手动获取熔断器
        private CircuitBreaker getCircuitBreaker(String resource) {
            DegradeRule rule = DegradeRuleManager.getRules(resource)
                .stream()
                .findFirst()
                .orElse(null);
            
            if (rule != null) {
                return DegradeRuleManager.getCircuitBreakers(resource)
                    .stream()
                    .findFirst()
                    .orElse(null);
            }
            return null;
        }
        
        // 手动切换熔断器状态
        public void manualSwitch(String resource, CircuitBreaker.State state) {
            CircuitBreaker circuitBreaker = getCircuitBreaker(resource);
            if (circuitBreaker != null) {
                if (state == CircuitBreaker.State.OPEN) {
                    circuitBreaker.tryOpen();
                } else if (state == CircuitBreaker.State.CLOSED) {
                    circuitBreaker.tryClose();
                } else if (state == CircuitBreaker.State.HALF_OPEN) {
                    circuitBreaker.tryHalfOpen();
                }
            }
        }
        
        // 检查熔断器状态
        public CircuitBreaker.State getState(String resource) {
            CircuitBreaker circuitBreaker = getCircuitBreaker(resource);
            return circuitBreaker != null ? circuitBreaker.currentState() : null;
        }
    }
}

### 2. **熔断降级实战**

```
import com.alibaba.csp.sentinel.annotation.SentinelResource;
import com.alibaba.csp.sentinel.slots.block.degrade.DegradeException;
import org.springframework.stereotype.Service;

@Service
public class PaymentService {
    
    /**
     * 熔断降级示例
     */
    @SentinelResource(
        value = "processPayment",
        blockHandler = "processPaymentBlockHandler",
        fallback = "processPaymentFallback",
        exceptionsToIgnore = {IllegalArgumentException.class}  // 忽略的异常
    )
    public PaymentResult processPayment(PaymentRequest request) {
        // 1. 参数校验
        validatePaymentRequest(request);
        
        // 2. 调用支付服务（可能失败）
        PaymentResponse response = paymentClient.pay(request);
        
        // 3. 处理响应
        if (response.isSuccess()) {
            return PaymentResult.success(response.getTransactionId());
        } else {
            throw new PaymentException("支付失败: " + response.getErrorMsg());
        }
    }
    
    /**
     * 流控/熔断处理
     */
    public PaymentResult processPaymentBlockHandler(PaymentRequest request, BlockException ex) {
        log.warn("支付服务限流/熔断, requestId: {}, rule: {}", 
            request.getRequestId(), ex.getRule());
        
        // 记录到降级日志
        degradeLogger.logPaymentBlock(request, ex);
        
        // 返回降级结果
        return PaymentResult.degraded("支付系统繁忙，请稍后重试");
    }
    
    /**
     * 业务异常降级处理
     */
    public PaymentResult processPaymentFallback(PaymentRequest request, Throwable t) {
        log.error("支付服务降级, requestId: {}, error: {}", 
            request.getRequestId(), t.getMessage(), t);
        
        // 根据异常类型返回不同的降级策略
        if (t instanceof TimeoutException) {
            return PaymentResult.degraded("支付超时，请确认是否扣款成功");
        } else if (t instanceof PaymentException) {
            return PaymentResult.degraded("支付失败: " + t.getMessage());
        } else {
            return PaymentResult.degraded("系统异常，请稍后重试");
        }
    }
    
    /**
     * 多级降级策略
     */
    @SentinelResource(
        value = "processPaymentV2",
        blockHandler = "processPaymentBlockHandler",
        fallback = "processPaymentFallbackV2",
        fallbackClass = PaymentFallbackHandler.class  // 外部降级类
    )
    public PaymentResult processPaymentV2(PaymentRequest request) {
        // 根据支付金额选择不同策略
        if (request.getAmount().compareTo(new BigDecimal("1000")) < 0) {
            // 小额支付：快速处理
            return fastPayment(request);
        } else {
            // 大额支付：安全处理
            return securePayment(request);
        }
    }
}

/**
 * 外部降级处理器（支持静态方法）
 */
public class PaymentFallbackHandler {
    
    // 静态方法，参数要一致，最后加Throwable参数
    public static PaymentResult processPaymentFallbackV2(
        PaymentRequest request, Throwable t) {
        
        // 一级降级：尝试本地缓存
        PaymentResult cachedResult = getCachedPaymentResult(request);
        if (cachedResult != null) {
            return cachedResult;
        }
        
        // 二级降级：返回兜底数据
        return PaymentResult.builder()
            .status(PaymentStatus.PENDING)
            .message("支付处理中，请稍后查询结果")
            .suggestRetryTime(System.currentTimeMillis() + 300000)  // 5分钟后重试
            .build();
    }
    
    // 三级降级：异步补偿
    public static void asyncCompensation(PaymentRequest request) {
        CompletableFuture.runAsync(() -> {
            try {
                // 延迟重试
                Thread.sleep(60000);
                retryPayment(request);
            } catch (Exception e) {
                log.error("支付补偿失败", e);
            }
        });
    }
}

import com.alibaba.csp.sentinel.slots.block.flow.param.ParamFlowRule;
import com.alibaba.csp.sentinel.slots.block.flow.param.ParamFlowRuleManager;

@Component
public class HotParamRuleInitializer {
    
    public void initHotParamRules() {
        List<ParamFlowRule> rules = new ArrayList<>();
        
        // 规则1：商品详情接口，针对不同商品ID进行限流
        ParamFlowRule productRule = new ParamFlowRule("getProductDetail")
            .setParamIdx(0)  // 参数索引，0表示第一个参数
            .setGrade(RuleConstant.FLOW_GRADE_QPS)
            .setCount(50);  // 热点参数阈值
        
        // 设置特定参数的独立限流值
        Map<Object, Integer> hotParamMap = new HashMap<>();
        hotParamMap.put(1001L, 100);  // 热门商品ID 1001，限流100 QPS
        hotParamMap.put(1002L, 200);  // 热门商品ID 1002，限流200 QPS
        productRule.setParamFlowItemList(hotParamMap);
        
        // 参数例外项配置
        ParamFlowItem item1 = new ParamFlowItem()
            .setObject(String.valueOf(1001L))
            .setClassType(Long.class.getName())
            .setCount(100);
        ParamFlowItem item2 = new ParamFlowItem()
            .setObject(String.valueOf(1002L))
            .setClassType(Long.class.getName())
            .setCount(200);
        productRule.setParamFlowItemList(Arrays.asList(item1, item2));
        
        rules.add(productRule);
        
        // 规则2：用户查询接口，针对用户ID和IP组合限流
        ParamFlowRule userRule = new ParamFlowRule("queryUserInfo")
            .setParamIdx(0, 1)  // 多个参数：用户ID和IP
            .setGrade(RuleConstant.FLOW_GRADE_QPS)
            .setCount(10)  // 默认阈值
            .setDurationInSec(1)  // 统计窗口
            .setControlBehavior(RuleConstant.CONTROL_BEHAVIOR_RATE_LIMITER)  // 匀速排队
            .setMaxQueueingTimeMs(500)  // 最大排队时间
            .setBurstCount(20);  // 突发流量额外配额
        
        rules.add(userRule);
        
        ParamFlowRuleManager.loadRules(rules);
        log.info("热点参数规则初始化完成，共{}条规则", rules.size());
    }
}

### 1. **系统规则配置**

```
import com.alibaba.csp.sentinel.slots.system.SystemRule;
import com.alibaba.csp.sentinel.slots.system.SystemRuleManager;

@Component
public class SystemRuleInitializer {
    
    public void initSystemRules() {
        List<SystemRule> rules = new ArrayList<>();
        
        // 规则1：系统负载保护
        SystemRule loadRule = new SystemRule();
        loadRule.setHighestSystemLoad(4.0);  // 系统最大负载4.0
        loadRule.setAvgRt(500);  // 所有入口流量平均RT阈值500ms
        loadRule.setMaxThread(500);  // 全局最大并发线程数500
        loadRule.setQps(1000.0);  // 全局入口QPS阈值1000
        loadRule.setHighestCpuUsage(0.8);  // CPU使用率阈值80%
        rules.add(loadRule);
        
        // 规则2：入口QPS保护
        SystemRule ingressQpsRule = new SystemRule();
        ingressQpsRule.setQps(2000.0);  // 入口QPS阈值2000
        ingressQpsRule.setResourceType(1);  // INGRESS类型
        rules.add(ingressQpsRule);
        
        // 规则3：全局并发保护
        SystemRule threadRule = new SystemRule();
        threadRule.setMaxThread(1000);  // 全局最大并发1000
        threadRule.setResourceType(0);  // COMMON类型
        rules.add(threadRule);
        
        SystemRuleManager.loadRules(rules);
        log.info("系统保护规则初始化完成，共{}条规则", rules.size());
    }
    
    /**
     * 动态调整系统规则
     */
    public void adjustSystemRulesBasedOnMetrics() {
        // 获取当前系统指标
        double systemLoad = OperatingSystemMXBean.getInstance().getSystemLoadAverage();
        long usedMemory = Runtime.getRuntime().totalMemory() - Runtime.getRuntime().freeMemory();
        long maxMemory = Runtime.getRuntime().maxMemory();
        double memoryUsage = (double) usedMemory / maxMemory;
        
        // 根据系统负载动态调整
        List<SystemRule> rules = SystemRuleManager.getRules();
        if (!rules.isEmpty()) {
            SystemRule rule = rules.get(0);
            
            if (systemLoad > 3.0) {
                // 高负载，降低QPS阈值
                rule.setQps(Math.max(500.0, rule.getQps() * 0.8));
                rule.setMaxThread((int) Math.max(200, rule.getMaxThread() * 0.8));
            } else if (systemLoad < 1.0) {
                // 低负载，提高QPS阈值
                rule.setQps(Math.min(5000.0, rule.getQps() * 1.2));
                rule.setMaxThread((int) Math.min(2000, rule.getMaxThread() * 1.2));
            }
            
            // 根据内存使用率调整
            if (memoryUsage > 0.8) {
                rule.setMaxThread((int) Math.max(100, rule.getMaxThread() * 0.7));
            }
            
            SystemRuleManager.loadRules(rules);
            log.info("系统规则已调整: QPS={}, MaxThread={}", rule.getQps(), rule.getMaxThread());
        }
    }

            // 1. 从请求头获取来源
            String origin = request.getHeader("X-Forwarded-For");
            if (StringUtils.isEmpty(origin)) {
                origin = request.getRemoteAddr();
            }
            
            // 2. 从Token中获取用户信息
            String token = request.getHeader("Authorization");
            if (StringUtils.hasText(token)) {
                String userId = extractUserIdFromToken(token);
                if (userId != null) {
                    // 组合来源：IP + 用户ID
                    return origin + ":" + userId;
                }
            }
            
            return origin;
        }

                // JWT解析或其他Token解析逻辑

            } catch (Exception e) {
                return null;
            }
        }

                Map<String, Object> response = new HashMap<>();
                response.put("code", 429);
                response.put("message", "请求过于频繁，请稍后重试");

                response.put("path", exchange.getRequest().getPath().value());

                    response.put("code", 430);
                    response.put("message", "热点参数限流");

                    response.put("code", 503);
                    response.put("message", "服务暂时不可用");

                    response.put("code", 403);
                    response.put("message", "访问被拒绝");
                } else if (t instanceof SystemBlockException) {
                    response.put("code", 429);
                    response.put("message", "系统保护模式");
                }
                
                return ServerResponse.status(HttpStatus.TOO_MANY_REQUESTS)
                    .contentType(MediaType.APPLICATION_JSON)
                    .body(BodyInserters.fromValue(response));
            }
        });

        });
        
        // 设置请求属性提取器（用于提取请求属性作为参数）
        GatewayCallbackManager.setRequestOriginParser(exchange -> {
            // 提取请求来源，用于黑白名单控制
            String origin = exchange.getRequest().getHeaders().getFirst("X-Forwarded-For");
            if (StringUtils.isEmpty(origin)) {
                origin = exchange.getRequest().getRemoteAddress().getAddress().getHostAddress();
            }
            return origin;
        });
    }
