# 请求限流全方位解决方案

限流是保护系统的第一道防线，用于控制请求流量，防止系统被突发流量冲垮。以下是完整的请求限流解决方案。

## 一、限流算法原理

### 1. **四大限流算法对比**

|算法|原理|优点|缺点|适用场景|
|---|---|---|---|---|
|**计数器算法**​|固定时间窗口计数|简单高效|临界点突发流量|简单限流场景|
|**滑动窗口**​|多个子窗口滑动统计|平滑限流|实现复杂|高精度限流|
|**漏桶算法**​|固定速率处理请求|平滑输出|无法应对突发|流量整形|
|**令牌桶算法**​|按速率生成令牌|允许突发|需要预热|主流应用|
|**自适应限流**​|动态调整阈值|智能控制|实现复杂|弹性系统|

### 1. **Guava RateLimiter**

```
import com.google.common.util.concurrent.RateLimiter;
import org.springframework.stereotype.Component;
import java.util.concurrent.ConcurrentHashMap;
import java.util.concurrent.TimeUnit;

@Component
@Slf4j
public class GuavaRateLimiter {
    
    // 存储不同资源的限流器
    private final ConcurrentHashMap<String, RateLimiter> limiters = 
        new ConcurrentHashMap<>();
    
    /**
     * 创建或获取限流器
     */
    public RateLimiter getRateLimiter(String resource, double permitsPerSecond) {
        return limiters.computeIfAbsent(resource, key -> 
            RateLimiter.create(permitsPerSecond)
        );
    }
    
    /**
     * 尝试获取令牌（非阻塞）
     */
    public boolean tryAcquire(String resource, double permitsPerSecond) {
        RateLimiter limiter = getRateLimiter(resource, permitsPerSecond);
        return limiter.tryAcquire();
    }
    
    /**
     * 等待获取令牌（阻塞）
     */
    public void acquire(String resource, double permitsPerSecond) {
        RateLimiter limiter = getRateLimiter(resource, permitsPerSecond);
        limiter.acquire();
    }
    
    /**
     * 带超时的尝试获取
     */
    public boolean tryAcquireWithTimeout(String resource, double permitsPerSecond, 
                                        long timeout, TimeUnit unit) {
        RateLimiter limiter = getRateLimiter(resource, permitsPerSecond);
        return limiter.tryAcquire(timeout, unit);
    }
    
    /**
     * 预热限流器
     */
    public RateLimiter createWarmupLimiter(String resource, double permitsPerSecond, 
                                          int warmupPeriod, TimeUnit unit) {
        return limiters.computeIfAbsent(resource, key -> 
            RateLimiter.create(permitsPerSecond, 
                unit.toSeconds(warmupPeriod), 
                TimeUnit.SECONDS)
        );
    }
    
    /**
     * 动态调整限流速率
     */
    public void setRate(String resource, double newPermitsPerSecond) {
        RateLimiter limiter = limiters.get(resource);
        if (limiter != null) {
            limiter.setRate(newPermitsPerSecond);
            log.info("限流器速率已调整: {} -> {}", resource, newPermitsPerSecond);
        }
    }
    
    /**
     * 获取限流器状态
     */
    public LimiterStatus getLimiterStatus(String resource) {
        RateLimiter limiter = limiters.get(resource);
        if (limiter != null) {
            return LimiterStatus.builder()
                .rate(limiter.getRate())
                .availablePermits(limiter.acquire(0)) // 非阻塞获取
                .build();
        }
        return null;
    }
}

### 2. **滑动窗口限流器**

```
@Component
@Slf4j
public class SlidingWindowRateLimiter {
    
    // 滑动窗口数据结构
    private static class SlidingWindow {
        private final int windowSize;  // 窗口大小（秒）
        private final int[] counters;  // 计数器数组
        private final long[] timestamps;  // 时间戳数组
        private volatile int index = 0;  // 当前索引
        
        public SlidingWindow(int windowSize) {
            this.windowSize = windowSize;
            this.counters = new int[windowSize];
            this.timestamps = new long[windowSize];
        }
        
        public synchronized boolean allow(int maxRequests) {
            long currentTime = System.currentTimeMillis() / 1000;  // 当前秒
            int currentIndex = (int) (currentTime % windowSize);
            
            // 如果时间窗口已移动，重置计数器
            if (timestamps[currentIndex] != currentTime) {
                counters[currentIndex] = 0;
                timestamps[currentIndex] = currentTime;
            }
            
            // 计算滑动窗口内的总请求数
            int totalRequests = 0;
            for (int i = 0; i < windowSize; i++) {
                if (currentTime - timestamps[i] < windowSize) {
                    totalRequests += counters[i];
                }
            }
            
            // 检查是否超过限制
            if (totalRequests < maxRequests) {
                counters[currentIndex]++;
                return true;
            }
            
            return false;
        }
    }
    
    // 资源-限流器映射
    private final ConcurrentHashMap<String, SlidingWindow> windows = 
        new ConcurrentHashMap<>();
    
    /**
     * 允许请求
     */
    public boolean allowRequest(String resource, int maxRequestsPerWindow, 
                               int windowSizeInSeconds) {
        SlidingWindow window = windows.computeIfAbsent(resource, 
            key -> new SlidingWindow(windowSizeInSeconds));
        
        return window.allow(maxRequestsPerWindow);
    }
    
    /**
     * 带权重的请求
     */
    public boolean allowRequest(String resource, int weight, 
                               int maxRequestsPerWindow, int windowSizeInSeconds) {
        if (weight <= 0) {
            return true;
        }
        
        // 获取当前窗口
        SlidingWindow window = windows.get(resource);
        if (window == null) {
            window = new SlidingWindow(windowSizeInSeconds);
            windows.put(resource, window);
        }
        
        // 模拟权重请求
        synchronized (window) {
            for (int i = 0; i < weight; i++) {
                if (!window.allow(maxRequestsPerWindow)) {
                    return false;
                }
            }
            return true;
        }
    }
}

### 1. **Redis + Lua 分布式限流**

```
@Component
@Slf4j
public class RedisRateLimiter {
    
    @Autowired
    private StringRedisTemplate redisTemplate;
    
    /**
     * 固定窗口限流
     */
    public boolean fixedWindowRateLimit(String key, int limit, int windowSeconds) {
        String redisKey = "rate:limit:fixed:" + key;
        
        // 使用Lua脚本保证原子性
        String luaScript = """
            local key = KEYS[1]
            local limit = tonumber(ARGV[1])
            local window = tonumber(ARGV[2])
            
            local current = redis.call('GET', key)
            if current == false then
                redis.call('SET', key, 1, 'EX', window)
                return 1
            end
            
            if tonumber(current) < limit then
                redis.call('INCR', key)
                return 1
            end
            
            return 0
            """;
        
        DefaultRedisScript<Long> script = new DefaultRedisScript<>();
        script.setScriptText(luaScript);
        script.setResultType(Long.class);
        
        Long result = redisTemplate.execute(
            script, 
            Collections.singletonList(redisKey), 
            String.valueOf(limit), 
            String.valueOf(windowSeconds)
        );
        
        return result != null && result == 1;
    }
    
    /**
     * 滑动窗口限流
     */
    public boolean slidingWindowRateLimit(String key, int limit, int windowSeconds) {
        String redisKey = "rate:limit:sliding:" + key;
        long currentTime = System.currentTimeMillis();
        
        String luaScript = """
            local key = KEYS[1]
            local limit = tonumber(ARGV[1])
            local window = tonumber(ARGV[2])
            local currentTime = tonumber(ARGV[3])
            
            -- 移除窗口外的数据
            local clearBefore = currentTime - window * 1000
            redis.call('ZREMRANGEBYSCORE', key, 0, clearBefore)
            
            -- 获取当前窗口内请求数
            local currentCount = redis.call('ZCOUNT', key, clearBefore, currentTime)
            
            if currentCount < limit then
                -- 添加当前请求
                redis.call('ZADD', key, currentTime, currentTime)
                redis.call('EXPIRE', key, window)
                return 1
            end
            
            return 0
            """;
        
        DefaultRedisScript<Long> script = new DefaultRedisScript<>();
        script.setScriptText(luaScript);
        script.setResultType(Long.class);
        
        Long result = redisTemplate.execute(
            script,
            Collections.singletonList(redisKey),
            String.valueOf(limit),
            String.valueOf(windowSeconds),
            String.valueOf(currentTime)
        );
        
        return result != null && result == 1;
    }
    
    /**
     * 令牌桶限流
     */
    public boolean tokenBucketRateLimit(String key, int capacity, 
                                       int refillRate, int refillInterval) {
        String redisKey = "rate:limit:token:" + key;
        long currentTime = System.currentTimeMillis();
        
        String luaScript = """
            local key = KEYS[1]
            local capacity = tonumber(ARGV[1])
            local refillRate = tonumber(ARGV[2])
            local refillInterval = tonumber(ARGV[3])
            local currentTime = tonumber(ARGV[4])
            local tokensRequested = tonumber(ARGV[5]) or 1
            
            local bucket = redis.call('HMGET', key, 'tokens', 'lastRefillTime')
            local tokens = 0
            local lastRefillTime = currentTime
            
            if bucket[1] ~= false then
                tokens = tonumber(bucket[1])
                lastRefillTime = tonumber(bucket[2])
            end
            
            -- 计算补充的令牌
            local timePassed = currentTime - lastRefillTime
            local refillCount = math.floor(timePassed / refillInterval) * refillRate
            
            if refillCount > 0 then
                tokens = math.min(capacity, tokens + refillCount)
                lastRefillTime = currentTime
            end
            
            -- 检查是否有足够令牌
            if tokens >= tokensRequested then
                tokens = tokens - tokensRequested
                redis.call('HMSET', key, 
                    'tokens', tokens, 
                    'lastRefillTime', lastRefillTime)
                redis.call('EXPIRE', key, math.floor(capacity / refillRate) * 2)
                return 1
            end
            
            return 0
            """;
        
        DefaultRedisScript<Long> script = new DefaultRedisScript<>();
        script.setScriptText(luaScript);
        script.setResultType(Long.class);
        
        Long result = redisTemplate.execute(
            script,
            Collections.singletonList(redisKey),
            String.valueOf(capacity),
            String.valueOf(refillRate),
            String.valueOf(refillInterval),
            String.valueOf(currentTime),
            "1"
        );
        
        return result != null && result == 1;
    }
    
    /**
     * 漏斗限流
     */
    public boolean leakyBucketRateLimit(String key, int capacity, 
                                       int leakRate, int leakInterval) {
        String redisKey = "rate:limit:leaky:" + key;
        long currentTime = System.currentTimeMillis();
        
        String luaScript = """
            local key = KEYS[1]
            local capacity = tonumber(ARGV[1])
            local leakRate = tonumber(ARGV[2])
            local leakInterval = tonumber(ARGV[3])
            local currentTime = tonumber(ARGV[4])
            
            local bucket = redis.call('HMGET', key, 'water', 'lastLeakTime')
            local water = 0
            local lastLeakTime = currentTime
            
            if bucket[1] ~= false then
                water = tonumber(bucket[1])
                lastLeakTime = tonumber(bucket[2])
            end
            
            -- 计算漏出的水量
            local timePassed = currentTime - lastLeakTime
            local leaked = math.floor(timePassed / leakInterval) * leakRate
            
            if leaked > 0 then
                water = math.max(0, water - leaked)
                lastLeakTime = currentTime
            end
            
            -- 检查是否可加水
            if water < capacity then
                water = water + 1
                redis.call('HMSET', key, 
                    'water', water, 
                    'lastLeakTime', lastLeakTime)
                redis.call('EXPIRE', key, math.floor(capacity / leakRate) * 2)
                return 1
            end
            
            return 0
            """;
        
        DefaultRedisScript<Long> script = new DefaultRedisScript<>();
        script.setScriptText(luaScript);
        script.setResultType(Long.class);
        
        Long result = redisTemplate.execute(
            script,
            Collections.singletonList(redisKey),
            String.valueOf(capacity),
            String.valueOf(leakRate),
            String.valueOf(leakInterval),
            String.valueOf(currentTime)
        );
        
        return result != null && result == 1;
    }
}

### 2. **Redisson 分布式限流**

```
@Component
@Slf4j
public class RedissonRateLimiter {
    
    @Autowired
    private RedissonClient redissonClient;
    
    // 限流器缓存
    private final ConcurrentHashMap<String, RRateLimiter> limiterCache = 
        new ConcurrentHashMap<>();
    
    /**
     * 获取限流器
     */
    public RRateLimiter getRateLimiter(String key, int rate, int rateInterval, 
                                      RateIntervalUnit unit) {
        String limiterKey = "rate:limiter:" + key;
        
        return limiterCache.computeIfAbsent(limiterKey, k -> {
            RRateLimiter limiter = redissonClient.getRateLimiter(limiterKey);
            
            // 初始化限流配置
            limiter.trySetRate(RateType.OVERALL, rate, rateInterval, unit);
            
            return limiter;
        });
    }
    
    /**
     * 尝试获取许可
     */
    public boolean tryAcquire(String key, int permits) {
        RRateLimiter limiter = getRateLimiter(key, 100, 1, RateIntervalUnit.SECONDS);
        return limiter.tryAcquire(permits);
    }
    
    /**
     * 带超时的尝试获取
     */
    public boolean tryAcquire(String key, int permits, long timeout, TimeUnit unit) {
        RRateLimiter limiter = getRateLimiter(key, 100, 1, RateIntervalUnit.SECONDS);
        return limiter.tryAcquire(permits, timeout, unit);
    }
    
    /**
     * 获取限流器统计信息
     */
    public RateLimiterStats getStats(String key) {
        RRateLimiter limiter = getRateLimiter(key, 100, 1, RateIntervalUnit.SECONDS);
        
        return RateLimiterStats.builder()
            .availablePermits(limiter.availablePermits())
            .tryAcquireCount(limiter.tryAcquireCount())
            .build();
    }
    
    /**
     * 集群限流 - 所有节点共享总限额
     */
    public boolean clusterRateLimit(String resource, int totalLimit, int windowSeconds) {
        String key = "cluster:rate:limit:" + resource;
        
        // 使用Redisson的RAtomicLong
        RAtomicLong counter = redissonClient.getAtomicLong(key);
        
        // 第一次设置过期时间
        if (counter.isExists()) {
            counter.expire(windowSeconds, TimeUnit.SECONDS);
        }
        
        // 递增并检查
        long count = counter.incrementAndGet();
        if (count == 1) {
            counter.expire(windowSeconds, TimeUnit.SECONDS);
        }
        
        return count <= totalLimit;
    }
}

### 2. **自定义限流键解析器**

```
@Configuration
public class RateLimitKeyResolverConfig {
    
    /**
     * IP限流键解析器
     */
    @Bean
    @Primary
    public KeyResolver ipKeyResolver() {
        return exchange -> {
            // 获取真实IP（考虑代理）
            String ip = getClientIp(exchange.getRequest());
            String path = exchange.getRequest().getPath().value();
            
            // 组合键：IP + 路径
            String key = ip + ":" + path;
            log.debug("IP限流键: {}", key);
            
            return Mono.just(key);
        };
    }
    
    /**
     * 用户限流键解析器
     */
    @Bean
    public KeyResolver userKeyResolver() {
        return exchange -> {
            // 从Header获取用户ID
            String userId = exchange.getRequest().getHeaders()
                .getFirst("X-User-Id");
            
            if (StringUtils.isEmpty(userId)) {
                // 从Token解析
                String token = exchange.getRequest().getHeaders()
                    .getFirst("Authorization");
                if (StringUtils.hasText(token) && token.startsWith("Bearer ")) {
                    userId = extractUserIdFromToken(token.substring(7));
                }
            }
            
            if (StringUtils.isEmpty(userId)) {
                userId = "anonymous";
            }
            
            String path = exchange.getRequest().getPath().value();
            String key = userId + ":" + path;
            
            log.debug("用户限流键: {}", key);
            return Mono.just(key);
        };
    }
    
    /**
     * API Key限流键解析器
     */
    @Bean
    public KeyResolver apiKeyResolver() {
        return exchange -> {
            // 从Header获取API Key
            String apiKey = exchange.getRequest().getHeaders()
                .getFirst("X-API-Key");
            
            // 从Query参数获取
            if (StringUtils.isEmpty(apiKey)) {
                apiKey = exchange.getRequest().getQueryParams()
                    .getFirst("api_key");
            }
            
            if (StringUtils.isEmpty(apiKey)) {
                apiKey = "default";
            }
            
            String path = exchange.getRequest().getPath().value();
            String key = apiKey + ":" + path;
            
            log.debug("API Key限流键: {}", key);
            return Mono.just(key);
        };
    }
    
    /**
     * 全局限流键解析器
     */
    @Bean
    public KeyResolver globalKeyResolver() {
        return exchange -> {
            // 使用固定键，实现全局限流
            return Mono.just("global");
        };
    }
    
    /**
     * 组合限流键解析器
     */
    @Bean
    public KeyResolver compositeKeyResolver() {
        return exchange -> {
            StringBuilder keyBuilder = new StringBuilder();
            
            // 添加IP
            String ip = getClientIp(exchange.getRequest());
            keyBuilder.append(ip).append(":");
            
            // 添加用户ID
            String userId = exchange.getRequest().getHeaders()
                .getFirst("X-User-Id");
            if (StringUtils.hasText(userId)) {
                keyBuilder.append(userId).append(":");
            } else {
                keyBuilder.append("anonymous:");
            }
            
            // 添加路径
            String path = exchange.getRequest().getPath().value();
            keyBuilder.append(path).append(":");
            
            // 添加方法
            String method = exchange.getRequest().getMethodValue();
            keyBuilder.append(method);
            
            String key = keyBuilder.toString();
            log.debug("组合限流键: {}", key);
            
            return Mono.just(key);
        };
    }
    
    /**
     * 获取客户端真实IP
     */
    private String getClientIp(ServerHttpRequest request) {
        // 检查代理头
        String xff = request.getHeaders().getFirst("X-Forwarded-For");
        if (StringUtils.hasText(xff)) {
            return xff.split(",")[0].trim();
        }
        
        String xri = request.getHeaders().getFirst("X-Real-IP");
        if (StringUtils.hasText(xri)) {
            return xri;
        }
        
        // 获取远程地址
        if (request.getRemoteAddress() != null) {
            return request.getRemoteAddress().getAddress().getHostAddress();
        }
        
        return "unknown";
    }
}

### 3. **自定义限流过滤器**

```
@Component
@Slf4j
public class CustomRateLimitFilter implements GlobalFilter, Ordered {
    
    @Autowired
    private RedisRateLimiter redisRateLimiter;
    
    // 限流配置
    private final Map<String, RateLimitConfig> limitConfigs = new HashMap<>();
    
    public CustomRateLimitFilter() {
        // 初始化限流配置
        initRateLimitConfigs();
    }
    
    private void initRateLimitConfigs() {
        // 按路径配置
        limitConfigs.put("/api/users/", new RateLimitConfig(100, 60));  // 100/分钟
        limitConfigs.put("/api/orders/", new RateLimitConfig(50, 60));  // 50/分钟
        limitConfigs.put("/api/products/", new RateLimitConfig(200, 60)); // 200/分钟
        limitConfigs.put("/admin/", new RateLimitConfig(10, 60));       // 10/分钟
    }
    
    @Override
    public Mono<Void> filter(ServerWebExchange exchange, GatewayFilterChain chain) {
        ServerHttpRequest request = exchange.getRequest();
        String path = request.getPath().value();
        String method = request.getMethodValue();
        
        // 查找匹配的限流配置
        RateLimitConfig config = findMatchingConfig(path);
        if (config == null) {
            return chain.filter(exchange);
        }
        
        // 生成限流键
        String rateLimitKey = generateRateLimitKey(exchange, config);
        
        // 执行限流检查
        boolean allowed = checkRateLimit(rateLimitKey, config);
        
        if (!allowed) {
            log.warn("请求被限流: {} {}, key: {}", method, path, rateLimitKey);
            return handleRateLimitExceeded(exchange, config);
        }
        
        // 添加限流头部信息
        ServerHttpResponse response = exchange.getResponse();
        response.getHeaders().add("X-RateLimit-Limit", 
            String.valueOf(config.getLimit()));
        response.getHeaders().add("X-RateLimit-Remaining", 
            String.valueOf(getRemaining(rateLimitKey, config)));
        response.getHeaders().add("X-RateLimit-Reset", 
            String.valueOf(getResetTime(rateLimitKey, config)));
        
        return chain.filter(exchange);
    }
    
    @Override
    public int getOrder() {
        return -1;  // 高优先级
    }
    
    /**
     * 检查限流
     */
    private boolean checkRateLimit(String key, RateLimitConfig config) {
        // 使用Redis滑动窗口限流
        return redisRateLimiter.slidingWindowRateLimit(
            key, 
            config.getLimit(), 
            config.getWindowSeconds()
        );
    }
    
    /**
     * 生成限流键
     */
    private String generateRateLimitKey(ServerWebExchange exchange, 
                                       RateLimitConfig config) {
        ServerHttpRequest request = exchange.getRequest();
        String path = request.getPath().value();
        
        StringBuilder keyBuilder = new StringBuilder("gateway:");
        
        // 根据配置决定限流维度
        switch (config.getDimension()) {
            case IP:
                String ip = getClientIp(request);
                keyBuilder.append("ip:").append(ip).append(":").append(path);
                break;
            case USER:
                String userId = getUserId(exchange);
                keyBuilder.append("user:").append(userId).append(":").append(path);
                break;
            case API_KEY:
                String apiKey = getApiKey(exchange);
                keyBuilder.append("apikey:").append(apiKey).append(":").append(path);
                break;
            case GLOBAL:
                keyBuilder.append("global:").append(path);
                break;
            default:
                keyBuilder.append("default:").append(path);
        }
        
        return keyBuilder.toString();
    }
    
    /**
     * 处理限流拒绝
     */
    private Mono<Void> handleRateLimitExceeded(ServerWebExchange exchange, 
                                              RateLimitConfig config) {
        ServerHttpResponse response = exchange.getResponse();
        response.setStatusCode(HttpStatus.TOO_MANY_REQUESTS);
        response.getHeaders().setContentType(MediaType.APPLICATION_JSON);
        
        Map<String, Object> errorResponse = new HashMap<>();
        errorResponse.put("code", 429);
        errorResponse.put("message", "请求过于频繁，请稍后重试");
        errorResponse.put("limit", config.getLimit());
        errorResponse.put("window", config.getWindowSeconds());
        errorResponse.put("retry_after", config.getWindowSeconds());
        
        String body = JSON.toJSONString(errorResponse);
        DataBuffer buffer = response.bufferFactory().wrap(body.getBytes(StandardCharsets.UTF_8));
        
        return response.writeWith(Mono.just(buffer));
    }
    
    @Data
    @Builder
    public static class RateLimitConfig {
        private int limit;           // 限制数量
        private int windowSeconds;   // 时间窗口（秒）
        private RateLimitDimension dimension = RateLimitDimension.IP;  // 限流维度
        
        public RateLimitConfig(int limit, int windowSeconds) {
            this.limit = limit;
            this.windowSeconds = windowSeconds;
        }
    }
    
    public enum RateLimitDimension {
        IP,      // 按IP限流
        USER,    // 按用户限流
        API_KEY, // 按API Key限流
        GLOBAL   // 全局限流
    }

public class MultiLevelRateLimiter {

        List<RateLimitLevel> orderServiceLevels = Arrays.asList(
            new RateLimitLevel("global", 500, 1, RateLimitType.GLOBAL),

        );

    }

     * 多级限流检查
     */
    public RateLimitResult checkMultiLevelRateLimit(ServerWebExchange exchange) {
        String path = exchange.getRequest().getPath().value();
        
        // 查找匹配的限流配置
        List<RateLimitLevel> levels = findMatchingLevels(path);
        if (levels == null || levels.isEmpty()) {
            return RateLimitResult.allowed();
        }
        
        // 按顺序检查各级限流
        for (RateLimitLevel level : levels) {
            RateLimitResult result = checkRateLimitLevel(exchange, level);
            if (!result.isAllowed()) {
                log.warn("限流触发: level={}, path={}, message={}", 
                    level.getName(), path, result.getMessage());
                return result;
            }
        }
        
        return RateLimitResult.allowed();
    }
    
    /**
     * 检查单个限流层级
     */
    private RateLimitResult checkRateLimitLevel(ServerWebExchange exchange, 
                                               RateLimitLevel level) {
        String key = generateLevelKey(exchange, level);
        
        boolean allowed;
        switch (level.getType()) {
            case LOCAL:
                // 本地限流
                allowed = guavaRateLimiter.tryAcquire(key, level.getLimit());
                break;
            case GLOBAL:
            case IP:
            case USER:
            case API:
                // 分布式限流
                allowed = redisRateLimiter.fixedWindowRateLimit(
                    key, level.getLimit(), level.getWindowSeconds());
                break;
            default:
                allowed = true;
        }
        
        if (!allowed) {
            return RateLimitResult.denied(
                String.format("%s限流触发: 限制 %d/%d秒", 
                    level.getName(), level.getLimit(), level.getWindowSeconds())
            );
        }
        
        return RateLimitResult.allowed();
    }
    
    /**
     * 生成层级限流键
     */
    private String generateLevelKey(ServerWebExchange exchange, 
                                   RateLimitLevel level) {
        ServerHttpRequest request = exchange.getRequest();
        String path = request.getPath().value();
        
        StringBuilder keyBuilder = new StringBuilder("ratelimit:");
        keyBuilder.append(level.getType().name().toLowerCase()).append(":");
        
        switch (level.getType()) {
            case GLOBAL:
                keyBuilder.append("global:");
                break;
            case IP:
                String ip = getClientIp(request);
                keyBuilder.append("ip:").append(ip).append(":");
                break;
            case USER:
                String userId = getUserId(exchange);
                keyBuilder.append("user:").append(userId).append(":");
                break;
            case API:
                String apiKey = getApiKey(exchange);
                keyBuilder.append("api:").append(apiKey).append(":");
                break;
        }
        
        keyBuilder.append(path);
        return keyBuilder.toString();
    }

    public static class RateLimitLevel {
        private String name;           // 层级名称
        private int limit;             // 限制数量
        private int windowSeconds;     // 时间窗口
        private RateLimitType type;    // 限流类型
    }
    
    public enum RateLimitType {

        GLOBAL,  // 全局限流
        IP,      // IP限流
        USER,    // 用户限流
        API      // API限流
    }
    
    @Data
    @Builder
    public static class RateLimitResult {
        private boolean allowed;
        private String message;
        private String level;
        
        public static RateLimitResult allowed() {
            return RateLimitResult.builder()
                .allowed(true)
                .build();
        }
        
        public static RateLimitResult denied(String message) {
            return RateLimitResult.builder()
                .allowed(false)
                .message(message)
                .build();
        }
    }

@Component
@Slf4j
public class AdaptiveRateLimiter {
    
    // 当前限流阈值
    private volatile double currentRate = 1000.0;
    
    // 配置参数
    private final double minRate = 100.0;
    private final double maxRate = 10000.0;
    private final double incrementFactor = 1.2;
    private final double decrementFactor = 0.8;
    
    // 监控指标
    private final Queue<Double> responseTimeQueue = new LinkedList<>();
    private final Queue<Double> errorRateQueue = new LinkedList<>();
    private final int windowSize = 60;  // 60个采样点
    
    /**
     * 定期调整限流阈值
     */
    @Scheduled(fixedRate = 10000)  // 每10秒调整一次
    public void adjustRateLimit() {
        // 获取系统指标
        SystemMetrics metrics = collectSystemMetrics();
        
        // 计算调整因子
        double adjustmentFactor = calculateAdjustmentFactor(metrics);
        
        // 调整限流阈值
        double newRate = currentRate * adjustmentFactor;
        
        // 限制在合理范围内
        newRate = Math.max(minRate, Math.min(maxRate, newRate));
        
        if (Math.abs(newRate - currentRate) / currentRate > 0.1) {
            // 变化超过10%，记录日志
            log.info("自适应限流调整: {} -> {}", currentRate, newRate);
            currentRate = newRate;
            
            // 应用新的限流阈值
            applyNewRateLimit(newRate);
        }
    }
    
    /**
     * 计算调整因子
     */
    private double calculateAdjustmentFactor(SystemMetrics metrics) {
        double factor = 1.0;
        
        // 基于CPU使用率调整
        if (metrics.getCpuUsage() > 0.8) {
            factor *= 0.7;  // CPU高，降低限流阈值
        } else if (metrics.getCpuUsage() < 0.3) {
            factor *= 1.3;  // CPU低，提高限流阈值
        }
        
        // 基于内存使用率调整
        if (metrics.getMemoryUsage() > 0.8) {
            factor *= 0.8;  // 内存高，降低限流阈值
        } else if (metrics.getMemoryUsage() < 0.4) {
            factor *= 1.2;  // 内存低，提高限流阈值
        }
        
        // 基于响应时间调整
        if (metrics.getAvgResponseTime() > 1000) {
            factor *= 0.6;  // 响应时间长，大幅降低
        } else if (metrics.getAvgResponseTime() < 100) {
            factor *= 1.4;  // 响应时间短，大幅提高
        }
        
        // 基于错误率调整
        if (metrics.getErrorRate() > 0.1) {
            factor *= 0.5;  // 错误率高，大幅降低
        } else if (metrics.getErrorRate() < 0.01) {
            factor *= 1.5;  // 错误率低，大幅提高
        }
        
        return factor;
    }
    
    /**
     * 获取当前限流阈值
     */
    public double getCurrentRate() {
        return currentRate;
    }
    
    /**
     * 获取自适应限流配置
     */
    public AdaptiveConfig getConfig() {
        return AdaptiveConfig.builder()
            .currentRate(currentRate)
            .minRate(minRate)
            .maxRate(maxRate)
            .incrementFactor(incrementFactor)
            .decrementFactor(decrementFactor)
            .build();
    }
    
    @Data
    @Builder
    public static class SystemMetrics {

        private double memoryUsage;      // 内存使用率
        private double avgResponseTime;  // 平均响应时间

@Component
@Slf4j
public class RateLimitMonitor {
    
    private final MeterRegistry meterRegistry;
    
    // 监控指标
    private final Counter totalRequestsCounter;
    private final Counter limitedRequestsCounter;
    private final Timer requestProcessingTimer;
    
    // 限流统计
    private final Map<String, RateLimitStats> statsMap = new ConcurrentHashMap<>();
    
    public RateLimitMonitor(MeterRegistry meterRegistry) {
        this.meterRegistry = meterRegistry;
        
        // 初始化监控指标
        this.totalRequestsCounter = Counter.builder("ratelimit.requests.total")
            .description("总请求数")
            .register(meterRegistry);
            
        this.limitedRequestsCounter = Counter.builder("ratelimit.requests.limited")
            .description("被限流请求数")
            .register(meterRegistry);
            
        this.requestProcessingTimer = Timer.builder("ratelimit.request.processing")
            .description("请求处理时间")
            .publishPercentiles(0.5, 0.95, 0.99)
            .register(meterRegistry);
    }
    
    /**
     * 记录请求
     */
    public void recordRequest(String resource, boolean limited, long processingTime) {
        // 记录到Micrometer
        totalRequestsCounter.increment();
        
        if (limited) {
            limitedRequestsCounter.increment();
            
            meterRegistry.counter("ratelimit.requests.limited.by.resource",
                "resource", resource
            ).increment();
        }
        
        requestProcessingTimer.record(processingTime, TimeUnit.MILLISECONDS);
        
        // 记录到内部统计
        RateLimitStats stats = statsMap.computeIfAbsent(
            resource, k -> new RateLimitStats());
        stats.recordRequest(limited, processingTime);
    }
    
    /**
     * 获取限流统计
     */
    public RateLimitStats getStats(String resource) {
        return statsMap.get(resource);
    }
    
    /**
     * 获取限流Top N
     */
    public List<ResourceStats> getTopLimitedResources(int limit) {
        return statsMap.entrySet().stream()
            .map(entry -> ResourceStats.builder()
                .resource(entry.getKey())
                .totalRequests(entry.getValue().getTotalRequests())
                .limitedRequests(entry.getValue().getLimitedRequests())
                .limitRate(entry.getValue().getLimitRate())
                .build())
            .sorted(Comparator.comparing(ResourceStats::getLimitedRequests).reversed())
            .limit(limit)
            .collect(Collectors.toList());
    }
    
    /**
     * 限流告警检查
     */
    @Scheduled(fixedRate = 30000)  // 每30秒检查一次
    public void checkRateLimitAlerts() {
        statsMap.forEach((resource, stats) -> {
            double limitRate = stats.getLimitRate();
            
            if (limitRate > 0.1) {  // 限流率超过10%
                log.warn("限流告警: {} 限流率 {}%", resource, limitRate * 100);
                sendRateLimitAlert(resource, limitRate);
            }
            
            if (limitRate > 0.3) {  // 限流率超过30%，严重告警
                log.error("严重限流告警: {} 限流率 {}%", resource, limitRate * 100);
                sendCriticalRateLimitAlert(resource, limitRate);
                
                // 自动调整限流阈值
                autoAdjustRateLimit(resource);
            }
        });
    }
    
    @Data
    public static class RateLimitStats {
        private final AtomicLong totalRequests = new AtomicLong(0);
        private final AtomicLong limitedRequests = new AtomicLong(0);
        private final AtomicLong totalProcessingTime = new AtomicLong(0);
        
        public void recordRequest(boolean limited, long processingTime) {
            totalRequests.incrementAndGet();
            if (limited) {
                limitedRequests.incrementAndGet();
            }
            totalProcessingTime.addAndGet(processingTime);
        }
        
        public double getLimitRate() {
            long total = totalRequests.get();
            if (total == 0) {
                return 0.0;
            }
            return (double) limitedRequests.get() / total;
        }
        
        public double getAvgProcessingTime() {
            long total = totalRequests.get();
            if (total == 0) {
                return 0.0;
            }
            return (double) totalProcessingTime.get() / total;
        }
    }
}
