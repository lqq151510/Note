# 网关路由属性详解

路由属性是定义网关如何将请求转发到后端服务的核心配置。以下是网关（以 Spring Cloud Gateway 为例）的主要路由属性。

## 一、基础路由属性

### 1. **ID**

路由的唯一标识符

```
routes:
  - id: user-service-route
```

### 2. **URI**

目标服务地址

```
uri: http://localhost:8080
# 或使用服务发现
uri: lb://user-service
# 或使用WebSocket
uri: ws://localhost:8080
```

### 3. **Predicates (断言)**

匹配请求的条件，所有断言都为真时路由生效

#### 常用断言：

```
routes:
  - id: example-route
    uri: lb://example-service
    predicates:
      # 1. 路径匹配
      - Path=/api/v1/**
      
      # 2. 方法匹配
      - Method=GET,POST
      
      # 3. 头部匹配
      - Header=X-Request-Id, \d+  # 值匹配正则
      - Header=Authorization      # 只需存在
      
      # 4. 查询参数匹配
      - Query=token, abc.*
      - Query=debug               # 只需存在
      
      # 5. Cookie匹配
      - Cookie=sessionId, [A-Za-z0-9]+
      
      # 6. 主机名匹配
      - Host=**.example.com
      - Host=api.example.com,admin.example.com
      
      # 7. 远程地址匹配
      - RemoteAddr=192.168.1.1/24
      - RemoteAddr=10.0.0.1
      
      # 8. 时间匹配
      - After=2024-01-20T17:42:47.789-07:00[America/Denver]
      - Before=2024-01-20T17:42:47.789-07:00[America/Denver]
      - Between=2024-01-20T17:42:47.789-07:00[America/Denver], 2024-01-21T17:42:47.789-07:00[America/Denver]
      
      # 9. 权重匹配 (常用于A/B测试)
      - Weight=group-a, 80
      - Weight=group-b, 20
      
      # 10. Cloud Foundry路由服务
      - CloudFoundryRouteService=route-service-id
      
      # 组合使用示例
      - Path=/api/users/**
      - Method=GET,POST
      - Header=X-API-Version=v1
```

### 4. **Filters (过滤器)**

在请求转发前后执行的逻辑

#### 常用过滤器：

```
routes:
  - id: example-route
    uri: lb://example-service
    predicates:
      - Path=/api/**
    filters:
      # 1. 路径处理
      - StripPrefix=1                    # 去掉路径的第一段，/api/user/1 → /user/1
      - PrefixPath=/backend              # 添加前缀，/user → /backend/user
      - RewritePath=/api/v1/(?<segment>.*), /$\{segment}  # 正则重写
      - SetPath=/new/path/{segment}      # 设置新路径
      
      # 2. 请求头处理
      - AddRequestHeader=X-Request-Foo, Bar
      - RemoveRequestHeader=X-Secret-Header
      - SetRequestHeader=X-Request-Id, 123
      - AddRequestHeadersIfNotPresent=X-Request-Id, 123
      - MapRequestHeader=X-Request-Id, X-Correlation-Id
      
      # 3. 响应头处理
      - AddResponseHeader=X-Response-Foo, Bar
      - RemoveResponseHeader=X-Secret-Header
      - SetResponseHeader=X-Response-Id, 456
      - DedupeResponseHeader=X-Correlation-Id, RETAIN_FIRST
      
      # 4. 参数处理
      - AddRequestParameter=foo, bar
      - RemoveRequestParameter=secret
      
      # 5. 请求体处理
      - ModifyRequestBody:  # 修改请求体
          in-class: String
          out-class: String
          rewrite-function: |
            (exchange, s) -> Mono.just("{\"modified\": true, \"original\": " + s + "}")
      
      - ModifyResponseBody:  # 修改响应体
          in-class: String
          out-class: String
          rewrite-function: |
            (exchange, s) -> Mono.just("{\"modified\": true, \"data\": " + s + "}")
      
      # 6. 重试机制
      - name: Retry
        args:
          retries: 3
          statuses: BAD_GATEWAY,INTERNAL_SERVER_ERROR
          methods: GET,POST
          series: SERVER_ERROR
          backoff:
            firstBackoff: 10ms
            maxBackoff: 100ms
            factor: 2
            basedOnPreviousValue: false
      
      # 7. 熔断器
      - name: CircuitBreaker
        args:
          name: myCircuitBreaker
          fallbackUri: forward:/fallback/user
          statusCodes: 500,502,503,504
      
      # 8. 限流
      - name: RequestRateLimiter
        args:
          redis-rate-limiter.replenishRate: 10    # 每秒令牌生成数
          redis-rate-limiter.burstCapacity: 20    # 令牌桶容量
          redis-rate-limiter.requestedTokens: 1   # 每次请求消耗令牌数
          key-resolver: "#{@userKeyResolver}"     # 限流键解析器
      
      # 9. 请求大小限制
      - name: RequestSize
        args:
          maxSize: 5MB
      
      # 10. 重定向
      - RedirectTo=302, https://example.com/new-location
      
      # 11. 保存会话
      - SaveSession
      
      # 12. 安全相关
      - SecureHeaders
      - SetRequestHostHeader=api.example.com
      
      # 13. 缓存响应体
      - CacheRequestBody: "true"
      
      # 14. 自定义过滤器
      - name: CustomFilter
        args:
          enabled: true
          threshold: 1000
      
      # 过滤器顺序执行，可指定order
      - AddRequestHeader=X-First, first
      - name: AddRequestHeader
        args:
          name: X-Second
          value: second
        order: 1
```

## 二、高级路由配置

### 1. **组合配置示例**

```
spring:
  cloud:
    gateway:
      routes:
        # 用户服务路由
        - id: user-service
          uri: lb://user-service
          predicates:
            - Path=/api/users/**
            - Method=GET,POST,PUT,DELETE
            - Header=X-API-Version, v1
            - After=2024-01-01T00:00:00+08:00[Asia/Shanghai]
          filters:
            - StripPrefix=1
            - AddRequestHeader=X-Real-IP, ${remoteAddr}
            - name: Retry
              args:
                retries: 3
                methods: GET
            - name: CircuitBreaker
              args:
                name: userServiceCB
                fallbackUri: forward:/fallback/user
            - name: RequestRateLimiter
              args:
                redis-rate-limiter.replenishRate: 100
                redis-rate-limiter.burstCapacity: 200
                key-resolver: "#{@ipKeyResolver}"
        
        # 订单服务路由（A/B测试）
        - id: order-service-v1
          uri: lb://order-service-v1
          predicates:
            - Path=/api/orders/**
            - Weight=group-a, 80
          filters:
            - StripPrefix=1
            - AddRequestHeader=X-Service-Version, v1
        
        - id: order-service-v2
          uri: lb://order-service-v2
          predicates:
            - Path=/api/orders/**
            - Weight=group-b, 20
          filters:
            - StripPrefix=1
            - AddRequestHeader=X-Service-Version, v2
        
        # 管理接口路由（需要认证）
        - id: admin-service
          uri: lb://admin-service
          predicates:
            - Path=/admin/**
            - Host=admin.example.com
          filters:
            - StripPrefix=1
            - AddRequestHeader=X-Admin-Access, true
            - name: JwtAuthFilter
              args:
                secret: ${jwt.secret}
        
        # 静态资源路由
        - id: static-resources
          uri: file:/var/www/html
          predicates:
            - Path=/static/**
          filters:
            - SetPath=/${segment}
        
        # WebSocket路由
        - id: websocket-route
          uri: lb:ws://websocket-service
          predicates:
            - Path=/ws/**
        
        # 重定向路由
        - id: legacy-redirect
          uri: no://op
          predicates:
            - Path=/old-api/**
          filters:
            - RedirectTo=301, https://new.example.com/api/{segment}
```

### 2. **全局过滤器配置**

```
spring:
  cloud:
    gateway:
      default-filters:  # 应用于所有路由
        - AddResponseHeader=X-Gateway-Response, Gateway
        - DedupeResponseHeader=Access-Control-Allow-Credentials Access-Control-Allow-Origin
      
      # 跨域配置
      globalcors:
        cors-configurations:
          '[/**]':
            allowed-origins: "https://example.com"
            allowed-methods: "*"
            allowed-headers: "*"
            allow-credentials: true
            max-age: 3600
```

### 3. **元数据配置**

```
routes:
  - id: monitored-route
    uri: lb://monitored-service
    predicates:
      - Path=/monitor/**
    metadata:
      management.port: 8081
      management.path: /actuator
      circuit-breaker.enabled: true
      rate-limiter.max-requests: 1000
      response-timeout: 5000
```

## 三、动态路由配置

### 1. **基于数据库的路由**

```
@Component
public class DynamicRouteDefinitionLocator implements RouteDefinitionLocator {
    
    @Autowired
    private RouteRepository routeRepository;
    
    @Override
    public Flux<RouteDefinition> getRouteDefinitions() {
        List<RouteDefinition> routeDefinitions = routeRepository.findAll()
            .stream()
            .map(this::convertToRouteDefinition)
            .collect(Collectors.toList());
        return Flux.fromIterable(routeDefinitions);
    }
    
    private RouteDefinition convertToRouteDefinition(RouteEntity entity) {
        RouteDefinition definition = new RouteDefinition();
        definition.setId(entity.getId());
        definition.setUri(URI.create(entity.getUri()));
        
        // 设置断言
        List<PredicateDefinition> predicates = JSON.parseArray(
            entity.getPredicates(), PredicateDefinition.class);
        definition.setPredicates(predicates);
        
        // 设置过滤器
        List<FilterDefinition> filters = JSON.parseArray(
            entity.getFilters(), FilterDefinition.class);
        definition.setFilters(filters);
        
        // 设置元数据
        definition.setMetadata(JSON.parseObject(entity.getMetadata()));
        
        return definition;
    }
}
```
