# 网关登录校验完整实现

网关登录校验是微服务架构中的核心安全组件。以下是基于 Spring Cloud Gateway 的完整登录校验实现方案。

## 一、整体架构设计

### 1. **技术栈选择**

- 认证方式：JWT（JSON Web Token）
    
- 会话管理：Redis 缓存会话
    
- 安全框架：Spring Security + Gateway 过滤器
    
- 权限控制：RBAC 模型
    

### 2. **校验流程**

```
客户端请求 → 网关校验 → 认证服务 → 业务服务
     ↓           ↓         ↓         ↓
   1.携带Token  2.验证Token 3.颁发Token 4.业务处理
```

## 二、核心过滤器实现

### 1. **JWT认证过滤器**

```
import io.jsonwebtoken.Claims;
import io.jsonwebtoken.Jwts;
import lombok.extern.slf4j.Slf4j;
import org.springframework.beans.factory.annotation.Value;
import org.springframework.cloud.gateway.filter.GatewayFilterChain;
import org.springframework.cloud.gateway.filter.GlobalFilter;
import org.springframework.core.Ordered;
import org.springframework.core.io.buffer.DataBuffer;
import org.springframework.http.HttpStatus;
import org.springframework.http.server.reactive.ServerHttpRequest;
import org.springframework.http.server.reactive.ServerHttpResponse;
import org.springframework.stereotype.Component;
import org.springframework.util.AntPathMatcher;
import org.springframework.util.PathMatcher;
import org.springframework.util.StringUtils;
import org.springframework.web.server.ServerWebExchange;
import reactor.core.publisher.Mono;

import java.nio.charset.StandardCharsets;
import java.util.List;

@Slf4j
@Component
public class JwtAuthFilter implements GlobalFilter, Ordered {
    
    @Value("${jwt.secret}")
    private String secret;
    
    @Value("${jwt.header:Authorization}")
    private String tokenHeader;
    
    @Value("${jwt.prefix:Bearer }")
    private String tokenPrefix;
    
    // 路径匹配器
    private final PathMatcher pathMatcher = new AntPathMatcher();
    
    // 白名单路径（不需要认证）
    private final List<String> whiteList = List.of(
        "/auth/login",
        "/auth/register",
        "/auth/refresh",
        "/auth/captcha",
        "/auth/logout",
        "/public/**",
        "/swagger-ui/**",
        "/v3/api-docs/**",
        "/webjars/**",
        "/doc.html",
        "/favicon.ico",
        "/actuator/health",
        "/actuator/info"
    );
    
    @Override
    public Mono<Void> filter(ServerWebExchange exchange, GatewayFilterChain chain) {
        ServerHttpRequest request = exchange.getRequest();
        ServerHttpResponse response = exchange.getResponse();
        String path = request.getPath().value();
        String method = request.getMethodValue();
        
        // 1. 检查是否为白名单路径
        if (isWhiteList(path)) {
            log.debug("白名单路径，放行: {} {}", method, path);
            return chain.filter(exchange);
        }
        
        // 2. 获取Token
        String token = extractToken(request);
        if (!StringUtils.hasText(token)) {
            log.warn("请求缺少Token: {} {}", method, path);
            return unauthorized(response, "未提供认证令牌");
        }
        
        try {
            // 3. 验证Token
            Claims claims = validateToken(token);
            if (claims == null) {
                log.warn("Token验证失败: {}", token);
                return unauthorized(response, "令牌无效或已过期");
            }
            
            // 4. 检查Token是否在黑名单中（注销的Token）
            if (isTokenBlacklisted(token)) {
                log.warn("Token已被注销: {}", token);
                return unauthorized(response, "令牌已失效");
            }
            
            // 5. 权限校验
            if (!hasPermission(claims, request)) {
                log.warn("权限不足: userId={}, path={}", 
                    claims.getSubject(), path);
                return forbidden(response, "权限不足");
            }
            
            // 6. 将用户信息添加到请求头
            ServerHttpRequest mutatedRequest = addUserHeaders(request, claims);
            
            // 7. 记录审计日志
            auditLog(request, claims, "SUCCESS");
            
            return chain.filter(exchange.mutate().request(mutatedRequest).build());
            
        } catch (Exception e) {
            log.error("Token验证异常: {}", e.getMessage(), e);
            return unauthorized(response, "令牌验证异常");
        }
    }
    
    @Override
    public int getOrder() {
        return -100; // 高优先级
    }
    
    /**
     * 提取Token
     */
    private String extractToken(ServerHttpRequest request) {
        // 1. 从Header获取
        String authHeader = request.getHeaders().getFirst(tokenHeader);
        if (StringUtils.hasText(authHeader) && authHeader.startsWith(tokenPrefix)) {
            return authHeader.substring(tokenPrefix.length());
        }
        
        // 2. 从Query参数获取
        String tokenParam = request.getQueryParams().getFirst("token");
        if (StringUtils.hasText(tokenParam)) {
            return tokenParam;
        }
        
        // 3. 从Cookie获取
        String cookieToken = request.getCookies().getFirst("token");
        if (cookieToken != null) {
            return cookieToken.getValue();
        }
        
        return null;
    }
    
    /**
     * 验证Token
     */
    private Claims validateToken(String token) {
        try {
            return Jwts.parser()
                .setSigningKey(secret.getBytes(StandardCharsets.UTF_8))
                .parseClaimsJws(token)
                .getBody();
        } catch (Exception e) {
            log.error("Token解析失败: {}", e.getMessage());
            return null;
        }
    }
    
    /**
     * 检查白名单
     */
    private boolean isWhiteList(String path) {
        return whiteList.stream().anyMatch(pattern -> 
            pathMatcher.match(pattern, path)
        );
    }
    
    /**
     * Token黑名单检查
     */
    private boolean isTokenBlacklisted(String token) {
        // 从Redis检查token是否在黑名单
        // String key = "token:blacklist:" + DigestUtils.md5DigestAsHex(token.getBytes());
        // return Boolean.TRUE.equals(redisTemplate.hasKey(key));
        return false; // 简化实现
    }
    
    /**
     * 权限校验
     */
    private boolean hasPermission(Claims claims, ServerHttpRequest request) {
        String path = request.getPath().value();
        String method = request.getMethodValue();
        String userId = claims.getSubject();
        String userRoles = claims.get("roles", String.class);
        
        // 1. 获取用户权限
        Set<String> permissions = getUserPermissions(userId);
        
        // 2. 检查路径权限
        String requiredPermission = buildPermissionKey(path, method);
        
        // 3. 超级管理员跳过权限检查
        if (isSuperAdmin(userRoles)) {
            return true;
        }
        
        return permissions.contains(requiredPermission);
    }
    
    /**
     * 添加用户信息到请求头
     */
    private ServerHttpRequest addUserHeaders(ServerHttpRequest request, Claims claims) {
        return request.mutate()
            .header("X-User-Id", claims.getSubject())
            .header("X-User-Name", claims.get("username", String.class))
            .header("X-User-Roles", claims.get("roles", String.class))
            .header("X-User-Permissions", claims.get("permissions", String.class))
            .header("X-Request-Id", UUID.randomUUID().toString())
            .build();
    }
    
    /**
     * 未授权响应
     */
    private Mono<Void> unauthorized(ServerHttpResponse response, String message) {
        response.setStatusCode(HttpStatus.UNAUTHORIZED);
        response.getHeaders().setContentType(MediaType.APPLICATION_JSON);
        
        String body = String.format("""
            {
                "code": 401,
                "message": "%s",
                "data": null,
                "timestamp": %d
            }
            """, message, System.currentTimeMillis());
        
        DataBuffer buffer = response.bufferFactory()
            .wrap(body.getBytes(StandardCharsets.UTF_8));
        return response.writeWith(Mono.just(buffer));
    }
    
    /**
     * 禁止访问响应
     */
    private Mono<Void> forbidden(ServerHttpResponse response, String message) {
        response.setStatusCode(HttpStatus.FORBIDDEN);
        response.getHeaders().setContentType(MediaType.APPLICATION_JSON);
        
        String body = String.format("""
            {
                "code": 403,
                "message": "%s",
                "data": null,
                "timestamp": %d
            }
            """, message, System.currentTimeMillis());
        
        DataBuffer buffer = response.bufferFactory()
            .wrap(body.getBytes(StandardCharsets.UTF_8));
        return response.writeWith(Mono.just(buffer));
    }
    
    /**
     * 审计日志
     */
    private void auditLog(ServerHttpRequest request, Claims claims, String status) {
        // 异步记录审计日志
        CompletableFuture.runAsync(() -> {
            try {
                AuditLog log = AuditLog.builder()
                    .userId(claims.getSubject())
                    .username(claims.get("username", String.class))
                    .ip(getClientIp(request))
                    .method(request.getMethodValue())
                    .path(request.getPath().value())
                    .params(request.getQueryParams().toString())
                    .userAgent(request.getHeaders().getFirst("User-Agent"))
                    .status(status)
                    .timestamp(System.currentTimeMillis())
                    .build();
                
                // 保存到数据库或发送到消息队列
                auditLogService.save(log);
                
            } catch (Exception e) {
                log.error("记录审计日志失败: {}", e.getMessage());
            }
        });
    }
    
    /**
     * 获取客户端IP
     */
    private String getClientIp(ServerHttpRequest request) {
        String xff = request.getHeaders().getFirst("X-Forwarded-For");
        if (StringUtils.hasText(xff)) {
            return xff.split(",")[0].trim();
        }
        
        String xri = request.getHeaders().getFirst("X-Real-IP");
        if (StringUtils.hasText(xri)) {
            return xri;
        }
        
        return request.getRemoteAddress() != null ? 
               request.getRemoteAddress().getAddress().getHostAddress() : 
               "unknown";
    }
    
    /**
     * 获取用户权限
     */
    private Set<String> getUserPermissions(String userId) {
        // 从Redis缓存获取
        String cacheKey = "user:permissions:" + userId;
        // Set<String> permissions = redisTemplate.opsForSet().members(cacheKey);
        
        // 缓存未命中则从数据库获取
        // List<Permission> permissionList = permissionService.getUserPermissions(userId);
        // Set<String> permissionSet = permissionList.stream()
        //     .map(p -> p.getCode())
        //     .collect(Collectors.toSet());
        // redisTemplate.opsForSet().add(cacheKey, permissionSet.toArray(new String[0]));
        // redisTemplate.expire(cacheKey, 1, TimeUnit.HOURS);
        
        // return permissionSet;
        
        return new HashSet<>(); // 简化实现
    }
    
    /**
     * 构建权限键
     */
    private String buildPermissionKey(String path, String method) {
        return method + ":" + path.replaceAll("\\{\\w+\\}", "*");
    }
    
    /**
     * 检查是否为超级管理员
     */
    private boolean isSuperAdmin(String roles) {
        return roles != null && roles.contains("ROLE_SUPER_ADMIN");
    }
}

### 2. **登录成功处理器**

```
@Component
public class LoginSuccessHandler {
    
    @Autowired
    private JwtTokenProvider tokenProvider;
    
    @Autowired
    private RedisTemplate<String, Object> redisTemplate;
    
    /**
     * 登录成功后生成Token
     */
    public Mono<LoginResponse> handleLoginSuccess(User user, String clientIp, String userAgent) {
        return Mono.fromCallable(() -> {
            // 1. 生成访问令牌
            String accessToken = tokenProvider.generateAccessToken(user);
            
            // 2. 生成刷新令牌
            String refreshToken = tokenProvider.generateRefreshToken(user);
            
            // 3. 保存用户会话
            saveUserSession(user, accessToken, clientIp, userAgent);
            
            // 4. 保存令牌到Redis
            cacheToken(user.getId(), accessToken, refreshToken);
            
            // 5. 记录登录日志
            logLogin(user, clientIp, "SUCCESS");
            
            return LoginResponse.builder()
                .accessToken(accessToken)
                .refreshToken(refreshToken)
                .tokenType("Bearer")
                .expiresIn(tokenProvider.getAccessTokenExpiration())
                .refreshExpiresIn(tokenProvider.getRefreshTokenExpiration())
                .user(UserInfo.fromEntity(user))
                .build();
        }).subscribeOn(Schedulers.boundedElastic());
    }
    
    /**
     * 保存用户会话
     */
    private void saveUserSession(User user, String token, String clientIp, String userAgent) {
        UserSession session = UserSession.builder()
            .userId(user.getId())
            .username(user.getUsername())
            .token(token)
            .loginIp(clientIp)
            .userAgent(userAgent)
            .loginTime(System.currentTimeMillis())
            .lastAccessTime(System.currentTimeMillis())
            .expireAt(System.currentTimeMillis() + 
                tokenProvider.getAccessTokenExpiration() * 1000)
            .isOnline(true)
            .build();
        
        String sessionKey = "user:session:" + user.getId();
        redisTemplate.opsForHash().put(sessionKey, token, session);
        redisTemplate.expire(sessionKey, 7, TimeUnit.DAYS);
    }
    
    /**
     * 缓存Token
     */
    private void cacheToken(String userId, String accessToken, String refreshToken) {
        // 缓存访问令牌
        String accessKey = "token:access:" + DigestUtils.md5DigestAsHex(accessToken.getBytes());
        redisTemplate.opsForValue().set(accessKey, userId, 
            tokenProvider.getAccessTokenExpiration(), TimeUnit.SECONDS);
        
        // 缓存刷新令牌
        String refreshKey = "token:refresh:" + DigestUtils.md5DigestAsHex(refreshToken.getBytes());
        redisTemplate.opsForValue().set(refreshKey, userId, 
            tokenProvider.getRefreshTokenExpiration(), TimeUnit.SECONDS);
    }
    
    /**
     * 记录登录日志
     */
    private void logLogin(User user, String clientIp, String status) {
        CompletableFuture.runAsync(() -> {
            LoginLog log = LoginLog.builder()
                .userId(user.getId())
                .username(user.getUsername())
                .loginIp(clientIp)
                .loginTime(LocalDateTime.now())
                .status(status)
                .userAgent(userAgent)
                .build();
            
            // 保存到数据库
            loginLogService.save(log);
        });
    }
}

## 四、刷新令牌实现

### 1. **刷新令牌过滤器**

```
@Component
public class RefreshTokenFilter implements GlobalFilter {
    
    @Override
    public Mono<Void> filter(ServerWebExchange exchange, GatewayFilterChain chain) {
        ServerHttpRequest request = exchange.getRequest();
        String path = request.getPath().value();
        
        // 只处理刷新令牌请求
        if (!path.equals("/auth/refresh")) {
            return chain.filter(exchange);
        }
        
        return request.getBody()
            .next()
            .flatMap(dataBuffer -> {
                byte[] bytes = new byte[dataBuffer.readableByteCount()];
                dataBuffer.read(bytes);
                String body = new String(bytes, StandardCharsets.UTF_8);
                
                // 解析刷新令牌
                JsonNode jsonNode = objectMapper.readTree(body);
                String refreshToken = jsonNode.get("refreshToken").asText();
                
                if (!StringUtils.hasText(refreshToken)) {
                    return badRequest(exchange, "刷新令牌不能为空");
                }
                
                // 验证刷新令牌
                return validateAndRefresh(refreshToken, exchange, chain);
            })
            .switchIfEmpty(badRequest(exchange, "请求体为空"));
    }
    
    /**
     * 验证并刷新令牌
     */
    private Mono<Void> validateAndRefresh(String refreshToken, 
                                        ServerWebExchange exchange, 
                                        GatewayFilterChain chain) {
        try {
            // 1. 验证刷新令牌
            Claims claims = jwtUtil.parseToken(refreshToken);
            if (claims == null) {
                return unauthorized(exchange, "刷新令牌无效");
            }
            
            // 2. 检查刷新令牌是否在缓存中
            String refreshKey = "token:refresh:" + 
                DigestUtils.md5DigestAsHex(refreshToken.getBytes());
            if (!Boolean.TRUE.equals(redisTemplate.hasKey(refreshKey))) {
                return unauthorized(exchange, "刷新令牌已失效");
            }
            
            String userId = claims.getSubject();
            
            // 3. 获取用户信息
            User user = userService.getById(userId);
            if (user == null) {
                return unauthorized(exchange, "用户不存在");
            }
            
            // 4. 生成新的访问令牌
            String newAccessToken = tokenProvider.generateAccessToken(user);
            
            // 5. 返回新的令牌
            return writeResponse(exchange, newAccessToken, refreshToken);
            
        } catch (Exception e) {
            log.error("刷新令牌失败: {}", e.getMessage());
            return unauthorized(exchange, "刷新令牌失败");
        }
    }
    
    /**
     * 写入响应
     */
    private Mono<Void> writeResponse(ServerWebExchange exchange, 
                                   String accessToken, String refreshToken) {
        ServerHttpResponse response = exchange.getResponse();
        response.setStatusCode(HttpStatus.OK);
        response.getHeaders().setContentType(MediaType.APPLICATION_JSON);
        
        String body = String.format("""
            {
                "code": 200,
                "message": "success",
                "data": {
                    "accessToken": "%s",
                    "refreshToken": "%s",
                    "tokenType": "Bearer",
                    "expiresIn": %d
                }
            }
            """, accessToken, refreshToken, tokenProvider.getAccessTokenExpiration());
        
        DataBuffer buffer = response.bufferFactory()
            .wrap(body.getBytes(StandardCharsets.UTF_8));
        return response.writeWith(Mono.just(buffer));
    }
}

@Configuration
public class PermissionConfig {
    
    @Bean
    public RouterFunction<ServerResponse> permissionRoutes() {
        return RouterFunctions.route()
            .GET("/admin/permissions", this::getAllPermissions)
            .POST("/admin/permissions", this::createPermission)
            .PUT("/admin/permissions/{id}", this::updatePermission)
            .DELETE("/admin/permissions/{id}", this::deletePermission)
            .POST("/admin/users/{userId}/permissions", this::assignUserPermissions)
            .POST("/admin/roles/{roleId}/permissions", this::assignRolePermissions)
            .build();
    }
    
    /**
     * 权限数据结构
     */
    @Data
    public static class PermissionDTO {
        private String id;
        private String name;
        private String code;        // 权限代码，如: user:create
        private String method;      // HTTP方法
        private String path;        // 接口路径
        private String description;
        private Integer orderNo;
        private String parentId;
        private Boolean enabled = true;
        private LocalDateTime createTime;
    }
    
    /**
     * 获取所有权限
     */
    public Mono<ServerResponse> getAllPermissions(ServerRequest request) {
        List<PermissionDTO> permissions = permissionService.getAllPermissions();
        return ServerResponse.ok()
            .contentType(MediaType.APPLICATION_JSON)
            .bodyValue(ApiResponse.success(permissions));
    }
    
    /**
     * 为用户分配权限
     */
    public Mono<ServerResponse> assignUserPermissions(ServerRequest request) {
        String userId = request.pathVariable("userId");
        return request.bodyToMono(PermissionAssignRequest.class)
            .flatMap(req -> {
                // 分配权限
                permissionService.assignPermissionsToUser(userId, req.getPermissionIds());
                
                // 清除用户权限缓存
                clearUserPermissionCache(userId);
                
                return ServerResponse.ok()
                    .bodyValue(ApiResponse.success("权限分配成功"));
            });
    }
    
    /**
     * 清除用户权限缓存
     */
    private void clearUserPermissionCache(String userId) {
        String cacheKey = "user:permissions:" + userId;
        redisTemplate.delete(cacheKey);
    }
}

### 1. **网关安全配置类**

```
@Configuration
@EnableConfigurationProperties(SecurityProperties.class)
public class GatewaySecurityConfig {
    
    @Autowired
    private SecurityProperties securityProperties;
    
    /**
     * 配置跨域
     */
    @Bean
    public CorsWebFilter corsWebFilter() {
        CorsConfiguration config = new CorsConfiguration();
        config.setAllowCredentials(true);
        config.setAllowedOrigins(securityProperties.getCors().getAllowedOrigins());
        config.setAllowedMethods(securityProperties.getCors().getAllowedMethods());
        config.setAllowedHeaders(securityProperties.getCors().getAllowedHeaders());
        config.setExposedHeaders(securityProperties.getCors().getExposedHeaders());
        config.setMaxAge(securityProperties.getCors().getMaxAge());
        
        UrlBasedCorsConfigurationSource source = new UrlBasedCorsConfigurationSource();
        source.registerCorsConfiguration("/**", config);
        
        return new CorsWebFilter(source);
    }
    
    /**
     * 限流键解析器
     */
    @Bean
    public KeyResolver ipKeyResolver() {
        return exchange -> {
            String ip = exchange.getRequest().getRemoteAddress().getAddress().getHostAddress();
            return Mono.just(ip);
        };
    }
    
    @Bean
    public KeyResolver userKeyResolver() {
        return exchange -> {
            String userId = exchange.getRequest().getHeaders()
                .getFirst("X-User-Id");
            if (userId == null) {
                userId = "anonymous";
            }
            return Mono.just(userId);
        };
    }
    
    /**
     * 安全过滤器顺序配置
     */
    @Bean
    public FilterRegistrationBean<JwtAuthFilter> jwtAuthFilterRegistration() {
        FilterRegistrationBean<JwtAuthFilter> registration = 
            new FilterRegistrationBean<>();
        registration.setFilter(new JwtAuthFilter());
        registration.addUrlPatterns("/*");
        registration.setName("jwtAuthFilter");
        registration.setOrder(1);
        return registration;
    }
    
    /**
     * 配置SSL（HTTPS）
     */
    @Bean
    public ServerConnector serverConnector() {
        ServerConnector connector = new ServerConnector(server);
        connector.setPort(8443);
        
        HttpConfiguration httpsConfig = new HttpConfiguration();
        httpsConfig.addCustomizer(new SecureRequestCustomizer());
        
        SslContextFactory.Server sslContextFactory = new SslContextFactory.Server();
        sslContextFactory.setKeyStorePath("classpath:keystore.p12");
        sslContextFactory.setKeyStorePassword("password");
        sslContextFactory.setKeyManagerPassword("password");
        
        connector.addConnectionFactory(
            new SslConnectionFactory(sslContextFactory, "http/1.1"));
        connector.addConnectionFactory(
            new HttpConnectionFactory(httpsConfig));
        
        return connector;
    }
}

@Component
public class SessionManager {
    
    @Autowired
    private RedisTemplate<String, Object> redisTemplate;
    
    /**
     * 创建会话
     */
    public UserSession createSession(User user, String token, String clientIp) {
        UserSession session = UserSession.builder()
            .sessionId(UUID.randomUUID().toString())
            .userId(user.getId())
            .username(user.getUsername())
            .token(token)
            .loginIp(clientIp)
            .loginTime(System.currentTimeMillis())
            .lastAccessTime(System.currentTimeMillis())
            .userAgent(request.getHeader("User-Agent"))
            .expireAt(System.currentTimeMillis() + 2 * 60 * 60 * 1000) // 2小时
            .isOnline(true)
            .build();
        
        // 保存到Redis
        String sessionKey = "session:" + session.getSessionId();
        redisTemplate.opsForValue().set(sessionKey, session, 2, TimeUnit.HOURS);
        
        // 用户-会话映射
        String userSessionsKey = "user:sessions:" + user.getId();
        redisTemplate.opsForSet().add(userSessionsKey, session.getSessionId());
        redisTemplate.expire(userSessionsKey, 2, TimeUnit.HOURS);
        
        return session;
    }
    
    /**
     * 获取会话
     */
    public UserSession getSession(String sessionId) {
        String key = "session:" + sessionId;
        return (UserSession) redisTemplate.opsForValue().get(key);
    }
    
    /**
     * 更新会话访问时间
     */
    public void updateAccessTime(String sessionId) {
        String key = "session:" + sessionId;
        UserSession session = (UserSession) redisTemplate.opsForValue().get(key);
        if (session != null) {
            session.setLastAccessTime(System.currentTimeMillis());
            redisTemplate.opsForValue().set(key, session, 2, TimeUnit.HOURS);
        }
    }
    
    /**
     * 销毁会话
     */
    public void destroySession(String sessionId) {
        String key = "session:" + sessionId;
        UserSession session = (UserSession) redisTemplate.opsForValue().get(key);
        if (session != null) {
            // 从用户会话集合中移除
            String userSessionsKey = "user:sessions:" + session.getUserId();
            redisTemplate.opsForSet().remove(userSessionsKey, sessionId);
            
            // 删除会话
            redisTemplate.delete(key);
            
            // 将Token加入黑名单
            addTokenToBlacklist(session.getToken());
        }
    }
    
    /**
     * 获取用户的所有会话
     */
    public List<UserSession> getUserSessions(String userId) {
        String key = "user:sessions:" + userId;
        Set<Object> sessionIds = redisTemplate.opsForSet().members(key);
        
        return sessionIds.stream()
            .map(sessionId -> getSession((String) sessionId))
            .filter(Objects::nonNull)
            .collect(Collectors.toList());
    }
    
    /**
     * 强制用户下线
     */
    public void forceLogout(String userId) {
        List<UserSession> sessions = getUserSessions(userId);
        sessions.forEach(session -> destroySession(session.getSessionId()));
    }
}

@Component
public class AuthMetrics {
    
    private final MeterRegistry meterRegistry;
    
    // 计数器
    private final Counter loginSuccessCounter;
    private final Counter loginFailureCounter;
    private final Counter tokenRefreshCounter;
    private final Counter accessDeniedCounter;
    
    // 计时器
    private final Timer tokenValidationTimer;
    
    public AuthMetrics(MeterRegistry meterRegistry) {
        this.meterRegistry = meterRegistry;
        
        // 初始化计数器
        this.loginSuccessCounter = Counter.builder("auth.login.success")
            .description("登录成功次数")
            .register(meterRegistry);
            
        this.loginFailureCounter = Counter.builder("auth.login.failure")
            .description("登录失败次数")
            .tag("reason", "INVALID_CREDENTIALS")
            .register(meterRegistry);
            
        this.tokenRefreshCounter = Counter.builder("auth.token.refresh")
            .description("令牌刷新次数")
            .register(meterRegistry);
            
        this.accessDeniedCounter = Counter.builder("auth.access.denied")
            .description("访问被拒次数")
            .tag("reason", "NO_PERMISSION")
            .register(meterRegistry);
            
        // 初始化计时器
        this.tokenValidationTimer = Timer.builder("auth.token.validation")
            .description("令牌验证耗时")
            .publishPercentiles(0.5, 0.95, 0.99)
            .register(meterRegistry);
    }
    
    /**
     * 记录登录成功
     */
    public void recordLoginSuccess(String username) {
        loginSuccessCounter.increment();
        meterRegistry.counter("auth.login.by.user", 
            "username", username).increment();
    }
    
    /**
     * 记录登录失败
     */
    public void recordLoginFailure(String username, String reason) {
        loginFailureCounter.increment();
        meterRegistry.counter("auth.login.failure.by.reason", 
            "reason", reason).increment();
    }
    
    /**
     * 记录令牌验证时间
     */
    public Timer.Sample startTokenValidation() {
        return Timer.start(meterRegistry);
    }
    
    public void endTokenValidation(Timer.Sample sample) {
        sample.stop(tokenValidationTimer);
    }
}
