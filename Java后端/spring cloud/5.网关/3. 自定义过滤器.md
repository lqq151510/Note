# Spring Cloud Gateway 自定义过滤器

自定义过滤器是网关的核心扩展点，允许你实现特定业务逻辑。以下是完整的自定义过滤器实现指南。

## 一、过滤器类型

### 1. **GlobalFilter (全局过滤器)**

对所有路由生效，执行顺序由 `@Order`注解或实现 `Ordered`接口控制。

### 2. **GatewayFilter (路由过滤器)**

对特定路由生效，通过配置文件或API定义。

## 二、全局过滤器实现

### 1. **基础认证过滤器**

```
import org.springframework.cloud.gateway.filter.GatewayFilterChain;
import org.springframework.cloud.gateway.filter.GlobalFilter;
import org.springframework.core.Ordered;
import org.springframework.http.HttpStatus;
import org.springframework.stereotype.Component;
import org.springframework.web.server.ServerWebExchange;
import reactor.core.publisher.Mono;

@Component
public class AuthGlobalFilter implements GlobalFilter, Ordered {
    
    private static final String AUTH_HEADER = "Authorization";
    private static final String AUTH_PREFIX = "Bearer ";
    
    @Override
    public Mono<Void> filter(ServerWebExchange exchange, GatewayFilterChain chain) {
        // 1. 获取请求
        ServerHttpRequest request = exchange.getRequest();
        ServerHttpResponse response = exchange.getResponse();
        
        // 2. 检查白名单路径（如登录、健康检查）
        String path = request.getPath().value();
        if (isWhitelist(path)) {
            return chain.filter(exchange);
        }
        
        // 3. 验证Token
        String token = extractToken(request);
        if (token == null || !validateToken(token)) {
            return unauthorized(response, "Missing or invalid token");
        }
        
        // 4. 权限校验
        if (!hasPermission(token, request)) {
            return forbidden(response, "Insufficient permissions");
        }
        
        // 5. 将用户信息传递到下游服务
        String userId = extractUserId(token);
        ServerHttpRequest mutatedRequest = request.mutate()
            .header("X-User-Id", userId)
            .header("X-User-Roles", getUserRoles(token))
            .build();
        
        return chain.filter(exchange.mutate().request(mutatedRequest).build());
    }
    
    @Override
    public int getOrder() {
        return -100; // 设置执行顺序，值越小优先级越高
    }
    
    private boolean isWhitelist(String path) {
        return path.startsWith("/auth/login") || 
               path.startsWith("/public/") || 
               path.equals("/health");
    }
    
    private String extractToken(ServerHttpRequest request) {
        String authHeader = request.getHeaders().getFirst(AUTH_HEADER);
        if (authHeader != null && authHeader.startsWith(AUTH_PREFIX)) {
            return authHeader.substring(AUTH_PREFIX.length());
        }
        return null;
    }
    
    private boolean validateToken(String token) {
        // 调用认证服务或本地验证JWT
        try {
            // JWT验证逻辑
            return true;
        } catch (Exception e) {
            return false;
        }
    }
    
    private Mono<Void> unauthorized(ServerHttpResponse response, String message) {
        response.setStatusCode(HttpStatus.UNAUTHORIZED);
        response.getHeaders().setContentType(MediaType.APPLICATION_JSON);
        String body = String.format("{\"code\": 401, \"message\": \"%s\"}", message);
        DataBuffer buffer = response.bufferFactory().wrap(body.getBytes(StandardCharsets.UTF_8));
        return response.writeWith(Mono.just(buffer));
    }
    
    private Mono<Void> forbidden(ServerHttpResponse response, String message) {
        response.setStatusCode(HttpStatus.FORBIDDEN);
        response.getHeaders().setContentType(MediaType.APPLICATION_JSON);
        String body = String.format("{\"code\": 403, \"message\": \"%s\"}", message);
        DataBuffer buffer = response.bufferFactory().wrap(body.getBytes(StandardCharsets.UTF_8));
        return response.writeWith(Mono.just(buffer));
    }

### 2. **请求日志过滤器**

```
import com.fasterxml.jackson.databind.ObjectMapper;
import lombok.extern.slf4j.Slf4j;
import org.reactivestreams.Publisher;
import org.springframework.cloud.gateway.filter.GatewayFilterChain;
import org.springframework.cloud.gateway.filter.GlobalFilter;
import org.springframework.core.Ordered;
import org.springframework.core.io.buffer.DataBuffer;
import org.springframework.core.io.buffer.DataBufferFactory;
import org.springframework.core.io.buffer.DataBufferUtils;
import org.springframework.core.io.buffer.DefaultDataBufferFactory;
import org.springframework.http.server.reactive.ServerHttpRequest;
import org.springframework.http.server.reactive.ServerHttpRequestDecorator;
import org.springframework.http.server.reactive.ServerHttpResponse;
import org.springframework.http.server.reactive.ServerHttpResponseDecorator;
import org.springframework.stereotype.Component;
import org.springframework.web.server.ServerWebExchange;
import reactor.core.publisher.Flux;
import reactor.core.publisher.Mono;

import java.nio.charset.StandardCharsets;
import java.util.HashMap;
import java.util.Map;
import java.util.UUID;

@Slf4j
@Component
public class LoggingGlobalFilter implements GlobalFilter, Ordered {
    
    private final ObjectMapper objectMapper = new ObjectMapper();
    
    @Override
    public Mono<Void> filter(ServerWebExchange exchange, GatewayFilterChain chain) {
        long startTime = System.currentTimeMillis();
        String requestId = UUID.randomUUID().toString();
        
        // 1. 记录请求
        ServerHttpRequest request = exchange.getRequest();
        LogInfo logInfo = LogInfo.builder()
            .requestId(requestId)
            .method(request.getMethodValue())
            .path(request.getPath().value())
            .query(request.getURI().getQuery())
            .clientIp(getClientIp(request))
            .userAgent(request.getHeaders().getFirst("User-Agent"))
            .build();
        
        // 2. 添加请求ID到header
        ServerHttpRequest mutatedRequest = request.mutate()
            .header("X-Request-ID", requestId)
            .header("X-Start-Time", String.valueOf(startTime))
            .build();
        
        // 3. 包装请求以记录请求体
        ServerHttpRequestDecorator requestDecorator = new ServerHttpRequestDecorator(mutatedRequest) {
            @Override
            public Flux<DataBuffer> getBody() {
                return super.getBody().doOnNext(dataBuffer -> {
                    // 记录请求体（注意：这会消费buffer，需要缓存）
                });
            }
        };
        
        // 4. 包装响应以记录响应体
        ServerHttpResponseDecorator responseDecorator = new ServerHttpResponseDecorator(exchange.getResponse()) {
            @Override
            public Mono<Void> writeWith(Publisher<? extends DataBuffer> body) {
                if (body instanceof Flux) {
                    Flux<? extends DataBuffer> fluxBody = (Flux<? extends DataBuffer>) body;
                    
                    return super.writeWith(fluxBody.buffer().map(dataBuffers -> {
                        // 合并多个buffer
                        DataBufferFactory bufferFactory = new DefaultDataBufferFactory();
                        DataBuffer join = bufferFactory.join(dataBuffers);
                        byte[] content = new byte[join.readableByteCount()];
                        join.read(content);
                        DataBufferUtils.release(join);
                        
                        // 记录响应
                        String responseBody = new String(content, StandardCharsets.UTF_8);
                        long endTime = System.currentTimeMillis();
                        
                        logInfo.setDuration(endTime - startTime);
                        logInfo.setStatus(this.getStatusCode().value());
                        logInfo.setResponseBody(maskSensitiveData(responseBody));
                        
                        // 异步记录完整日志
                        logAsync(logInfo);
                        
                        return bufferFactory.wrap(content);
                    }));
                }
                return super.writeWith(body);
            }
        };
        
        // 5. 记录请求开始日志
        log.info("Request started: {}", logInfo.toJson());
        
        return chain.filter(exchange.mutate()
            .request(requestDecorator)
            .response(responseDecorator)
            .build());
    }
    
    @Override
    public int getOrder() {
        return Ordered.HIGHEST_PRECEDENCE; // 最先执行
    }
    
    private String getClientIp(ServerHttpRequest request) {
        String xForwardedFor = request.getHeaders().getFirst("X-Forwarded-For");
        if (xForwardedFor != null && !xForwardedFor.isEmpty()) {
            return xForwardedFor.split(",")[0].trim();
        }
        return request.getRemoteAddress() != null ? 
               request.getRemoteAddress().getAddress().getHostAddress() : "unknown";
    }
    
    private void logAsync(LogInfo logInfo) {
        Mono.fromRunnable(() -> {
            try {
                Map<String, Object> logMap = new HashMap<>();
                logMap.put("type", "gateway_access");
                logMap.put("requestId", logInfo.getRequestId());
                logMap.put("method", logInfo.getMethod());
                logMap.put("path", logInfo.getPath());
                logMap.put("status", logInfo.getStatus());
                logMap.put("duration", logInfo.getDuration());
                logMap.put("timestamp", System.currentTimeMillis());
                
                // 结构化日志输出
                String jsonLog = objectMapper.writeValueAsString(logMap);
                
                if (logInfo.getStatus() >= 400) {
                    log.warn(jsonLog);
                } else {
                    log.info(jsonLog);
                }
                
                // 发送到监控系统
                sendToMetrics(logInfo);
                
            } catch (Exception e) {
                log.error("Failed to log request", e);
            }
        }).subscribeOn(Schedulers.boundedElastic()).subscribe();
    }
    
    @lombok.Builder
    @lombok.Data
    private static class LogInfo {
        private String requestId;
        private String method;
        private String path;
        private String query;
        private String clientIp;
        private String userAgent;
        private Integer status;
        private Long duration;
        private String requestBody;
        private String responseBody;
        
        public String toJson() {
            try {
                return objectMapper.writeValueAsString(this);
            } catch (Exception e) {
                return "{\"error\": \"Failed to serialize log\"}";
            }
        }
    }
}

### 3. **限流过滤器**

```
import com.google.common.util.concurrent.RateLimiter;
import org.springframework.cloud.gateway.filter.GatewayFilterChain;
import org.springframework.cloud.gateway.filter.GlobalFilter;
import org.springframework.core.Ordered;
import org.springframework.core.annotation.Order;
import org.springframework.http.HttpStatus;
import org.springframework.stereotype.Component;
import org.springframework.web.server.ServerWebExchange;
import reactor.core.publisher.Mono;

import java.util.Map;
import java.util.concurrent.ConcurrentHashMap;
import java.util.concurrent.TimeUnit;

@Component
@Order(Ordered.HIGHEST_PRECEDENCE + 1)
public class RateLimitGlobalFilter implements GlobalFilter {
    
    // 基于Guava的限流器
    private final Map<String, RateLimiter> limiters = new ConcurrentHashMap<>();
    
    // 配置：每秒令牌数
    private static final double DEFAULT_RATE = 100.0; // 默认100 QPS
    private final Map<String, Double> rateConfig = Map.of(
        "/api/user/", 50.0,    // 用户接口 50 QPS
        "/api/order/", 30.0,   // 订单接口 30 QPS
        "/api/payment/", 20.0  // 支付接口 20 QPS
    );
    
    @Override
    public Mono<Void> filter(ServerWebExchange exchange, GatewayFilterChain chain) {
        String path = exchange.getRequest().getPath().value();
        String clientIp = getClientIp(exchange.getRequest());
        
        // 1. 获取限流器
        RateLimiter limiter = getRateLimiter(path, clientIp);
        
        // 2. 尝试获取令牌
        if (limiter.tryAcquire(1, TimeUnit.MILLISECONDS.toMillis(100))) {
            // 添加限流头部信息
            exchange.getResponse().getHeaders().add("X-RateLimit-Limit", 
                String.valueOf((int) limiter.getRate()));
            exchange.getResponse().getHeaders().add("X-RateLimit-Remaining", 
                String.valueOf((int) (limiter.getRate() - limiter.acquire())));
            
            return chain.filter(exchange);
        } else {
            // 3. 限流处理
            return tooManyRequests(exchange);
        }
    }
    
    private RateLimiter getRateLimiter(String path, String key) {
        String limiterKey = key + ":" + getRateKey(path);
        return limiters.computeIfAbsent(limiterKey, k -> {
            double rate = getRateForPath(path);
            return RateLimiter.create(rate);
        });
    }
    
    private String getRateKey(String path) {
        for (String pattern : rateConfig.keySet()) {
            if (path.startsWith(pattern)) {
                return pattern;
            }
        }
        return "default";
    }
    
    private double getRateForPath(String path) {
        return rateConfig.entrySet().stream()
            .filter(entry -> path.startsWith(entry.getKey()))
            .map(Map.Entry::getValue)
            .findFirst()
            .orElse(DEFAULT_RATE);
    }
    
    private Mono<Void> tooManyRequests(ServerWebExchange exchange) {
        exchange.getResponse().setStatusCode(HttpStatus.TOO_MANY_REQUESTS);
        exchange.getResponse().getHeaders().setContentType(MediaType.APPLICATION_JSON);
        
        String body = "{\"code\": 429, \"message\": \"请求过于频繁，请稍后重试\"}";
        DataBuffer buffer = exchange.getResponse().bufferFactory()
            .wrap(body.getBytes(StandardCharsets.UTF_8));
        
        return exchange.getResponse().writeWith(Mono.just(buffer));
    }
    
    // 动态更新限流配置
    public void updateRate(String path, double newRate) {
        rateConfig.put(path, newRate);
        // 更新现有限流器
        limiters.entrySet().stream()
            .filter(entry -> entry.getKey().endsWith(":" + path))
            .forEach(entry -> entry.getValue().setRate(newRate));
    }
}

### 1. **自定义参数验证过滤器**

```
import org.springframework.cloud.gateway.filter.GatewayFilter;
import org.springframework.cloud.gateway.filter.factory.AbstractGatewayFilterFactory;
import org.springframework.http.HttpStatus;
import org.springframework.stereotype.Component;
import org.springframework.util.StringUtils;
import org.springframework.validation.annotation.Validated;

import javax.validation.constraints.NotEmpty;
import java.util.Arrays;
import java.util.List;

@Component
public class ValidateParamsGatewayFilterFactory extends 
    AbstractGatewayFilterFactory<ValidateParamsGatewayFilterFactory.Config> {
    
    public ValidateParamsGatewayFilterFactory() {
        super(Config.class);
    }
    
    @Override
    public List<String> shortcutFieldOrder() {
        return Arrays.asList("requiredParams", "errorMessage");
    }
    
    @Override
    public GatewayFilter apply(Config config) {
        return (exchange, chain) -> {
            // 验证必填参数
            for (String param : config.getRequiredParams()) {
                String value = exchange.getRequest().getQueryParams().getFirst(param);
                if (!StringUtils.hasText(value)) {
                    return badRequest(exchange, 
                        String.format(config.getErrorMessage(), param));
                }
            }
            
            // 验证参数格式
            if (config.getRegexPatterns() != null) {
                for (ParamPattern pattern : config.getRegexPatterns()) {
                    String value = exchange.getRequest().getQueryParams()
                        .getFirst(pattern.getParamName());
                    if (value != null && !value.matches(pattern.getRegex())) {
                        return badRequest(exchange, 
                            String.format("参数 %s 格式错误", pattern.getParamName()));
                    }
                }
            }
            
            return chain.filter(exchange);
        };
    }
    
    @Validated
    public static class Config {
        @NotEmpty
        private List<String> requiredParams;
        private String errorMessage = "参数 %s 为必填项";
        private List<ParamPattern> regexPatterns;
        
        // Getters and Setters
        public List<String> getRequiredParams() { return requiredParams; }
        public void setRequiredParams(List<String> requiredParams) { 
            this.requiredParams = requiredParams; 
        }
        
        public String getErrorMessage() { return errorMessage; }
        public void setErrorMessage(String errorMessage) { 
            this.errorMessage = errorMessage; 
        }
        
        public List<ParamPattern> getRegexPatterns() { return regexPatterns; }
        public void setRegexPatterns(List<ParamPattern> regexPatterns) { 
            this.regexPatterns = regexPatterns; 
        }
    }
    
    @lombok.Data
    public static class ParamPattern {
        private String paramName;
        private String regex;
    }
}

### 2. **API版本控制过滤器**

```
@Component
public class ApiVersionGatewayFilterFactory extends 
    AbstractGatewayFilterFactory<ApiVersionGatewayFilterFactory.Config> {
    
    public ApiVersionGatewayFilterFactory() {
        super(Config.class);
    }
    
    @Override
    public GatewayFilter apply(Config config) {
        return (exchange, chain) -> {
            ServerHttpRequest request = exchange.getRequest();
            
            // 1. 从header获取版本
            String version = request.getHeaders().getFirst("X-API-Version");
            
            // 2. 从查询参数获取版本
            if (version == null) {
                version = request.getQueryParams().getFirst("v");
            }
            
            // 3. 从路径获取版本
            if (version == null) {
                String path = request.getPath().value();
                if (path.matches("/v\\d+/.*")) {
                    version = path.split("/")[1].substring(1);
                }
            }
            
            // 4. 默认版本
            if (version == null) {
                version = config.getDefaultVersion();
            }
            
            // 5. 验证版本
            if (!config.getSupportedVersions().contains(version)) {
                return badRequest(exchange, 
                    String.format("不支持的API版本: %s，支持的版本: %s", 
                        version, config.getSupportedVersions()));
            }
            
            // 6. 重写路径
            if (config.isRewritePath()) {
                String newPath = rewritePath(request.getPath().value(), version);
                ServerHttpRequest mutatedRequest = request.mutate()
                    .path(newPath)
                    .build();
                exchange = exchange.mutate().request(mutatedRequest).build();
            }
            
            // 7. 添加版本头到下游
            ServerHttpRequest finalRequest = exchange.getRequest().mutate()
                .header("X-API-Version-Internal", version)
                .build();
            
            return chain.filter(exchange.mutate().request(finalRequest).build());
        };
    }
    
    private String rewritePath(String path, String version) {
        // 移除路径中的版本号
        if (path.matches("/v\\d+/.*")) {
            return "/" + path.split("/", 3)[2];
        }
        return path;
    }
    
    public static class Config {
        private String defaultVersion = "v1";
        private List<String> supportedVersions = Arrays.asList("v1", "v2");
        private boolean rewritePath = true;
        
        // Getters and Setters
    }
}

### 1. **请求/响应修改过滤器**

```
@Component
public class ModifyRequestFilter implements GlobalFilter {
    
    @Override
    public Mono<Void> filter(ServerWebExchange exchange, GatewayFilterChain chain) {
        // 1. 修改请求体
        ServerHttpRequest request = exchange.getRequest();
        MediaType contentType = request.getHeaders().getContentType();
        
        if (contentType != null && contentType.includes(MediaType.APPLICATION_JSON)) {
            // 使用CachedBodyOutputMessage缓存请求体
            return DataBufferUtils.join(request.getBody())
                .flatMap(dataBuffer -> {
                    DataBufferUtils.retain(dataBuffer);
                    Flux<DataBuffer> cachedFlux = Flux
                        .defer(() -> Flux.just(dataBuffer.slice(0, dataBuffer.readableByteCount())));
                    
                    ServerHttpRequest mutatedRequest = new ServerHttpRequestDecorator(request) {
                        @Override
                        public Flux<DataBuffer> getBody() {
                            return cachedFlux;
                        }
                    };
                    
                    // 修改请求体
                    return ServerRequest.create(exchange.mutate().request(mutatedRequest).build(), 
                        messageReaders)
                        .bodyToMono(String.class)
                        .flatMap(body -> {
                            // 修改JSON
                            String modifiedBody = modifyJson(body);
                            
                            // 创建新请求
                            byte[] newBody = modifiedBody.getBytes(StandardCharsets.UTF_8);
                            ServerHttpRequest newRequest = request.mutate()
                                .header("Content-Length", String.valueOf(newBody.length))
                                .build();
                            
                            // 重新包装exchange
                            exchange.getAttributes().put(CACHED_REQUEST_BODY_ATTR, newBody);
                            
                            return chain.filter(exchange.mutate().request(newRequest).build());
                        });
                });
        }
        
        return chain.filter(exchange);
    }
    
    private String modifyJson(String original) {
        try {
            JsonNode root = objectMapper.readTree(original);
            ((ObjectNode) root).put("gateway_processed", true);
            ((ObjectNode) root).put("timestamp", System.currentTimeMillis());
            return objectMapper.writeValueAsString(root);
        } catch (Exception e) {
            return original;
        }
    }
}

### 2. **缓存过滤器**

```
@Component
public class CacheResponseFilter implements GlobalFilter {
    
    private final Cache<String, CachedResponse> cache = 
        Caffeine.newBuilder()
            .maximumSize(1000)
            .expireAfterWrite(5, TimeUnit.MINUTES)
            .recordStats()
            .build();
    
    @Override
    public Mono<Void> filter(ServerWebExchange exchange, GatewayFilterChain chain) {
        ServerHttpRequest request = exchange.getRequest();
        
        // 1. 检查是否应该缓存
        if (!shouldCache(request)) {
            return chain.filter(exchange);
        }
        
        // 2. 生成缓存键
        String cacheKey = generateCacheKey(request);
        
        // 3. 检查缓存
        CachedResponse cached = cache.getIfPresent(cacheKey);
        if (cached != null) {
            // 返回缓存响应
            ServerHttpResponse response = exchange.getResponse();
            response.setStatusCode(HttpStatus.OK);
            response.getHeaders().putAll(cached.getHeaders());
            response.getHeaders().set("X-Cache", "HIT");
            
            DataBuffer buffer = response.bufferFactory()
                .wrap(cached.getBody());
            return response.writeWith(Mono.just(buffer));
        }
        
        // 4. 缓存响应
        ServerHttpResponse originalResponse = exchange.getResponse();
        ServerHttpResponseDecorator decoratedResponse = new ServerHttpResponseDecorator(originalResponse) {
            private byte[] bodyBytes;
            
            @Override
            public Mono<Void> writeWith(Publisher<? extends DataBuffer> body) {
                return super.writeWith(Flux.from(body).map(dataBuffer -> {
                    // 保存响应体
                    byte[] bytes = new byte[dataBuffer.readableByteCount()];
                    dataBuffer.read(bytes);
                    bodyBytes = bytes;
                    
                    // 放入缓存
                    if (originalResponse.getStatusCode() == HttpStatus.OK) {
                        CachedResponse cachedResponse = new CachedResponse(
                            bodyBytes,
                            originalResponse.getHeaders(),
                            System.currentTimeMillis()
                        );
                        cache.put(cacheKey, cachedResponse);
                    }
                    
                    return originalResponse.bufferFactory().wrap(bytes);
                }));
            }
        };
        
        return chain.filter(exchange.mutate().response(decoratedResponse).build());
    }
    
    private boolean shouldCache(ServerHttpRequest request) {
        String method = request.getMethodValue();
        String path = request.getPath().value();
        
        // 只缓存GET请求
        if (!"GET".equals(method)) {
            return false;
        }
        
        // 检查路径是否在白名单
        List<String> cachePaths = Arrays.asList(
            "/api/products/",
            "/api/categories/",
            "/api/config/"
        );
        
        return cachePaths.stream().anyMatch(path::startsWith);
    }
    
    private String generateCacheKey(ServerHttpRequest request) {
        StringBuilder key = new StringBuilder();
        key.append(request.getMethodValue())
           .append(":")
           .append(request.getPath().value());
        
        // 包含查询参数
        if (request.getQueryParams() != null && !request.getQueryParams().isEmpty()) {
            key.append("?").append(request.getQueryParams().toString());
        }
        
        // 包含认证信息
        String userId = request.getHeaders().getFirst("X-User-Id");
        if (userId != null) {
            key.append(":user:").append(userId);
        }
        
        return key.toString();
    }
    
    @lombok.Data
    @lombok.AllArgsConstructor
    private static class CachedResponse {
        private byte[] body;
        private HttpHeaders headers;
        private long timestamp;
    }
}

@Component
public class OptimizedFilter implements GlobalFilter {

    private final LoadingCache<String, Boolean> permissionCache =

            .maximumSize(10000)
            .expireAfterWrite(5, TimeUnit.MINUTES)

    @Override
    public Mono<Void> filter(ServerWebExchange exchange, GatewayFilterChain chain) {

            // 异步验证

@Component
public class MetricsFilter implements GlobalFilter {
    
    private final MeterRegistry meterRegistry;
    private final Timer timer;
    private final Counter successCounter;
    private final Counter errorCounter;
    
    public MetricsFilter(MeterRegistry meterRegistry) {
        this.meterRegistry = meterRegistry;
        
        this.timer = Timer.builder("gateway.request.duration")
            .description("Gateway request processing time")
            .publishPercentiles(0.5, 0.95, 0.99)
            .register(meterRegistry);
        
        this.successCounter = Counter.builder("gateway.requests")
            .tag("status", "success")
            .register(meterRegistry);
        
        this.errorCounter = Counter.builder("gateway.requests")
            .tag("status", "error")
            .register(meterRegistry);
    }
    
    @Override
    public Mono<Void> filter(ServerWebExchange exchange, GatewayFilterChain chain) {
        long startTime = System.currentTimeMillis();
        
        return chain.filter(exchange)
            .doOnSuccess(v -> {
                long duration = System.currentTimeMillis() - startTime;
                timer.record(duration, TimeUnit.MILLISECONDS);
                successCounter.increment();
                
                // 记录更多指标
                meterRegistry.counter("gateway.requests.by.path",
                    "path", exchange.getRequest().getPath().value(),
                    "method", exchange.getRequest().getMethodValue(),
                    "status", String.valueOf(exchange.getResponse().getStatusCode().value())
                ).increment();
            })
            .doOnError(e -> {
                errorCounter.increment();
                meterRegistry.counter("gateway.errors.by.type",
                    "type", e.getClass().getSimpleName()
                ).increment();
            });
    }
}

### 2. **请求体只能读取一次**

```
// 使用CachedBodyOutputMessage缓存
class CachingFilter implements GlobalFilter {
    @Override
    public Mono<Void> filter(ServerWebExchange exchange, GatewayFilterChain chain) {
        if (exchange.getRequest().getHeaders().getContentType() == null) {
            return chain.filter(exchange);
        }
        
        return DataBufferUtils.join(exchange.getRequest().getBody())
            .flatMap(dataBuffer -> {
                DataBufferUtils.retain(dataBuffer);
                Flux<DataBuffer> cachedFlux = Flux.just(dataBuffer);
                
                ServerHttpRequest mutatedRequest = new ServerHttpRequestDecorator(
                    exchange.getRequest()) {
                    @Override
                    public Flux<DataBuffer> getBody() {
                        return cachedFlux;
                    }
                };
                
                return chain.filter(exchange.mutate().request(mutatedRequest).build());
            });
    }
}
