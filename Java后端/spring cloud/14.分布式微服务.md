```
// 负载均衡配置
@Configuration
public class LoadBalancerConfig {
    
    /**
     * 随机负载均衡策略
     */
    @Bean
    public ReactorLoadBalancer<ServiceInstance> randomLoadBalancer(
        Environment environment,
        LoadBalancerClientFactory loadBalancerClientFactory) {
        
        String name = environment.getProperty(LoadBalancerClientFactory.PROPERTY_NAME);
        return new RandomLoadBalancer(
            loadBalancerClientFactory.getLazyProvider(name, ServiceInstanceListSupplier.class),
            name
        );
    }
    
    /**
     * 加权轮询负载均衡
     */
    @Bean
    public ReactorLoadBalancer<ServiceInstance> weightedRoundRobinLoadBalancer(
        Environment environment,
        LoadBalancerClientFactory loadBalancerClientFactory) {
        
        String name = environment.getProperty(LoadBalancerClientFactory.PROPERTY_NAME);
        return new WeightedRoundRobinLoadBalancer(
            loadBalancerClientFactory.getLazyProvider(name, ServiceInstanceListSupplier.class),
            name
        );
    }
    
    /**
     * 自定义负载均衡器
     */
    public class WeightedRoundRobinLoadBalancer 
        implements ReactorLoadBalancer<ServiceInstance> {
        
        private final ObjectProvider<ServiceInstanceListSupplier> supplierProvider;
        private final String serviceId;
        private final AtomicInteger position = new AtomicInteger(0);
        
        public WeightedRoundRobinLoadBalancer(
            ObjectProvider<ServiceInstanceListSupplier> supplierProvider,
            String serviceId) {
            this.supplierProvider = supplierProvider;
            this.serviceId = serviceId;
        }
        
        @Override
        public Mono<Response<ServiceInstance>> choose(Request request) {
            ServiceInstanceListSupplier supplier = supplierProvider
                .getIfAvailable(NoopServiceInstanceListSupplier::new);
            
            return supplier.get(request).next()
                .map(instances -> {
                    if (instances.isEmpty()) {
                        return new EmptyResponse();
                    }
                    
                    // 按权重选择实例
                    List<ServiceInstance> weightedInstances = new ArrayList<>();
                    for (ServiceInstance instance : instances) {
                        int weight = Integer.parseInt(
                            instance.getMetadata().getOrDefault("weight", "1"));
                        for (int i = 0; i < weight; i++) {
                            weightedInstances.add(instance);
                        }
                    }
                    
                    int pos = position.getAndUpdate(
                        p -> (p + 1) % weightedInstances.size());
                    return new DefaultResponse(weightedInstances.get(pos));
                });
        }
    }
}
```

```
@Component
@Slf4j
@RefreshScope
public class DynamicConfigManager {
    
    @Value("${app.config.timeout:5000}")
    private Integer timeout;
    
    @Value("${app.config.retry.count:3}")
    private Integer retryCount;
    
    @Value("${app.config.cache.ttl:300}")
    private Integer cacheTtl;
    
    /**
     * 配置变更监听
     */
    @EventListener
    public void onEnvironmentChange(EnvironmentChangeEvent event) {
        Set<String> changedKeys = event.getKeys();
        log.info("配置变更: {}", changedKeys);
        
        changedKeys.forEach(key -> {
            String value = environment.getProperty(key);
            log.info("配置变更 - {} = {}", key, value);
            
            // 根据不同配置执行不同处理
            handleConfigChange(key, value);
        });
    }
    
    /**
     * 处理配置变更
     */
    private void handleConfigChange(String key, String newValue) {
        if (key.startsWith("app.config.")) {
            switch (key) {
                case "app.config.timeout":
                    updateTimeout(Integer.parseInt(newValue));
                    break;
                case "app.config.retry.count":
                    updateRetryCount(Integer.parseInt(newValue));
                    break;
                case "app.config.cache.ttl":
                    updateCacheTtl(Integer.parseInt(newValue));
                    break;
            }
        }
    }
    
    /**
     * 手动刷新配置
     */
    @PostMapping("/config/refresh")
    public ApiResult<RefreshResult> refreshConfig() {
        try {
            Set<String> refreshedKeys = contextRefresher.refresh();
            
            return ApiResult.success(
                RefreshResult.builder()
                    .success(true)
                    .refreshedKeys(refreshedKeys)
                    .timestamp(System.currentTimeMillis())
                    .message("配置刷新成功")
                    .build()
            );
        } catch (Exception e) {
            return ApiResult.error(500, "配置刷新失败: " + e.getMessage());
        }
    }
}
```

### 2. **自定义网关过滤器**

```
@Component
@Order(-1)
@Slf4j
public class GlobalAuthFilter implements GlobalFilter {
    
    @Autowired
    private RedisTemplate<String, Object> redisTemplate;
    
    @Autowired
    private JwtTokenProvider tokenProvider;
    
    // 白名单路径
    private static final Set<String> WHITE_LIST = Set.of(
        "/auth/login",
        "/auth/register",
        "/auth/refresh",
        "/public/",
        "/static/",
        "/health",
        "/actuator/health"
    );
    
    @Override
    public Mono<Void> filter(ServerWebExchange exchange, GatewayFilterChain chain) {
        ServerHttpRequest request = exchange.getRequest();
        ServerHttpResponse response = exchange.getResponse();
        
        String path = request.getPath().value();
        String method = request.getMethodValue();
        
        // 1. 白名单检查
        if (isWhiteList(path)) {
            return chain.filter(exchange);
        }
        
        // 2. 提取Token
        String token = extractToken(request);
        if (!StringUtils.hasText(token)) {
            return unauthorized(response, "未提供认证令牌");
        }
        
        // 3. 验证Token
        if (!tokenProvider.validateToken(token)) {
            return unauthorized(response, "令牌无效或已过期");
        }
        
        // 4. 检查Token是否在黑名单
        if (isTokenBlacklisted(token)) {
            return unauthorized(response, "令牌已失效");
        }
        
        // 5. 获取用户信息
        Claims claims = tokenProvider.parseToken(token);
        String userId = claims.getSubject();
        String username = claims.get("username", String.class);
        String roles = claims.get("roles", String.class);
        
        // 6. 权限检查
        if (!hasPermission(path, method, roles)) {
            return forbidden(response, "权限不足");
        }
        
        // 7. 限流检查
        if (isRateLimited(userId, path)) {
            return tooManyRequests(response, "请求过于频繁");
        }
        
        // 8. 添加用户信息到请求头
        ServerHttpRequest mutatedRequest = addUserHeaders(request, claims);
        
        // 9. 记录审计日志
        auditLog(request, claims, "SUCCESS");
        
        return chain.filter(exchange.mutate().request(mutatedRequest).build());
    }
    
    /**
     * 添加用户信息到请求头
     */
    private ServerHttpRequest addUserHeaders(ServerHttpRequest request, Claims claims) {
        return request.mutate()
            .header("X-User-Id", claims.getSubject())
            .header("X-User-Name", claims.get("username", String.class))
            .header("X-User-Roles", claims.get("roles", String.class))
            .header("X-User-Permissions", claims.get("permissions", String.class))
            .header("X-Request-Id", UUID.randomUUID().toString())
            .header("X-Forwarded-For", getClientIp(request))
            .build();
    }
    
    /**
     * 权限检查
     */
    private boolean hasPermission(String path, String method, String roles) {
        // 获取用户权限
        Set<String> permissions = getUserPermissions(roles);
        
        // 构建需要的权限
        String requiredPermission = method + ":" + normalizePath(path);
        
        return permissions.contains(requiredPermission) || 
               permissions.contains("ALL");
    }
    
    /**
     * 限流检查
     */
    private boolean isRateLimited(String userId, String path) {
        String key = "rate:limit:" + userId + ":" + path;
        Long count = redisTemplate.opsForValue().increment(key);
        
        if (count == 1) {
            redisTemplate.expire(key, 1, TimeUnit.SECONDS);
        }
        
        return count != null && count > 100; // 每秒100次限制
    }
    
    /**
     * 审计日志
     */
    private void auditLog(ServerHttpRequest request, Claims claims, String status) {
        CompletableFuture.runAsync(() -> {
            try {
                AuditLog log = AuditLog.builder()
                    .userId(claims.getSubject())
                    .username(claims.get("username", String.class))
                    .method(request.getMethodValue())
                    .path(request.getPath().value())
                    .ip(getClientIp(request))
                    .userAgent(request.getHeaders().getFirst("User-Agent"))
                    .params(request.getQueryParams().toString())
                    .status(status)
                    .timestamp(System.currentTimeMillis())
                    .build();
                
                // 发送到消息队列
                auditLogProducer.send(log);
            } catch (Exception e) {
                log.error("记录审计日志失败", e);
            }
        });
    }
}
```

```
@Service
@Slf4j
public class OrderService {
    
    @Autowired
    private OrderMapper orderMapper;
    
    @Autowired
    private AccountFeignClient accountFeignClient;
    
    @Autowired
    private ProductFeignClient productFeignClient;
    
    @Autowired
    private MessageProducer messageProducer;
    
    /**
     * 创建订单（分布式事务）
     */
    @GlobalTransactional(name = "createOrder", timeoutMills = 30000)
    public ApiResult<OrderDTO> createOrder(OrderCreateRequest request) {
        // 1. 创建订单
        Order order = buildOrder(request);
        orderMapper.insert(order);
        log.info("订单创建成功: {}", order.getOrderNo());
        
        // 2. 扣减库存
        ProductDeductRequest deductRequest = new ProductDeductRequest();
        deductRequest.setProductId(request.getProductId());
        deductRequest.setQuantity(request.getQuantity());
        
        ApiResult<Void> deductResult = productFeignClient.deductStock(deductRequest);
        if (!deductResult.isSuccess()) {
            throw new BusinessException("库存不足");
        }
        log.info("库存扣减成功");
        
        // 3. 扣减余额
        BalanceDeductRequest balanceRequest = new BalanceDeductRequest();
        balanceRequest.setUserId(request.getUserId());
        balanceRequest.setAmount(request.getTotalAmount());
        
        ApiResult<Void> balanceResult = accountFeignClient.deductBalance(balanceRequest);
        if (!balanceResult.isSuccess()) {
            throw new BusinessException("余额不足");
        }
        log.info("余额扣减成功");
        
        // 4. 更新订单状态
        order.setStatus(OrderStatus.PAID);
        order.setPayTime(new Date());
        orderMapper.updateById(order);
        
        // 5. 发送订单创建消息
        sendOrderCreatedMessage(order);
        
        return ApiResult.success(OrderDTO.fromEntity(order));
    }
    
    /**
     * TCC模式实现
     */
    @TwoPhaseBusinessAction(name = "createOrderTcc", 
                           commitMethod = "commit", 
                           rollbackMethod = "rollback")
    public boolean prepareCreateOrder(BusinessActionContext actionContext, 
                                     OrderCreateRequest request) {
        // Try 阶段
        Order order = buildOrder(request);
        order.setStatus(OrderStatus.PREPARE);
        orderMapper.insert(order);
        
        // 保存上下文
        actionContext.getActionContext().put("order", order);
        
        return true;
    }
    
    public boolean commit(BusinessActionContext actionContext) {
        // Confirm 阶段
        Order order = (Order) actionContext.getActionContext().get("order");
        order.setStatus(OrderStatus.PAID);
        orderMapper.updateById(order);
        
        // 发送消息
        sendOrderCreatedMessage(order);
        
        return true;
    }
    
    public boolean rollback(BusinessActionContext actionContext) {
        // Cancel 阶段
        Order order = (Order) actionContext.getActionContext().get("order");
        
        if (order != null) {
            // 删除订单
            orderMapper.deleteById(order.getId());
            
            // 发送订单取消消息
            sendOrderCanceledMessage(order);
        }
        
        return true;
    }
    
    /**
     * Saga模式实现
     */
    @SagaStart
    public void createOrderSaga(OrderCreateRequest request) {
        Saga saga = Saga.newSaga("order.create")
            .activity("createOrder", this::createOrderActivity)
            .activity("deductStock", this::deductStockActivity)
            .activity("deductBalance", this::deductBalanceActivity)
            .compensation("cancelOrder", this::cancelOrderActivity)
            .build();
        
        saga.execute();
    }
    
    private boolean createOrderActivity(SagaContext context) {
        Order order = buildOrder((OrderCreateRequest) context.getParam("request"));
        orderMapper.insert(order);
        context.setParam("order", order);
        return true;
    }
    
    private boolean deductStockActivity(SagaContext context) {
        Order order = (Order) context.getParam("order");
        ProductDeductRequest request = new ProductDeductRequest();
        request.setProductId(order.getProductId());
        request.setQuantity(order.getQuantity());
        
        return productFeignClient.deductStock(request).isSuccess();
    }
    
    private boolean deductBalanceActivity(SagaContext context) {
        Order order = (Order) context.getParam("order");
        BalanceDeductRequest request = new BalanceDeductRequest();
        request.setUserId(order.getUserId());
        request.setAmount(order.getTotalAmount());
        
        return accountFeignClient.deductBalance(request).isSuccess();
    }
    
    private boolean cancelOrderActivity(SagaContext context) {
        Order order = (Order) context.getParam("order");
        if (order != null) {
            orderMapper.deleteById(order.getId());
        }
        return true;
    }
}
```

```
@Component
@Slf4j
public class RedisDistributedLock {
    
    @Autowired
    private RedissonClient redissonClient;
    
    @Autowired
    private StringRedisTemplate redisTemplate;
    
    /**
     * Redisson分布式锁
     */
    public <T> T executeWithLock(String lockKey, Supplier<T> supplier, 
                                long waitTime, long leaseTime, TimeUnit unit) {
        RLock lock = redissonClient.getLock(lockKey);
        
        try {
            boolean locked = lock.tryLock(waitTime, leaseTime, unit);
            if (!locked) {
                throw new DistributedLockException("获取锁失败: " + lockKey);
            }
            
            try {
                return supplier.get();
            } finally {
                if (lock.isHeldByCurrentThread()) {
                    lock.unlock();
                }
            }
        } catch (InterruptedException e) {
            Thread.currentThread().interrupt();
            throw new DistributedLockException("锁被中断", e);
        }
    }
    
    /**
     * Lua脚本实现分布式锁
     */
    public boolean tryLockWithLua(String lockKey, String requestId, 
                                 long expireTime, TimeUnit unit) {
        
        String luaScript = """
            if redis.call('setnx', KEYS[1], ARGV[1]) == 1 then
                redis.call('pexpire', KEYS[1], ARGV[2])
                return 1
            else
                if redis.call('get', KEYS[1]) == ARGV[1] then
                    redis.call('pexpire', KEYS[1], ARGV[2])
                    return 1
                end
                return 0
            end
            """;
        
        DefaultRedisScript<Long> script = new DefaultRedisScript<>();
        script.setScriptText(luaScript);
        script.setResultType(Long.class);
        
        Long result = redisTemplate.execute(
            script,
            Collections.singletonList(lockKey),
            requestId,
            String.valueOf(unit.toMillis(expireTime))
        );
        
        return result != null && result == 1;
    }
    
    /**
     * 释放锁
     */
    public boolean unlockWithLua(String lockKey, String requestId) {
        String luaScript = """
            if redis.call('get', KEYS[1]) == ARGV[1] then
                return redis.call('del', KEYS[1])
            else
                return 0
            end
            """;
        
        DefaultRedisScript<Long> script = new DefaultRedisScript<>();
        script.setScriptText(luaScript);
        script.setResultType(Long.class);
        
        Long result = redisTemplate.execute(
            script,
            Collections.singletonList(lockKey),
            requestId
        );
        
        return result != null && result == 1;
    }
    
    /**
     * 分段锁
     */
    public <T> T executeWithSegmentLock(String baseKey, String segmentKey, 
                                       Supplier<T> supplier, int segmentCount) {
        String lockKey = baseKey + ":" + (Math.abs(segmentKey.hashCode()) % segmentCount);
        return executeWithLock(lockKey, supplier, 3000, 10000, TimeUnit.MILLISECONDS);
    }
    
    /**
     * 读写锁
     */
    public <T> T executeWithReadWriteLock(String lockKey, Supplier<T> supplier, 
                                         boolean isWrite) {
        RReadWriteLock readWriteLock = redissonClient.getReadWriteLock(lockKey);
        RLock lock = isWrite ? readWriteLock.writeLock() : readWriteLock.readLock();
        
        try {
            lock.lock(10, TimeUnit.SECONDS);
            return supplier.get();
        } finally {
            if (lock.isHeldByCurrentThread()) {
                lock.unlock();
            }
        }
    }
    
    /**
     * 联锁
     */
    public <T> T executeWithMultiLock(List<String> lockKeys, Supplier<T> supplier) {
        List<RLock> locks = lockKeys.stream()
            .map(redissonClient::getLock)
            .collect(Collectors.toList());
        
        RLock multiLock = redissonClient.getMultiLock(locks.toArray(new RLock[0]));
        
        try {
            multiLock.lock(10, TimeUnit.SECONDS);
            return supplier.get();
        } finally {
            if (multiLock.isHeldByCurrentThread()) {
                multiLock.unlock();
            }
        }
    }
```

```
@Component
@Slf4j
public class MultiLevelCache {
    
    @Autowired
    private RedisTemplate<String, Object> redisTemplate;
    
    @Autowired
    private CaffeineCacheManager caffeineCacheManager;
    
    // 本地缓存配置
    private final Cache<String, CacheWrapper> localCache = Caffeine.newBuilder()
        .maximumSize(10000)
        .expireAfterWrite(60, TimeUnit.SECONDS)
        .recordStats()
        .build();
    
    /**
     * 多级缓存获取
     */
    public <T> T get(String key, Class<T> clazz, Supplier<T> loader, 
                    Duration localTtl, Duration redisTtl) {
        
        // 1. 尝试本地缓存
        CacheWrapper wrapper = localCache.getIfPresent(key);
        if (wrapper != null && !wrapper.isExpired()) {
            log.debug("命中本地缓存: {}", key);
            return (T) wrapper.getValue();
        }
        
        // 2. 尝试Redis缓存
        T value = getFromRedis(key, clazz);
        if (value != null) {
            // 回写到本地缓存
            localCache.put(key, new CacheWrapper(value, localTtl.toMillis()));
            return value;
        }
        
        // 3. 加分布式锁，防止缓存击穿
        String lockKey = "cache:lock:" + key;
        RLock lock = redissonClient.getLock(lockKey);
        
        try {
            if (lock.tryLock(3, 10, TimeUnit.SECONDS)) {
                // 双重检查
                value = getFromRedis(key, clazz);
                if (value != null) {
                    return value;
                }
                
                // 4. 从数据源加载
                value = loader.get();
                
                if (value != null) {
                    // 5. 异步写入多级缓存
                    asyncWriteToCaches(key, value, localTtl, redisTtl);
                } else {
                    // 缓存空值防止缓存穿透
                    cacheNullValue(key, localTtl, redisTtl);
                }
                
                return value;
            } else {
                // 获取锁失败，返回最后一次成功的缓存
                return getLastSuccessfulCache(key, clazz);
            }
        } catch (InterruptedException e) {
            Thread.currentThread().interrupt();
            return getFallbackValue(key, clazz);
        } finally {
            if (lock.isHeldByCurrentThread()) {
                lock.unlock();
            }
        }
    }
    
    /**
     * 异步写入多级缓存
     */
    private <T> void asyncWriteToCaches(String key, T value, 
                                       Duration localTtl, Duration redisTtl) {
        CompletableFuture.runAsync(() -> {
            try {
                // 写入本地缓存
                localCache.put(key, new CacheWrapper(value, localTtl.toMillis()));
                
                // 写入Redis
                redisTemplate.opsForValue().set(
                    key, 
                    value, 
                    redisTtl.toSeconds(), 
                    TimeUnit.SECONDS
                );
                
                // 写入二级缓存
                writeToL2Cache(key, value, redisTtl);
            } catch (Exception e) {
                log.error("写入缓存失败", e);
            }
        });
    }
    
    /**
     * 缓存预热
     */
    @Scheduled(fixedDelay = 300000)  // 每5分钟预热一次
    public void preloadHotData() {
        List<String> hotKeys = getHotKeys();
        
        hotKeys.parallelStream().forEach(key -> {
            if (!localCache.asMap().containsKey(key)) {
                // 异步预热
                CompletableFuture.runAsync(() -> {
                    try {
                        Object value = loadData(key);
                        if (value != null) {
                            localCache.put(key, new CacheWrapper(value, 300000L));
                        }
                    } catch (Exception e) {
                        log.error("缓存预热失败: {}", key, e);
                    }
                });
            }
        });
    }
    
    /**
     * 缓存一致性维护
     */
    @EventListener
    public void handleCacheInvalidation(CacheInvalidationEvent event) {
        String key = event.getKey();
        
        // 删除本地缓存
        localCache.invalidate(key);
        
        // 删除Redis缓存
        redisTemplate.delete(key);
        
        // 发布缓存失效消息
        redisTemplate.convertAndSend("cache:invalidation", key);
        
        log.info("缓存失效: {}", key);
    }
    
    /**
     * 缓存监控
     */

        com.github.benmanes.caffeine.cache.stats.CacheStats stats = localCache.stats();
        
        return CacheMetrics.builder()
            .hitCount(stats.hitCount())
            .missCount(stats.missCount())
            .loadSuccessCount(stats.loadSuccessCount())
            .loadFailureCount(stats.loadFailureCount())
            .totalLoadTime(stats.totalLoadTime())
            .evictionCount(stats.evictionCount())
            .evictionWeight(stats.evictionWeight())
            .build();
    }
    
    @Data
    @Builder
    public static class CacheWrapper {
        private Object value;
        private long timestamp;
        private long ttl;
        
        public boolean isExpired() {
            return System.currentTimeMillis() - timestamp > ttl;
        }
```

```
                        future.completeExceptionally(throwable);

                        future.complete(result);
                    }
                });

                ContextManager.stopSpan();
```
