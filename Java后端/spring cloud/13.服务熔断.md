# 服务熔断全方位解决方案

服务熔断是防止服务雪崩的关键技术，当某个服务失败率达到阈值时，自动熔断对它的调用，避免级联故障。

## 一、熔断器核心原理

### 1. **熔断器状态机**

```
public enum CircuitBreakerState {
    CLOSED,      // 关闭状态：正常请求
    OPEN,        // 开启状态：拒绝请求
    HALF_OPEN    // 半开状态：试探性请求
}
```

## 二、Resilience4j 熔断实现

### 1. **基础熔断配置**

```
import io.github.resilience4j.circuitbreaker.CircuitBreaker;
import io.github.resilience4j.circuitbreaker.CircuitBreakerConfig;
import io.github.resilience4j.circuitbreaker.CircuitBreakerRegistry;
import org.springframework.context.annotation.Bean;
import org.springframework.context.annotation.Configuration;
import java.time.Duration;

@Configuration
public class CircuitBreakerConfiguration {
    
    /**
     * 全局熔断器配置
     */
    @Bean
    public CircuitBreakerRegistry circuitBreakerRegistry() {
        // 默认配置
        CircuitBreakerConfig defaultConfig = CircuitBreakerConfig.custom()
            .slidingWindowType(CircuitBreakerConfig.SlidingWindowType.COUNT_BASED)  // 计数窗口
            .slidingWindowSize(100)                                                 // 滑动窗口大小
            .failureRateThreshold(50)                                               // 失败率阈值50%
            .slowCallRateThreshold(100)                                             // 慢调用率阈值100%
            .slowCallDurationThreshold(Duration.ofSeconds(2))                       // 慢调用时间阈值
            .waitDurationInOpenState(Duration.ofSeconds(30))                        // 熔断持续时间
            .permittedNumberOfCallsInHalfOpenState(10)                              // 半开状态允许请求数
            .minimumNumberOfCalls(20)                                               // 最小调用次数
            .recordExceptions(                                                      // 记录为失败的异常
                IOException.class, 
                TimeoutException.class,
                CallNotPermittedException.class
            )
            .ignoreExceptions(                                                      // 忽略的异常
                BusinessException.class,
                IllegalArgumentException.class
            )
            .enableAutomaticTransitionFromOpenToHalfOpen()                          // 自动切换到半开
            .writableStackTraceEnabled(false)                                       // 不写入堆栈跟踪
            .build();
        
        return CircuitBreakerRegistry.of(defaultConfig);
    }
    
    /**
     * 服务专用熔断器
     */
    @Bean
    public CircuitBreaker userServiceCircuitBreaker(CircuitBreakerRegistry registry) {
        return registry.circuitBreaker("user-service", 
            CircuitBreakerConfig.custom()
                .slidingWindowSize(50)
                .failureRateThreshold(30)
                .waitDurationInOpenState(Duration.ofSeconds(10))
                .build()
        );
    }
    
    /**
     * 外部API熔断器（更严格）
     */
    @Bean
    public CircuitBreaker externalApiCircuitBreaker(CircuitBreakerRegistry registry) {
        return registry.circuitBreaker("external-api",
            CircuitBreakerConfig.custom()
                .slidingWindowSize(20)
                .failureRateThreshold(20)
                .slowCallRateThreshold(50)
                .slowCallDurationThreshold(Duration.ofSeconds(3))
                .waitDurationInOpenState(Duration.ofSeconds(60))
                .minimumNumberOfCalls(5)
                .build()
        );
    }
}

### 2. **熔断器使用示例**

```
import io.github.resilience4j.circuitbreaker.CircuitBreaker;
import io.github.resilience4j.circuitbreaker.CallNotPermittedException;
import org.springframework.stereotype.Service;
import org.springframework.web.client.RestTemplate;
import java.util.function.Supplier;

@Service
@Slf4j
public class CircuitBreakerService {
    
    @Autowired
    private CircuitBreaker userServiceCircuitBreaker;
    
    @Autowired
    private RestTemplate restTemplate;
    
    @Autowired
    private UserCacheService userCacheService;
    
    /**
     * 基本熔断使用
     */
    public UserDTO getUserWithCircuitBreaker(Long userId) {
        Supplier<UserDTO> supplier = CircuitBreaker
            .decorateSupplier(userServiceCircuitBreaker, () -> {
                // 实际的远程调用
                return callUserService(userId);
            });
        
        try {
            return supplier.get();
        } catch (CallNotPermittedException e) {
            // 熔断器已打开
            log.warn("熔断器已打开，拒绝请求，userId: {}", userId);
            return getFallbackUser(userId);
        } catch (Exception e) {
            // 业务异常
            log.error("调用用户服务异常", e);
            return getFallbackUser(userId);
        }
    }
    
    /**
     * 带降级的熔断
     */
    public UserDTO getUserWithFallback(Long userId) {
        Supplier<UserDTO> decoratedSupplier = CircuitBreaker
            .decorateSupplier(userServiceCircuitBreaker, () -> callUserService(userId));
        
        return Try.ofSupplier(decoratedSupplier)
            .recover(CallNotPermittedException.class, e -> {
                log.warn("熔断降级，返回缓存数据");
                return userCacheService.getUserFromCache(userId);
            })
            .recover(TimeoutException.class, e -> {
                log.warn("超时降级");
                return UserDTO.builder()
                    .id(userId)
                    .name("请求超时")
                    .message("用户服务响应超时")
                    .build();
            })
            .recover(Throwable.class, e -> {
                log.error("其他异常降级", e);
                return getDefaultUser(userId);
            })
            .get();
    }
    
    /**
     * 异步熔断
     */
    public CompletableFuture<UserDTO> getUserAsync(Long userId) {
        Supplier<CompletableFuture<UserDTO>> supplier = 
            CircuitBreaker.decorateSupplier(userServiceCircuitBreaker, 
                () -> CompletableFuture.supplyAsync(() -> callUserService(userId)));
        
        return CompletableFuture.supplyAsync(() -> Try.ofSupplier(supplier)
            .recover(Throwable.class, e -> {
                log.error("异步调用失败", e);
                return CompletableFuture.completedFuture(getFallbackUser(userId));
            })
            .get())
            .thenCompose(Function.identity());
    }
    
    /**
     * 熔断器状态监控
     */
    public CircuitBreakerMetrics getCircuitBreakerMetrics() {
        CircuitBreaker.Metrics metrics = userServiceCircuitBreaker.getMetrics();
        CircuitBreaker.State state = userServiceCircuitBreaker.getState();
        
        return CircuitBreakerMetrics.builder()
            .name("user-service")
            .state(state.name())
            .failureRate(metrics.getFailureRate())
            .slowCallRate(metrics.getSlowCallRate())
            .totalCalls(metrics.getNumberOfSuccessfulCalls() + 
                       metrics.getNumberOfFailedCalls())
            .successfulCalls(metrics.getNumberOfSuccessfulCalls())
            .failedCalls(metrics.getNumberOfFailedCalls())
            .notPermittedCalls(metrics.getNumberOfNotPermittedCalls())
            .slowCalls(metrics.getNumberOfSlowCalls())
            .totalDuration(metrics.getTotalDuration())
            .build();
    }
    
    /**
     * 手动控制熔断器
     */
    public void manualControlCircuitBreaker(String action) {
        switch (action) {
            case "open":
                userServiceCircuitBreaker.transitionToOpenState();
                log.info("手动打开熔断器");
                break;
            case "close":
                userServiceCircuitBreaker.transitionToClosedState();
                log.info("手动关闭熔断器");
                break;
            case "half-open":
                userServiceCircuitBreaker.transitionToHalfOpenState();
                log.info("手动切换到半开状态");
                break;
            case "reset":
                userServiceCircuitBreaker.reset();
                log.info("重置熔断器");
                break;
        }
    }
    
    @Data
    @Builder
    public static class CircuitBreakerMetrics {
        private String name;
        private String state;
        private float failureRate;
        private float slowCallRate;
        private int totalCalls;
        private int successfulCalls;
        private int failedCalls;
        private int notPermittedCalls;
        private int slowCalls;
        private Duration totalDuration;
    }
}
```

### 3. **熔断器事件监听**

```
@Component
@Slf4j
public class CircuitBreakerEventListener {
    
    @Autowired
    private CircuitBreakerRegistry circuitBreakerRegistry;
    
    @PostConstruct
    public void registerEventListeners() {
        circuitBreakerRegistry.getAllCircuitBreakers()
            .forEach(this::registerListeners);
    }
    
    private void registerListeners(CircuitBreaker circuitBreaker) {
        circuitBreaker.getEventPublisher()
            .onSuccess(event -> {
                log.debug("熔断器成功事件: {}", event.getEventType());
            })
            .onError(event -> {
                log.warn("熔断器错误事件: {}, 异常: {}", 
                    event.getEventType(), event.getThrowable().getMessage());
            })
            .onStateTransition(event -> {
                log.info("熔断器状态变更: {} -> {}, 名称: {}", 
                    event.getStateTransition().getFromState(),
                    event.getStateTransition().getToState(),
                    event.getCircuitBreakerName());
                
                // 发送状态变更通知
                sendStateTransitionAlert(event);
            })
            .onCallNotPermitted(event -> {
                log.warn("熔断器拒绝调用: {}", event.getCircuitBreakerName());
                
                // 记录拒绝指标
                recordCallNotPermitted(event);
            })
            .onSlowCallRateExceeded(event -> {
                log.warn("慢调用率超过阈值: {}, 当前慢调用率: {}", 
                    event.getCircuitBreakerName(), 
                    event.getSlowCallRate());
            })
            .onFailureRateExceeded(event -> {
                log.error("失败率超过阈值: {}, 当前失败率: {}", 
                    event.getCircuitBreakerName(), 
                    event.getFailureRate());
                
                // 发送告警
                sendFailureRateAlert(event);
            });
    }
    
    /**
     * 发送状态变更告警
     */
    private void sendStateTransitionAlert(CircuitBreakerEvent event) {
        if (event instanceof CircuitBreakerOnStateTransitionEvent) {
            CircuitBreakerOnStateTransitionEvent transitionEvent = 
                (CircuitBreakerOnStateTransitionEvent) event;
            
            String message = String.format(
                "熔断器状态变更: %s [%s -> %s]", 
                event.getCircuitBreakerName(),
                transitionEvent.getStateTransition().getFromState(),
                transitionEvent.getStateTransition().getToState()
            );
            
            // 发送到监控系统
            alertService.sendAlert("熔断器状态变更", message, 
                transitionEvent.getStateTransition().getToState().name());
        }
    }
}

### 2. **Sentinel 熔断注解使用**

```
import com.alibaba.csp.sentinel.annotation.SentinelResource;
import com.alibaba.csp.sentinel.slots.block.BlockException;
import com.alibaba.csp.sentinel.slots.block.degrade.DegradeException;
import org.springframework.stereotype.Service;
import org.springframework.web.client.RestTemplate;

@Service
@Slf4j
public class SentinelCircuitBreakerService {
    
    @Autowired
    private RestTemplate restTemplate;
    
    @Autowired
    private UserCacheService userCacheService;
    
    /**
     * 慢调用比例熔断
     */
    @SentinelResource(
        value = "getUserById",
        blockHandler = "getUserBlockHandler",
        fallback = "getUserFallback",
        blockHandlerClass = SentinelBlockHandler.class
    )
    public UserDTO getUserById(Long userId) {
        // 模拟慢调用
        try {
            Thread.sleep(1500);  // 1.5秒
        } catch (InterruptedException e) {
            Thread.currentThread().interrupt();
        }
        
        return restTemplate.getForObject(
            "http://user-service/api/users/" + userId, 
            UserDTO.class
        );
    }
    
    /**
     * 异常比例熔断
     */
    @SentinelResource(
        value = "updateUser",
        blockHandler = "updateUserBlockHandler",
        fallback = "updateUserFallback"
    )
    public ApiResult updateUser(UserUpdateRequest request) {
        // 模拟随机异常
        if (Math.random() > 0.7) {
            throw new RuntimeException("随机异常");
        }
        
        return restTemplate.postForObject(
            "http://user-service/api/users/update",
            request,
            ApiResult.class
        );
    }
    
    /**
     * 异常数熔断
     */
    @SentinelResource(
        value = "deleteUser",
        blockHandler = "deleteUserBlockHandler"
    )
    public ApiResult deleteUser(Long userId) {
        if (userId <= 0) {
            throw new IllegalArgumentException("用户ID无效");
        }
        
        return restTemplate.postForObject(
            "http://user-service/api/users/delete",
            userId,
            ApiResult.class
        );
    }
    
    /**
     * 外部BlockHandler类
     */
    public static class SentinelBlockHandler {
        
        public static UserDTO getUserBlockHandler(Long userId, BlockException ex) {
            log.warn("Sentinel熔断降级，userId: {}, rule: {}", 
                userId, ex.getRule());
            
            if (ex instanceof DegradeException) {
                return UserDTO.builder()
                    .id(userId)
                    .name("熔断降级")
                    .message("用户服务暂时不可用")
                    .build();
            }
            
            return UserDTO.builder()
                .id(userId)
                .name("限流降级")
                .message("系统繁忙，请稍后重试")
                .build();
        }
    }
}

### 3. **Sentinel 动态规则管理**

```
@Component
@Slf4j
public class SentinelRuleManager {
    
    @Autowired
    private DegradeRuleManager degradeRuleManager;
    
    @Autowired
    private FlowRuleManager flowRuleManager;
    
    /**
     * 动态添加熔断规则
     */
    public void addDegradeRule(DegradeRule rule) {
        List<DegradeRule> rules = degradeRuleManager.getRules();
        rules.add(rule);
        degradeRuleManager.loadRules(rules);
        
        log.info("添加熔断规则: {}", rule);
    }
    
    /**
     * 动态更新熔断规则
     */
    public void updateDegradeRule(String resource, DegradeRule newRule) {
        List<DegradeRule> rules = degradeRuleManager.getRules();
        
        rules.removeIf(rule -> rule.getResource().equals(resource));
        rules.add(newRule);
        
        degradeRuleManager.loadRules(rules);
        
        log.info("更新熔断规则: {} -> {}", resource, newRule);
    }
    
    /**
     * 根据指标自动调整熔断规则
     */
    @Scheduled(fixedDelay = 30000)  // 每30秒调整一次
    public void autoAdjustDegradeRules() {
        degradeRuleManager.getRules().forEach(rule -> {
            try {
                // 获取资源统计
                DegradeRule degradeRule = rule;
                String resource = degradeRule.getResource();
                
                // 获取当前指标
                double failureRate = getFailureRate(resource);
                double slowCallRate = getSlowCallRate(resource);
                
                // 根据指标调整规则
                if (failureRate > 0.3) {
                    // 失败率高，降低阈值
                    degradeRule.setCount(degradeRule.getCount() * 0.8);
                    log.info("调整熔断阈值: {} -> {}", 
                        resource, degradeRule.getCount());
                } else if (failureRate < 0.1) {
                    // 失败率低，提高阈值
                    degradeRule.setCount(degradeRule.getCount() * 1.2);
                }
                
                // 重新加载规则
                List<DegradeRule> rules = degradeRuleManager.getRules();
                degradeRuleManager.loadRules(rules);
                
            } catch (Exception e) {
                log.error("调整熔断规则失败", e);
            }
        });
    }
    
    /**
     * 获取熔断器状态
     */
    public Map<String, CircuitBreakerStatus> getCircuitBreakerStatus() {
        Map<String, CircuitBreakerStatus> statusMap = new HashMap<>();
        
        degradeRuleManager.getRules().forEach(rule -> {
            try {
                // 获取熔断器
                CircuitBreaker circuitBreaker = DegradeRuleManager.getCircuitBreakers(
                    rule.getResource()).stream().findFirst().orElse(null);
                
                if (circuitBreaker != null) {
                    CircuitBreaker.Status status = circuitBreaker.currentState();
                    CircuitBreaker.Metrics metrics = circuitBreaker.getMetrics();
                    
                    CircuitBreakerStatus cbStatus = CircuitBreakerStatus.builder()
                        .resource(rule.getResource())
                        .state(status.name())
                        .failureRate(metrics.getFailureRate())
                        .slowCallRate(metrics.getSlowCallRate())
                        .totalCalls(metrics.getNumberOfSuccessfulCalls() + 
                                   metrics.getNumberOfFailedCalls())
                        .failedCalls(metrics.getNumberOfFailedCalls())
                        .slowCalls(metrics.getNumberOfSlowCalls())
                        .notPermittedCalls(metrics.getNumberOfNotPermittedCalls())
                        .build();
                    
                    statusMap.put(rule.getResource(), cbStatus);
                }
            } catch (Exception e) {
                log.error("获取熔断器状态失败", e);
            }
        });
        
        return statusMap;
    }
}

import org.springframework.cloud.client.circuitbreaker.CircuitBreakerFactory;

                    return getDefaultUser(userId);

### 3. **Circuit Breaker 监控端点**

```
@RestController
@RequestMapping("/actuator")
@Slf4j
public class CircuitBreakerEndpoint {
    
    @Autowired
    private CircuitBreakerFactory circuitBreakerFactory;
    
    @Autowired
    private HealthEndpoint healthEndpoint;
    
    /**
     * 获取熔断器状态
     */
    @GetMapping("/circuitbreakers")
    public Map<String, CircuitBreakerStatus> getCircuitBreakers() {
        Map<String, CircuitBreakerStatus> statusMap = new HashMap<>();
        
        // 获取健康状态
        Health health = healthEndpoint.health();
        Map<String, HealthComponent> components = health.getComponents();
        
        if (components != null && components.containsKey("circuitBreakers")) {
            HealthComponent cbHealth = components.get("circuitBreakers");
            if (cbHealth instanceof CompositeHealth) {
                CompositeHealth composite = (CompositeHealth) cbHealth;
                composite.getComponents().forEach((name, component) -> {
                    CircuitBreakerStatus status = CircuitBreakerStatus.builder()
                        .name(name)
                        .status(component.getStatus().getCode())
                        .details(component.getDetails())
                        .build();
                    statusMap.put(name, status);
                });
            }
        }
        
        return statusMap;
    }
    
    /**
     * 手动切换熔断器状态
     */
    @PostMapping("/circuitbreakers/{name}/state")
    public ApiResult<String> changeCircuitBreakerState(
            @PathVariable String name,
            @RequestParam String state) {
        
        try {
            // 这里需要根据具体的CircuitBreaker实现来切换状态
            // 以Resilience4j为例
            CircuitBreaker circuitBreaker = circuitBreakerFactory.create(name);
            if (circuitBreaker instanceof Resilience4JCircuitBreaker) {
                io.github.resilience4j.circuitbreaker.CircuitBreaker resilience4jBreaker = 
                    ((Resilience4JCircuitBreaker) circuitBreaker).getCircuitBreaker();
                
                switch (state.toUpperCase()) {
                    case "OPEN":
                        resilience4jBreaker.transitionToOpenState();
                        break;
                    case "CLOSED":
                        resilience4jBreaker.transitionToClosedState();
                        break;
                    case "HALF_OPEN":
                        resilience4jBreaker.transitionToHalfOpenState();
                        break;
                    case "FORCED_OPEN":
                        resilience4jBreaker.transitionToForcedOpenState();
                        break;
                    case "DISABLED":
                        resilience4jBreaker.transitionToDisabledState();
                        break;
                    default:
                        return ApiResult.error(400, "不支持的状态: " + state);
                }
                
                return ApiResult.success("熔断器状态已切换: " + state);
            }
            
            return ApiResult.error(400, "不支持的CircuitBreaker实现");
        } catch (Exception e) {
            return ApiResult.error(500, "切换状态失败: " + e.getMessage());
        }
    }
    
    /**
     * 获取熔断器指标
     */
    @GetMapping("/circuitbreakers/{name}/metrics")
    public CircuitBreakerMetrics getCircuitBreakerMetrics(@PathVariable String name) {
        CircuitBreaker circuitBreaker = circuitBreakerFactory.create(name);
        
        if (circuitBreaker instanceof Resilience4JCircuitBreaker) {
            io.github.resilience4j.circuitbreaker.CircuitBreaker resilience4jBreaker = 
                ((Resilience4JCircuitBreaker) circuitBreaker).getCircuitBreaker();
            
            io.github.resilience4j.circuitbreaker.CircuitBreaker.Metrics metrics = 
                resilience4jBreaker.getMetrics();
            
            return CircuitBreakerMetrics.builder()
                .name(name)
                .state(resilience4jBreaker.getState().name())
                .failureRate(metrics.getFailureRate())
                .slowCallRate(metrics.getSlowCallRate())
                .totalCalls(metrics.getNumberOfSuccessfulCalls() + 
                           metrics.getNumberOfFailedCalls())
                .successfulCalls(metrics.getNumberOfSuccessfulCalls())
                .failedCalls(metrics.getNumberOfFailedCalls())
                .notPermittedCalls(metrics.getNumberOfNotPermittedCalls())
                .slowCalls(metrics.getNumberOfSlowCalls())
                .build();
        }
        
        return null;
    }

## 五、熔断器监控与告警

### 1. **熔断器监控指标**

```
@Component
@Slf4j
public class CircuitBreakerMonitor {
    
    private final MeterRegistry meterRegistry;
    
    // 监控指标
    private final Map<String, CircuitBreakerStats> statsMap = new ConcurrentHashMap<>();
    
    public CircuitBreakerMonitor(MeterRegistry meterRegistry) {
        this.meterRegistry = meterRegistry;
        initMetrics();
    }
    
    private void initMetrics() {
        // 注册全局熔断器指标
        Gauge.builder("circuitbreaker.state", this::getOpenCircuitBreakerCount)
            .description("打开的熔断器数量")
            .register(meterRegistry);
            
        Gauge.builder("circuitbreaker.failure.rate", this::getAverageFailureRate)
            .description("平均失败率")
            .register(meterRegistry);
    }
    
    /**
     * 记录熔断事件
     */
    public void recordCircuitBreakerEvent(String circuitBreakerName, 
                                         String eventType, 
                                         Map<String, Object> metadata) {
        
        // 记录到Micrometer
        Counter.builder("circuitbreaker.events")
            .tag("name", circuitBreakerName)
            .tag("type", eventType)
            .register(meterRegistry)
            .increment();
        
        // 记录内部统计
        CircuitBreakerStats stats = statsMap.computeIfAbsent(
            circuitBreakerName, k -> new CircuitBreakerStats());
        
        stats.recordEvent(eventType, metadata);
        
        // 检查告警条件
        checkCircuitBreakerAlerts(circuitBreakerName, stats);
    }
    
    /**
     * 检查熔断器告警
     */
    private void checkCircuitBreakerAlerts(String circuitBreakerName, 
                                          CircuitBreakerStats stats) {
        
        // 检查失败率
        if (stats.getFailureRate() > 0.3) {
            log.warn("熔断器失败率高: {} - {}%", 
                circuitBreakerName, stats.getFailureRate() * 100);
            
            sendCircuitBreakerAlert(circuitBreakerName, "HIGH_FAILURE_RATE", 
                stats.getFailureRate());
        }
        
        // 检查状态变化频率
        if (stats.getStateChangeCount() > 5) {  // 短时间内状态变化频繁
            log.error("熔断器状态频繁变化: {} - 变化次数: {}", 
                circuitBreakerName, stats.getStateChangeCount());
            
            sendCircuitBreakerAlert(circuitBreakerName, "FREQUENT_STATE_CHANGES", 
                stats.getStateChangeCount());
        }
        
        // 检查长时间打开
        if (stats.isOpen() && stats.getOpenDuration() > 300000) {  // 5分钟
            log.error("熔断器长时间打开: {} - 持续时间: {}ms", 
                circuitBreakerName, stats.getOpenDuration());
            
            sendCircuitBreakerAlert(circuitBreakerName, "LONG_OPEN_DURATION", 
                stats.getOpenDuration());
        }
    }
    
    /**
     * 获取打开的熔断器数量
     */
    private int getOpenCircuitBreakerCount() {
        return (int) statsMap.values().stream()
            .filter(CircuitBreakerStats::isOpen)
            .count();
    }
    
    /**
     * 获取平均失败率
     */
    private double getAverageFailureRate() {
        return statsMap.values().stream()
            .mapToDouble(CircuitBreakerStats::getFailureRate)
            .average()
            .orElse(0.0);
    }
    
    /**
     * 获取Top N有问题的熔断器
     */
    public List<CircuitBreakerIssue> getTopCircuitBreakerIssues(int limit) {
        return statsMap.entrySet().stream()
            .map(entry -> CircuitBreakerIssue.builder()
                .name(entry.getKey())
                .failureRate(entry.getValue().getFailureRate())
                .stateChangeCount(entry.getValue().getStateChangeCount())
                .openDuration(entry.getValue().getOpenDuration())
                .isOpen(entry.getValue().isOpen())
                .build())
            .sorted(Comparator.comparing(CircuitBreakerIssue::getFailureRate).reversed())
            .limit(limit)
            .collect(Collectors.toList());
    }
    
    @Data
    public static class CircuitBreakerStats {
        private volatile boolean open = false;
        private long openTime = 0;
        private long stateChangeCount = 0;
        private long totalCalls = 0;
        private long failedCalls = 0;
        
        public void recordEvent(String eventType, Map<String, Object> metadata) {
            switch (eventType) {
                case "STATE_CHANGE":
                    stateChangeCount++;
                    if ("OPEN".equals(metadata.get("to"))) {
                        open = true;
                        openTime = System.currentTimeMillis();
                    } else if ("CLOSED".equals(metadata.get("to"))) {
                        open = false;
                        openTime = 0;
                    }
                    break;
                case "CALL_FAILED":
                    totalCalls++;
                    failedCalls++;
                    break;
                case "CALL_SUCCESS":
                    totalCalls++;
                    break;
            }
        }
        
        public double getFailureRate() {
            if (totalCalls == 0) {
                return 0.0;
            }
            return (double) failedCalls / totalCalls;
        }
        
        public long getOpenDuration() {
            if (open && openTime > 0) {
                return System.currentTimeMillis() - openTime;
            }
            return 0;
        }
    }
}
