# RabbitMQ 消费者失败重试机制

消费者失败重试机制是保证消息**可靠处理**的核心组件，确保在业务逻辑失败时能够**自动重试**，提高系统健壮性。

## 一、 基础重试机制

### 1. Spring Retry 集成配置

```
@Configuration
@EnableRetry
public class BasicRetryConfig {
    
    /**
     * 基础重试容器工厂
     */
    @Bean
    public SimpleRabbitListenerContainerFactory retryContainerFactory(
            ConnectionFactory connectionFactory) {
        
        SimpleRabbitListenerContainerFactory factory = new SimpleRabbitListenerContainerFactory();
        factory.setConnectionFactory(connectionFactory);
        
        // 1. 确认模式
        factory.setAcknowledgeMode(AcknowledgeMode.MANUAL);
        
        // 2. 配置重试拦截器
        factory.setAdviceChain(retryInterceptor());
        
        // 3. 预取数量
        factory.setPrefetchCount(10);
        
        return factory;
    }
    
    /**
     * 重试拦截器配置
     */
    @Bean
    public Advice retryInterceptor() {
        return RetryInterceptorBuilder.stateless()
            .maxAttempts(3)                     // 最大重试次数
            .backOffOptions(1000, 2.0, 10000)  // 初始1秒，倍数2，最大10秒
            .recoverer(new RejectAndDontRequeueRecoverer()) // 重试耗尽后的处理
            .build();
    }
    
    /**
     * 基础重试消费者
     */
    @Component
    public class BasicRetryConsumer {
        
        @RabbitListener(queues = "basic.retry.queue", 
                       containerFactory = "retryContainerFactory")
        public void handleWithRetry(String message,
                                   Channel channel,
                                   @Header(AmqpHeaders.DELIVERY_TAG) long deliveryTag) 
                throws IOException {
            
            System.out.println("处理消息: " + message);
            
            try {
                // 业务处理
                processMessage(message);
                
                // 确认消息
                channel.basicAck(deliveryTag, false);
                System.out.println("处理成功");
                
            } catch (Exception e) {
                System.err.println("处理失败: " + e.getMessage());
                // 抛出异常触发重试
                throw e;
            }
        }
        
        /**
         * 使用 @Retryable 注解
         */
        @RabbitListener(queues = "annotation.retry.queue")
        @Retryable(
            value = {Exception.class},
            maxAttempts = 3,
            backoff = @Backoff(delay = 1000, multiplier = 2.0)
        )
        public void handleWithAnnotation(String message,
                                        Channel channel,
                                        @Header(AmqpHeaders.DELIVERY_TAG) long deliveryTag) 
                throws IOException {
            
            System.out.println("@Retryable 处理: " + message);
            
            try {
                processMessage(message);
                channel.basicAck(deliveryTag, false);
                
            } catch (Exception e) {
                System.err.println("处理失败，触发重试: " + e.getMessage());
                throw e;
            }
        }
        
        /**
         * 恢复方法
         */
        @Recover
        public void recover(Exception e, String message, 
                          Channel channel, long deliveryTag) throws IOException {
            
            System.err.println("重试耗尽，发送到死信队列: " + message);
            
            // 拒绝消息，不重新入队
            channel.basicNack(deliveryTag, false, false);
            
            // 记录失败
            saveToFailedMessages(message, e);
        }
        
        private void processMessage(String message) throws Exception {
            // 模拟业务处理
            if (message.contains("fail")) {
                throw new Exception("模拟处理失败");
            }
            
            // 正常处理
            Thread.sleep(100);
        }
        
        private void saveToFailedMessages(String message, Exception e) {
            // 保存失败消息
            System.err.println("保存失败消息: " + message + ", 错误: " + e.getMessage());
        }
    }
}
```

### 2. 手动重试实现

```
@Component
public class ManualRetryHandler {
    
    private final Map<String, RetryContext> retryContexts = new ConcurrentHashMap<>();
    
    /**
     * 手动重试消费者
     */
    @RabbitListener(queues = "manual.retry.queue", 
                   ackMode = "MANUAL")
    public void handleWithManualRetry(String message,
                                     Channel channel,
                                     @Header(AmqpHeaders.DELIVERY_TAG) long deliveryTag,
                                     @Header(AmqpHeaders.MESSAGE_ID) String messageId) 
            throws IOException {
        
        // 获取或创建重试上下文
        RetryContext context = retryContexts.computeIfAbsent(
            messageId, k -> new RetryContext(message, 0, 3));
        
        try {
            System.out.println("第" + (context.getRetryCount() + 1) + "次尝试处理: " + message);
            
            // 业务处理
            processWithRetry(message, context.getRetryCount());
            
            // 处理成功
            channel.basicAck(deliveryTag, false);
            retryContexts.remove(messageId);
            
            System.out.println("处理成功，清理重试上下文");
            
        } catch (Exception e) {
            // 处理失败
            context.incrementRetry();
            System.err.println("处理失败，重试次数: " + context.getRetryCount());
            
            if (context.getRetryCount() >= context.getMaxRetries()) {
                // 达到最大重试次数
                handleMaxRetries(message, context, channel, deliveryTag, e);
                retryContexts.remove(messageId);
                
            } else {
                // 重新入队等待重试
                channel.basicNack(deliveryTag, false, true);
                
                // 记录重试延迟
                scheduleRetryMonitor(messageId, context);
            }
        }
    }
    
    /**
     * 处理达到最大重试次数
     */
    private void handleMaxRetries(String message, RetryContext context, 
                                 Channel channel, long deliveryTag, 
                                 Exception e) throws IOException {
        
        System.err.println("达到最大重试次数: " + message);
        
        // 发送到死信队列
        channel.basicNack(deliveryTag, false, false);
        
        // 记录失败
        saveToMaxRetries(message, context, e);
        
        // 发送告警
        sendRetryAlert(message, context, e);
    }
    
    /**
     * 重试监控调度
     */
    private void scheduleRetryMonitor(String messageId, RetryContext context) {
        ScheduledExecutorService scheduler = Executors.newSingleThreadScheduledExecutor();
        
        long delay = calculateRetryDelay(context.getRetryCount());
        
        scheduler.schedule(() -> {
            if (retryContexts.containsKey(messageId)) {
                System.out.println("重试监控: 消息 " + messageId + " 仍在重试中，次数: " + 
                                 context.getRetryCount());
                
                // 检查是否超时
                if (System.currentTimeMillis() - context.getCreateTime() > 300000) { // 5分钟
                    System.err.println("重试超时: " + messageId);
                    retryContexts.remove(messageId);
                }
            }
        }, delay, TimeUnit.MILLISECONDS);
    }
    
    /**
     * 计算重试延迟
     */
    private long calculateRetryDelay(int retryCount) {
        // 指数退避算法
        return Math.min(1000L * (1L << retryCount), 30000L); // 最大30秒
    }
    
    private void processWithRetry(String message, int retryCount) throws Exception {
        // 带重试的业务处理
        if (retryCount > 0) {
            // 可以根据重试次数调整处理逻辑
            System.out.println("第" + (retryCount + 1) + "次重试处理");
        }
        
        // 模拟处理
        if (Math.random() < 0.3) { // 30%失败率
            throw new Exception("模拟随机失败");
        }
        
        Thread.sleep(100);
    }
    
    private void saveToMaxRetries(String message, RetryContext context, Exception e) {
        // 保存达到最大重试的消息
        System.err.println("保存最大重试失败: " + message + 
                          ", 重试次数: " + context.getRetryCount());
    }
    
    private void sendRetryAlert(String message, RetryContext context, Exception e) {
        // 发送重试告警
        System.err.println("发送重试告警: 消息 " + message + 
                          " 达到最大重试次数 " + context.getRetryCount());
    }
    
    /**
     * 重试上下文
     */
    @Data
    public static class RetryContext {
        private final String message;
        private int retryCount;
        private final int maxRetries;
        private final long createTime = System.currentTimeMillis();
        private long lastRetryTime = System.currentTimeMillis();
        
        public RetryContext(String message, int retryCount, int maxRetries) {
            this.message = message;
            this.retryCount = retryCount;
            this.maxRetries = maxRetries;
        }
        
        public void incrementRetry() {
            this.retryCount++;
            this.lastRetryTime = System.currentTimeMillis();
        }
    }
}
```

## 二、 高级重试策略

### 1. 智能重试策略

```
@Component
public class SmartRetryStrategy {
    
    private final Map<String, MessageRetryStats> retryStats = new ConcurrentHashMap<>();
    
    /**
     * 智能重试消费者
     */
    @RabbitListener(queues = "smart.retry.queue", 
                   ackMode = "MANUAL")
    public void handleWithSmartRetry(String message,
                                    Channel channel,
                                    @Header(AmqpHeaders.DELIVERY_TAG) long deliveryTag,
                                    @Header(AmqpHeaders.CONSUMER_QUEUE) String queue,
                                    @Header(AmqpHeaders.MESSAGE_ID) String messageId) 
            throws IOException {
        
        MessageRetryStats stats = retryStats.computeIfAbsent(
            messageId, k -> new MessageRetryStats(messageId, queue));
        
        try {
            // 检查是否应该重试
            if (!shouldRetry(message, stats)) {
                handleNoRetry(message, stats, channel, deliveryTag);
                return;
            }
            
            // 智能处理
            processIntelligently(message, stats);
            
            // 处理成功
            channel.basicAck(deliveryTag, false);
            stats.recordSuccess();
            retryStats.remove(messageId);
            
        } catch (Exception e) {
            // 处理失败
            stats.recordFailure(e);
            
            // 智能决策：是否重新入队
            RetryDecision decision = makeRetryDecision(message, e, stats);
            
            switch (decision.getAction()) {
                case RETRY:
                    // 重新入队
                    channel.basicNack(deliveryTag, false, true);
                    stats.incrementRetryCount();
                    
                    // 记录重试
                    logRetry(stats, decision.getDelay());
                    break;
                    
                case DEAD_LETTER:
                    // 发送到死信队列
                    channel.basicNack(deliveryTag, false, false);
                    retryStats.remove(messageId);
                    
                    // 记录死信
                    logDeadLetter(message, stats, e);
                    break;
                    
                case DISCARD:
                    // 丢弃消息
                    channel.basicAck(deliveryTag, false);
                    retryStats.remove(messageId);
                    
                    // 记录丢弃
                    logDiscard(message, stats, e);
                    break;
            }
        }
    }
    
    /**
     * 智能判断是否重试
     */
    private boolean shouldRetry(String message, MessageRetryStats stats) {
        // 1. 检查最大重试次数
        if (stats.getRetryCount() >= getMaxRetriesForMessage(message)) {
            return false;
        }
        
        // 2. 检查异常类型
        if (stats.getLastException() instanceof IllegalArgumentException) {
            return false; // 参数错误不重试
        }
        
        // 3. 检查消息内容
        if (message.contains("no-retry")) {
            return false; // 特定消息不重试
        }
        
        // 4. 检查重试频率
        if (stats.getRetryCount() > 0) {
            long timeSinceLastRetry = System.currentTimeMillis() - stats.getLastRetryTime();
            long minInterval = getMinRetryInterval(stats.getRetryCount());
            if (timeSinceLastRetry < minInterval) {
                return false; // 重试间隔太短
            }
        }
        
        return true;
    }
    
    /**
     * 智能决策
     */
    private RetryDecision makeRetryDecision(String message, Exception e, 
                                          MessageRetryStats stats) {
        
        // 1. 基于异常类型
        if (e instanceof IllegalArgumentException) {
            return new RetryDecision(Action.DISCARD, 0, "参数错误");
        }
        
        if (e instanceof IOException || e instanceof TimeoutException) {
            return new RetryDecision(Action.RETRY, 
                calculateRetryDelay(stats.getRetryCount()), "网络异常");
        }
        
        // 2. 基于重试次数
        if (stats.getRetryCount() >= 5) {
            return new RetryDecision(Action.DEAD_LETTER, 0, "达到最大重试次数");
        }
        
        // 3. 基于消息内容
        if (message.contains("important")) {
            return new RetryDecision(Action.RETRY, 
                calculateRetryDelay(stats.getRetryCount()), "重要消息");
        }
        
        // 4. 基于时间
        if (System.currentTimeMillis() - stats.getCreateTime() > 300000) { // 5分钟
            return new RetryDecision(Action.DEAD_LETTER, 0, "处理超时");
        }
        
        // 默认重试
        return new RetryDecision(Action.RETRY, 
            calculateRetryDelay(stats.getRetryCount()), "默认重试");
    }
    
    /**
     * 智能处理
     */
    private void processIntelligently(String message, MessageRetryStats stats) 
            throws Exception {
        
        int retryCount = stats.getRetryCount();
        
        if (retryCount > 0) {
            // 如果是重试，可以调整处理策略
            System.out.println("第" + (retryCount + 1) + "次重试处理");
            
            // 可以降级处理
            if (retryCount >= 2) {
                processWithDegradation(message);
                return;
            }
        }
        
        // 正常处理
        processMessage(message);
    }
    
    private int getMaxRetriesForMessage(String message) {
        // 根据消息内容决定最大重试次数
        if (message.contains("urgent")) {
            return 3;
        } else if (message.contains("normal")) {
            return 5;
        } else {
            return 10;
        }
    }
    
    private long getMinRetryInterval(int retryCount) {
        // 最小重试间隔
        return Math.min(1000L * retryCount, 10000L);
    }
    
    private long calculateRetryDelay(int retryCount) {
        // 指数退避 + 随机抖动
        long baseDelay = 1000L * (1L << Math.min(retryCount, 10));
        long jitter = (long) (Math.random() * 1000);
        return Math.min(baseDelay + jitter, 30000L);
    }
    
    private void handleNoRetry(String message, MessageRetryStats stats, 
                              Channel channel, long deliveryTag) throws IOException {
        
        System.err.println("不重试消息: " + message);
        
        if (stats.getLastException() instanceof IllegalArgumentException) {
            // 参数错误，丢弃
            channel.basicAck(deliveryTag, false);
        } else {
            // 其他错误，发送到死信
            channel.basicNack(deliveryTag, false, false);
        }
        
        retryStats.remove(stats.getMessageId());
    }
    
    private void processMessage(String message) throws Exception {
        // 正常处理逻辑
        if (Math.random() < 0.2) {
            throw new IOException("模拟处理失败");
        }
        Thread.sleep(100);
    }
    
    private void processWithDegradation(String message) throws Exception {
        // 降级处理逻辑
        System.out.println("使用降级处理: " + message);
        Thread.sleep(50);
    }
    
    private void logRetry(MessageRetryStats stats, long delay) {
        System.out.println("安排重试: " + stats.getMessageId() + 
                          ", 第" + stats.getRetryCount() + "次, 延迟" + delay + "ms");
    }
    
    private void logDeadLetter(String message, MessageRetryStats stats, Exception e) {
        System.err.println("发送到死信队列: " + message + 
                          ", 重试次数: " + stats.getRetryCount());
    }
    
    private void logDiscard(String message, MessageRetryStats stats, Exception e) {
        System.err.println("丢弃消息: " + message + 
                          ", 原因: " + e.getMessage());
    }
    
    /**
     * 重试决策
     */
    @Data
    @AllArgsConstructor
    public static class RetryDecision {
        private Action action;
        private long delay; // 延迟时间(ms)
        private String reason;
        
        public enum Action {
            RETRY,      // 重试
            DEAD_LETTER, // 发送到死信队列
            DISCARD     // 丢弃
        }
    }
    
    /**
     * 消息重试统计
     */
    @Data
    public static class MessageRetryStats {
        private final String messageId;
        private final String queue;
        private int retryCount = 0;
        private Exception lastException;
        private final long createTime = System.currentTimeMillis();
        private long lastRetryTime = 0;
        
        public void recordSuccess() {
            this.retryCount = 0;
            this.lastException = null;
        }
        
        public void recordFailure(Exception e) {
            this.lastException = e;
        }
        
        public void incrementRetryCount() {
            this.retryCount++;
            this.lastRetryTime = System.currentTimeMillis();
        }
    }
}
```

### 2. 基于时间的重试策略

```
@Component
public class TimeBasedRetryStrategy {
    
    private final ScheduledExecutorService retryScheduler = 
        Executors.newScheduledThreadPool(5);
    
    private final Map<String, ScheduledFuture<?>> scheduledRetries = 
        new ConcurrentHashMap<>();
    
    /**
     * 时间重试消费者
     */
    @RabbitListener(queues = "time.retry.queue", 
                   ackMode = "MANUAL")
    public void handleWithTimeRetry(String message,
                                   Channel channel,
                                   @Header(AmqpHeaders.DELIVERY_TAG) long deliveryTag,
                                   @Header(AmqpHeaders.MESSAGE_ID) String messageId) 
            throws IOException {
        
        try {
            processMessage(message);
            channel.basicAck(deliveryTag, false);
            
            // 取消可能的预定重试
            cancelScheduledRetry(messageId);
            
        } catch (Exception e) {
            System.err.println("处理失败: " + e.getMessage());
            
            // 立即拒绝，不重新入队
            channel.basicNack(deliveryTag, false, false);
            
            // 安排延迟重试
            scheduleDelayedRetry(message, messageId, e, 1);
        }
    }
    
    /**
     * 安排延迟重试
     */
    private void scheduleDelayedRetry(String message, String messageId, 
                                     Exception e, int retryCount) {
        
        if (retryCount > 5) { // 最大重试5次
            System.err.println("达到最大重试次数: " + message);
            saveToPermanentFailure(message, e, retryCount);
            return;
        }
        
        long delay = calculateTimeBasedDelay(retryCount);
        
        System.out.println("安排延迟重试: " + messageId + 
                          ", 第" + retryCount + "次, 延迟" + delay + "ms");
        
        ScheduledFuture<?> future = retryScheduler.schedule(() -> {
            try {
                executeDelayedRetry(message, messageId, retryCount);
            } catch (Exception ex) {
                System.err.println("延迟重试执行失败: " + ex.getMessage());
            } finally {
                scheduledRetries.remove(messageId);
            }
        }, delay, TimeUnit.MILLISECONDS);
        
        scheduledRetries.put(messageId, future);
    }
    
    /**
     * 执行延迟重试
     */
    private void executeDelayedRetry(String message, String messageId, int retryCount) {
        System.out.println("执行延迟重试: " + messageId + ", 第" + retryCount + "次");
        
        try {
            processWithRetry(message, retryCount);
            System.out.println("延迟重试成功: " + messageId);
            
        } catch (Exception e) {
            System.err.println("延迟重试失败: " + e.getMessage());
            
            // 安排下一次重试
            scheduleDelayedRetry(message, messageId, e, retryCount + 1);
        }
    }
    
    /**
     * 基于时间的延迟计算
     */
    private long calculateTimeBasedDelay(int retryCount) {
        // 策略1: 固定间隔
        // return 5000L; // 固定5秒
        
        // 策略2: 线性增长
        // return 1000L * retryCount; // 1秒 * 重试次数
        
        // 策略3: 指数退避
        long baseDelay = 1000L * (1L << Math.min(retryCount, 6)); // 最大64秒
        
        // 策略4: 工作日/周末不同策略
        Calendar cal = Calendar.getInstance();
        int dayOfWeek = cal.get(Calendar.DAY_OF_WEEK);
        
        if (dayOfWeek == Calendar.SATURDAY || dayOfWeek == Calendar.SUNDAY) {
            baseDelay *= 2; // 周末加倍
        }
        
        // 策略5: 基于时间段的策略
        int hour = cal.get(Calendar.HOUR_OF_DAY);
        if (hour >= 9 && hour <= 18) { // 工作时间
            baseDelay = Math.min(baseDelay, 10000L); // 最大10秒
        } else { // 非工作时间
            baseDelay = Math.min(baseDelay, 30000L); // 最大30秒
        }
        
        return Math.min(baseDelay, 300000L); // 最大5分钟
    }
    
    /**
     * 取消预定重试
     */
    private void cancelScheduledRetry(String messageId) {
        ScheduledFuture<?> future = scheduledRetries.remove(messageId);
        if (future != null && !future.isDone()) {
            future.cancel(false);
            System.out.println("取消预定重试: " + messageId);
        }
    }
    
    private void processMessage(String message) throws Exception {
        if (Math.random() < 0.4) {
            throw new Exception("模拟处理失败");
        }
        Thread.sleep(100);
    }
    
    private void processWithRetry(String message, int retryCount) throws Exception {
        System.out.println("第" + retryCount + "次重试处理");
        
        if (retryCount >= 3) {
            // 第三次重试使用降级策略
            processWithDegradation(message);
        } else {
            processMessage(message);
        }
    }
    
    private void processWithDegradation(String message) throws Exception {
        // 降级处理逻辑
        System.out.println("使用降级处理: " + message);
        Thread.sleep(50);
    }
    
    private void saveToPermanentFailure(String message, Exception e, int retryCount) {
        // 保存到永久失败记录
        System.err.println("保存永久失败: " + message + 
                          ", 重试次数: " + retryCount);
    }
    
    /**
     * 清理过期的重试任务
     */
    @Scheduled(fixedDelay = 300000) // 每5分钟清理一次
    public void cleanupExpiredRetries() {
        long now = System.currentTimeMillis();
        
        scheduledRetries.entrySet().removeIf(entry -> {
            ScheduledFuture<?> future = entry.getValue();
            if (future.isDone() || future.isCancelled()) {
                return true;
            }
            
            // 可以添加更复杂的清理逻辑
            return false;
        });
    }
}
```

## 三、 重试与死信队列集成

### 1. 重试+死信完整方案

```
@Configuration
public class RetryWithDLQConfig {
    
    /**
     * 主队列配置（带重试）
     */
    @Bean
    public Queue retryMainQueue() {
        Map<String, Object> args = new HashMap<>();
        
        // 死信配置
```

```
@Component
public class RetryMonitorSystem {
    
    private final Map<String, QueueRetryStats> queueStats = new ConcurrentHashMap<>();
    private final MeterRegistry meterRegistry;
    
    private final Counter retryCounter;
    private final Counter successAfterRetryCounter;
    private final Counter failureAfterRetryCounter;
    private final Timer retryTimer;
    
    public RetryMonitorSystem(MeterRegistry meterRegistry) {
        this.meterRegistry = meterRegistry;
        
        // 注册指标
        this.retryCounter = Counter.builder("rabbitmq.retry.count")
            .description("消息重试次数")
            .register(meterRegistry);
        
        this.successAfterRetryCounter = Counter.builder("rabbitmq.retry.success.count")
            .description("重试后成功次数")
            .register(meterRegistry);
        
        this.failureAfterRetryCounter = Counter.builder("rabbitmq.retry.failure.count")
            .description("重试后失败次数")
            .register(meterRegistry);
        
        this.retryTimer = Timer.builder("rabbitmq.retry.duration")
            .description("重试耗时")
            .register(meterRegistry);
    }
    
    /**
     * 记录重试开始
     */
    public RetryContext startRetry(String messageId, String queue, int retryCount) {
        RetryContext context = new RetryContext(messageId, queue, retryCount);
        
        QueueRetryStats stats = queueStats.computeIfAbsent(queue, k -> new QueueRetryStats());
        stats.recordRetryStart();
        
        retryCounter.increment();
        
        return context;
    }
    
    /**
     * 记录重试结果
     */
    public void recordRetryResult(RetryContext context, boolean success, 
                                 Exception exception, long duration) {
        
        QueueRetryStats stats = queueStats.get(context.getQueue());
        if (stats != null) {
            stats.recordRetryResult(success, duration);
        }
        
        if (success) {
            successAfterRetryCounter.increment();
        } else {
            failureAfterRetryCounter.increment();
        }
        
        retryTimer.record(duration, TimeUnit.MILLISECONDS);
        
        // 记录详细日志
        logRetryResult(context, success, exception, duration);
        
        // 检查是否需要告警
        checkForAlerts(context.getQueue(), stats);
    }
    
    /**
     * 获取队列重试统计
     */
    public QueueRetryMetrics getQueueMetrics(String queue) {
        QueueRetryStats stats = queueStats.get(queue);
        if (stats != null) {
            return stats.getMetrics();
        }
        return new QueueRetryMetrics();
    }
    
    /**
     * 健康检查
     */
    public Health retryHealthCheck() {
        Map<String, Object> details = new HashMap<>();
        
        double overallSuccessRate = queueStats.values().stream()
            .mapToDouble(s -> s.getMetrics().getSuccessRate())
            .average()
            .orElse(1.0);
        
        details.put("overallRetrySuccessRate", overallSuccessRate);
        details.put("totalRetries", retryCounter.count());
        
        queueStats.forEach((queue, stats) -> {
            QueueRetryMetrics metrics = stats.getMetrics();
            details.put(queue + ".retryCount", metrics.getRetryCount());
            details.put(queue + ".successRate", metrics.getSuccessRate());
            details.put(queue + ".avgRetryDuration", metrics.getAvgRetryDuration());
        });
        
        // 检查重试成功率
        if (overallSuccessRate < 0.5) {
            return Health.down()
                .withDetail("error", "重试成功率过低: " + overallSuccessRate)
                .withDetails(details)
                .build();
        }
        
        return Health.up().withDetails(details).build();
    }
    
    /**
     * 检查告警
     */
    private void checkForAlerts(String queue, QueueRetryStats stats) {
        QueueRetryMetrics metrics = stats.getMetrics();
        
        // 1. 检查重试频率
        if (metrics.getRetryRatePerMinute() > 10) {
            sendAlert(queue, "HIGH_RETRY_RATE", 
                "队列 " + queue + " 重试频率过高: " + metrics.getRetryRatePerMinute() + "/分钟");
        }
        
        // 2. 检查重试成功率
        if (metrics.getSuccessRate() < 0.3) {
            sendAlert(queue, "LOW_RETRY_SUCCESS_RATE", 
                "队列 " + queue + " 重试成功率过低: " + metrics.getSuccessRate());
        }
        
        // 3. 检查平均重试时长
        if (metrics.getAvgRetryDuration() > 10000) { // 超过10秒
            sendAlert(queue, "HIGH_RETRY_DURATION", 
                "队列 " + queue + " 平均重试时长过高: " + metrics.getAvgRetryDuration() + "ms");
        }
    }
    
    private void logRetryResult(RetryContext context, boolean success, 
                               Exception exception, long duration) {
        
        String level = success ? "INFO" : "ERROR";
        System.out.printf("[%s] 重试结果 - 消息: %s, 队列: %s, 重试次数: %d, 结果: %s, 耗时: %dms, 错误: %s%n",
            level, context.getMessageId(), context.getQueue(), 
            context.getRetryCount(), success ? "成功" : "失败", 
            duration, exception != null ? exception.getMessage() : "N/A");
    }

        // alertSystem.sendAlert(queue, type, message);

     * 重试上下文
     */
    @Data
    @AllArgsConstructor
    public static class RetryContext {
        private String messageId;
        private String queue;
        private int retryCount;
        private long startTime = System.currentTimeMillis();
    }
    
    /**
     * 队列重试统计
     */
    public static class QueueRetryStats {
        private final AtomicInteger retryCount = new AtomicInteger();
        private final AtomicInteger successCount = new AtomicInteger();
        private final AtomicLong totalRetryDuration = new AtomicLong();
        private final AtomicLong lastMinuteRetryCount = new AtomicLong();
        private final AtomicLong lastMinuteStart = new AtomicLong(System.currentTimeMillis());
        
        synchronized void recordRetryStart() {
            retryCount.incrementAndGet();
            lastMinuteRetryCount.incrementAndGet();
            
            // 每分钟重置
            if (System.currentTimeMillis() - lastMinuteStart.get() > 60000) {
                lastMinuteRetryCount.set(1);
                lastMinuteStart.set(System.currentTimeMillis());
            }
        }
        
        synchronized void recordRetryResult(boolean success, long duration) {
            if (success) {
                successCount.incrementAndGet();
            }
            totalRetryDuration.addAndGet(duration);
        }
        
        QueueRetryMetrics getMetrics() {
            QueueRetryMetrics metrics = new QueueRetryMetrics();
            metrics.setRetryCount(retryCount.get());
            metrics.setSuccessCount(successCount.get());
            
            if (retryCount.get() > 0) {
                metrics.setSuccessRate((double) successCount.get() / retryCount.get());
                metrics.setAvgRetryDuration(totalRetryDuration.get() / retryCount.get());
            } else {
                metrics.setSuccessRate(1.0);
                metrics.setAvgRetryDuration(0);
            }
            
            metrics.setRetryRatePerMinute(lastMinuteRetryCount.get());
            
            return metrics;
        }
    }
    
    /**
     * 队列重试指标
     */
    @Data
    public static class QueueRetryMetrics {
        private int retryCount = 0;
        private int successCount = 0;
        private double successRate = 1.0;
        private long avgRetryDuration = 0;
        private long retryRatePerMinute = 0;
    }
```

```
public class ProductionRetryConfig {

        retryableExceptions.put(IOException.class, true);
        retryableExceptions.put(TimeoutException.class, true);
        retryableExceptions.put(SocketException.class, true);
        
        // 数据库异常重试
        retryableExceptions.put(SQLException.class, true);
        retryableExceptions.put(DataAccessException.class, true);
        
        // 业务异常不重试
        retryableExceptions.put(IllegalArgumentException.class, false);
        retryableExceptions.put(IllegalStateException.class, false);
        
        SimpleRetryPolicy policy = new SimpleRetryPolicy(3, retryableExceptions, true);
        policy.setMaxAttempts(3);
        
        return policy;

     * 生产环境错误处理器

    private ErrorHandler productionErrorHandler() {
        return new ConditionalRejectingErrorHandler(

                @Override
                public boolean isFatal(Throwable t) {
                    // 这些异常不重试，直接发送到死信

        public void recover(Message message, Throwable cause) {

            String queue = (String) message.getMessageProperties().getHeaders()

                retryMonitor.recordRetryResult(context, false, 
                    cause instanceof Exception ? (Exception) cause : new Exception(cause), 0);
            }

            // 记录到监控

    @Service

         * 生产环境处理方法
         */

                       containerFactory = "productionRetryContainerFactory")
        public void handleProductionWithRetry(String message,
                                             Channel channel,
                                             @Header(AmqpHeaders.DELIVERY_TAG) long deliveryTag,
                                             @Header(AmqpHeaders.CONSUMER_QUEUE) String queue,

            try {
                // 业务处理
                processBusiness(message);
                
                // 确认消息
                channel.basicAck(deliveryTag, false);
                
                // 记录成功
                if (context != null) {
                    retryMonitor.recordRetryResult(context, true, null, 
                        System.currentTimeMillis() - startTime);
                }
                
            } catch (Exception e) {
                // 记录重试开始
                context = retryMonitor.startRetry(messageId, queue, 1);

                throw e;

        @RabbitListener(queues = "batch.production.retry.queue", 
                       containerFactory = "productionRetryContainerFactory")
        public void handleBatchWithRetry(List<Message> messages, Channel channel) 
                throws IOException {
            
            List<Long> successTags = new ArrayList<>();
            List<Long> failureTags = new ArrayList<>();
            
            for (Message message : messages) {
                long deliveryTag = message.getMessageProperties().getDeliveryTag();
                String messageId = message.getMessageProperties().getMessageId();
                
                try {
                    String body = new String(message.getBody());
                    processBusiness(body);
                    successTags.add(deliveryTag);

                channel.basicAck(lastTag, true);
            }
            
            // 批量拒绝失败（触发重试）
            if (!failureTags.isEmpty()) {
                long lastTag = failureTags.get(failureTags.size() - 1);
                channel.basicNack(lastTag, true, true);
            }
        }
        
        private void processBusiness(String message) throws Exception {

            }
```

## 六、 常见问题

### Q1: 重试次数设置多少合适？

- 网络异常：3-5次
    
- 业务异常：1-3次
    
- 重要消息：5-10次
    
- 实时消息：1-2次
    

### Q2: 如何避免无限重试？

- 设置最大重试次数
    
- 使用死信队列
    
- 监控重试次数
    
- 设置消息TTL
