# RabbitMQ 发送者重连机制

发送者重连机制是确保 RabbitMQ 消息可靠性的重要部分，包括**连接恢复、消息重试、故障转移**等功能。以下是完整的发送者重连实现方案。

## 一、 基础重连配置

### 1. ConnectionFactory 自动重连配置

```
public class ReconnectConfig {

     * 配置自动重连的连接工厂
     */
    @Bean
    public ConnectionFactory connectionFactory() {
        CachingConnectionFactory factory = new CachingConnectionFactory();

        factory.setHost("localhost");

        factory.setPassword("guest");
        factory.setVirtualHost("/");

        factory.setAutomaticRecoveryEnabled(true);

        factory.setNetworkRecoveryInterval(5000); // 5秒重试一次
        
        // 3. 连接超时（毫秒）
        factory.setConnectionTimeout(30000); // 30秒
        
        // 4. 握手超时（毫秒）
        factory.setHandshakeTimeout(10000); // 10秒
        
        // 5. 通道检查超时（毫秒）
        factory.setChannelCheckoutTimeout(2000);
        
        // 6. 连接心跳（秒）
        factory.setRequestedHeartbeat(60); // 60秒心跳
        
        // 7. 拓扑恢复（自动重新声明交换机和队列）
        factory.setTopologyRecoveryEnabled(true);

        factory.addConnectionListener(new ConnectionListener() {
            @Override
            public void onCreate(Connection connection) {
                System.out.println("连接创建成功: " + connection);
            }
            
            @Override
            public void onClose(Connection connection) {
                System.out.println("连接关闭: " + connection);
            }
            
            @Override
            public void onShutDown(ShutdownSignalException signal) {
                System.out.println("连接异常关闭: " + signal.getReason());
            }
        });
        
        // 9. 阻塞监听器
        factory.addBlockedListener(new BlockedListener() {
            @Override
            public void handleBlocked(String reason) {
                System.out.warn("连接被阻塞: " + reason);
            }
            
            @Override
            public void handleUnblocked() {
                System.out.info("连接解除阻塞");
            }
        });
        
        // 10. 通道缓存配置
        factory.setChannelCacheSize(25); // 通道缓存大小
        factory.setChannelCheckoutTimeout(2000); // 获取通道超时时间
        
        return factory;
    }
    
    /**
     * 配置重试策略
     */
    @Bean
    public RetryTemplate retryTemplate() {
        RetryTemplate retryTemplate = new RetryTemplate();
        
        // 重试策略：最多重试3次

        retryPolicy.setMaxAttempts(3);
        retryTemplate.setRetryPolicy(retryPolicy);
        
        // 退避策略：指数退避
        ExponentialBackOffPolicy backOffPolicy = new ExponentialBackOffPolicy();
        backOffPolicy.setInitialInterval(1000);  // 初始间隔1秒
        backOffPolicy.setMultiplier(2.0);       // 倍数
        backOffPolicy.setMaxInterval(10000);    // 最大间隔10秒
        retryTemplate.setBackOffPolicy(backOffPolicy);
        
        return retryTemplate;
    }
```

### 2. 生产者重连模板

```
@Component
public class ReconnectProducer {
    
    @Autowired
    private RabbitTemplate rabbitTemplate;
    
    @Autowired
    private RetryTemplate retryTemplate;
    
    /**
     * 基本发送（带重试）
     */
    public void sendWithRetry(String exchange, String routingKey, Object message) {
        retryTemplate.execute(context -> {
            try {
                System.out.println("尝试发送消息，重试次数: " + context.getRetryCount());
                rabbitTemplate.convertAndSend(exchange, routingKey, message);
                System.out.println("消息发送成功");
                return null;
            } catch (Exception e) {
                System.err.println("消息发送失败: " + e.getMessage());
                throw e; // 抛出异常触发重试
            }
        });
    }
    
    /**
     * 发送带确认的消息
     */
    public void sendWithConfirm(String exchange, String routingKey, Object message) 
            throws Exception {
        
        // 设置确认模式
        rabbitTemplate.setConfirmCallback((correlationData, ack, cause) -> {
            if (ack) {
                System.out.println("消息发送到交换机成功: " + correlationData);
            } else {
                System.err.println("消息发送到交换机失败: " + cause);
                // 失败处理逻辑
                handleSendFailure(correlationData, cause);
            }
        });
        
        // 设置返回模式
        rabbitTemplate.setReturnsCallback(returned -> {
            System.err.println("消息路由到队列失败: " + returned.getReplyText());
            handleReturnedMessage(returned);
        });
        
        // 发送消息
        CorrelationData correlationData = new CorrelationData(UUID.randomUUID().toString());
        rabbitTemplate.convertAndSend(exchange, routingKey, message, correlationData);
        
        // 等待确认（可选）
        if (correlationData.getFuture() != null) {
            CorrelationData.Confirm confirm = correlationData.getFuture().get(5, TimeUnit.SECONDS);
            if (!confirm.isAck()) {
                throw new RuntimeException("消息未确认: " + confirm.getReason());
            }
        }
    }
    
    /**
     * 发送带延迟重试的消息
     */
    public void sendWithDelayedRetry(String exchange, String routingKey, 
                                    Object message, int maxRetries) {
        
        for (int attempt = 1; attempt <= maxRetries; attempt++) {
            try {
                rabbitTemplate.convertAndSend(exchange, routingKey, message);
                System.out.println("消息发送成功，尝试次数: " + attempt);
                return;
            } catch (Exception e) {
                System.err.println("发送失败，第" + attempt + "次尝试: " + e.getMessage());
                
                if (attempt == maxRetries) {
                    // 达到最大重试次数
                    handleMaxRetriesReached(message, e);
                    break;
                }
                
                // 延迟重试
                try {
                    long delay = calculateDelay(attempt);
                    System.out.println("等待 " + delay + "ms 后重试...");
                    Thread.sleep(delay);
                } catch (InterruptedException ie) {
                    Thread.currentThread().interrupt();
                    break;
                }
            }
        }
    }
    
    private long calculateDelay(int attempt) {
        // 指数退避算法
        return Math.min(1000L * (1L << attempt), 30000L); // 最大30秒
    }
    
    private void handleSendFailure(CorrelationData correlationData, String cause) {
        // 发送失败处理逻辑
        System.err.println("处理发送失败: " + correlationData + ", 原因: " + cause);
        // 可以记录日志、发送告警、保存到数据库等
    }
    
    private void handleReturnedMessage(ReturnedMessage returned) {
        // 消息返回处理逻辑
        System.err.println("处理返回消息: " + returned);
        // 可以重新发送、记录到死信队列等
    }
    
    private void handleMaxRetriesReached(Object message, Exception e) {
        // 达到最大重试次数处理
        System.err.println("达到最大重试次数，消息: " + message);
        // 保存到数据库、发送告警等
    }
}
```

## 二、 高级重连策略

### 1. 智能重连策略

```
@Component
public class SmartReconnectStrategy {
    
    private final Map<String, ReconnectStats> stats = new ConcurrentHashMap<>();
    private final ScheduledExecutorService scheduler = Executors.newScheduledThreadPool(1);
    
    /**
     * 智能发送
     */
    public void smartSend(RabbitTemplate template, String exchange, 
                         String routingKey, Object message) {
        
        String destination = exchange + ":" + routingKey;
        ReconnectStats stat = stats.computeIfAbsent(destination, k -> new ReconnectStats());
        
        for (int attempt = 0; attempt < getMaxAttempts(stat); attempt++) {
            try {
                template.convertAndSend(exchange, routingKey, message);
                
                // 发送成功，更新统计
                stat.recordSuccess();
                System.out.println("智能发送成功: " + destination);
                return;
                
            } catch (Exception e) {
                // 发送失败，更新统计
                stat.recordFailure();
                System.err.println("智能发送失败 [" + destination + "]: " + e.getMessage());
                
                if (shouldGiveUp(attempt, stat)) {
                    // 放弃发送
                    handleGiveUp(destination, message, e);
                    break;
                }
                
                // 计算延迟
                long delay = calculateSmartDelay(attempt, stat);
                if (delay > 0) {
                    try {
                        Thread.sleep(delay);
                    } catch (InterruptedException ie) {
                        Thread.currentThread().interrupt();
                        break;
                    }
                }
                
                // 检查连接状态
                if (isConnectionBroken(template)) {
                    System.out.println("检测到连接断开，尝试重建...");
                    tryReconnect(template);
                }
            }
        }
    }
    
    /**
     * 连接健康检查
     */
    public boolean isConnectionHealthy(RabbitTemplate template) {
        try {
            // 尝试执行一个轻量级操作检查连接
            template.execute(channel -> {
                channel.queueDeclarePassive("health.check.queue");
                return null;
            });
            return true;
        } catch (Exception e) {
            return false;
        }
    }
    
    /**
     * 连接状态监控
     */
    public void startConnectionMonitor(RabbitTemplate template) {
        scheduler.scheduleAtFixedRate(() -> {
            try {
                if (!isConnectionHealthy(template)) {
                    System.warn("连接健康检查失败，尝试恢复...");
                    tryReconnect(template);
                }
            } catch (Exception e) {
                System.err.println("连接监控异常: " + e.getMessage());
            }
        }, 30, 30, TimeUnit.SECONDS); // 每30秒检查一次
    }
    
    /**
     * 尝试重连
     */
    private void tryReconnect(RabbitTemplate template) {
        ConnectionFactory factory = template.getConnectionFactory();
        if (factory instanceof CachingConnectionFactory) {
            CachingConnectionFactory ccf = (CachingConnectionFactory) factory;
            try {
                ccf.destroy(); // 销毁旧连接
                ccf.createConnection(); // 创建新连接
                System.out.println("连接重建成功");
            } catch (Exception e) {
                System.err.println("连接重建失败: " + e.getMessage());
            }
        }
    }
    
    private int getMaxAttempts(ReconnectStats stat) {
        // 根据历史成功率动态调整最大尝试次数
        double successRate = stat.getSuccessRate();
        if (successRate > 0.9) return 3;     // 成功率>90%，重试3次
        if (successRate > 0.7) return 5;     // 成功率>70%，重试5次
        return 8;                            // 默认重试8次
    }
    
    private long calculateSmartDelay(int attempt, ReconnectStats stat) {
        // 基于尝试次数和统计的智能延迟
        double successRate = stat.getSuccessRate();
        
        long baseDelay = 1000L * (1L << attempt); // 指数退避
        
        if (successRate < 0.5) {
            baseDelay *= 2; // 成功率低，增加延迟
        }
        
        return Math.min(baseDelay, 60000L); // 最大1分钟
    }
    
    private boolean shouldGiveUp(int attempt, ReconnectStats stat) {
        // 是否应该放弃
        if (attempt >= 10) return true; // 超过10次尝试
        if (stat.getConsecutiveFailures() > 20) return true; // 连续失败20次
        return false;
    }
    
    private boolean isConnectionBroken(RabbitTemplate template) {
        try {
            template.execute(channel -> null);
            return false;
        } catch (Exception e) {
            return e.getMessage().contains("connection") || 
                   e.getMessage().contains("channel");
        }
    }
    
    private void handleGiveUp(String destination, Object message, Exception e) {
        // 放弃发送的处理
        System.err.println("放弃发送消息到: " + destination);
        saveFailedMessage(destination, message, e);
        sendAlert(destination, e);
    }
    
    private void saveFailedMessage(String destination, Object message, Exception e) {
        // 保存失败消息到数据库
        System.out.println("保存失败消息: " + destination);
    }
    
    private void sendAlert(String destination, Exception e) {
        // 发送告警
        System.err.println("发送告警: " + destination + " - " + e.getMessage());
    }
    
    /**
     * 重连统计
     */
    @Data
    public static class ReconnectStats {
        private int totalAttempts = 0;
        private int successfulAttempts = 0;
        private int consecutiveFailures = 0;
        private long lastSuccessTime = 0;
        private long lastFailureTime = 0;
        
        public synchronized void recordSuccess() {
            totalAttempts++;
            successfulAttempts++;
            consecutiveFailures = 0;
            lastSuccessTime = System.currentTimeMillis();
        }
        
        public synchronized void recordFailure() {
            totalAttempts++;
            consecutiveFailures++;
            lastFailureTime = System.currentTimeMillis();
        }
        
        public synchronized double getSuccessRate() {
            if (totalAttempts == 0) return 1.0;
            return (double) successfulAttempts / totalAttempts;
        }
    }
}
```

### 2. 集群重连策略

```
@Component
public class ClusterReconnectStrategy {
    
    private List<String> addresses = Arrays.asList(
        "rabbitmq1:5672", 
        "rabbitmq2:5672", 
        "rabbitmq3:5672"
    );
    
    private volatile int currentIndex = 0;
    private final AtomicInteger failureCount = new AtomicInteger(0);
    private final Object lock = new Object();
    
    /**
     * 配置集群连接工厂
     */
    @Bean
    public ConnectionFactory clusterConnectionFactory() {
        // 创建地址列表
        com.rabbitmq.client.Address[] addressArray = addresses.stream()
            .map(addr -> {
                String[] parts = addr.split(":");
                return new com.rabbitmq.client.Address(parts[0], Integer.parseInt(parts[1]));
            })
            .toArray(com.rabbitmq.client.Address[]::new);
        
        ConnectionFactory factory = new ConnectionFactory();
        factory.setUsername("guest");
        factory.setPassword("guest");
        factory.setVirtualHost("/");
        
        // 集群重连配置
        factory.setAutomaticRecoveryEnabled(true);
        factory.setTopologyRecoveryEnabled(true);
        factory.setNetworkRecoveryInterval(5000);
        factory.setRequestedHeartbeat(60);
        
        // 创建集群连接
        return new CachingConnectionFactory(factory, addressArray);
    }
    
    /**
     * 集群感知的 RabbitTemplate
     */
    @Bean
    public RabbitTemplate clusterRabbitTemplate() {
        RabbitTemplate template = new RabbitTemplate(clusterConnectionFactory());
        
        // 添加集群故障转移监听器
        template.addListener(new ChannelListener() {
            @Override
            public void onCreate(Channel channel, boolean transactional) {
                System.out.println("通道创建: " + channel);
            }
            
            @Override
            public void onShutDown(ShutdownSignalException signal) {
                System.err.println("通道关闭: " + signal.getReason());
                handleClusterFailure();
            }
        });
        
        return template;
    }
    
    /**
     * 集群发送
     */
    public void sendWithClusterFallback(String exchange, String routingKey, 
                                       Object message) {
        
        for (int i = 0; i < addresses.size(); i++) {
            try {
                String currentAddress = getCurrentAddress();
                System.out.println("尝试发送到: " + currentAddress);
                
                sendToCurrentNode(exchange, routingKey, message);
                
                // 发送成功，重置失败计数
                failureCount.set(0);
                return;
                
            } catch (Exception e) {
                System.err.println("发送失败: " + e.getMessage());
                
                // 记录失败
                failureCount.incrementAndGet();
                
                if (shouldSwitchNode()) {
                    switchToNextNode();
                }
                
                if (i == addresses.size() - 1) {
                    // 所有节点都尝试失败
                    handleAllNodesFailed(message, e);
                }
            }
        }
    }
    
    /**
     * 节点健康检查
     */
    public void startClusterHealthCheck() {
        ScheduledExecutorService scheduler = Executors.newSingleThreadScheduledExecutor();
        
        scheduler.scheduleAtFixedRate(() -> {
            for (String address : addresses) {
                if (isNodeHealthy(address)) {
                    System.out.println("节点健康: " + address);
                } else {
                    System.err.println("节点不健康: " + address);
                    // 可以从地址列表中临时移除
                }
            }
        }, 0, 30, TimeUnit.SECONDS);
    }
    
    private String getCurrentAddress() {
        synchronized (lock) {
            return addresses.get(currentIndex);
        }
    }
    
    private void sendToCurrentNode(String exchange, String routingKey, Object message) {
        // 使用当前节点的连接发送
        RabbitTemplate template = createTemplateForCurrentNode();
        template.convertAndSend(exchange, routingKey, message);
    }
    
    private RabbitTemplate createTemplateForCurrentNode() {
        String address = getCurrentAddress();
        String[] parts = address.split(":");
        
        CachingConnectionFactory factory = new CachingConnectionFactory(parts[0], 
                                                                       Integer.parseInt(parts[1]));
        factory.setUsername("guest");
        factory.setPassword("guest");
        
        return new RabbitTemplate(factory);
    }
    
    private boolean shouldSwitchNode() {
        // 根据失败次数决定是否切换节点
        return failureCount.get() >= 3;
    }
    
    private void switchToNextNode() {
        synchronized (lock) {
            currentIndex = (currentIndex + 1) % addresses.size();
            failureCount.set(0);
            System.out.println("切换到节点: " + addresses.get(currentIndex));
        }
    }
    
    private boolean isNodeHealthy(String address) {
        String[] parts = address.split(":");
        try (ConnectionFactory factory = new ConnectionFactory()) {
            factory.setHost(parts[0]);
            factory.setPort(Integer.parseInt(parts[1]));
            factory.setConnectionTimeout(5000);
            
            try (Connection connection = factory.newConnection();
                 Channel channel = connection.createChannel()) {
                
                // 执行一个简单的操作检查节点
                channel.queueDeclarePassive("amq.rabbitmq.health-check");
                return true;
            }
        } catch (Exception e) {
            return false;
        }
    }
    
    private void handleClusterFailure() {
        System.err.println("集群故障，尝试故障转移...");
        switchToNextNode();
    }
    
    private void handleAllNodesFailed(Object message, Exception e) {
        System.err.println("所有集群节点都失败");
        saveForLaterRetry(message);
        sendClusterAlert(e);
    }
    
    private void saveForLaterRetry(Object message) {
        // 保存消息以便稍后重试
        System.out.println("保存消息以便稍后重试");
    }
    
    private void sendClusterAlert(Exception e) {
        // 发送集群故障告警
        System.err.println("发送集群故障告警: " + e.getMessage());
    }
}
```

```
@Component
public class TransactionalSender {
    
    @Autowired
    private RabbitTemplate rabbitTemplate;
    
    /**
     * 配置事务管理器
     */
    @Bean
    public RabbitTransactionManager transactionManager(ConnectionFactory connectionFactory) {

     */

    public void sendTransactional(String exchange, String routingKey, 
                                 Object message, boolean shouldRollback) {

        rabbitTemplate.setChannelTransacted(true);
        
        try {
            // 发送消息1
            rabbitTemplate.convertAndSend(exchange, routingKey + ".1", message);
            System.out.println("消息1发送成功");

            // 模拟业务操作
            doBusinessOperation();
            
            if (shouldRollback) {
                throw new RuntimeException("模拟异常，触发回滚");
            }
            
        } catch (Exception e) {
            // 事务会自动回滚
            System.err.println("发送失败，事务回滚: " + e.getMessage());
            throw e;
        }
    }

     * 手动事务控制
     */
    public void sendWithManualTransaction(String exchange, String routingKey, 
                                         Object message) {
        
        rabbitTemplate.setChannelTransacted(true);
        
        rabbitTemplate.execute(channel -> {
            try {
                // 开始事务
                channel.txSelect();
                
                // 发送消息
                rabbitTemplate.convertAndSend(exchange, routingKey, message);
                
                // 其他操作
                doBusinessOperation();
                
                // 提交事务
                channel.txCommit();
                System.out.println("事务提交成功");
                
            } catch (Exception e) {
                // 回滚事务
                channel.txRollback();
                System.err.println("事务回滚: " + e.getMessage());
                throw e;
            }
            return null;
        });
    }
    
    /**
     * 补偿事务
     */
    public void sendWithCompensation(String exchange, String routingKey, 
                                    Object message) {
        
        boolean success = false;
        try {
            // 尝试发送
            rabbitTemplate.convertAndSend(exchange, routingKey, message);
            success = true;

            System.err.println("发送失败，尝试补偿: " + e.getMessage());

        } finally {
            if (!success) {
                // 记录失败
                recordFailure(exchange, routingKey, message);
            }
        }
    }

        // 模拟业务操作

        // 或者保存到数据库稍后重试

    private void saveForCompensation(String exchange, String routingKey,

    }
```

### 2. 消息持久化与重试

```
@Component
public class PersistentMessageSender {
    
    @Autowired
    private RabbitTemplate rabbitTemplate;
    
    @Autowired
    private MessagePersistenceService persistenceService;
    
    /**
     * 持久化发送
     */
    public void sendWithPersistence(String exchange, String routingKey, 
                                   Object message) {
        
        String messageId = UUID.randomUUID().toString();
        MessageStatus status = new MessageStatus(messageId, exchange, 
                                                routingKey, message);
        
        try {
            // 1. 保存到本地数据库
            persistenceService.saveMessage(status);
            
            // 2. 发送到 RabbitMQ
            sendToRabbitMQ(exchange, routingKey, message, messageId);
            
            // 3. 更新状态为已发送
            persistenceService.updateStatus(messageId, MessageStatus.Status.SENT);
            
        } catch (Exception e) {
            // 4. 更新状态为失败
            persistenceService.updateStatus(messageId, MessageStatus.Status.FAILED);
            
            // 5. 触发重试机制
            scheduleRetry(status, e);
        }
    }
    
    /**
     * 可靠发送（最终一致性）
     */
    public void sendReliable(String exchange, String routingKey, Object message) {
        String messageId = UUID.randomUUID().toString();
        
        // 创建可靠消息
        ReliableMessage reliableMessage = new ReliableMessage();
        reliableMessage.setMessageId(messageId);
        reliableMessage.setExchange(exchange);
        reliableMessage.setRoutingKey(routingKey);
        reliableMessage.setMessage(message);
        reliableMessage.setStatus(ReliableMessage.Status.PENDING);
        reliableMessage.setCreateTime(new Date());
        reliableMessage.setRetryCount(0);
        reliableMessage.setMaxRetries(5);
        
        // 保存到数据库
        saveReliableMessage(reliableMessage);
        
        // 尝试发送
        trySendReliableMessage(reliableMessage);
    }
    
    /**
     * 定时重试任务
     */
    @Scheduled(fixedDelay = 60000) // 每分钟执行一次
    public void retryFailedMessages() {
        List<ReliableMessage> failedMessages = findFailedMessages();
        
        for (ReliableMessage message : failedMessages) {
            if (shouldRetry(message)) {
                try {
                    retryMessage(message);
                } catch (Exception e) {
                    handleRetryFailure(message, e);
                }
            } else {
                handleMaxRetriesReached(message);
            }
        }
    }
    
    private void sendToRabbitMQ(String exchange, String routingKey, 
                               Object message, String messageId) {
        
        // 设置消息属性
        MessagePostProcessor processor = msg -> {
            MessageProperties props = msg.getMessageProperties();
            props.setMessageId(messageId);
            props.setCorrelationId(messageId);
            props.setDeliveryMode(MessageDeliveryMode.PERSISTENT);
            props.setTimestamp(new Date());
            props.setHeader("persistent", true);
            return msg;
        };
        
        rabbitTemplate.convertAndSend(exchange, routingKey, message, processor);
    }
    
    private void scheduleRetry(MessageStatus status, Exception e) {
        // 安排重试
        System.out.println("安排重试: " + status.getMessageId());
        
        ScheduledExecutorService scheduler = Executors.newSingleThreadScheduledExecutor();
        scheduler.schedule(() -> {
            try {
                retrySend(status);
            } catch (Exception ex) {
                System.err.println("重试失败: " + ex.getMessage());
            }
        }, calculateRetryDelay(status.getRetryCount()), TimeUnit.MILLISECONDS);
    }
    
    private long calculateRetryDelay(int retryCount) {
        // 指数退避
        return Math.min(1000L * (1L << retryCount), 300000L); // 最大5分钟
    }
    
    private void retrySend(MessageStatus status) {
        status.incrementRetryCount();
        
        if (status.getRetryCount() > status.getMaxRetries()) {
            handleMaxRetries(status);
            return;
        }
        
        try {
            sendToRabbitMQ(status.getExchange(), status.getRoutingKey(), 
                          status.getMessage(), status.getMessageId());
            persistenceService.updateStatus(status.getMessageId(), 
                                           MessageStatus.Status.SENT);
        } catch (Exception e) {
            persistenceService.updateStatus(status.getMessageId(), 
                                           MessageStatus.Status.FAILED);
            scheduleRetry(status, e);
        }
    }
    
    private void handleMaxRetries(MessageStatus status) {
        System.err.println("达到最大重试次数: " + status.getMessageId());
        persistenceService.updateStatus(status.getMessageId(), 
                                       MessageStatus.Status.MAX_RETRIES);
        sendAlert(status);
    }
    
    private void sendAlert(MessageStatus status) {
        // 发送告警
        System.err.println("发送告警: 消息" + status.getMessageId() + "达到最大重试次数");
    }
    
    private void saveReliableMessage(ReliableMessage message) {
        // 保存到数据库
        System.out.println("保存可靠消息: " + message.getMessageId());
    }
    
    private void trySendReliableMessage(ReliableMessage message) {
        // 尝试发送可靠消息
        try {
            sendToRabbitMQ(message.getExchange(), message.getRoutingKey(), 
                          message.getMessage(), message.getMessageId());
            
            message.setStatus(ReliableMessage.Status.SENT);
            updateReliableMessage(message);
            
        } catch (Exception e) {
            message.setStatus(ReliableMessage.Status.FAILED);
            message.setLastError(e.getMessage());
            updateReliableMessage(message);
        }
    }
    
    private List<ReliableMessage> findFailedMessages() {
        // 查询失败消息
        return Collections.emptyList();
    }
    
    private boolean shouldRetry(ReliableMessage message) {
        return message.getRetryCount() < message.getMaxRetries() &&
               message.getStatus() == ReliableMessage.Status.FAILED;
    }
    
    private void retryMessage(ReliableMessage message) {
        message.incrementRetryCount();
        trySendReliableMessage(message);
    }
    
    private void handleRetryFailure(ReliableMessage message, Exception e) {
        message.setLastError(e.getMessage());
        updateReliableMessage(message);
    }
    
    private void handleMaxRetriesReached(ReliableMessage message) {
        message.setStatus(ReliableMessage.Status.MAX_RETRIES);
        updateReliableMessage(message);
        sendMaxRetriesAlert(message);
    }
    
    private void updateReliableMessage(ReliableMessage message) {
        // 更新数据库
        System.out.println("更新可靠消息状态: " + message.getMessageId());
    }
    
    private void sendMaxRetriesAlert(ReliableMessage message) {
        // 发送最大重试告警
        System.err.println("消息达到最大重试次数: " + message.getMessageId());
    }
    
    /**
     * 消息状态
     */
    @Data
    public static class MessageStatus {
        private String messageId;
        private String exchange;
        private String routingKey;
        private Object message;
        private Status status = Status.PENDING;
        private int retryCount = 0;
        private int maxRetries = 5;
        private Date createTime = new Date();
        private Date lastTryTime;
        
        public enum Status {
            PENDING, SENT, FAILED, MAX_RETRIES
        }
        
        public void incrementRetryCount() {
            this.retryCount++;
            this.lastTryTime = new Date();
        }
    }
}
```

```
@Component
public class SendMonitor {
    
    private final MeterRegistry meterRegistry;
    private final Counter successCounter;
    private final Counter failureCounter;
    private final Timer sendTimer;
    
    private final Map<String, SendStats> statsMap = new ConcurrentHashMap<>();
    
    public SendMonitor(MeterRegistry meterRegistry) {
        this.meterRegistry = meterRegistry;
        
        // 创建指标
        this.successCounter = Counter.builder("rabbitmq.send.success")
            .description("消息发送成功次数")
            .register(meterRegistry);
        
        this.failureCounter = Counter.builder("rabbitmq.send.failure")
            .description("消息发送失败次数")
            .register(meterRegistry);
        
        this.sendTimer = Timer.builder("rabbitmq.send.duration")
            .description("消息发送耗时")
            .register(meterRegistry);
    }
    
    /**
     * 监控发送
     */
    public <T> T monitorSend(Callable<T> sendOperation, String destination) {
        SendStats stats = statsMap.computeIfAbsent(destination, k -> new SendStats());
        
        long startTime = System.currentTimeMillis();
        try {
            T result = sendTimer.record(() -> {
                try {
                    return sendOperation.call();
                } catch (Exception e) {
                    throw new RuntimeException(e);
                }
            });
            
            // 成功
            long duration = System.currentTimeMillis() - startTime;
            stats.recordSuccess(duration);
            successCounter.increment();
            
            meterRegistry.counter("rabbitmq.send.success", 
                "destination", destination).increment();
            
            return result;
            
        } catch (Exception e) {
            // 失败
            stats.recordFailure();
            failureCounter.increment();
            
            meterRegistry.counter("rabbitmq.send.failure", 
                "destination", destination, 
                "error", e.getClass().getSimpleName()).increment();
            
            throw e;
        }
    }
    
    /**
     * 获取发送统计
     */
    public SendStats getStats(String destination) {
        return statsMap.get(destination);
    }
    
    /**
     * 健康检查
     */
    public Health healthCheck() {
        Map<String, Object> details = new HashMap<>();
        
        statsMap.forEach((dest, stats) -> {
            Map<String, Object> statDetails = new HashMap<>();
            statDetails.put("total", stats.getTotalCount());
            statDetails.put("success", stats.getSuccessCount());
            statDetails.put("failure", stats.getFailureCount());
            statDetails.put("successRate", stats.getSuccessRate());
            statDetails.put("avgDuration", stats.getAverageDuration());
            details.put(dest, statDetails);
        });
        
        // 检查是否有连续失败
        boolean healthy = statsMap.values().stream()
            .noneMatch(stats -> stats.getConsecutiveFailures() > 10);
        
        if (healthy) {
            return Health.up().withDetails(details).build();
        } else {
            return Health.down().withDetails(details).build();
        }
    }
    
    /**
     * 发送统计
     */
    @Data
    public static class SendStats {
        private AtomicLong totalCount = new AtomicLong();
        private AtomicLong successCount = new AtomicLong();
        private AtomicLong failureCount = new AtomicLong();
        private AtomicLong totalDuration = new AtomicLong();
        private AtomicInteger consecutiveFailures = new AtomicInteger();
        
        public void recordSuccess(long duration) {
            totalCount.incrementAndGet();
            successCount.incrementAndGet();
            totalDuration.addAndGet(duration);
            consecutiveFailures.set(0);
        }
        
        public void recordFailure() {
            totalCount.incrementAndGet();
            failureCount.incrementAndGet();
            consecutiveFailures.incrementAndGet();
        }
        
        public double getSuccessRate() {
            long total = totalCount.get();
            if (total == 0) return 1.0;
            return (double) successCount.get() / total;
        }
        
        public long getAverageDuration() {
            long success = successCount.get();
            if (success == 0) return 0;
            return totalDuration.get() / success;
        }
    }
}
```

```
                    retryTemplate.execute(context -> {
                        int attempt = context.getRetryCount() + 1;
                        log.info("发送消息，尝试次数: {}, 消息ID: {}", attempt, messageId);

                props.setCorrelationId(messageId);
                props.setDeliveryMode(MessageDeliveryMode.PERSISTENT);
                props.setTimestamp(new Date());
                props.setHeader("send-time", System.currentTimeMillis());
                return msg;

                CachingConnectionFactory ccf = (CachingConnectionFactory) factory;
                if (!ccf.isRunning()) {

                    try {
                        ccf.resetConnection();
                    } catch (Exception e) {
                        log.error("重置连接失败: {}", e.getMessage());
                    }
                }
            }
        }

            // 发送告警

        }
    }
```

## 七、 常见问题

### Q1: 重连太频繁怎么办？

- 增加重连间隔
    
- 使用指数退避
    
- 添加抖动（jitter）避免同时重连
    

### Q2: 如何避免消息重复？

- 使用消息ID去重
    
- 实现幂等性消费
    
- 数据库唯一约束
    

### Q3: 连接池配置？

- 适当大小的连接池
    
- 及时释放连接
    
- 监控连接使用情况
    

### Q4: 监控指标？

- 成功/失败次数
    
- 平均发送时间
    
- 重试次数分布
    
- 连接状态
    

发送者重连是确保消息可靠性的关键，需要结合业务需求、网络环境和系统资源来设计合适的重连策略。
