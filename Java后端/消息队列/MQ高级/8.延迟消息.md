# RabbitMQ 延迟消息

延迟消息是 RabbitMQ 中实现**定时任务、延迟处理、计划执行**的核心功能。有两种主要实现方式：**TTL+死信队列**和**延迟消息插件**。

## 一、 TTL+死信队列方案

### 1. 基础延迟队列配置

```
@Configuration
public class BasicDelayedQueueConfig {
    
    /**
     * 死信交换机
     */
    @Bean
    public DirectExchange dlxExchange() {
        return new DirectExchange("dlx.exchange", true, false);
    }
    
    /**
     * 死信队列
     */

    public Queue dlxQueue() {
        return QueueBuilder.durable("dlx.queue")
                .build();
    }
    
    /**
     * 死信绑定
     */

        return BindingBuilder.bind(dlxQueue())

     */

                .withArguments(args)
                .build();

            System.out.println("收到延迟消息: " + message);
            System.out.println("当前时间: " + LocalDateTime.now());
        }
    }
}
```

```
    public Queue delay5mQueue() {
        return createDelayedQueue("delay.5m.queue", 300000, "process.exchange", "process.key");
    }
    
    @Bean
    public Queue delay30mQueue() {
        return createDelayedQueue("delay.30m.queue", 1800000, "process.exchange", "process.key");
    }
    
    /**
     * 处理交换机
     */
    @Bean
    public DirectExchange processExchange() {
        return new DirectExchange("process.exchange", true, false);
    }
    
    /**
     * 处理队列
     */
    @Bean
    public Queue processQueue() {
        return QueueBuilder.durable("process.queue")
                .build();
    }
    
    /**
     * 处理绑定
     */
    @Bean
    public Binding processBinding() {
        return BindingBuilder.bind(processQueue())
                .to(processExchange())
                .with("process.key");
    }
    
    /**
     * 创建延迟队列的辅助方法
     */
    private Queue createDelayedQueue(String queueName, long ttl, 
                                    String dlxExchange, String dlxRoutingKey) {
        Map<String, Object> args = new HashMap<>();
        args.put("x-dead-letter-exchange", dlxExchange);
        args.put("x-dead-letter-routing-key", dlxRoutingKey);
        args.put("x-message-ttl", ttl);
        
        return QueueBuilder.durable(queueName)
                .withArguments(args)
                .build();
    }
    
    /**
     * 多级延迟生产者
     */
    @Component
    public class MultiLevelProducer {
        
        @Autowired
        private RabbitTemplate rabbitTemplate;
        
        /**
         * 发送延迟消息
         */
        public void sendDelayedMessage(String message, DelayLevel level) {
            String queueName = getQueueByLevel(level);
            rabbitTemplate.convertAndSend("", queueName, message);
            
            System.out.println("发送延迟消息: " + message);
            System.out.println("延迟级别: " + level + ", 队列: " + queueName);
            System.out.println("发送时间: " + LocalDateTime.now());
        }
        
        /**
         * 根据延迟级别选择队列
         */
        private String getQueueByLevel(DelayLevel level) {
            switch (level) {
                case LEVEL_5S: return "delay.5s.queue";
                case LEVEL_30S: return "delay.30s.queue";
                case LEVEL_5M: return "delay.5m.queue";
                case LEVEL_30M: return "delay.30m.queue";
                default: return "delay.5s.queue";
            }
        }
        
        /**
         * 发送动态延迟消息
         */
        public void sendDynamicDelay(String message, long delayMillis) {
            // 选择最接近的延迟队列
            DelayLevel level = calculateDelayLevel(delayMillis);
            sendDelayedMessage(message, level);
        }
        
        private DelayLevel calculateDelayLevel(long delayMillis) {
            if (delayMillis <= 5000) {
                return DelayLevel.LEVEL_5S;
            } else if (delayMillis <= 30000) {
                return DelayLevel.LEVEL_30S;
            } else if (delayMillis <= 300000) {
                return DelayLevel.LEVEL_5M;
            } else {
                return DelayLevel.LEVEL_30M;
            }
        }
        
        /**
         * 延迟级别枚举
         */
        public enum DelayLevel {
            LEVEL_5S,    // 5秒
            LEVEL_30S,   // 30秒
            LEVEL_5M,    // 5分钟
            LEVEL_30M    // 30分钟
        }
    }
    
    /**
     * 处理队列消费者
     */
    @Component
    public class ProcessQueueConsumer {
        
        @RabbitListener(queues = "process.queue")
        public void handleProcessMessage(String message) {
            System.out.println("处理延迟消息: " + message);
            System.out.println("处理时间: " + LocalDateTime.now());
        }
    }
}
```

### 3. 动态TTL延迟消息

```
@Component
public class DynamicTTLDelayedMessage {
    
    @Autowired
    private RabbitTemplate rabbitTemplate;
    
    /**
     * 发送带消息级别TTL的延迟消息
     */
    public void sendWithMessageTTL(String exchange, String routingKey, 
                                  Object message, long ttlMillis) {
        
        // 设置消息级别的TTL
        rabbitTemplate.convertAndSend(exchange, routingKey, message, 
            msg -> {
                msg.getMessageProperties().setExpiration(String.valueOf(ttlMillis));
                msg.getMessageProperties().setHeader("delay-time", ttlMillis);
                msg.getMessageProperties().setHeader("send-time", System.currentTimeMillis());
                return msg;
            });
        
        System.out.println("发送延迟消息，TTL: " + ttlMillis + "ms");
    }
    
    /**
     * 发送到延迟队列
     */
    public void sendToDelayedQueue(String message, long delayMillis) {
        // 创建消息
        MessageProperties properties = new MessageProperties();
        properties.setExpiration(String.valueOf(delayMillis));
        properties.setHeader("original-exchange", "delayed.exchange");
        properties.setHeader("original-routing-key", "delayed.key");
        
        Message msg = new Message(message.getBytes(), properties);
        
        // 发送到延迟队列
        rabbitTemplate.send("", "delayed.queue", msg);
    }
    
    /**
     * 批量发送延迟消息
     */
    public void batchSendWithDelays(List<DelayedMessage> messages) {
        for (DelayedMessage delayedMessage : messages) {
            sendWithMessageTTL(
                delayedMessage.getExchange(),
                delayedMessage.getRoutingKey(),
                delayedMessage.getMessage(),
                delayedMessage.getDelayMillis()
            );
        }
    }
    
    /**
     * 延迟调度器
     */
    public void scheduleMessage(String message, LocalDateTime scheduleTime) {
        long delayMillis = calculateDelay(scheduleTime);
        
        if (delayMillis > 0) {
            sendWithMessageTTL("scheduler.exchange", "scheduler.key", message, delayMillis);
            System.out.println("安排定时任务: " + scheduleTime);
        }
    }
    
    /**
     * 循环延迟任务
     */
    public void scheduleRecurringTask(String taskId, Runnable task, 
                                     long intervalMillis, int maxTimes) {
        
        for (int i = 1; i <= maxTimes; i++) {
            long delay = intervalMillis * i;
            
            // 发送延迟消息执行任务
            String message = String.format("任务%s-第%d次执行", taskId, i);
            
            sendWithMessageTTL("recurring.exchange", "recurring.key", 
                new TaskMessage(taskId, message, i), delay);
        }
    }
    
    private long calculateDelay(LocalDateTime scheduleTime) {
        LocalDateTime now = LocalDateTime.now();
        Duration duration = Duration.between(now, scheduleTime);
        return duration.toMillis();
    }
    
    /**
     * 延迟消息
     */
    @Data
    @AllArgsConstructor
    public static class DelayedMessage {
        private String exchange;
        private String routingKey;
        private Object message;
        private long delayMillis;
    }
    
    /**
     * 任务消息
     */
    @Data
    @AllArgsConstructor
    public static class TaskMessage {
        private String taskId;
        private String message;
        private int executionCount;
    }
}
```

## 二、 延迟消息插件方案

> 需要安装 RabbitMQ 延迟消息插件：`rabbitmq_delayed_message_exchange`

### 1. 延迟交换机配置

```
@Configuration
public class DelayedPluginConfig {
    
    /**
     * 自定义延迟交换机
     */
    @Bean
    public CustomExchange delayedExchange() {
        Map<String, Object> args = new HashMap<>();
        
        // 设置交换机类型为 x-delayed-message
        args.put("x-delayed-type", "direct");
        
        return new CustomExchange(
            "delayed.exchange",  // 交换机名称
            "x-delayed-message", // 交换机类型
            true,                // 是否持久化
            false,               // 是否自动删除
            args                 // 参数
        );
    }
    
    /**
     * 延迟队列
     */
    @Bean
    public Queue delayedPluginQueue() {
        return QueueBuilder.durable("delayed.plugin.queue")
                .build();
    }
    
    /**
     * 绑定延迟队列
     */
    @Bean
    public Binding delayedBinding() {
        return BindingBuilder.bind(delayedPluginQueue())
                .to(delayedExchange())
                .with("delayed.key")
                .noargs();
    }
    
    /**
     * 延迟消息生产者
     */
    @Component
    public class DelayedPluginProducer {
        
        @Autowired
        private RabbitTemplate rabbitTemplate;
        
        /**
         * 发送延迟消息
         */
        public void sendDelayedMessage(String message, long delayMillis) {
            rabbitTemplate.convertAndSend("delayed.exchange", "delayed.key", message,
                msg -> {
                    // 设置延迟头
                    msg.getMessageProperties().setHeader("x-delay", delayMillis);
                    msg.getMessageProperties().setHeader("send-time", System.currentTimeMillis());
                    msg.getMessageProperties().setHeader("expected-delivery", 
                        System.currentTimeMillis() + delayMillis);
                    return msg;
                });
            
            System.out.println("发送插件延迟消息: " + message);
            System.out.println("延迟: " + delayMillis + "ms");
        }
        
        /**
         * 发送定时消息
         */
        public void sendScheduledMessage(String message, LocalDateTime scheduleTime) {
            long delay = calculateDelay(scheduleTime);
            
            if (delay > 0) {
                sendDelayedMessage(message, delay);
            } else {
                // 如果时间已过，立即发送
                sendDelayedMessage(message, 0);
            }
        }
        
        /**
         * 批量发送延迟消息
         */
        public void batchSendDelayed(List<DelayedItem> items) {
            for (DelayedItem item : items) {
                sendDelayedMessage(item.getMessage(), item.getDelayMillis());
            }
        }
        
        private long calculateDelay(LocalDateTime scheduleTime) {
            return Duration.between(LocalDateTime.now(), scheduleTime).toMillis();
        }
    }
    
    /**
     * 延迟消息消费者
     */
    @Component
    public class DelayedPluginConsumer {
        
        @RabbitListener(queues = "delayed.plugin.queue")
        public void handleDelayedMessage(String message,
                                        @Header(value = "x-delay", required = false) Integer delay,
                                        @Header("send-time") Long sendTime) {
            
            long now = System.currentTimeMillis();
            long actualDelay = now - sendTime;
            
            System.out.println("收到延迟消息: " + message);
            System.out.println("预期延迟: " + delay + "ms");
            System.out.println("实际延迟: " + actualDelay + "ms");
            System.out.println("延迟误差: " + (actualDelay - delay) + "ms");
            System.out.println("接收时间: " + LocalDateTime.now());
        }
    }
    
    /**
     * 延迟项
     */
    @Data
    @AllArgsConstructor
    public static class DelayedItem {
        private String message;
        private long delayMillis;
    }
}
```

## 三、 实际应用场景

### 场景1：订单超时取消

```
@Component
public class OrderTimeoutCancel {
    
    @Autowired
    private RabbitTemplate rabbitTemplate;
    
    /**
     * 订单状态
     */
    public enum OrderStatus {
        CREATED,        // 已创建
        PAID,           // 已支付
        CANCELLED,      // 已取消
        COMPLETED       // 已完成
    }
    
    /**
     * 创建订单
     */
    public Order createOrder(Order order) {
        // 保存订单
        order.setStatus(OrderStatus.CREATED);
        order.setCreateTime(LocalDateTime.now());
        
        // 发送15分钟超时延迟消息
        sendOrderTimeoutMessage(order.getOrderId(), 15 * 60 * 1000); // 15分钟
        
        System.out.println("创建订单: " + order.getOrderId());
        System.out.println("设置15分钟后超时检查");
        
        return order;
    }
    
    /**
     * 支付订单
     */
    public boolean payOrder(String orderId) {
        // 更新订单状态
        updateOrderStatus(orderId, OrderStatus.PAID);
        
        // 取消超时检查
        cancelOrderTimeout(orderId);
        
        System.out.println("订单已支付: " + orderId);
        System.out.println("取消超时检查");
        
        return true;
    }
    
    /**
     * 发送订单超时消息
     */
    private void sendOrderTimeoutMessage(String orderId, long delayMillis) {
        OrderTimeoutMessage message = new OrderTimeoutMessage(orderId, delayMillis);
        
        rabbitTemplate.convertAndSend("order.timeout.exchange", "order.timeout.key", 
            message, msg -> {
                msg.getMessageProperties().setHeader("x-delay", delayMillis);
                msg.getMessageProperties().setHeader("order-id", orderId);
                msg.getMessageProperties().setHeader("expected-cancel-time", 
                    System.currentTimeMillis() + delayMillis);
                return msg;
            });
    }
    
    /**
     * 取消订单超时检查
     */
    private void cancelOrderTimeout(String orderId) {
        // 在实际应用中，可能需要存储超时消息ID，然后取消
        // 这里简化处理
        System.out.println("取消订单超时检查: " + orderId);
    }
    
    /**
     * 处理订单超时
     */
    @RabbitListener(queues = "order.timeout.queue")
    public void handleOrderTimeout(OrderTimeoutMessage message) {
        String orderId = message.getOrderId();
        
        // 检查订单状态
        Order order = getOrder(orderId);
        
        if (order.getStatus() == OrderStatus.CREATED) {
            // 订单未支付，执行取消
            cancelOrder(orderId);
            
            System.out.println("订单超时取消: " + orderId);
            System.out.println("取消时间: " + LocalDateTime.now());
            
            // 发送取消通知
            sendCancelNotification(order);
        } else {
            System.out.println("订单状态已更新，无需取消: " + orderId);
        }
    }
    
    /**
     * 批量订单超时处理
     */
    public void batchCreateOrders(List<Order> orders) {
        for (Order order : orders) {
            createOrder(order);
        }
    }
    
    /**
     * 可配置的超时时间
     */
    public void createOrderWithConfigurableTimeout(Order order, String timeoutConfig) {
        long timeoutMillis = getTimeoutMillis(timeoutConfig);
        createOrder(order);
        
        // 可以根据配置发送不同的延迟消息
        sendOrderTimeoutMessage(order.getOrderId(), timeoutMillis);
    }
    
    private long getTimeoutMillis(String timeoutConfig) {
        switch (timeoutConfig) {
            case "FAST": return 5 * 60 * 1000;      // 5分钟
            case "NORMAL": return 15 * 60 * 1000;   // 15分钟
            case "SLOW": return 30 * 60 * 1000;     // 30分钟
            default: return 15 * 60 * 1000;
        }
    }
    
    private void updateOrderStatus(String orderId, OrderStatus status) {
        // 更新订单状态
    }
    
    private void cancelOrder(String orderId) {
        // 取消订单
        updateOrderStatus(orderId, OrderStatus.CANCELLED);
    }
    
    private Order getOrder(String orderId) {
        // 获取订单
        return new Order();
    }
    
    private void sendCancelNotification(Order order) {
        // 发送取消通知
        System.out.println("发送订单取消通知: " + order.getOrderId());
    }
    
    /**
     * 订单超时消息
     */
    @Data
    @AllArgsConstructor
    public static class OrderTimeoutMessage {
        private String orderId;
        private long timeoutMillis;
    }
    
    /**
     * 订单实体
     */
    @Data
    public static class Order {
        private String orderId;
        private BigDecimal amount;
        private OrderStatus status;
        private LocalDateTime createTime;
        private LocalDateTime payTime;
    }
}
```

### 场景2：消息重试与延迟重试

```
@Component
public class DelayedRetrySystem {
    
    @Autowired
    private RabbitTemplate rabbitTemplate;
    
    /**
     * 重试级别配置
     */
    public enum RetryLevel {
        LEVEL_1(1000),      // 1秒
        LEVEL_2(5000),      // 5秒
        LEVEL_3(30000),     // 30秒
        LEVEL_4(60000),     // 1分钟
        LEVEL_5(300000);    // 5分钟
        
        private final long delayMillis;
        
        RetryLevel(long delayMillis) {
            this.delayMillis = delayMillis;
        }
        
        public long getDelayMillis() {
            return delayMillis;
        }
    }
    
    /**
     * 发送需要重试的消息
     */
    public void sendWithRetry(String exchange, String routingKey, 
                             Object message, String messageId) {
        
        // 发送原始消息
        rabbitTemplate.convertAndSend(exchange, routingKey, message,
            msg -> {
                msg.getMessageProperties().setMessageId(messageId);
                msg.getMessageProperties().setHeader("retry-count", 0);
                msg.getMessageProperties().setHeader("max-retries", 3);
                return msg;
            });
        
        System.out.println("发送原始消息: " + messageId);
    }
    
    /**
     * 处理消息（带重试）
     */
    @RabbitListener(queues = "retry.queue")
    public void handleWithRetry(String message,
                               Channel channel,
                               @Header(AmqpHeaders.DELIVERY_TAG) long deliveryTag,
                               @Header(AmqpHeaders.MESSAGE_ID) String messageId,
                               @Header("retry-count") Integer retryCount,
                               @Header("max-retries") Integer maxRetries) 
            throws IOException {
        
        System.out.println("处理消息: " + messageId);
        System.out.println("重试次数: " + retryCount + "/" + maxRetries);
        
        try {
            // 处理消息
            processMessage(message);
            
            // 确认消息
            channel.basicAck(deliveryTag, false);
            System.out.println("处理成功: " + messageId);
            
        } catch (Exception e) {
            System.err.println("处理失败: " + e.getMessage());
            
            if (retryCount < maxRetries) {
                // 安排延迟重试
                scheduleRetry(message, messageId, retryCount + 1, maxRetries);
                
                // 确认原始消息
                channel.basicAck(deliveryTag, false);
                
                System.out.println("安排重试: 第" + (retryCount + 1) + "次");
            } else {
                // 达到最大重试次数
                sendToDeadLetter(message, messageId, retryCount, e);
                
                // 确认消息
                channel.basicAck(deliveryTag, false);
                
                System.err.println("达到最大重试次数: " + messageId);
            }
        }
    }
    
    /**
     * 安排延迟重试
     */
    private void scheduleRetry(String message, String messageId, 
                              int retryCount, int maxRetries) {
        
        // 计算重试延迟
        long delay = calculateRetryDelay(retryCount);
        
        // 发送到延迟重试队列
        rabbitTemplate.convertAndSend("retry.delayed.exchange", "retry.delayed.key", 
            message, msg -> {
                msg.getMessageProperties().setMessageId(messageId);
                msg.getMessageProperties().setHeader("x-delay", delay);
                msg.getMessageProperties().setHeader("retry-count", retryCount);
                msg.getMessageProperties().setHeader("max-retries", maxRetries);
                msg.getMessageProperties().setHeader("retry-delay", delay);
                return msg;
            });
    }
    
    /**
     * 延迟重试消费者
     */
    @RabbitListener(queues = "retry.delayed.queue")
    public void handleDelayedRetry(String message,
                                  @Header(AmqpHeaders.MESSAGE_ID) String messageId,
                                  @Header("retry-count") Integer retryCount,
                                  @Header("max-retries") Integer maxRetries,
                                  @Header("retry-delay") Long retryDelay) {
        
        System.out.println("执行延迟重试: " + messageId);
        System.out.println("重试次数: " + retryCount);
        System.out.println("重试延迟: " + retryDelay + "ms");

        rabbitTemplate.convertAndSend("retry.exchange", "retry.key", message,
            msg -> {

                msg.getMessageProperties().setHeader("retry-count", retryCount);
                msg.getMessageProperties().setHeader("max-retries", maxRetries);
                return msg;
            });
    }
    
    /**
     * 计算重试延迟
     */
    private long calculateRetryDelay(int retryCount) {
        // 指数退避算法
        RetryLevel level = getRetryLevel(retryCount);
        return level.getDelayMillis();
    }
    
    private RetryLevel getRetryLevel(int retryCount) {
        switch (retryCount) {
            case 1: return RetryLevel.LEVEL_1;
            case 2: return RetryLevel.LEVEL_2;
            case 3: return RetryLevel.LEVEL_3;
            case 4: return RetryLevel.LEVEL_4;
            default: return RetryLevel.LEVEL_5;
        }
    }
    
    /**
     * 发送到死信队列
     */
    private void sendToDeadLetter(String message, String messageId, 
                                 int retryCount, Exception e) {
        
        DeadLetterMessage deadLetter = new DeadLetterMessage(
            messageId, message, retryCount, e.getMessage(), System.currentTimeMillis());
        
        rabbitTemplate.convertAndSend("dead.letter.exchange", "dead.letter.key", 
            deadLetter);
    }
    
    /**
     * 智能重试策略

    public void sendWithSmartRetry(String exchange, String routingKey,

                                  RetryStrategy strategy) {

                msg.getMessageProperties().setHeader("max-retries", strategy.getMaxRetries());

            });
    }

        if (Math.random() < 0.3) { // 30%失败率
```

```
                generateMonitoringReport();
            } catch (Exception e) {

    private void monitorDelayedMessages() {

        messageStats.values().removeIf(stats -> {
            if (stats.getStatus() == DeliveryStatus.DELIVERED) {
                // 已投递，检查是否需要清理
                if (now - stats.getActualDelivery() > 3600000) { // 1小时
                    return true;
                }
            } else if (stats.getStatus() == DeliveryStatus.PENDING) {
                // 检查是否过期
                if (now > stats.getExpectedDelivery() + 300000) { // 预期时间+5分钟
                    stats.setStatus(DeliveryStatus.EXPIRED);
                    handleExpiredMessage(stats);
                    return true;
                }

            return false;
```

```
                    msg.getMessageProperties().setHeader("x-delay", delayMillis);
```

## 七、 常见问题

### Q1: TTL方案 vs 延迟插件？

|特性|TTL+死信队列|延迟消息插件|
|---|---|---|
|安装要求|RabbitMQ自带|需要安装插件|
|延迟精度|较低|较高|
|灵活性|需要多个队列|单个队列支持多种延迟|
|性能|较好|较好|
|推荐|简单场景|生产环境推荐|

### Q2: 最大延迟时间？

- TTL方案：最多约49天（2^32-1毫秒）
    
- 延迟插件：理论无限制，但建议不超过30天
    
- 实际限制：业务需求、消息存储时间
    

### Q3: 延迟精度如何？

- RabbitMQ不保证精确延迟
    
- 实际延迟受负载、网络等因素影响
    
- 建议：业务允许±10%误差
    

### Q4: 如何监控延迟？

- 记录发送时间和预期投递时间
    
- 监控实际投递时间
    
- 计算延迟误差
    
- 设置告警阈值
    

延迟消息是RabbitMQ的重要功能，合理使用可以实现复杂的定时和调度需求，提高系统灵活性和用户体验。
