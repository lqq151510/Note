## 一、 核心架构设计

## 二、 基础实现方案

### 方案1：TTL+死信队列（推荐）

```
@Configuration
public class OrderTimeoutTTLConfig {
    
    // 1. 死信交换机配置
    @Bean
    public DirectExchange orderDlxExchange() {
        return new DirectExchange("order.dlx.exchange", true, false);
    }
    
    // 2. 死信队列
    @Bean
    public Queue orderDlxQueue() {
        return QueueBuilder.durable("order.dlx.queue")
                .build();
    }
    
    // 3. 死信绑定
    @Bean
    public Binding orderDlxBinding() {
        return BindingBuilder.bind(orderDlxQueue())
                .to(orderDlxExchange())
                .with("order.dlx.key");
    }
    
    // 4. 订单延迟队列（TTL=15分钟）
    @Bean
    public Queue orderDelayQueue() {
        Map<String, Object> args = new HashMap<>();
        
        // 设置死信交换机
        args.put("x-dead-letter-exchange", "order.dlx.exchange");
        // 设置死信路由键
        args.put("x-dead-letter-routing-key", "order.dlx.key");
        // 设置队列TTL为15分钟（900000毫秒）
        args.put("x-message-ttl", 900000);
        
        return QueueBuilder.durable("order.delay.queue")
                .withArguments(args)
                .build();
    }
    
    // 5. 订单服务
    @Service
    public class OrderService {
        
        @Autowired
        private RabbitTemplate rabbitTemplate;
        
        @Autowired
        private OrderRepository orderRepository;
        
        // 订单状态
        public enum OrderStatus {
            CREATED,    // 已创建
            PAID,       // 已支付
            CANCELLED,  // 已取消
            COMPLETED   // 已完成
        }
        
        /**
         * 创建订单（发送延迟消息）
         */
        @Transactional
        public Order createOrder(CreateOrderRequest request) {
            // 生成订单号
            String orderNo = generateOrderNo();
            
            // 创建订单实体
            Order order = new Order();
            order.setOrderNo(orderNo);
            order.setUserId(request.getUserId());
            order.setAmount(request.getAmount());
            order.setStatus(OrderStatus.CREATED);
            order.setCreateTime(LocalDateTime.now());
            
            // 保存订单
            orderRepository.save(order);
            
            // 发送延迟消息（15分钟后超时）
            sendOrderTimeoutMessage(orderNo);
            
            // 记录日志
            log.info("订单创建成功，订单号: {}, 设置15分钟超时", orderNo);
            
            return order;
        }
        
        /**
         * 支付订单（消费延迟消息）
         */
        @Transactional
        public boolean payOrder(String orderNo, PayOrderRequest request) {
            // 1. 检查订单状态
            Order order = orderRepository.findByOrderNo(orderNo)
                .orElseThrow(() -> new BusinessException("订单不存在"));
            
            if (order.getStatus() != OrderStatus.CREATED) {
                throw new BusinessException("订单状态不正确");
            }
            
            // 2. 更新订单状态
            order.setStatus(OrderStatus.PAID);
            order.setPayTime(LocalDateTime.now());
            order.setPayMethod(request.getPayMethod());
            order.setTransactionId(request.getTransactionId());
            
            orderRepository.save(order);
            
            // 3. 记录支付日志
            log.info("订单支付成功，订单号: {}, 交易号: {}", orderNo, request.getTransactionId());
            
            return true;
        }
        
        /**
         * 发送订单超时消息
         */
        private void sendOrderTimeoutMessage(String orderNo) {
            OrderTimeoutMessage message = new OrderTimeoutMessage();
            message.setOrderNo(orderNo);
            message.setSendTime(System.currentTimeMillis());
            message.setExpectedCancelTime(System.currentTimeMillis() + 900000);
            
            rabbitTemplate.convertAndSend("", "order.delay.queue", message);
        }
        
        /**
         * 处理订单超时
         */
        @RabbitListener(queues = "order.dlx.queue")
        @Transactional
        public void handleOrderTimeout(OrderTimeoutMessage message) {
            String orderNo = message.getOrderNo();
            
            log.info("收到订单超时消息，订单号: {}", orderNo);
            
            try {
                // 查询订单
                Order order = orderRepository.findByOrderNo(orderNo)
                    .orElseThrow(() -> {
                        log.warn("订单不存在，订单号: {}", orderNo);
                        return new BusinessException("订单不存在");
                    });
                
                // 检查订单状态
                if (order.getStatus() == OrderStatus.CREATED) {
                    // 执行取消逻辑
                    cancelOrder(order);
                } else if (order.getStatus() == OrderStatus.PAID) {
                    log.info("订单已支付，忽略超时取消，订单号: {}", orderNo);
                } else {
                    log.warn("订单状态异常，状态: {}, 订单号: {}", order.getStatus(), orderNo);
                }
                
            } catch (Exception e) {
                log.error("处理订单超时失败，订单号: {}, 错误: {}", orderNo, e.getMessage(), e);
                // 可以记录到死信或者重试
            }
        }
        
        /**
         * 取消订单逻辑
         */
        private void cancelOrder(Order order) {
            // 1. 更新订单状态
            order.setStatus(OrderStatus.CANCELLED);
            order.setCancelTime(LocalDateTime.now());
            order.setCancelReason("超时未支付");
            
            orderRepository.save(order);
            
            // 2. 释放库存（如果有的话）
            releaseInventory(order);
            
            // 3. 发送取消通知
            sendCancelNotification(order);
            
            // 4. 记录取消日志
            log.info("订单超时取消成功，订单号: {}, 取消时间: {}", 
                order.getOrderNo(), order.getCancelTime());
        }
        
        private String generateOrderNo() {
            // 生成订单号逻辑
            return "ORD" + System.currentTimeMillis() + RandomUtil.randomNumbers(4);
        }
        
        private void releaseInventory(Order order) {
            // 释放库存逻辑
            log.info("释放订单库存，订单号: {}", order.getOrderNo());
        }
        
        private void sendCancelNotification(Order order) {
            // 发送取消通知
            log.info("发送订单取消通知，订单号: {}", order.getOrderNo());
        }
    }
    
    // 订单超时消息实体
    @Data
    public static class OrderTimeoutMessage {
        private String orderNo;
        private Long sendTime;
        private Long expectedCancelTime;
    }
}
```

```
    }
    
    // 4. 使用延迟插件的订单服务
    @Service
    public class PluginOrderService {
        
        @Autowired
        private RabbitTemplate rabbitTemplate;
        
        /**
         * 发送精确延迟消息
         */
        public void sendPreciseTimeoutMessage(String orderNo, long timeoutMinutes) {
            long delayMillis = timeoutMinutes * 60 * 1000;
            
            OrderTimeoutMessage message = new OrderTimeoutMessage();
            message.setOrderNo(orderNo);
            message.setTimeoutMinutes(timeoutMinutes);
            message.setSendTime(System.currentTimeMillis());
            
            rabbitTemplate.convertAndSend("order.delayed.exchange", 
                "order.delayed.key", message,
                msg -> {
                    // 设置延迟头
                    msg.getMessageProperties().setHeader("x-delay", delayMillis);
                    msg.getMessageProperties().setHeader("order-no", orderNo);
                    return msg;
                });
            
            log.info("发送精确延迟消息，订单号: {}, 延迟: {}分钟", orderNo, timeoutMinutes);
        }
        
        /**
         * 可配置的延迟时间
         */
        public void sendConfigurableTimeout(String orderNo, String orderType) {
            long timeoutMinutes = getTimeoutByOrderType(orderType);
            sendPreciseTimeoutMessage(orderNo, timeoutMinutes);
        }
        
        private long getTimeoutByOrderType(String orderType) {
            switch (orderType) {
                case "NORMAL": return 15;    // 普通订单15分钟
                case "FLASH": return 5;      // 闪购订单5分钟
                case "GROUP": return 30;     // 团购订单30分钟
                default: return 15;
            }
        }
    }
}
```

```
@Component
public class ConfigurableTimeoutService {
    
    @Autowired
    private RabbitTemplate rabbitTemplate;
    
    @Autowired
    private OrderTimeoutConfigRepository configRepository;
    
    /**
     * 根据配置发送延迟消息
     */
    public void sendConfigurableTimeout(String orderNo, String orderType, Long userId) {
        // 1. 获取配置的超时时间
        TimeoutConfig config = getTimeoutConfig(orderType, userId);
        
        // 2. 计算实际超时时间
        long timeoutMillis = calculateTimeout(config);
        
        // 3. 发送延迟消息
        sendDelayedMessage(orderNo, timeoutMillis, config);
    }
    
    /**
     * 获取超时配置
     */
    private TimeoutConfig getTimeoutConfig(String orderType, Long userId) {
        // 1. 用户级别配置
        TimeoutConfig userConfig = configRepository.findByUserIdAndOrderType(userId, orderType);
        if (userConfig != null) {
            return userConfig;
        }
        
        // 2. 订单类型配置
        TimeoutConfig typeConfig = configRepository.findByOrderType(orderType);
        if (typeConfig != null) {
            return typeConfig;
        }
        
        // 3. 默认配置
        return getDefaultConfig();
    }
    
    /**
     * 计算实际超时时间
     */
    private long calculateTimeout(TimeoutConfig config) {
        long baseTimeout = config.getBaseTimeout();
        
        // 考虑时间因素
        LocalDateTime now = LocalDateTime.now();
        int hour = now.getHour();
        
        // 夜间延长超时时间
        if (hour >= 22 || hour <= 6) {
            baseTimeout = (long) (baseTimeout * 1.5); // 延长50%
        }
        
        // 周末延长
        DayOfWeek dayOfWeek = now.getDayOfWeek();
        if (dayOfWeek == DayOfWeek.SATURDAY || dayOfWeek == DayOfWeek.SUNDAY) {
            baseTimeout = (long) (baseTimeout * 1.2); // 延长20%
        }
        
        return Math.min(baseTimeout, config.getMaxTimeout());
    }
    
    /**
     * 发送延迟消息
     */
    private void sendDelayedMessage(String orderNo, long timeoutMillis, TimeoutConfig config) {
        OrderTimeoutMessage message = new OrderTimeoutMessage();
        message.setOrderNo(orderNo);
        message.setTimeoutMillis(timeoutMillis);
        message.setConfigId(config.getId());
        message.setSendTime(System.currentTimeMillis());
        
        rabbitTemplate.convertAndSend("order.delayed.exchange", 
            "order.delayed.key", message,
            msg -> {
                msg.getMessageProperties().setHeader("x-delay", timeoutMillis);
                msg.getMessageProperties().setHeader("order-no", orderNo);
                msg.getMessageProperties().setHeader("config-id", config.getId());
                return msg;
            });
        
        log.info("发送配置化延迟消息，订单号: {}, 配置ID: {}, 延迟: {}ms", 
            orderNo, config.getId(), timeoutMillis);
    }
    
    /**
     * 动态更新超时时间
     */
    public boolean updateOrderTimeout(String orderNo, long newTimeoutMillis) {
        // 1. 检查订单状态
        Order order = getOrder(orderNo);
        if (order.getStatus() != OrderStatus.CREATED) {
            return false;
        }
        
        // 2. 取消原有的延迟消息
        cancelTimeoutMessage(orderNo);
        
        // 3. 发送新的延迟消息
        sendDelayedMessage(orderNo, newTimeoutMillis, getDefaultConfig());
        
        log.info("更新订单超时时间，订单号: {}, 新超时: {}ms", orderNo, newTimeoutMillis);
        return true;
    }
    
    /**
     * 取消超时消息
     */
    private void cancelTimeoutMessage(String orderNo) {
        // 在实际应用中，需要记录消息ID然后取消
        // 这里简化处理
        log.info("取消订单超时消息，订单号: {}", orderNo);
    }
    
    private Order getOrder(String orderNo) {
        // 获取订单
        return new Order();
    }
    
    private TimeoutConfig getDefaultConfig() {
        TimeoutConfig config = new TimeoutConfig();
        config.setId(1L);
        config.setBaseTimeout(900000L);  // 15分钟
        config.setMaxTimeout(3600000L);  // 1小时
        return config;
    }
    
    // 超时配置实体
    @Data
    public static class TimeoutConfig {
        private Long id;
        private String orderType;
        private Long userId;
        private Long baseTimeout;  // 基础超时时间（毫秒）
        private Long maxTimeout;   // 最大超时时间（毫秒）
        private String description;
    }
}
```

### 2. 消息幂等性处理

```
@Component
public class IdempotentOrderTimeout {
    
    @Autowired
    private RedisTemplate<String, String> redisTemplate;
    
    @Autowired
    private OrderRepository orderRepository;
    
    /**
     * 幂等性处理订单超时
     */
    @RabbitListener(queues = "order.dlx.queue")
    @Transactional
    public void handleOrderTimeoutIdempotent(OrderTimeoutMessage message) {
        String orderNo = message.getOrderNo();
        
        // 1. 幂等性检查
        String processedKey = "order:timeout:processed:" + orderNo;
        Boolean notProcessed = redisTemplate.opsForValue()
            .setIfAbsent(processedKey, "1", 24, TimeUnit.HOURS);
        
        if (Boolean.FALSE.equals(notProcessed)) {
            log.info("订单超时消息已处理，跳过，订单号: {}", orderNo);
            return;
        }
        
        try {
            // 2. 查询订单
            Order order = orderRepository.findByOrderNoForUpdate(orderNo)  // 使用悲观锁
                .orElseThrow(() -> {
                    log.warn("订单不存在，订单号: {}", orderNo);
                    return new BusinessException("订单不存在");
                });
            
            // 3. 状态检查
            if (order.getStatus() != OrderStatus.CREATED) {
                log.info("订单状态已更新，跳过取消，订单号: {}, 状态: {}", 
                    orderNo, order.getStatus());
                return;
            }
            
            // 4. 执行取消
            cancelOrderWithRetry(order);
            
        } catch (Exception e) {
            // 5. 处理失败，清除幂等标记
            redisTemplate.delete(processedKey);
            throw e;
        }
    }
    
    /**
     * 带重试的取消逻辑
     */
    private void cancelOrderWithRetry(Order order) {
        int maxRetries = 3;
        
        for (int i = 1; i <= maxRetries; i++) {
            try {
                cancelOrder(order);
                log.info("订单取消成功，订单号: {}, 重试次数: {}", order.getOrderNo(), i-1);
                return;
                
            } catch (Exception e) {
                if (i == maxRetries) {
                    throw e;
                }
                
                log.warn("订单取消失败，准备重试，订单号: {}, 重试次数: {}", 
                    order.getOrderNo(), i);
                
                // 指数退避
                try {
                    Thread.sleep(1000L * (1L << (i-1)));
                } catch (InterruptedException ie) {
                    Thread.currentThread().interrupt();
                    throw new RuntimeException(ie);
                }
            }
        }
    }
    
    /**
     * 分布式锁取消
     */
    public void cancelOrderWithDistributedLock(Order order) {
        String lockKey = "order:cancel:lock:" + order.getOrderNo();
        
        // 获取分布式锁
        Boolean lockAcquired = redisTemplate.opsForValue()
            .setIfAbsent(lockKey, "1", 10, TimeUnit.SECONDS);
        
        if (Boolean.FALSE.equals(lockAcquired)) {
            throw new BusinessException("订单正在处理中，请稍后");
        }
        
        try {
            // 双重检查
            Order currentOrder = orderRepository.findByOrderNo(order.getOrderNo())
                .orElseThrow(() -> new BusinessException("订单不存在"));
            
            if (currentOrder.getStatus() != OrderStatus.CREATED) {
                return;
            }
            
            // 执行取消
            cancelOrder(currentOrder);
            
        } finally {
            // 释放锁
            redisTemplate.delete(lockKey);
        }
    }
    
    private void cancelOrder(Order order) {
        // 取消订单逻辑
        order.setStatus(OrderStatus.CANCELLED);
        order.setCancelTime(LocalDateTime.now());
        order.setCancelReason("超时未支付");
        
        orderRepository.save(order);
    }
}
```

```
@Component
public class BatchOrderTimeout {
    
    @Autowired
    private RabbitTemplate rabbitTemplate;
    
    /**
     * 批量创建订单
     */
    public List<String> batchCreateOrders(List<CreateOrderRequest> requests) {
        List<String> orderNos = new ArrayList<>();
        
        for (CreateOrderRequest request : requests) {
            try {
                String orderNo = createOrderWithTimeout(request);
                orderNos.add(orderNo);
            } catch (Exception e) {
                log.error("创建订单失败: {}", e.getMessage(), e);
            }
        }
        
        return orderNos;
    }
    
    /**
     * 批量取消超时订单
     */
    @Scheduled(fixedDelay = 60000)  // 每分钟执行一次
    public void batchCancelTimeoutOrders() {
        // 1. 查询超时订单
        LocalDateTime timeoutTime = LocalDateTime.now().minusMinutes(15);
        List<Order> timeoutOrders = orderRepository.findTimeoutOrders(timeoutTime);
        
        if (timeoutOrders.isEmpty()) {
            return;
        }
        
        log.info("发现{}个超时订单需要处理", timeoutOrders.size());
        
        // 2. 分批处理
        int batchSize = 100;
        List<List<Order>> batches = ListUtil.partition(timeoutOrders, batchSize);
        
        for (List<Order> batch : batches) {
            processBatch(batch);
        }
    }
    
    /**
     * 处理批次
     */
    private void processBatch(List<Order> orders) {
        ExecutorService executor = Executors.newFixedThreadPool(10);
        List<CompletableFuture<Void>> futures = new ArrayList<>();
        
        for (Order order : orders) {
            CompletableFuture<Void> future = CompletableFuture.runAsync(() -> {
                try {
                    cancelOrderIfTimeout(order);
                } catch (Exception e) {
                    log.error("取消订单失败，订单号: {}", order.getOrderNo(), e);
                }
            }, executor);
            
            futures.add(future);
        }
        
        // 等待所有完成
        CompletableFuture.allOf(futures.toArray(new CompletableFuture[0]))
            .exceptionally(e -> {
                log.error("批次处理异常", e);
                return null;
            })
            .join();
        
        executor.shutdown();
    }
    
    /**
     * 检查并取消订单

        // 双重检查

        Duration duration = Duration.between(order.getCreateTime(), now);
        
        if (duration.toMinutes() >= 15) {
            cancelOrder(order);
            log.info("批量取消超时订单，订单号: {}", order.getOrderNo());
        }
    }

    }

    }
```

### 1. 订单超时监控

```
@Component
public class OrderTimeoutMonitor {
    
    @Autowired
    private OrderRepository orderRepository;
    
    @Autowired
    private RabbitTemplate rabbitTemplate;
    
    private final ScheduledExecutorService monitorExecutor = 
        Executors.newScheduledThreadPool(1);
    
    /**
     * 启动监控
     */
    @PostConstruct
    public void startMonitoring() {
        monitorExecutor.scheduleAtFixedRate(() -> {
            try {
                monitorTimeoutOrders();
                checkProcessingStatus();
                generateReport();
            } catch (Exception e) {
                log.error("订单超时监控异常", e);
            }
        }, 0, 60, TimeUnit.SECONDS);  // 每分钟执行一次
    }
    
    /**
     * 监控超时订单
     */
    private void monitorTimeoutOrders() {
        // 1. 查询即将超时的订单
        LocalDateTime warningTime = LocalDateTime.now().plusMinutes(5);
        List<Order> warningOrders = orderRepository.findOrdersAboutToTimeout(
            OrderStatus.CREATED, warningTime);
        
        if (!warningOrders.isEmpty()) {
            // 发送预警通知
            sendWarningNotification(warningOrders);
        }
        
        // 2. 查询已超时但未处理的订单
        LocalDateTime timeoutTime = LocalDateTime.now().minusMinutes(15);
        List<Order> timeoutOrders = orderRepository.findOrdersByStatusAndCreateTimeBefore(
            OrderStatus.CREATED, timeoutTime);
        
        if (!timeoutOrders.isEmpty()) {
            // 发送告警
            sendTimeoutAlert(timeoutOrders);
            
            // 尝试自动取消
            autoCancelOrders(timeoutOrders);
        }
    }
    
    /**
     * 检查处理状态
     */
    private void checkProcessingStatus() {
        // 检查最近取消的订单
        LocalDateTime checkTime = LocalDateTime.now().minusMinutes(5);
        List<Order> recentCancelled = orderRepository.findRecentCancelledOrders(checkTime);
        
        for (Order order : recentCancelled) {
            // 验证取消是否成功
            verifyCancelSuccess(order);
        }
    }
    
    /**
     * 发送预警通知
     */
    private void sendWarningNotification(List<Order> orders) {
        for (Order order : orders) {
            // 计算剩余时间
            Duration remaining = Duration.between(LocalDateTime.now(), 
                order.getCreateTime().plusMinutes(15));
            long remainingMinutes = remaining.toMinutes();
            
            if (remainingMinutes <= 5) {
                // 发送短信/推送通知
                sendNotification(order.getUserId(), 
                    String.format("您的订单即将超时，剩余%d分钟", remainingMinutes));
```

```
public class CompleteOrderTimeoutService {

    private RabbitTemplate rabbitTemplate;

    private OrderRepository orderRepository;
    
    @Autowired
    private RedisTemplate<String, String> redisTemplate;

    /**
     * 创建订单（完整流程）
     */
    @Transactional(rollbackFor = Exception.class)
    public OrderDTO createOrder(CreateOrderDTO request) {
        // 1. 参数验证
        validateCreateRequest(request);
        
        // 2. 生成订单
        Order order = generateOrder(request);
        
        // 3. 保存订单
        orderRepository.save(order);
        
        // 4. 发送延迟消息
        String messageId = sendTimeoutMessage(order);
        
        // 5. 记录发送记录
        recordMessageSend(order.getOrderNo(), messageId);
        
        // 6. 发送创建成功事件
        sendOrderCreatedEvent(order);
        
        // 7. 返回结果
        return convertToDTO(order);
    }
    
    /**
     * 支付订单
     */
    @Transactional(rollbackFor = Exception.class)
    public PayOrderResult payOrder(PayOrderRequest request) {
        // 1. 幂等性检查
        if (!checkPaymentIdempotent(request.getPaymentNo())) {
            return PayOrderResult.alreadyPaid();
        }
        
        // 2. 查询订单
        Order order = getAndLockOrder(request.getOrderNo());
        
        // 3. 验证订单状态
        validateOrderForPayment(order);
        
        // 4. 执行支付
        boolean paymentSuccess = processPayment(order, request);
        
        if (paymentSuccess) {
            // 5. 更新订单状态
            updateOrderAsPaid(order, request);
            
            // 6. 取消超时消息
            cancelTimeoutMessage(order.getOrderNo());
            
            // 7. 发送支付成功事件
            sendPaymentSuccessEvent(order);
            
            return PayOrderResult.success(order.getOrderNo());
        } else {
            // 支付失败处理
            handlePaymentFailure(order, request);
            return PayOrderResult.failed("支付失败");
        }
    }
    
    /**
     * 处理订单超时
     */
    @RabbitListener(
        queues = "${order.delay-message.dlx-queue:order.dlx.queue}",
        containerFactory = "orderContainerFactory"
    )
    @Transactional(rollbackFor = Exception.class)
    public void processOrderTimeout(OrderTimeoutMessage message) {
        String orderNo = message.getOrderNo();
        long startTime = System.currentTimeMillis();
        
        try {
            log.info("开始处理订单超时，订单号: {}", orderNo);
            
            // 1. 幂等性检查
            if (!checkTimeoutIdempotent(orderNo)) {
                log.info("订单超时已处理，跳过，订单号: {}", orderNo);
                return;
            }
            
            // 2. 查询并锁定订单
            Order order = getAndLockOrder(orderNo);
            
            // 3. 检查订单状态
            if (order.getStatus() != OrderStatus.CREATED) {
                log.info("订单状态已更新，无需取消，订单号: {}, 状态: {}", 
                    orderNo, order.getStatus());
                return;
            }
            
            // 4. 执行取消
            cancelOrder(order);
            
            // 5. 记录处理结果
            recordCancelResult(order, true, null);
            
            long costTime = System.currentTimeMillis() - startTime;
            log.info("订单超时处理完成，订单号: {}, 耗时: {}ms", orderNo, costTime);
            
        } catch (Exception e) {
            // 6. 记录失败
            recordCancelResult(null, false, e.getMessage());
            
            long costTime = System.currentTimeMillis() - startTime;
            log.error("订单超时处理失败，订单号: {}, 耗时: {}ms, 错误: {}", 
                orderNo, costTime, e.getMessage(), e);
            
            // 7. 根据异常类型决定是否重试
            if (shouldRetry(e)) {
                throw e; // 抛出异常触发重试
            }
        }
    }

        order.setStatus(OrderStatus.CANCELLED);
        order.setCancelTime(LocalDateTime.now());
        order.setCancelReason("超时未支付");
        orderRepository.save(order);
        
        // 2. 释放库存
        releaseInventory(order);
        
        // 3. 退还优惠券
        refundCoupon(order);
        
        // 4. 记录取消日志
        logCancelOperation(order);
        
        // 5. 发送取消通知
        sendCancelNotification(order);
        
        // 6. 发送取消事件
        sendOrderCancelledEvent(order);
    }
    
    /**

    private String sendTimeoutMessage(Order order) {
        long delayMillis = defaultTimeoutMinutes * 60 * 1000L;
        
        OrderTimeoutMessage message = new OrderTimeoutMessage();
        message.setOrderNo(order.getOrderNo());
        message.setUserId(order.getUserId());
        message.setAmount(order.getAmount());
        message.setTimeoutMinutes(defaultTimeoutMinutes);
        message.setSendTime(System.currentTimeMillis());
        
        CorrelationData correlationData = new CorrelationData(UUID.randomUUID().toString());
        
        rabbitTemplate.convertAndSend(
            "${order.delay-message.exchange}",
            "${order.delay-message.routing-key}",
            message,

                msg.getMessageProperties().setHeader("x-delay", delayMillis);
                msg.getMessageProperties().setMessageId(correlationData.getId());
                msg.getMessageProperties().setCorrelationId(order.getOrderNo());
                msg.getMessageProperties().setHeader("order-no", order.getOrderNo());
                msg.getMessageProperties().setHeader("send-time", System.currentTimeMillis());
                msg.getMessageProperties().setDeliveryMode(MessageDeliveryMode.PERSISTENT);
                return msg;
            },
            correlationData
        );
        
        return correlationData.getId();
    }
    
    /**

     */
```

```
@Component
public class DuplicateConsumptionSolution {
    
    @Autowired
    private RedisTemplate<String, String> redisTemplate;
    
    /**
     * 解决方案1：Redis幂等性
     */
    public boolean processOrderWithRedisIdempotent(String orderNo, Runnable processor) {
        String key = "order:processed:" + orderNo;
        
        // SETNX原子操作
        Boolean success = redisTemplate.opsForValue()
            .setIfAbsent(key, "processing", 1, TimeUnit.HOURS);
        
        if (Boolean.FALSE.equals(success)) {
            log.warn("订单正在处理或已处理: {}", orderNo);
            return false;
        }
        
        try {
            processor.run();
            
            // 更新状态
            redisTemplate.opsForValue().set(key, "processed", 24, TimeUnit.HOURS);
            return true;
            
        } catch (Exception e) {
            // 处理失败，删除key
            redisTemplate.delete(key);
            throw e;
        }
    }
    
    /**
     * 解决方案2：数据库乐观锁
     */
    @Transactional
    public boolean processOrderWithOptimisticLock(String orderNo, Runnable processor) {
        Order order = orderRepository.findByOrderNo(orderNo)
            .orElseThrow(() -> new BusinessException("订单不存在"));
        
        int currentVersion = order.getVersion();
        
        // 尝试更新版本
        int updated = orderRepository.updateOrderVersion(orderNo, currentVersion);
        
        if (updated == 0) {
            // 版本冲突，说明其他进程已处理
            log.warn("订单版本冲突: {}", orderNo);
            return false;
        }
        
        processor.run();
        return true;
    }
    
    /**
     * 解决方案3：业务状态机
     */
    public boolean processWithStateMachine(String orderNo, OrderStatus targetStatus) {
        Order order = getOrder(orderNo);
        
        // 检查状态转换是否允许
        if (!order.canTransitionTo(targetStatus)) {
            log.warn("无效状态转换: {} -> {}", order.getStatus(), targetStatus);
            return false;
        }

            orderNo, order.getStatus(), targetStatus);
```

## 总结

取消超时订单是电商系统的核心功能，通过RabbitMQ延迟消息可以优雅地实现。关键点：

1. **选择合适的方案**：TTL+死信队列简单可靠，延迟消息插件更精确
    
2. **保证幂等性**：防止重复取消
    
3. **完善监控**：及时发现问题
    
4. **考虑性能**：批量处理、异步化
    
5. **容错处理**：重试机制、补偿任务
    

这个方案可以扩展到其他超时场景，如：优惠券过期、会话超时、预约超时等。
