# RabbitMQ 发送者确认（Publisher Confirms）

发送者确认机制是 RabbitMQ 保证消息可靠投递的核心特性，确保消息成功到达 RabbitMQ 服务器。

## 一、 基础确认机制配置

### 1. 开启确认机制

```
@Configuration
public class PublisherConfirmConfig {
    
    /**
     * 配置支持确认机制的连接工厂
     */
    @Bean
    public ConnectionFactory confirmConnectionFactory() {
        CachingConnectionFactory factory = new CachingConnectionFactory();
        factory.setHost("localhost");
        factory.setPort(5672);
        
        // ========== 确认机制配置 ==========
        
        // 1. 开启发布者确认
        factory.setPublisherConfirmType(CachingConnectionFactory.ConfirmType.CORRELATED);
        
        // 2. 开启发布者返回（消息路由失败返回）
        factory.setPublisherReturns(true);
        
        // 3. 确认超时时间（毫秒）
        factory.setPublisherConfirmType(CachingConnectionFactory.ConfirmType.CORRELATED);
        
        // 4. 其他相关配置
        factory.setAutomaticRecoveryEnabled(true);
        factory.setRequestedHeartbeat(60);
        factory.setConnectionTimeout(30000);
        
        return factory;
    }
    
    /**
     * 配置支持确认的 RabbitTemplate
     */
    @Bean
    public RabbitTemplate confirmRabbitTemplate(ConnectionFactory connectionFactory) {
        RabbitTemplate template = new RabbitTemplate(connectionFactory);
        
        // 设置消息转换器
        template.setMessageConverter(new Jackson2JsonMessageConverter());
        
        // ========== 确认回调配置 ==========
        
        // 1. 确认回调（消息到达交换机）
        template.setConfirmCallback((correlationData, ack, cause) -> {
            if (ack) {
                System.out.println("消息发送到交换机成功: " + correlationData);
            } else {
                System.err.println("消息发送到交换机失败: " + cause);
                // 处理发送失败
                handleConfirmFailure(correlationData, cause);
            }
        });
        
        // 2. 返回回调（消息路由到队列失败）
        template.setReturnsCallback(returned -> {
            System.err.println("消息路由到队列失败: " + returned);
            // 处理返回消息
            handleReturnedMessage(returned);
        });
        
        // 3. 开启强制模式（消息路由失败时触发返回）
        template.setMandatory(true);
        
        // 4. 设置确认超时
        template.setReplyTimeout(5000);
        
        return template;
    }
    
    private void handleConfirmFailure(CorrelationData correlationData, String cause) {
        // 处理确认失败
        System.err.println("处理确认失败: " + correlationData + ", 原因: " + cause);
    }
    
    private void handleReturnedMessage(ReturnedMessage returned) {
        // 处理返回消息
        System.err.println("处理返回消息: " + returned);
    }
}

## 二、 CorrelationData 管理

### 1. 增强的 CorrelationData

```
@Component
public class EnhancedCorrelationData extends CorrelationData {
    
    private final String messageId;
    private final Object message;
    private final String exchange;
    private final String routingKey;
    private final long sendTime;
    private volatile ConfirmStatus status = ConfirmStatus.PENDING;
    private volatile String errorCause;
    private volatile ReturnedMessage returnedMessage;
    
    public EnhancedCorrelationData(String messageId, Object message, 
                                  String exchange, String routingKey) {
        super(messageId);
        this.messageId = messageId;
        this.message = message;
        this.exchange = exchange;
        this.routingKey = routingKey;
        this.sendTime = System.currentTimeMillis();
    }
    
    /**
     * 确认成功回调
     */
    public void onAck() {
        this.status = ConfirmStatus.ACK;
        this.errorCause = null;
        logConfirm("确认成功");
    }
    
    /**
     * 确认失败回调
     */
    public void onNack(String cause) {
        this.status = ConfirmStatus.NACK;
        this.errorCause = cause;
        logConfirm("确认失败: " + cause);
    }
    
    /**
     * 消息返回回调
     */
    public void onReturn(ReturnedMessage returned) {
        this.status = ConfirmStatus.RETURNED;
        this.returnedMessage = returned;
        logConfirm("消息返回: " + returned.getReplyText());
    }
    
    /**
     * 超时回调
     */
    public void onTimeout() {
        this.status = ConfirmStatus.TIMEOUT;
        this.errorCause = "确认超时";
        logConfirm("确认超时");
    }
    
    /**
     * 获取确认等待时间
     */
    public long getConfirmDuration() {
        return System.currentTimeMillis() - sendTime;
    }
    
    /**
     * 是否确认完成
     */
    public boolean isConfirmed() {
        return status == ConfirmStatus.ACK;
    }
    
    /**
     * 是否失败
     */
    public boolean isFailed() {
        return status == ConfirmStatus.NACK || 
               status == ConfirmStatus.RETURNED || 
               status == ConfirmStatus.TIMEOUT;
    }
    
    /**
     * 获取失败原因
     */
    public String getFailureReason() {
        if (status == ConfirmStatus.NACK || status == ConfirmStatus.TIMEOUT) {
            return errorCause;
        } else if (status == ConfirmStatus.RETURNED && returnedMessage != null) {
            return returnedMessage.getReplyText();
        }
        return null;
    }
    
    private void logConfirm(String message) {
        System.out.println("消息确认 [" + messageId + "]: " + message + 
                          ", 耗时: " + getConfirmDuration() + "ms");
    }
    
    /**
     * 确认状态枚举
     */
    public enum ConfirmStatus {
        PENDING,    // 等待确认
        ACK,        // 确认成功
        NACK,       // 确认失败
        RETURNED,   // 消息返回
        TIMEOUT     // 确认超时
    }
    
    // Getter 方法
    public String getMessageId() { return messageId; }
    public Object getMessage() { return message; }
    public String getExchange() { return exchange; }
    public String getRoutingKey() { return routingKey; }
    public ConfirmStatus getStatus() { return status; }
    public String getErrorCause() { return errorCause; }
    public ReturnedMessage getReturnedMessage() { return returnedMessage; }
}

### 2. CorrelationData 管理器

```
@Component
public class CorrelationDataManager {
    
    private final ConcurrentMap<String, EnhancedCorrelationData> pendingConfirms = 
        new ConcurrentHashMap<>();
    
    private final ScheduledExecutorService timeoutScheduler = 
        Executors.newSingleThreadScheduledExecutor();
    
    /**
     * 创建并注册 CorrelationData
     */
    public EnhancedCorrelationData createCorrelationData(String exchange, 
                                                         String routingKey, 
                                                         Object message) {
        String messageId = UUID.randomUUID().toString();
        EnhancedCorrelationData correlationData = 
            new EnhancedCorrelationData(messageId, message, exchange, routingKey);
        
        // 注册到管理器
        pendingConfirms.put(messageId, correlationData);
        
        // 设置超时检查
        scheduleTimeoutCheck(correlationData, 5000); // 5秒超时
        
        return correlationData;
    }
    
    /**
     * 处理确认回调
     */
    public void handleConfirm(CorrelationData correlationData, boolean ack, String cause) {
        if (correlationData instanceof EnhancedCorrelationData) {
            EnhancedCorrelationData enhancedData = (EnhancedCorrelationData) correlationData;
            
            if (ack) {
                enhancedData.onAck();
            } else {
                enhancedData.onNack(cause);
            }
            
            // 从等待列表中移除
            pendingConfirms.remove(enhancedData.getMessageId());
            
            // 触发确认完成回调
            notifyConfirmComplete(enhancedData);
        }
    }
    
    /**
     * 处理返回回调
     */
    public void handleReturn(ReturnedMessage returned) {
        String messageId = returned.getMessage().getMessageProperties().getMessageId();
        if (messageId != null) {
            EnhancedCorrelationData correlationData = pendingConfirms.get(messageId);
            if (correlationData != null) {
                correlationData.onReturn(returned);
                pendingConfirms.remove(messageId);
                notifyConfirmComplete(correlationData);
            }
        }
    }
    
    /**
     * 检查超时消息
     */
    public void checkTimeouts() {
        long now = System.currentTimeMillis();
        List<EnhancedCorrelationData> timeouts = new ArrayList<>();
        
        for (EnhancedCorrelationData data : pendingConfirms.values()) {
            if (data.getConfirmDuration() > 10000) { // 超过10秒
                timeouts.add(data);
            }
        }
        
        for (EnhancedCorrelationData data : timeouts) {
            data.onTimeout();
            pendingConfirms.remove(data.getMessageId());
            notifyConfirmComplete(data);
        }
    }
    
    /**
     * 获取未确认消息统计
     */
    public ConfirmStats getConfirmStats() {
        ConfirmStats stats = new ConfirmStats();
        
        for (EnhancedCorrelationData data : pendingConfirms.values()) {
            stats.total++;
            long duration = data.getConfirmDuration();
            if (duration > stats.maxDuration) {
                stats.maxDuration = duration;
            }
            stats.avgDurationSum += duration;
        }
        
        if (stats.total > 0) {
            stats.avgDuration = stats.avgDurationSum / stats.total;
        }
        
        return stats;
    }
    
    /**
     * 清理管理器
     */
    public void cleanup() {
        pendingConfirms.clear();
        timeoutScheduler.shutdown();
    }
    
    private void scheduleTimeoutCheck(EnhancedCorrelationData data, long timeoutMillis) {
        timeoutScheduler.schedule(() -> {
            if (pendingConfirms.containsKey(data.getMessageId())) {
                data.onTimeout();
                pendingConfirms.remove(data.getMessageId());
                notifyConfirmComplete(data);
            }
        }, timeoutMillis, TimeUnit.MILLISECONDS);
    }
    
    private void notifyConfirmComplete(EnhancedCorrelationData data) {
        // 可以触发事件、记录日志、更新监控等
        System.out.println("确认完成: " + data.getMessageId() + 
                          ", 状态: " + data.getStatus());
    }
    
    /**
     * 确认统计
     */
    public static class ConfirmStats {
        public int total = 0;
        public long maxDuration = 0;
        public long avgDuration = 0;
        public long avgDurationSum = 0;
    }
}

## 三、 同步确认模式

### 1. 同步等待确认

```
@Component
public class SyncConfirmSender {
    
    @Autowired
    private RabbitTemplate rabbitTemplate;
    
    @Autowired
    private CorrelationDataManager correlationDataManager;
    
    /**
     * 同步发送并等待确认
     */
    public boolean sendAndConfirm(String exchange, String routingKey, 
                                 Object message, long timeout, TimeUnit unit) 
            throws Exception {
        
        // 创建 CorrelationData
        EnhancedCorrelationData correlationData = 
            correlationDataManager.createCorrelationData(exchange, routingKey, message);
        
        try {
            // 发送消息
            rabbitTemplate.convertAndSend(exchange, routingKey, message, 
                createMessagePostProcessor(correlationData), correlationData);
            
            // 等待确认
            return waitForConfirm(correlationData, timeout, unit);
            
        } catch (Exception e) {
            // 发送异常
            correlationData.onNack(e.getMessage());
            throw e;
        }
    }
    
    /**
     * 等待确认结果
     */
    private boolean waitForConfirm(EnhancedCorrelationData correlationData, 
                                  long timeout, TimeUnit unit) 
            throws InterruptedException, TimeoutException {
        
        long endTime = System.currentTimeMillis() + unit.toMillis(timeout);
        
        while (System.currentTimeMillis() < endTime) {
            EnhancedCorrelationData.ConfirmStatus status = correlationData.getStatus();
            
            if (status == EnhancedCorrelationData.ConfirmStatus.ACK) {
                return true; // 确认成功
            } else if (status == EnhancedCorrelationData.ConfirmStatus.NACK || 
                      status == EnhancedCorrelationData.ConfirmStatus.RETURNED || 
                      status == EnhancedCorrelationData.ConfirmStatus.TIMEOUT) {
                return false; // 确认失败
            }
            
            // 等待一小段时间
            Thread.sleep(100);
        }
        
        // 超时
        correlationData.onTimeout();
        throw new TimeoutException("确认超时");
    }
    
    /**
     * 批量同步发送
     */
    public Map<String, Boolean> batchSendAndConfirm(List<MessageItem> messages, 
                                                   long timeout, TimeUnit unit) {
        
        Map<String, Boolean> results = new HashMap<>();
        List<Future<Boolean>> futures = new ArrayList<>();
        ExecutorService executor = Executors.newFixedThreadPool(5);
        
        for (MessageItem item : messages) {
            Future<Boolean> future = executor.submit(() -> 
                sendAndConfirm(item.exchange, item.routingKey, 
                              item.message, timeout, unit));
            futures.add(future);
        }
        
        for (int i = 0; i < messages.size(); i++) {
            try {
                MessageItem item = messages.get(i);
                Future<Boolean> future = futures.get(i);
                
                boolean success = future.get(timeout, unit);
                results.put(item.messageId, success);
                
            } catch (Exception e) {
                results.put(messages.get(i).messageId, false);
            }
        }
        
        executor.shutdown();
        return results;
    }
    
    /**
     * 带重试的同步发送
     */
    public boolean sendWithRetryAndConfirm(String exchange, String routingKey, 
                                          Object message, int maxRetries) 
            throws Exception {
        
        for (int attempt = 1; attempt <= maxRetries; attempt++) {
            try {
                System.out.println("发送尝试 " + attempt + "/" + maxRetries);
                
                boolean confirmed = sendAndConfirm(exchange, routingKey, message, 5, TimeUnit.SECONDS);
                
                if (confirmed) {
                    System.out.println("发送并确认成功，尝试次数: " + attempt);
                    return true;
                } else {
                    System.err.println("发送失败，等待重试...");
                }
                
            } catch (Exception e) {
                System.err.println("发送异常: " + e.getMessage());
            }
            
            if (attempt < maxRetries) {
                // 指数退避等待
                long delay = calculateRetryDelay(attempt);
                Thread.sleep(delay);
            }
        }
        
        System.err.println("达到最大重试次数，发送失败");
        return false;
    }
    
    private MessagePostProcessor createMessagePostProcessor(EnhancedCorrelationData correlationData) {
        return message -> {
            MessageProperties props = message.getMessageProperties();
            
            // 设置消息ID
            props.setMessageId(correlationData.getMessageId());
            props.setCorrelationId(correlationData.getId());
            
            // 设置持久化
            props.setDeliveryMode(MessageDeliveryMode.PERSISTENT);
            props.setTimestamp(new Date());
            
            // 自定义头
            props.setHeader("send-time", System.currentTimeMillis());
            props.setHeader("attempt", 1);
            
            return message;
        };
    }
    
    private long calculateRetryDelay(int attempt) {
        // 指数退避
        return Math.min(1000L * (1L << (attempt - 1)), 30000L);
    }
    
    /**
     * 消息项
     */
    public static class MessageItem {
        public final String messageId;
        public final String exchange;
        public final String routingKey;
        public final Object message;
        
        public MessageItem(String exchange, String routingKey, Object message) {
            this.messageId = UUID.randomUUID().toString();
            this.exchange = exchange;
            this.routingKey = routingKey;
            this.message = message;
        }
    }
}

## 四、 异步确认模式

### 1. 异步确认处理器

```
@Component
public class AsyncConfirmHandler {
    
    private final Map<String, CompletableFuture<ConfirmResult>> pendingFutures = 
        new ConcurrentHashMap<>();
    
    private final ScheduledExecutorService timeoutExecutor = 
        Executors.newScheduledThreadPool(1);
    
    /**
     * 异步发送
     */
    public CompletableFuture<ConfirmResult> sendAsync(RabbitTemplate rabbitTemplate,
                                                     String exchange, 
                                                     String routingKey, 
                                                     Object message) {
        
        String messageId = UUID.randomUUID().toString();
        CompletableFuture<ConfirmResult> future = new CompletableFuture<>();
        
        // 注册等待的Future
        pendingFutures.put(messageId, future);
        
        // 创建CorrelationData
        CorrelationData correlationData = new CorrelationData(messageId);
        
        // 设置超时
        scheduleTimeout(messageId, 5000);
        
        try {
            // 异步发送
            rabbitTemplate.convertAndSend(exchange, routingKey, message, 
                createMessagePostProcessor(messageId), correlationData);
            
        } catch (Exception e) {
            // 发送立即失败
            future.completeExceptionally(e);
            pendingFutures.remove(messageId);
        }
        
        return future;
    }
    
    /**
     * 确认回调
     */
    public void onConfirm(CorrelationData correlationData, boolean ack, String cause) {
        String messageId = correlationData != null ? correlationData.getId() : null;
        if (messageId != null) {
            CompletableFuture<ConfirmResult> future = pendingFutures.remove(messageId);
            if (future != null) {
                if (ack) {
                    future.complete(new ConfirmResult(true, "ACK", null));
                } else {
                    future.complete(new ConfirmResult(false, "NACK", cause));
                }
            }
        }
    }
    
    /**
     * 返回回调
     */
    public void onReturn(ReturnedMessage returned) {
        String messageId = returned.getMessage().getMessageProperties().getMessageId();
        if (messageId != null) {
            CompletableFuture<ConfirmResult> future = pendingFutures.remove(messageId);
            if (future != null) {
                future.complete(new ConfirmResult(false, "RETURNED", 
                    returned.getReplyText()));
            }
        }
    }
    
    /**
     * 批量异步发送
     */
    public List<CompletableFuture<ConfirmResult>> batchSendAsync(
            RabbitTemplate rabbitTemplate, List<MessageItem> messages) {
        
        return messages.stream()
            .map(item -> sendAsync(rabbitTemplate, item.exchange, 
                                 item.routingKey, item.message))
            .collect(Collectors.toList());
    }
    
    /**
     * 等待所有确认
     */
    public CompletableFuture<List<ConfirmResult>> waitAllAsync(
            List<CompletableFuture<ConfirmResult>> futures) {
        
        CompletableFuture<Void> allDoneFuture = 
            CompletableFuture.allOf(futures.toArray(new CompletableFuture[0]));
        
        return allDoneFuture.thenApply(v -> 
            futures.stream()
                .map(CompletableFuture::join)
                .collect(Collectors.toList())
        );
    }
    
    /**
     * 带回调的异步发送
     */
    public void sendWithCallback(RabbitTemplate rabbitTemplate,
                                String exchange, String routingKey, 
                                Object message, ConfirmCallback callback) {
        
        sendAsync(rabbitTemplate, exchange, routingKey, message)
            .whenComplete((result, throwable) -> {
                if (throwable != null) {
                    callback.onError(throwable);
                } else {
                    if (result.success) {
                        callback.onSuccess(result);
                    } else {
                        callback.onFailure(result);
                    }
                }
            });
    }
    
    private MessagePostProcessor createMessagePostProcessor(String messageId) {
        return message -> {
            MessageProperties props = message.getMessageProperties();
            props.setMessageId(messageId);
            props.setCorrelationId(messageId);
            return message;
        };
    }
    
    private void scheduleTimeout(String messageId, long timeoutMillis) {
        timeoutExecutor.schedule(() -> {
            CompletableFuture<ConfirmResult> future = pendingFutures.remove(messageId);
            if (future != null && !future.isDone()) {
                future.complete(new ConfirmResult(false, "TIMEOUT", "确认超时"));
            }
        }, timeoutMillis, TimeUnit.MILLISECONDS);
    }
    
    /**
     * 确认结果
     */
    public static class ConfirmResult {
        public final boolean success;
        public final String type; // ACK, NACK, RETURNED, TIMEOUT
        public final String reason;
        public final long timestamp = System.currentTimeMillis();
        
        public ConfirmResult(boolean success, String type, String reason) {
            this.success = success;
            this.type = type;
            this.reason = reason;
        }
    }
    
    /**
     * 确认回调接口
     */
    public interface ConfirmCallback {
        void onSuccess(ConfirmResult result);
        void onFailure(ConfirmResult result);
        void onError(Throwable throwable);
    }
}

## 五、 批量确认模式

### 1. 批量消息处理器

```
@Component
public class BatchConfirmHandler {
    
    private final BlockingQueue<BatchMessage> messageQueue = new LinkedBlockingQueue<>();
    private final Map<String, CompletableFuture<ConfirmResult>> pendingConfirms = 
        new ConcurrentHashMap<>();
    
    private final ScheduledExecutorService batchExecutor = 
        Executors.newSingleThreadScheduledExecutor();
    
    private volatile boolean running = false;
    
    /**
     * 初始化批量处理器
     */
    @PostConstruct
    public void init() {
        startBatchProcessor();
    }
    
    /**
     * 发送批量消息
     */
    public CompletableFuture<ConfirmResult> sendBatchMessage(String exchange, 
                                                           String routingKey, 
                                                           Object message) {
        
        String messageId = UUID.randomUUID().toString();
        CompletableFuture<ConfirmResult> future = new CompletableFuture<>();
        
        pendingConfirms.put(messageId, future);
        
        BatchMessage batchMessage = new BatchMessage(messageId, exchange, 
                                                    routingKey, message, future);
        
        try {
            messageQueue.put(batchMessage);
            return future;
        } catch (InterruptedException e) {
            future.completeExceptionally(e);
            pendingConfirms.remove(messageId);
            Thread.currentThread().interrupt();
            return future;
        }
    }
    
    /**
     * 启动批量处理器
     */
    public void startBatchProcessor() {
        if (!running) {
            running = true;
            batchExecutor.scheduleWithFixedDelay(this::processBatch, 
                100, 100, TimeUnit.MILLISECONDS);
        }
    }
    
    /**
     * 处理批量消息
     */
    private void processBatch() {
        if (!running) return;
        
        List<BatchMessage> batch = new ArrayList<>();
        
        // 收集一批消息
        messageQueue.drainTo(batch, 100);
        
        if (!batch.isEmpty()) {
            sendBatch(batch);
        }
    }
    
    /**
     * 发送批量
     */
    private void sendBatch(List<BatchMessage> batch) {
        RabbitTemplate template = createRabbitTemplate();
        
        // 设置批量确认回调
        template.setConfirmCallback((correlationData, ack, cause) -> {
            if (correlationData != null) {
                handleBatchConfirm(correlationData.getId(), ack, cause);
            }
        });
        
        // 发送批量消息
        for (BatchMessage message : batch) {
            try {
                CorrelationData correlationData = new CorrelationData(message.messageId);
                template.convertAndSend(message.exchange, message.routingKey, 
                    message.message, correlationData);
                
            } catch (Exception e) {
                message.future.completeExceptionally(e);
                pendingConfirms.remove(message.messageId);
            }
        }
    }
    
    /**
     * 处理批量确认
     */
    private void handleBatchConfirm(String messageId, boolean ack, String cause) {
        CompletableFuture<ConfirmResult> future = pendingConfirms.remove(messageId);
        if (future != null) {
            if (ack) {
                future.complete(new ConfirmResult(true, "ACK", null));
            } else {
                future.complete(new ConfirmResult(false, "NACK", cause));
            }
        }
    }
    
    /**
     * 获取批量统计
     */
    public BatchStats getBatchStats() {
        BatchStats stats = new BatchStats();
        stats.pendingCount = messageQueue.size();
        stats.confirmingCount = pendingConfirms.size();
        return stats;
    }
    
    private RabbitTemplate createRabbitTemplate() {
        CachingConnectionFactory factory = new CachingConnectionFactory();
        factory.setPublisherConfirmType(CachingConnectionFactory.ConfirmType.CORRELATED);
        
        RabbitTemplate template = new RabbitTemplate(factory);
        template.setMandatory(true);
        return template;
    }
    
    /**
     * 停止批量处理器
     */
    @PreDestroy
    public void destroy() {
        running = false;
        batchExecutor.shutdown();
        
        // 清理未处理的消息
        messageQueue.clear();
        pendingConfirms.values().forEach(future -> 
            future.completeExceptionally(new RuntimeException("处理器关闭")));
        pendingConfirms.clear();
    }
    
    /**
     * 批量消息
     */
    private static class BatchMessage {
        final String messageId;

            .register(meterRegistry);

            .description("确认耗时统计")

    public ConfirmContext startConfirm(String messageId, String destination) {

     * 记录确认结果
     */
    public void recordConfirm(ConfirmContext context, boolean ack, String reason) {
        long duration = System.currentTimeMillis() - context.startTime;
        
        // 记录耗时
        confirmTimer.record(duration, TimeUnit.MILLISECONDS);
        confirmDurationSummary.record(duration);
        
        // 更新计数器
        if (ack) {
            ackCounter.increment();
        } else {
            nackCounter.increment();
            
            // 按原因分类
            if ("RETURNED".equals(reason)) {
                returnedCounter.increment();
            } else if ("TIMEOUT".equals(reason)) {
                timeoutCounter.increment();
            }
        }
        
        // 更新目标统计
        ConfirmMetrics metrics = destinationMetrics.computeIfAbsent(
            context.destination, k -> new ConfirmMetrics());
        
        metrics.recordConfirm(ack, duration);
        
        // 记录到日志
        logConfirm(context.messageId, context.destination, ack, reason, duration);
    }
    
    /**
     * 获取确认统计
     */
    public ConfirmStats getConfirmStats(String destination) {
        ConfirmMetrics metrics = destinationMetrics.get(destination);
        if (metrics != null) {
            return metrics.getStats();
        }
        return new ConfirmStats();
    }
    
    /**
     * 健康检查
     */
    public Health confirmHealthCheck() {
        Map<String, Object> details = new HashMap<>();
        
        // 计算总体成功率
        long totalAck = ackCounter.count();
        long totalNack = nackCounter.count();
        long totalReturned = returnedCounter.count();
        long totalTimeout = timeoutCounter.count();
        long total = totalAck + totalNack + totalReturned + totalTimeout;
        
        double successRate = total > 0 ? (double) totalAck / total : 1.0;
        
        details.put("successRate", successRate);
        details.put("totalAck", totalAck);
        details.put("totalNack", totalNack);
        details.put("totalReturned", totalReturned);
        details.put("totalTimeout", totalTimeout);
        
        // 检查是否有连续失败
        boolean hasCriticalFailure = destinationMetrics.values().stream()
            .anyMatch(metrics -> metrics.getConsecutiveFailures() > 10);
        
        if (successRate > 0.95 && !hasCriticalFailure) {
            return Health.up().withDetails(details).build();
        } else {
            return Health.down().withDetails(details).build();
        }
    }

                           boolean ack, String reason, long duration) {
        
        String level = ack ? "INFO" : "ERROR";
        System.out.printf("[%s] 确认结果 - 消息: %s, 目标: %s, 结果: %s, 原因: %s, 耗时: %dms%n",
            level, messageId, destination, ack ? "成功" : "失败", 
            reason != null ? reason : "N/A", duration);
    }

        private final AtomicInteger consecutiveFailures = new AtomicInteger();
        
        synchronized void recordConfirm(boolean ack, long duration) {
            if (ack) {
                ackCount.incrementAndGet();
                totalDuration.addAndGet(duration);
                consecutiveFailures.set(0);
            } else {
                nackCount.incrementAndGet();
                consecutiveFailures.incrementAndGet();

            stats.successRate = stats.total > 0 ?

            stats.consecutiveFailures = consecutiveFailures.get();
            stats.avgDuration = stats.totalAck > 0 ? 
                totalDuration.get() / stats.totalAck : 0;
            return stats;
        }
        
        int getConsecutiveFailures() {
            return consecutiveFailures.get();
        }
    }
    
    /**
     * 确认统计

        public long totalNack = 0;

        public long avgDuration = 0;
    }

@Component
public class SmartConfirmStrategy {
    
    @Autowired
    private ConfirmMonitor confirmMonitor;
    
    private final Map<String, DestinationStats> destinationStats = new ConcurrentHashMap<>();
    private final ScheduledExecutorService statsUpdater = 
        Executors.newSingleThreadScheduledExecutor();
    
    /**
     * 初始化策略
     */
    @PostConstruct
    public void init() {
        // 定期更新统计
        statsUpdater.scheduleAtFixedRate(this::updateDestinationStats, 
            10, 10, TimeUnit.SECONDS);
    }
    
    /**
     * 智能发送
     */
    public <T> T smartSend(RabbitTemplate template, String exchange, 
                          String routingKey, Object message, 
                          SendOperation<T> operation) throws Exception {
        
        String destination = exchange + ":" + routingKey;
        DestinationStats stats = getDestinationStats(destination);
        
        // 根据历史成功率选择发送策略
        SendStrategy strategy = chooseStrategy(stats);
        
        switch (strategy) {
            case IMMEDIATE:
                return sendImmediate(template, exchange, routingKey, message, operation);
                
            case RETRY_ON_FAILURE:
                return sendWithRetry(template, exchange, routingKey, message, operation, 3);
                
            case PERSISTENT_FIRST:
                return sendPersistentFirst(template, exchange, routingKey, message, operation);
                
            case BATCH_OPTIMIZED:
                return sendBatchOptimized(template, exchange, routingKey, message, operation);
                
            default:
                return sendImmediate(template, exchange, routingKey, message, operation);
        }
    }
    
    /**
     * 立即发送
     */
    private <T> T sendImmediate(RabbitTemplate template, String exchange, 
                               String routingKey, Object message, 
                               SendOperation<T> operation) throws Exception {
        
        String messageId = UUID.randomUUID().toString();
        ConfirmMonitor.ConfirmContext context = 
            confirmMonitor.startConfirm(messageId, exchange + ":" + routingKey);
        
        try {
            T result = operation.send(template, exchange, routingKey, message, messageId);
            
            // 假设成功，实际应在确认回调中记录
            confirmMonitor.recordConfirm(context, true, null);
            return result;
            
        } catch (Exception e) {
            confirmMonitor.recordConfirm(context, false, e.getMessage());
            throw e;
        }
    }
    
    /**
     * 带重试发送
     */
    private <T> T sendWithRetry(RabbitTemplate template, String exchange, 
                               String routingKey, Object message, 
                               SendOperation<T> operation, int maxRetries) throws Exception {
        
        Exception lastException = null;
        
        for (int attempt = 1; attempt <= maxRetries; attempt++) {
            try {
                return sendImmediate(template, exchange, routingKey, message, operation);
            } catch (Exception e) {
                lastException = e;
                
                if (attempt < maxRetries) {
                    long delay = calculateRetryDelay(attempt);
                    Thread.sleep(delay);
                }
            }
        }
        
        throw lastException;
    }
    
    /**
     * 持久化优先发送
     */
    private <T> T sendPersistentFirst(RabbitTemplate template, String exchange, 
                                     String routingKey, Object message, 
                                     SendOperation<T> operation) throws Exception {
        
        // 先保存到本地
        saveToLocalStorage(exchange, routingKey, message);
        
        try {
            // 尝试发送
            T result = sendImmediate(template, exchange, routingKey, message, operation);
            
            // 发送成功，清理本地存储
            cleanupLocalStorage(exchange, routingKey, message);
            return result;
            
        } catch (Exception e) {
            // 发送失败，稍后重试
            scheduleRetry(exchange, routingKey, message, e);
            throw e;
        }
    }
    
    /**
     * 批量优化发送
     */
    private <T> T sendBatchOptimized(RabbitTemplate template, String exchange, 
                                    String routingKey, Object message, 
                                    SendOperation<T> operation) throws Exception {
        
        // 收集到批量处理器
        return addToBatchProcessor(exchange, routingKey, message);
    }
    
    /**
     * 选择发送策略
     */
    private SendStrategy chooseStrategy(DestinationStats stats) {
        double successRate = stats.getSuccessRate();
        
        if (successRate > 0.99) {
            return SendStrategy.IMMEDIATE; // 高成功率，立即发送
        } else if (successRate > 0.9) {
            return SendStrategy.RETRY_ON_FAILURE; // 较高成功率，失败重试
        } else if (successRate > 0.7) {
            return SendStrategy.PERSISTENT_FIRST; // 中等成功率，持久化优先
        } else {
            return SendStrategy.BATCH_OPTIMIZED; // 低成功率，批量优化
        }
    }
    
    /**
     * 更新目标统计
     */
    private void updateDestinationStats() {
        destinationStats.keySet().forEach(destination -> {
            ConfirmMonitor.ConfirmStats stats = confirmMonitor.getConfirmStats(destination);
            DestinationStats destStats = getDestinationStats(destination);
            destStats.update(stats);
        });
    }
    
    private DestinationStats getDestinationStats(String destination) {
        return destinationStats.computeIfAbsent(destination, k -> new DestinationStats());
    }
    
    private long calculateRetryDelay(int attempt) {
        return Math.min(1000L * (1L << attempt), 30000L);
    }
    
    private void saveToLocalStorage(String exchange, String routingKey, Object message) {
        // 保存到本地数据库
        System.out.println("保存到本地存储: " + exchange + ":" + routingKey);
    }
    
    private void cleanupLocalStorage(String exchange, String routingKey, Object message) {
        // 清理本地存储
        System.out.println("清理本地存储: " + exchange + ":" + routingKey);
    }
    
    private void scheduleRetry(String exchange, String routingKey, Object message, Exception e) {
        // 安排重试
        System.out.println("安排重试: " + exchange + ":" + routingKey);
    }
    
    private <T> T addToBatchProcessor(String exchange, String routingKey, Object message) {
        // 添加到批量处理器
        System.out.println("添加到批量处理器: " + exchange + ":" + routingKey);
        return null;
    }
    
    /**
     * 发送策略枚举
     */
    public enum SendStrategy {
        IMMEDIATE,          // 立即发送
        RETRY_ON_FAILURE,   // 失败重试
        PERSISTENT_FIRST,   // 持久化优先
        BATCH_OPTIMIZED     // 批量优化
    }
    
    /**
     * 发送操作接口
     */
    public interface SendOperation<T> {
        T send(RabbitTemplate template, String exchange, String routingKey, 
              Object message, String messageId) throws Exception;
    }

     */

        private volatile double successRate = 1.0;
        private volatile long avgDuration = 0;
        private volatile int consecutiveFailures = 0;
        private volatile long lastUpdate = System.currentTimeMillis();
        
        synchronized void update(ConfirmMonitor.ConfirmStats stats) {
            this.successRate = stats.successRate;
            this.avgDuration = stats.avgDuration;
            this.consecutiveFailures = stats.consecutiveFailures;
            this.lastUpdate = System.currentTimeMillis();
        }
        
        double getSuccessRate() {
            return successRate;
        }
        
        long getAvgDuration() {
            return avgDuration;
        }
        
        int getConsecutiveFailures() {
            return consecutiveFailures;

    }

        public String sendReliably(String exchange, String routingKey, Object message)

                    messageIds.add(messageId);
