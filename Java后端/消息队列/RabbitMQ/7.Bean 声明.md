# 基于 Bean 声明队列和交换机（Spring Boot）

在 Spring Boot 中，通过 Bean 声明的方式配置 RabbitMQ 是最佳实践，它提供了**类型安全、易于管理、支持依赖注入**的优势。

## 一、 基础配置方式

### 1. 基本 Bean 声明

```
@Configuration
public class RabbitMQBeanConfig {
    
    // ==================== 1. 交换机定义 ====================
    
    /**
     * Direct 交换机
     */
    @Bean
    public DirectExchange directExchange() {
        return new DirectExchange("direct.exchange", true, false);
        // 参数: 交换机名, 是否持久化, 是否自动删除
    }
    
    /**
     * Topic 交换机
     */
    @Bean
    public TopicExchange topicExchange() {
        return ExchangeBuilder.topicExchange("topic.exchange")
                .durable(true)      // 持久化
                .alternate("alternate.exchange") // 备用交换机
                .build();
    }
    
    /**
     * Fanout 交换机
     */
    @Bean
    public FanoutExchange fanoutExchange() {
        return ExchangeBuilder.fanoutExchange("fanout.exchange")
                .durable(true)
                .build();
    }
    
    /**
     * Headers 交换机
     */
    @Bean
    public HeadersExchange headersExchange() {
        return new HeadersExchange("headers.exchange", true, false);
    }
    
    // ==================== 2. 队列定义 ====================
    
    /**
     * 持久化队列
     */
    @Bean
    public Queue durableQueue() {
        return new Queue("durable.queue", true); // true=持久化
    }
    
    /**
     * 非持久化队列
     */
    @Bean
    public Queue nonDurableQueue() {
        return new Queue("non.durable.queue", false);
    }
    
    /**
     * 排他队列（连接关闭时自动删除）
     */
    @Bean
    public Queue exclusiveQueue() {
        return new Queue("exclusive.queue", true, true, true);
        // 参数: 队列名, 是否持久化, 是否排他, 是否自动删除
    }
    
    /**
     * 自动删除队列
     */
    @Bean
    public Queue autoDeleteQueue() {
        return new Queue("auto.delete.queue", false, false, true);
    }
    
    // ==================== 3. 绑定定义 ====================
    
    /**
     * Direct 绑定
     */
    @Bean
    public Binding directBinding() {
        return BindingBuilder.bind(durableQueue())
                .to(directExchange())
                .with("direct.routing.key");
    }
    
    /**
     * Topic 绑定
     */
    @Bean
    public Binding topicBinding() {
        return BindingBuilder.bind(durableQueue())
                .to(topicExchange())
                .with("topic.*.key");
    }
    
    /**
     * Fanout 绑定
     */
    @Bean
    public Binding fanoutBinding() {
        return BindingBuilder.bind(durableQueue())
                .to(fanoutExchange());
    }
    
    /**
     * Headers 绑定
     */
    @Bean
    public Binding headersBinding() {
        Map<String, Object> headers = new HashMap<>();
        headers.put("x-match", "all"); // all: 全部匹配, any: 任一匹配
        headers.put("type", "user");
        headers.put("action", "create");
        
        return BindingBuilder.bind(durableQueue())
                .to(headersExchange())
                .whereAll(headers)
                .match();
    }
}
```

### 1. 高级队列配置

```
@Configuration
public class AdvancedQueueConfig {
    
    // ==================== 队列高级特性 ====================
    
    /**
     * 带参数的队列
     */
    @Bean
    public Queue advancedQueue() {
        Map<String, Object> args = new HashMap<>();
        
        // 消息TTL（毫秒）
        args.put("x-message-ttl", 60000);
        
        // 队列最大长度
        args.put("x-max-length", 10000);
        
        // 队列最大字节数
        args.put("x-max-length-bytes", 104857600); // 100MB
        
        // 死信交换机
        args.put("x-dead-letter-exchange", "dlx.exchange");
        
        // 死信路由键
        args.put("x-dead-letter-routing-key", "dlx.routing.key");
        
        // 最大优先级
        args.put("x-max-priority", 10);
        
        // 惰性队列（消息存磁盘）
        args.put("x-queue-mode", "lazy");
        
        // 队列过期时间（队列空闲多久后删除）
        args.put("x-expires", 86400000); // 24小时
        
        // 单个消息大小限制
        args.put("x-max-in-memory-length", 1000);
        
        return QueueBuilder.durable("advanced.queue")
                .withArguments(args)
                .build();
    }
    
    /**
     * 延迟队列（通过TTL+死信实现）
     */
    @Bean
    public Queue delayedQueue() {
        Map<String, Object> args = new HashMap<>();
        args.put("x-dead-letter-exchange", "process.exchange");
        args.put("x-dead-letter-routing-key", "process.routing.key");
        args.put("x-message-ttl", 10000); // 10秒延迟
        
        return QueueBuilder.durable("delayed.queue")
                .withArguments(args)
                .build();
    }
    
    /**
     * 优先级队列
     */
    @Bean
    public Queue priorityQueue() {
        Map<String, Object> args = new HashMap<>();
        args.put("x-max-priority", 10);
        
        return QueueBuilder.durable("priority.queue")
                .withArguments(args)
                .build();
    }
    
    /**
     * 单一活跃消费者队列
     */
    @Bean
    public Queue singleActiveConsumerQueue() {
        Map<String, Object> args = new HashMap<>();
        args.put("x-single-active-consumer", true);
        
        return QueueBuilder.durable("single.active.queue")
                .withArguments(args)
                .build();
    }
    
    /**
     * 仲裁队列（高可用队列）
     */
    @Bean
    public Queue quorumQueue() {
        Map<String, Object> args = new HashMap<>();
        args.put("x-queue-type", "quorum");
        
        return QueueBuilder.durable("quorum.queue")
                .withArguments(args)
                .build();
    }
    
    // ==================== 交换机高级特性 ====================
    
    /**
     * 带参数的交换机
     */
    @Bean
    public DirectExchange advancedExchange() {
        Map<String, Object> args = new HashMap<>();
        args.put("alternate-exchange", "ae.exchange"); // 备用交换机
        
        return ExchangeBuilder.directExchange("advanced.exchange")
                .durable(true)
                .withArguments(args)
                .build();
    }
    
    /**
     * 内部交换机
     */
    @Bean
    public DirectExchange internalExchange() {
        return ExchangeBuilder.directExchange("internal.exchange")
                .durable(true)
                .internal()  // 标记为内部交换机
                .build();
    }
    
    /**
     * 延迟消息插件交换机
     */
    @Bean
    public CustomExchange delayedMessageExchange() {
        Map<String, Object> args = new HashMap<>();
        args.put("x-delayed-type", "direct"); // 底层类型
        
        return new CustomExchange("delayed.exchange", 
                "x-delayed-message",  // 使用延迟插件
                true, false, args);
    }
}
```

```
@Configuration
public class ConditionalBeanConfig {
    
    /**
     * 根据属性配置决定是否创建Bean
     */
    @Bean
    @ConditionalOnProperty(name = "rabbitmq.queue.enabled", havingValue = "true")
    public Queue conditionalQueue() {
        return new Queue("conditional.queue", true);
    }
    
    /**
     * 根据类路径存在性决定
     */
    @Bean
    @ConditionalOnClass(name = "com.example.AnotherService")
    public Queue classConditionQueue() {
        return new Queue("class.condition.queue", true);
    }
    
    /**
     * 根据Bean存在性决定
     */
    @Bean
    @ConditionalOnBean(name = "someService")
    public Queue beanConditionQueue() {
        return new Queue("bean.condition.queue", true);
    }
    
    /**
     * 根据表达式决定
     */
    @Bean
    @ConditionalOnExpression("'${rabbitmq.mode}' == 'cluster'")
    public Queue clusterQueue() {
        return QueueBuilder.durable("cluster.queue")
                .build();
    }
    
    /**
     * 多条件组合
     */
    @Bean
    @ConditionalOnProperty(name = "rabbitmq.feature.audit", havingValue = "true")
    @ConditionalOnBean(DataSource.class)
    public Queue auditQueue() {
        return new Queue("audit.queue", true);
    }
}
```

## 四、 动态 Bean 声明

### 1. 基于配置文件的动态声明

```
@Configuration
public class DynamicBeanConfig {
    
    @Autowired
    private Environment environment;
    
    /**
     * 从配置文件读取队列配置
     */
    @Bean
    public Queue configurableQueue() {
        String queueName = environment.getProperty("rabbitmq.queue.name", "default.queue");
        boolean durable = environment.getProperty("rabbitmq.queue.durable", Boolean.class, true);
        boolean exclusive = environment.getProperty("rabbitmq.queue.exclusive", Boolean.class, false);
        boolean autoDelete = environment.getProperty("rabbitmq.queue.autoDelete", Boolean.class, false);
        
        return new Queue(queueName, durable, exclusive, autoDelete);
    }
    
    /**
     * 批量创建队列
     */
    @Bean
    public List<Queue> queueList() {
        List<Queue> queues = new ArrayList<>();
        
        String queueNames = environment.getProperty("rabbitmq.queues", "");
        if (StringUtils.hasText(queueNames)) {
            String[] names = queueNames.split(",");
            for (String name : names) {
                queues.add(new Queue(name.trim(), true));
            }
        }
        
        return queues;
    }
    
    /**
     * 创建带参数的队列Map
     */
    @Bean
    public Map<String, Queue> queueMap() {
        Map<String, Queue> queueMap = new HashMap<>();
        
        // 读取配置
        String queueConfig = environment.getProperty("rabbitmq.queue.config");
        if (StringUtils.hasText(queueConfig)) {
            // 格式: queue1:10000,queue2:30000
            String[] configs = queueConfig.split(",");
            for (String config : configs) {
                String[] parts = config.split(":");
                if (parts.length == 2) {
                    String queueName = parts[0].trim();
                    int ttl = Integer.parseInt(parts[1].trim());
                    
                    Map<String, Object> args = new HashMap<>();
                    args.put("x-message-ttl", ttl);
                    
                    Queue queue = QueueBuilder.durable(queueName)
                            .withArguments(args)
                            .build();
                    
                    queueMap.put(queueName, queue);
                }
            }
        }
        
        return queueMap;
    }
}
```

### 2. 编程式 Bean 注册

```
@Configuration
public class ProgrammaticBeanConfig implements BeanDefinitionRegistryPostProcessor {
    
    @Autowired
    private ApplicationContext applicationContext;
    
    @Override
    public void postProcessBeanDefinitionRegistry(BeanDefinitionRegistry registry) throws BeansException {
        // 动态注册队列Bean
        registerQueueBeans(registry);
        
        // 动态注册交换机Bean
        registerExchangeBeans(registry);
        
        // 动态注册绑定Bean
        registerBindingBeans(registry);
    }
    
    @Override
    public void postProcessApplicationContext(ConfigurableApplicationContext applicationContext) throws BeansException {
        // 应用上下文处理
    }
    
    private void registerQueueBeans(BeanDefinitionRegistry registry) {
        // 从数据库或配置中心读取队列配置
        List<QueueConfig> queueConfigs = loadQueueConfigs();
        
        for (QueueConfig config : queueConfigs) {
            GenericBeanDefinition beanDefinition = new GenericBeanDefinition();
            beanDefinition.setBeanClass(Queue.class);
            
            // 设置构造器参数
            ConstructorArgumentValues args = new ConstructorArgumentValues();
            args.addIndexedArgumentValue(0, config.getName());
            args.addIndexedArgumentValue(1, config.isDurable());
            args.addIndexedArgumentValue(2, config.isExclusive());
            args.addIndexedArgumentValue(3, config.isAutoDelete());
            
            if (config.getArguments() != null && !config.getArguments().isEmpty()) {
                args.addIndexedArgumentValue(4, config.getArguments());
            }
            
            beanDefinition.setConstructorArgumentValues(args);
            
            // 注册Bean
            String beanName = config.getName() + "Queue";
            registry.registerBeanDefinition(beanName, beanDefinition);
        }
    }
    
    private void registerExchangeBeans(BeanDefinitionRegistry registry) {
        // 类似方式注册交换机
    }
    
    private void registerBindingBeans(BeanDefinitionRegistry registry) {
        // 类似方式注册绑定
    }
    
    private List<QueueConfig> loadQueueConfigs() {
        // 从配置文件、数据库等加载配置
        List<QueueConfig> configs = new ArrayList<>();
        
        // 示例配置
        Map<String, Object> args = new HashMap<>();
        args.put("x-message-ttl", 60000);
        
        configs.add(new QueueConfig("dynamic.queue1", true, false, false, args));
        configs.add(new QueueConfig("dynamic.queue2", true, false, false, null));
        
        return configs;
    }
    
    @Data
    public static class QueueConfig {
        private String name;
        private boolean durable;
        private boolean exclusive;
        private boolean autoDelete;
        private Map<String, Object> arguments;
        
        public QueueConfig(String name, boolean durable, boolean exclusive, 
                          boolean autoDelete, Map<String, Object> arguments) {
            this.name = name;
            this.durable = durable;
            this.exclusive = exclusive;
            this.autoDelete = autoDelete;
            this.arguments = arguments;
        }
    }
}
```

```
@Configuration
public class AnnotationBasedConfig {
    
    /**
     * 在 @RabbitListener 中声明队列、交换机、绑定
     * 这种方式更简洁，适合简单的配置
     */
    @RabbitListener(bindings = @QueueBinding(
        value = @Queue(
            value = "annotation.queue",
            durable = "true",
            arguments = {
                @Argument(name = "x-message-ttl", value = "60000", type = "java.lang.Integer"),
                @Argument(name = "x-dead-letter-exchange", value = "dlx.exchange"),
                @Argument(name = "x-max-length", value = "10000", type = "java.lang.Integer")
            }
        ),
        exchange = @Exchange(
            value = "annotation.exchange",
            type = ExchangeTypes.DIRECT,
            durable = "true"
        ),
        key = "annotation.routing.key"
    ))
    public void handleAnnotationMessage(String message) {
        System.out.println("收到注解配置的消息: " + message);
    }
    
    /**
     * 多个绑定
     */
    @RabbitListener(bindings = {
        @QueueBinding(
            value = @Queue("multi.queue1"),
            exchange = @Exchange(value = "multi.exchange", type = ExchangeTypes.TOPIC),
            key = "key1"
        ),
        @QueueBinding(
            value = @Queue("multi.queue2"),
            exchange = @Exchange(value = "multi.exchange", type = ExchangeTypes.TOPIC),
            key = "key2"
        )
    })
    public void handleMultiBinding(String message,
                                  @Header(AmqpHeaders.RECEIVED_ROUTING_KEY) String routingKey) {
        System.out.println("路由键: " + routingKey + ", 消息: " + message);
    }
    
    /**
     * 临时队列
     */
    @RabbitListener(bindings = @QueueBinding(
        value = @Queue,  // 不指定名称，自动生成
        exchange = @Exchange(value = "temp.exchange", type = ExchangeTypes.FANOUT)
    ))
    public void handleTempQueue(String message) {
        System.out.println("临时队列消息: " + message);
    }
    
    /**
     * 带条件的队列
     */
    @RabbitListener(
        bindings = @QueueBinding(
            value = @Queue(
                value = "${rabbitmq.queue.name}",
                durable = "${rabbitmq.queue.durable:true}"
            ),
            exchange = @Exchange(
                value = "${rabbitmq.exchange.name}",
                type = ExchangeTypes.DIRECT
            ),
            key = "${rabbitmq.routing.key}"
        ),
        containerFactory = "customContainerFactory"
    )
    public void handleConfigurableMessage(String message) {
        System.out.println("配置化消息: " + message);
    }
}
```

### 2. 自定义注解

```
// 自定义注解
@Target(ElementType.TYPE)
@Retention(RetentionPolicy.RUNTIME)
@Import(RabbitMQRegistrar.class)
public @interface EnableRabbitModule {
    String module() default "default";
    boolean enableDlx() default true;
}

// 注册器
public class RabbitMQRegistrar implements ImportBeanDefinitionRegistrar {
    
    @Override
    public void registerBeanDefinitions(AnnotationMetadata metadata, 
                                       BeanDefinitionRegistry registry) {
        
        Map<String, Object> attributes = metadata.getAnnotationAttributes(
            EnableRabbitModule.class.getName());
        
        String module = (String) attributes.get("module");
        boolean enableDlx = (Boolean) attributes.get("enableDlx");
        
        // 动态注册Bean
        registerModuleBeans(module, enableDlx, registry);
    }
    
    private void registerModuleBeans(String module, boolean enableDlx, 
                                    BeanDefinitionRegistry registry) {
        // 注册交换机
        registerExchange(module, registry);
        
        // 注册队列
        registerQueues(module, registry);
        
        // 注册绑定
        registerBindings(module, registry);
        
        // 注册死信队列
```

### 1. 配置管理

```
@Configuration
public class BestPracticeConfig {
    
    // 1. 使用常量定义名称
    public static final String ORDER_EXCHANGE = "order.exchange";
    public static final String ORDER_QUEUE = "order.queue";
    public static final String ORDER_ROUTING_KEY = "order.created";
    
    // 2. 集中管理配置
    @Bean
    public RabbitMQConfigProperties rabbitMQConfig() {
        RabbitMQConfigProperties config = new RabbitMQConfigProperties();
        
        // 队列配置
        Map<String, QueueConfig> queues = new HashMap<>();
        queues.put("order", new QueueConfig("order.queue", 60000, 10000));
        queues.put("user", new QueueConfig("user.queue", 30000, 5000));
        config.setQueues(queues);
        
        // 交换机配置
        Map<String, ExchangeConfig> exchanges = new HashMap<>();
        exchanges.put("order", new ExchangeConfig("order.exchange", "direct"));
        exchanges.put("user", new ExchangeConfig("user.exchange", "topic"));
        config.setExchanges(exchanges);
        
        return config;
    }
    
    // 3. 根据配置动态创建Bean
    @Bean
    public List<Declarable> declarables(RabbitMQConfigProperties config) {
        List<Declarable> declarables = new ArrayList<>();
        
        // 创建交换机
        config.getExchanges().forEach((name, exchangeConfig) -> {
            AbstractExchange exchange = createExchange(exchangeConfig);
            declarables.add(exchange);
        });
        
        // 创建队列
        config.getQueues().forEach((name, queueConfig) -> {
            Queue queue = createQueue(queueConfig);
            declarables.add(queue);
        });
        
        // 创建绑定
        config.getBindings().forEach(bindingConfig -> {
            Binding binding = BindingBuilder.bind(new Queue(bindingConfig.getQueue()))
                    .to(new DirectExchange(bindingConfig.getExchange()))
                    .with(bindingConfig.getRoutingKey());
            declarables.add(binding);
        });
        
        return declarables;
    }
    
    private AbstractExchange createExchange(ExchangeConfig config) {
        switch (config.getType().toLowerCase()) {
            case "direct":
                return ExchangeBuilder.directExchange(config.getName())
                        .durable(true)
                        .build();
            case "topic":
                return ExchangeBuilder.topicExchange(config.getName())
                        .durable(true)
                        .build();
            case "fanout":
                return ExchangeBuilder.fanoutExchange(config.getName())
                        .durable(true)
                        .build();
            default:
                throw new IllegalArgumentException("不支持的交换机类型: " + config.getType());
        }
    }
    
    private Queue createQueue(QueueConfig config) {
        Map<String, Object> args = new HashMap<>();
        if (config.getTtl() > 0) {
            args.put("x-message-ttl", config.getTtl());
        }
        if (config.getMaxLength() > 0) {
            args.put("x-max-length", config.getMaxLength());
        }
        
        return QueueBuilder.durable(config.getName())
                .withArguments(args)
                .build();
    }
    
    // 配置类
    @Data
    public static class RabbitMQConfigProperties {
        private Map<String, QueueConfig> queues = new HashMap<>();
        private Map<String, ExchangeConfig> exchanges = new HashMap<>();
        private List<BindingConfig> bindings = new ArrayList<>();
    }
    
    @Data
    public static class QueueConfig {
        private String name;
        private long ttl;
        private int maxLength;
        
        public QueueConfig(String name, long ttl, int maxLength) {
            this.name = name;
            this.ttl = ttl;
            this.maxLength = maxLength;
        }
    }
    
    @Data
    public static class ExchangeConfig {
        private String name;
        private String type;
        
        public ExchangeConfig(String name, String type) {
            this.name = name;
            this.type = type;
        }
    }
    
    @Data
    public static class BindingConfig {
        private String queue;
        private String exchange;
        private String routingKey;
    }
}
```

## 九、 常见问题

### Q1: Bean 声明 vs 注解声明？

- **Bean 声明**：更灵活，支持复杂配置，易于管理
    
- **注解声明**：更简洁，适合简单场景，代码更集中
    

### Q2: 配置应该放在哪里？

- 简单配置：使用 `@RabbitListener`注解
    
- 复杂配置：使用 `@Bean`声明
    
- 动态配置：实现 `BeanDefinitionRegistryPostProcessor`
    

### Q3: 如何避免重复声明？

- 使用 `@ConditionalOnMissingBean`
    
- 统一配置管理
    
- 模块化配置
    

### Q4: 配置的加载顺序？

1. 自动配置
    
2. `@Bean`声明
    
3. `@RabbitListener`注解
    
4. 动态注册
    

基于 Bean 声明的方式提供了最大的灵活性和控制力，适合企业级应用和复杂场景。通过合理的配置组织，可以构建出清晰、可维护、易于扩展的 RabbitMQ 配置体系。
